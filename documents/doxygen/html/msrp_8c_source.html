145 struct msrp_attribute *msrp_lookup(struct msrp_attribute *rattrib)
146 {
147  struct msrp_attribute *attrib;
148  int mac_eq;
149 
150  attrib = MSRP_db->attrib_list;
151  while (NULL != attrib) {
152  if ( (rattrib->type == attrib->type) ||
153  ((MSRP_TALKER_ADV_TYPE == rattrib->type) && (MSRP_TALKER_FAILED_TYPE == attrib->type)) ||
154  ((MSRP_TALKER_FAILED_TYPE == rattrib->type) && (MSRP_TALKER_ADV_TYPE == attrib->type)))
155  {
156  if (MSRP_DOMAIN_TYPE == attrib->type) {
157  if (attrib->attribute.domain.SRclassID ==
158  rattrib->attribute.domain.SRclassID)
159  return attrib;
160  } else {
161  /* compare on the stream ID */
162  mac_eq =
163  memcmp(attrib->attribute.talk_listen.
164  StreamID,
165  rattrib->attribute.talk_listen.
166  StreamID, 8);
167  if (0 == mac_eq)
168  return attrib;
169  }
170  }
171  attrib = attrib->next;
172  }
173  return NULL;

192 int msrp_add(struct msrp_attribute *rattrib)
193 {
194  struct msrp_attribute *attrib;
195  struct msrp_attribute *attrib_tail;
196  int mac_eq;
197 
198  /* XXX do a lookup first to guarantee uniqueness? */
199 
200  attrib_tail = attrib = MSRP_db->attrib_list;
201 
202  while (NULL != attrib) {
203  /* sort list into types, then sorted in order within types */
204  if (rattrib->type == attrib->type) {
205  if (MSRP_DOMAIN_TYPE == attrib->type) {
206  mac_eq = memcmp(&(attrib->attribute.domain),
207  &(rattrib->attribute.domain),
208  sizeof(msrpdu_domain_t));
209 
210  if (mac_eq < 0) {
211  /* possible tail insertion ... */
212 
213  /*
214  * see if there is a another of the same following
215  * not likely since valid values are only 0 and 1
216  */
217  if (NULL != attrib->next) {
218  if (attrib->type ==
219  attrib->next->type) {
220  attrib = attrib->next;
221  continue;
222  }
223  }
224 
225  rattrib->next = attrib->next;
226  rattrib->prev = attrib;
227  attrib->next = rattrib;
228  if (rattrib->next)
229  rattrib->next->prev = rattrib;
230  return 0;
231 
232  } else {
233  /* head insertion ... */
234  rattrib->next = attrib;
235  rattrib->prev = attrib->prev;
236  attrib->prev = rattrib;
237  if (NULL != rattrib->prev)
238  rattrib->prev->next = rattrib;
239  else
240  MSRP_db->attrib_list = rattrib;
241 
242  return 0;
243  }
244  } else {
245  mac_eq =
246  memcmp(attrib->attribute.talk_listen.
247  StreamID,
248  rattrib->attribute.talk_listen.
249  StreamID, 8);
250 
251  if (mac_eq < 0) {
252  /* possible tail insertion ... */
253 
254  if (NULL != attrib->next) {
255  if (attrib->type ==
256  attrib->next->type) {
257  attrib = attrib->next;
258  continue;
259  }
260  }
261 
262  rattrib->next = attrib->next;
263  rattrib->prev = attrib;
264  attrib->next = rattrib;
265  if (rattrib->next)
266  rattrib->next->prev = rattrib;
267  return 0;
268 
269  } else {
270  /* head insertion ... */
271  rattrib->next = attrib;
272  rattrib->prev = attrib->prev;
273  attrib->prev = rattrib;
274  if (NULL != rattrib->prev)
275  rattrib->prev->next = rattrib;
276  else
277  MSRP_db->attrib_list = rattrib;
278 
279  return 0;
280  }
281  }
282  }
283  attrib_tail = attrib;
284  attrib = attrib->next;
285  }
286 
287  /* if we are here we didn't need to stitch in a a sorted entry
288  * so append it onto the tail (if it exists)
289  */
290 
291  if (NULL == attrib_tail) {
292  rattrib->next = NULL;
293  rattrib->prev = NULL;
294  MSRP_db->attrib_list = rattrib;
295  } else {
296  rattrib->next = NULL;
297  rattrib->prev = attrib_tail;
298  attrib_tail->next = rattrib;
299  }
300 
301  return 0;
302 }

304 int msrp_merge(struct msrp_attribute *rattrib)
305 {
306  struct msrp_attribute *talker_attrib;
307  struct msrp_attribute *attrib;
308  int talker_missing;
309 
310  attrib = msrp_lookup(rattrib);
311 
312  if (NULL == attrib)
313  return -1; /* shouldn't happen */
314 
315  /* we always update the last mac address state for diagnostics */
316  memcpy(attrib->registrar.macaddr, rattrib->registrar.macaddr, 6);
317 
318  /* some attribute types carry updated fault condition information
319  * from the bridges indicating either insufficient bandwidth or
320  * excessive latency between the talker and listener. We merge
321  * these status conditions with the existing attribute without
322  * necessarily changing the MRP state (we let the FSM functions
323  * handle these)
324  */
325  switch (attrib->type) {
326  case MSRP_TALKER_ADV_TYPE:
327  case MSRP_TALKER_FAILED_TYPE:
328  /* support merging
329  * TalkerFailed <- TalkerAdvertise and
330  * TalkerAdvertise <- TalkerFailed
331  */
332  if (rattrib->operation == attrib->operation) {
333 
334  attrib->attribute.talk_listen.FailureInformation.FailureCode =
335  rattrib->attribute.talk_listen.FailureInformation.
336  FailureCode;
337  memcpy(attrib->attribute.talk_listen.FailureInformation.
338  BridgeID,
339  rattrib->attribute.talk_listen.FailureInformation.
340  BridgeID, 8);
341 #ifdef ENABLE_MERGED_LATENCY
342  attrib->attribute.talk_listen.AccumulatedLatency =
343  rattrib->attribute.talk_listen.AccumulatedLatency;
344 #endif
345  if (attrib->type != rattrib->type) {
346  attrib->type = rattrib->type;
347  attrib->registrar.mrp_state = MRP_MT_STATE; /* ugly - force a notify */
348  }
349  }
350  break;
351  case MSRP_LISTENER_TYPE:
352  if (attrib->substate != rattrib->substate) {
353  /* If transitioning from AskingFailed to Ready or ReadyFailed, need to
354  * check a matching talkerID has been declared.
355  */
356  if (MSRP_LISTENER_ASKFAILED == attrib->substate) {
357  talker_attrib = msrp_alloc();
358  if (NULL == talker_attrib)
359  break;
360  talker_attrib->type =
361  MSRP_TALKER_ADV_TYPE;
362  memcpy(talker_attrib->attribute.
363  talk_listen.StreamID,
364  attrib->attribute.
365  talk_listen.StreamID, 8);
366  talker_missing = (NULL == msrp_lookup(talker_attrib));
367  free(talker_attrib);
368  if (talker_missing)
369  break;
370  }
371  /*
372  * Listener attributes declared locally have
373  * attrib->operation set to MSRP_OPERATION_DECLARE.
374  *
375  * Listener attributes declared externally have
376  * attrib->operation set to MSRP_OPERATION_REGISTER.
377  *
378  * Support merging the substate only when the directions match.
379  */
380  if (rattrib->operation == attrib->operation) {
381  attrib->substate = rattrib->substate;
382  }
383  attrib->registrar.mrp_state = MRP_MT_STATE; /* ugly - force a notify */
384  }
385  break;
386  case MSRP_DOMAIN_TYPE:
387  /*
388  * If the device has the capability of updating the AVTP packet priority,
389  * then ENABLE_MERGED_DOMAIN_PRIORITY can be defined.
390  */
391  if (
392  (attrib->attribute.domain.SRclassPriority !=
393  rattrib->attribute.domain.SRclassPriority) ||
394  (attrib->attribute.domain.SRclassVID !=
395  rattrib->attribute.domain.SRclassVID)) {
396 #ifdef ENABLE_MERGED_DOMAIN_PRIORITY
397  attrib->attribute.domain.SRclassPriority =
398  rattrib->attribute.domain.SRclassPriority;
399 #endif
400  attrib->attribute.domain.neighborSRclassPriority =
401  rattrib->attribute.domain.SRclassPriority;
402  attrib->attribute.domain.SRclassVID =
403  rattrib->attribute.domain.SRclassVID;
404  attrib->registrar.mrp_state = MRP_MT_STATE; /* ugly - force a notify */
405  }
406  break;
407  default:
408  break;
409  }
410  return 0;
411 }

414 int msrp_event_orig(int event, struct msrp_attribute *rattrib)
415 #else
416 int msrp_event(int event, struct msrp_attribute *rattrib)
417 #endif
418 {
419  struct msrp_attribute *attrib;
420  int count = 0;
421  int rc;
422  int is_talker_attrib = 0;
423  int interested = 1;
424 
425  if (NULL != rattrib) {
426  /* only check talker attributes as listener attributes are filtered by the AVB switch (802.1Q 2011, 35.2.4.4) */
427  if (rattrib->type == MSRP_TALKER_ADV_TYPE || rattrib->type == MSRP_TALKER_FAILED_TYPE ) {
428  is_talker_attrib = 1;
429  }
430  }
431 
432  switch (event) {
433  case MRP_EVENT_LVATIMER:
434  mrp_lvatimer_stop(&(MSRP_db->mrp_db));
435  mrp_jointimer_stop(&(MSRP_db->mrp_db));
436  /* update state */
437  attrib = MSRP_db->attrib_list;
438 
439  while (NULL != attrib) {
440  mrp_applicant_fsm(&(MSRP_db->mrp_db),
441  &(attrib->applicant), MRP_EVENT_TXLA,
442  mrp_registrar_in(&(attrib->registrar)));
443  mrp_registrar_fsm(&(attrib->registrar),
444  &(MSRP_db->mrp_db), MRP_EVENT_TXLA);
445 #if LOG_MSRP
446  msrp_print_debug_info(event, attrib);
447 #endif
448  attrib = attrib->next;
449  }
450 
451  mrp_lvatimer_fsm(&(MSRP_db->mrp_db), MRP_EVENT_LVATIMER);
452 
453  MSRP_db->send_empty_LeaveAll_flag = 1;
454  /*
455  * We are going to emit a PDU here, so call lva FSM with
456  * TX event so as to set the lva.tx=1 flag prior to sending
457  * the PDU. After the pdu is sent, lva.tx=0 and the lva
458  * LSM is back to the passive state.
459  */
460  mrp_lvatimer_fsm(&(MSRP_db->mrp_db), MRP_EVENT_TX);
461  msrp_txpdu();
462  MSRP_db->send_empty_LeaveAll_flag = 0;
463  break;
464  case MRP_EVENT_RLA:
465  mrp_jointimer_start(&(MSRP_db->mrp_db));
466  if (NULL == rattrib)
467  return -1; /* XXX internal fault */
468 
469  /* update state */
470  attrib = MSRP_db->attrib_list;
471 
472  while (NULL != attrib) {
473  if (attrib->type == rattrib->type) {
474  mrp_applicant_fsm(&(MSRP_db->mrp_db),
475  &(attrib->applicant), MRP_EVENT_RLA,
476  mrp_registrar_in(&(attrib->registrar)));
477  mrp_registrar_fsm(&(attrib->registrar),
478  &(MSRP_db->mrp_db), MRP_EVENT_RLA);
479 #if LOG_MSRP
480  msrp_print_debug_info(event, attrib);
481 #endif
482  }
483  attrib = attrib->next;
484  }
485 
486  mrp_lvatimer_fsm(&(MSRP_db->mrp_db), MRP_EVENT_RLA);
487 
488  break;
489  case MRP_EVENT_TX:
490  mrp_jointimer_stop(&(MSRP_db->mrp_db));
491  attrib = MSRP_db->attrib_list;
492 
493  while (NULL != attrib) {
494  mrp_applicant_fsm(&(MSRP_db->mrp_db),
495  &(attrib->applicant), MRP_EVENT_TX,
496  mrp_registrar_in(&(attrib->registrar)));
497 #if LOG_MSRP
498  msrp_print_debug_info(event, attrib);
499 #endif
500  count += mrp_applicant_state_transition_implies_tx(&(attrib->applicant));
501  attrib = attrib->next;
502  }
503 
504  /*
505  * Don't need to call mrp_lvatimer_fsm(..., MRP_EVENT_TX) from
506  * here because the TX event only ever does anything in the
507  * lva FSM when the start is active, and it is only active
508  * momentarily after a LVATIMER event.
509  */
510  msrp_txpdu();
511 
512  /*
513  * Certain state transitions imply we need to request another tx
514  * opportunity.
515  */
516  if (count) {
517  mrp_jointimer_start(&(MSRP_db->mrp_db));
518  }
519 
520  break;
521  case MRP_EVENT_LVTIMER:
522  mrp_lvtimer_stop(&(MSRP_db->mrp_db));
523  attrib = MSRP_db->attrib_list;
524 
525  while (NULL != attrib) {
526  mrp_registrar_fsm(&(attrib->registrar),
527  &(MSRP_db->mrp_db),
528  MRP_EVENT_LVTIMER);
529 
530 #if LOG_MSRP
531  msrp_print_debug_info(event, attrib);
532 #endif
533  attrib = msrp_conditional_reclaim(attrib);
534  }
535  break;
536  case MRP_EVENT_PERIODIC:
537  /*
538  * Periodic timer is ignored in MSRP application.
539  */
540  break;
541  case MRP_EVENT_NEW:
542  case MRP_EVENT_JOIN:
543  case MRP_EVENT_RNEW:
544  case MRP_EVENT_RJOININ:
545  case MRP_EVENT_RJOINMT:
546  case MRP_EVENT_LV:
547  case MRP_EVENT_RIN:
548  case MRP_EVENT_RMT:
549  case MRP_EVENT_RLV:
550  if (NULL == rattrib)
551  return -1; /* XXX internal fault */
552 
553  /* are we interested? Assume yes */
554  interested=1;
555 
556  if( is_talker_attrib && MSRP_db->enable_pruning_of_uninteresting_ids ) {
557  struct msrp_attribute listener_lookup = *rattrib;
558 
559  /*
560  * Having a matching listener declaration in the MSRP database automatically
561  * makes this ID interesting.
562  */
563  listener_lookup.type = MSRP_LISTENER_TYPE;
564 
565  /* check for uninteresting stream IDs */
566  if ((eui64set_find(&MSRP_db->interesting_stream_ids,
567  eui64_read(rattrib->attribute.talk_listen.StreamID)) == 0)
568  &&
569  (msrp_lookup(&listener_lookup) == 0)) {
570 
571  /* Not interested in this talker's stream id */
572  interested = 0;
573  }
574  }
575 
576  if( interested ) {
577  /* only start a join timer if we are interested */
578  mrp_jointimer_start(&(MSRP_db->mrp_db));
579 
580  /* update state */
581  attrib = msrp_lookup(rattrib);
582 
583  if (NULL == attrib) {
584  /* ignore rMT! if attribute does not already exist */
585  if (MRP_EVENT_RMT == event) {
586  free(rattrib);
587  return 0;
588  }
589  msrp_add(rattrib);
590  attrib = rattrib;
591  } else {
592  msrp_merge(rattrib);
593  free(rattrib);
594  }
595 
596  mrp_applicant_fsm(&(MSRP_db->mrp_db), &(attrib->applicant),
597  event,
598  mrp_registrar_in(&(attrib->registrar)));
599 
600  /* remap local requests into registrar events */
601  switch (event) {
602  case MRP_EVENT_NEW:
603  mrp_registrar_fsm(&(attrib->registrar),
604  &(MSRP_db->mrp_db), MRP_EVENT_BEGIN);
605  attrib->registrar.notify = MRP_NOTIFY_NEW;
606  break;
607  case MRP_EVENT_JOIN:
608  if (MRP_IN_STATE == attrib->registrar.mrp_state)
609  mrp_registrar_fsm(&(attrib->registrar),
610  &(MSRP_db->mrp_db),
611  MRP_EVENT_RJOININ);
612  else
613  mrp_registrar_fsm(&(attrib->registrar),
614  &(MSRP_db->mrp_db),
615  MRP_EVENT_RJOINMT);
616  break;
617  case MRP_EVENT_LV:
618  mrp_registrar_fsm(&(attrib->registrar),
619  &(MSRP_db->mrp_db), MRP_EVENT_RLV);
620  break;
621  default:
622  rc = mrp_registrar_fsm(&(attrib->registrar),
623  &(MSRP_db->mrp_db), event);
624  if (-1 == rc) {
625  printf
626  ("MSRP registrar error on attrib->type = %s (%d)\n",
627  msrp_attrib_type_string(attrib->type),
628  attrib->type);
629  }
630  break;
631  }
632  msrp_conditional_reclaim(attrib);
633 #if LOG_MSRP
634  msrp_print_debug_info(event, attrib);
635 #endif
636  } else {
637  /* free this attrib if we are not interested */
638  free(rattrib);
639  }
640 
641  break;
642  default:
643  break;
644  }
645 
646  /*
647  * XXX should honor the MSRP_db->mrp_db.registration and
648  * MSRP_db->mrp_db.participant controls
649  */
650 
651  /* generate local notifications */
652  attrib = MSRP_db->attrib_list;
653 
654  while (NULL != attrib) {
655  if (MRP_NOTIFY_NONE != attrib->registrar.notify) {
656  msrp_send_notifications(attrib,
657  attrib->registrar.notify);
658  attrib->registrar.notify = MRP_NOTIFY_NONE;
659  }
660  attrib = attrib->next;
661  }
662 
663  return 0;
664 }

667 {

710  int attrib_type,
711  uint8_t *pkt_attrib_list_len,
712  uint8_t *pkt_eof)
713 {
714  uint32_t list_len = (uint32_t)pkt_attrib_list_len[0] << 8 |
715  (uint32_t)pkt_attrib_list_len[1];
716  uint32_t attrib_len = (uint32_t)pkt_attrib_list_len[-1];
717 #if !LOG_MSRP
718  (void)(attrib_type);
719 #endif
720

722  * Check for list length < attribute len + header_size (2) + event (1)
723  * Note: this test is more of a sanity test than anything else. It does
724  * not attempt to use number of values to compute 3packed and 4packed byte
725  * count.

728 #if LOG_MSRP

733  attrib_len);

740 #if LOG_MSRP

742  ("MSRP msrp_recv_msg_check_attrib_list_len(%s) FAILED EOF list len %d pkt %p eof %p\n",

746  pkt_eof);
747 #endif
748  return 0;
749  }

752  if ((0 != pkt_attrib_list_len[list_len]) || (0 != pkt_attrib_list_len[list_len + 1])) {
753 #if LOG_MSRP

755  ("MSRP msrp_recv_msg_check_attrib_list_len(%s) FAILED ENDMARK list len %d pkt %p\n",
756  msrp_attrib_type_string(attrib_type),

758  pkt_attrib_list_len,
