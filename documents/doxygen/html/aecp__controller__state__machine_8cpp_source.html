71 void aecp_controller_state_machine::common_hdr_init(int message_type, struct jdksavdecc_frame * cmd_frame, uint64_t target_entity_id, uint32_t cd_len)
72 {
73  struct jdksavdecc_aecpdu_common_control_header aecpdu_common_ctrl_hdr;
74  ssize_t aecpdu_common_ctrl_hdr_returned;
75 
76  /************************************** 1722 Protocol Header **************************************/
77  aecpdu_common_ctrl_hdr.cd = 1;
78  aecpdu_common_ctrl_hdr.subtype = JDKSAVDECC_SUBTYPE_AECP;
79  aecpdu_common_ctrl_hdr.sv = 0;
80  aecpdu_common_ctrl_hdr.version = 0;
81  aecpdu_common_ctrl_hdr.message_type = (uint8_t)message_type;
82  aecpdu_common_ctrl_hdr.status = JDKSAVDECC_AEM_STATUS_SUCCESS;
83  aecpdu_common_ctrl_hdr.control_data_length = (uint16_t)cd_len;
84  jdksavdecc_uint64_write(target_entity_id, &aecpdu_common_ctrl_hdr.target_entity_id, 0, sizeof(uint64_t));
85 
86  /********************** Fill frame payload with AECP Common Control Header information *********************/
87  aecpdu_common_ctrl_hdr_returned = jdksavdecc_aecpdu_common_control_header_write(&aecpdu_common_ctrl_hdr,
88  cmd_frame->payload,
89  ETHER_HDR_SIZE, // Offset to write the field to
90  sizeof(cmd_frame->payload));
91 
92  if (aecpdu_common_ctrl_hdr_returned < 0)
93  {
94  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "common_hdr_init error");
95  assert(aecpdu_common_ctrl_hdr_returned >= 0);
96  }
97 }

99 int aecp_controller_state_machine::tx_cmd(void * notification_id, uint32_t notification_flag, struct jdksavdecc_frame * cmd_frame, bool resend)
100 {
101  int send_frame_returned;
102 
103  if (!resend)
104  {
105  uint16_t current_seq_id = aecp_seq_id;
106 
107  jdksavdecc_aecpdu_common_set_sequence_id(aecp_seq_id++, cmd_frame->payload, ETHER_HDR_SIZE);
108  inflight in_flight = inflight(cmd_frame,
109  current_seq_id,
110  notification_id,
111  notification_flag,
112  AVDECC_MSG_TIMEOUT_MS);
113  in_flight.start_timer();
114  inflight_cmds.push_back(in_flight);
115  }
116  else
117  {
118  uint16_t resend_with_seq_id = jdksavdecc_aecpdu_common_get_sequence_id(cmd_frame->payload, ETHER_HDR_SIZE);
119  std::vector<inflight>::iterator j =
120  std::find_if(inflight_cmds.begin(), inflight_cmds.end(), SeqIdComp(resend_with_seq_id));
121 
122  if (j != inflight_cmds.end()) // found?
123  {
124  j->start_timer();
125  }
126  }
127 
128  send_frame_returned = net_interface_ref->send_frame(cmd_frame->payload, cmd_frame->length);
129  if (send_frame_returned < 0)
130  {
131  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "netif_send_frame error");
132  assert(send_frame_returned >= 0);
133  }
134 
135  callback(notification_id, notification_flag, cmd_frame->payload);
136 
137  return 0;
138 }

185 void aecp_controller_state_machine::state_timeout(uint32_t inflight_cmd_index)
186 {
187  struct jdksavdecc_frame frame = inflight_cmds.at(inflight_cmd_index).frame();
188  bool is_retried = inflight_cmds.at(inflight_cmd_index).retried();
189  uint32_t notification_flag = inflight_cmds.at(inflight_cmd_index).notification_flag();
190 
191  if (is_retried)
192  {
193  jdksavdecc_eui64 id = jdksavdecc_common_control_header_get_stream_id(frame.payload, ETHER_HDR_SIZE);
194  uint16_t cmd_type = jdksavdecc_aecpdu_aem_get_command_type(frame.payload, ETHER_HDR_SIZE);
195  cmd_type &= 0x7FFF;
196  uint16_t desc_type = jdksavdecc_aem_command_read_descriptor_get_descriptor_type(frame.payload, ETHER_HDR_SIZE);
197  uint16_t desc_index = jdksavdecc_aem_command_read_descriptor_get_descriptor_index(frame.payload, ETHER_HDR_SIZE);
198 
199  notification_imp_ref->post_notification_msg(COMMAND_TIMEOUT,
200  jdksavdecc_uint64_get(&id, 0),
201  cmd_type,
202  desc_type,
203  desc_index,
204  UINT_MAX,
205  inflight_cmds.at(inflight_cmd_index).cmd_notification_id);
206 
207  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR,
208  "Command Timeout, 0x%llx, %s, %s, %d, %d",
209  jdksavdecc_uint64_get(&id, 0),
210  utility::aem_cmd_value_to_name(cmd_type),
211  utility::aem_desc_value_to_name(desc_type),
212  desc_index,
213  inflight_cmds.at(inflight_cmd_index).cmd_seq_id);
214 
215  inflight_cmds.erase(inflight_cmds.begin() + inflight_cmd_index);
216  }
217  else
218  {
219  log_imp_ref->post_log_msg(LOGGING_LEVEL_DEBUG,
220  "Resend the command with sequence id = %d",
221  inflight_cmds.at(inflight_cmd_index).cmd_seq_id);
222 
223  tx_cmd(inflight_cmds.at(inflight_cmd_index).cmd_notification_id,
224  notification_flag,
225  &frame,
226  true);
227  }

242 {
243  switch (msg_type)
244  {
245  case JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_RESPONSE:
246  case JDKSAVDECC_AECP_MESSAGE_TYPE_ADDRESS_ACCESS_RESPONSE: // Fallthrough intentional
247  if (u_field)

258  return -1;
259  }
260 
261  return 0;

317 int aecp_controller_state_machine::callback(void * notification_id, uint32_t notification_flag, uint8_t * frame)
318 {

320 
321  uint16_t cmd_type = jdksavdecc_aecpdu_aem_get_command_type(frame, ETHER_HDR_SIZE);
322  uint32_t status = jdksavdecc_common_control_header_get_status(frame, ETHER_HDR_SIZE);
323  uint16_t desc_type = 0;
324  uint16_t desc_index = 0;
325  bool is_unsolicited = cmd_type >> 15 & 0x01;
326  cmd_type &= 0x7FFF;
327 
328  switch (cmd_type)
329  {
330  case JDKSAVDECC_AEM_COMMAND_ACQUIRE_ENTITY:

442  default:

444  break;
445  }
446 
447  jdksavdecc_eui64 id = jdksavdecc_common_control_header_get_stream_id(frame, ETHER_HDR_SIZE);
448  if ((notification_flag == CMD_WITH_NOTIFICATION) &&
449  ((msg_type == JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_RESPONSE) ||
450  (msg_type == JDKSAVDECC_AECP_MESSAGE_TYPE_ADDRESS_ACCESS_RESPONSE)))
451  {

453  jdksavdecc_uint64_get(&id, 0),
454  cmd_type,
455  desc_type,
456  desc_index,
457  status,
458  notification_id);
459 
460  if (status != AEM_STATUS_SUCCESS)
461  {
462  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR,
463  "RESPONSE_RECEIVED, 0x%llx, %s, %s, %d, %d, %s",
464  jdksavdecc_uint64_get(&id, 0),
465  utility::aem_cmd_value_to_name(cmd_type),
466  utility::aem_desc_value_to_name(desc_type),
467  desc_index,

469  utility::aem_cmd_status_value_to_name(status));
470  }
471  }
472  else if (((notification_flag == CMD_WITH_NOTIFICATION) || (notification_flag == CMD_WITHOUT_NOTIFICATION)) &&
473  ((msg_type == JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_COMMAND) || (msg_type == JDKSAVDECC_AECP_MESSAGE_TYPE_ADDRESS_ACCESS_COMMAND)))
474  {
475  log_imp_ref->post_log_msg(LOGGING_LEVEL_DEBUG,
476  "COMMAND_SENT, 0x%llx, %s, %s, %d, %d",
