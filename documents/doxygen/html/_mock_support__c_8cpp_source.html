501 static MockValue_c getMockValueCFromNamedValue(const MockNamedValue& namedValue)
502 {
503  MockValue_c returnValue;
504  if (SimpleString::StrCmp(namedValue.getType().asCharString(), "bool") == 0) {
505  returnValue.type = MOCKVALUETYPE_BOOL;
506  returnValue.value.boolValue = namedValue.getBoolValue() ? 1 : 0;
507  }
508  else if (SimpleString::StrCmp(namedValue.getType().asCharString(), "int") == 0) {
509  returnValue.type = MOCKVALUETYPE_INTEGER;
510  returnValue.value.intValue = namedValue.getIntValue();
511  }
512  else if (SimpleString::StrCmp(namedValue.getType().asCharString(), "unsigned int") == 0) {
513  returnValue.type = MOCKVALUETYPE_UNSIGNED_INTEGER;
514  returnValue.value.unsignedIntValue = namedValue.getUnsignedIntValue();
515  }
516  else if (SimpleString::StrCmp(namedValue.getType().asCharString(), "long int") == 0) {
517  returnValue.type = MOCKVALUETYPE_LONG_INTEGER;
518  returnValue.value.longIntValue = namedValue.getLongIntValue();
519  }
520  else if (SimpleString::StrCmp(namedValue.getType().asCharString(), "unsigned long int") == 0) {
521  returnValue.type = MOCKVALUETYPE_UNSIGNED_LONG_INTEGER;
522  returnValue.value.unsignedLongIntValue = namedValue.getUnsignedLongIntValue();
523  }
524  else if (SimpleString::StrCmp(namedValue.getType().asCharString(), "double") == 0) {
525  returnValue.type = MOCKVALUETYPE_DOUBLE;
526  returnValue.value.doubleValue = namedValue.getDoubleValue();
527  }
528  else if (SimpleString::StrCmp(namedValue.getType().asCharString(), "const char*") == 0) {
529  returnValue.type = MOCKVALUETYPE_STRING;
530  returnValue.value.stringValue = namedValue.getStringValue();
531  }
532  else if (SimpleString::StrCmp(namedValue.getType().asCharString(), "void*") == 0) {
533  returnValue.type = MOCKVALUETYPE_POINTER;
534  returnValue.value.pointerValue = namedValue.getPointerValue();
535  }
536  else if (SimpleString::StrCmp(namedValue.getType().asCharString(), "const void*") == 0) {
537  returnValue.type = MOCKVALUETYPE_CONST_POINTER;
538  returnValue.value.constPointerValue = namedValue.getConstPointerValue();
539  }
540  else if (SimpleString::StrCmp(namedValue.getType().asCharString(), "void (*)()") == 0) {
541  returnValue.type = MOCKVALUETYPE_FUNCTIONPOINTER;
542  returnValue.value.functionPointerValue = (void (*)()) namedValue.getFunctionPointerValue();
543  }
544  else if (SimpleString::StrCmp(namedValue.getType().asCharString(), "const unsigned char*") == 0) {
545  returnValue.type = MOCKVALUETYPE_MEMORYBUFFER;
546  returnValue.value.memoryBufferValue = namedValue.getMemoryBuffer();
547  }
548  else {
549  returnValue.type = MOCKVALUETYPE_OBJECT;
550  returnValue.value.objectValue = namedValue.getObjectPointer();
551  }
552  return returnValue;
553 }
