152 int end_station_imp::send_read_desc_cmd_with_flag(void * notification_id, uint32_t notification_flag, uint16_t desc_type, uint16_t desc_index)
153 {
154  struct jdksavdecc_frame cmd_frame;
155  struct jdksavdecc_aem_command_read_descriptor aem_command_read_desc;
156  memset(&aem_command_read_desc, 0, sizeof(aem_command_read_desc));
157 
158  /***************************** AECP Common Data ****************************/
159  aem_command_read_desc.aem_header.aecpdu_header.controller_entity_id = adp_ref->get_controller_entity_id();
160  // Fill aem_command_read_desc.sequence_id in AEM Controller State Machine
161  aem_command_read_desc.aem_header.command_type = JDKSAVDECC_AEM_COMMAND_READ_DESCRIPTOR;
162 
163  /******************************************************** AECP Message Specific Data ********************************************************/
164  aem_command_read_desc.configuration_index = (desc_type == JDKSAVDECC_DESCRIPTOR_ENTITY || desc_type == JDKSAVDECC_DESCRIPTOR_CONFIGURATION) ? 0 : entity_desc_vec.at(current_entity_desc)->current_configuration();
165  aem_command_read_desc.descriptor_type = desc_type;
166  aem_command_read_desc.descriptor_index = desc_index;
167 
168  /************************** Fill frame payload with AECP data and send the frame *************************/
169  aecp_controller_state_machine_ref->ether_frame_init(end_station_mac, &cmd_frame,
170  ETHER_HDR_SIZE + JDKSAVDECC_AEM_COMMAND_READ_DESCRIPTOR_COMMAND_LEN);
171  ssize_t write_return_val = jdksavdecc_aem_command_read_descriptor_write(&aem_command_read_desc,
172  cmd_frame.payload,
173  ETHER_HDR_SIZE,
174  sizeof(cmd_frame.payload));
175 
176  if (write_return_val < 0)
177  {
178  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "aem_cmd_read_desc_write error");
179  return -1;
180  }
181 
182  aecp_controller_state_machine_ref->common_hdr_init(JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_COMMAND,
183  &cmd_frame,
184  end_station_entity_id,
185  JDKSAVDECC_AEM_COMMAND_READ_DESCRIPTOR_COMMAND_LEN -
186  JDKSAVDECC_COMMON_CONTROL_HEADER_LEN);
187  system_queue_tx(notification_id, notification_flag, cmd_frame.payload, cmd_frame.length);
188  return 0;
189 }

191 int end_station_imp::proc_read_desc_resp(void *& notification_id, const uint8_t * frame, size_t frame_len, int & status)
192 {
193  const int read_desc_offset = ETHER_HDR_SIZE + JDKSAVDECC_AEM_COMMAND_READ_DESCRIPTOR_RESPONSE_LEN;
194  struct jdksavdecc_frame cmd_frame;
195  struct jdksavdecc_aem_command_read_descriptor_response aem_cmd_read_desc_resp;
196  ssize_t aem_cmd_read_desc_resp_returned;
197  uint32_t msg_type;
198  bool u_field;
199  uint16_t desc_type;
200  configuration_descriptor_imp * config_desc_imp_ref = NULL;
201  memset(&aem_cmd_read_desc_resp, 0, sizeof(aem_cmd_read_desc_resp));
202 
203  if (entity_desc_vec.size() >= 1 && entity_desc_vec.at(current_entity_desc)->config_desc_count() >= 1)
204  {
205  config_desc_imp_ref = dynamic_cast<configuration_descriptor_imp *>(entity_desc_vec.at(current_entity_desc)->get_config_desc_by_index(current_config_desc));
206 
207  if (!config_desc_imp_ref)
208  {
209  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "Dynamic cast from base configuration_descriptor to derived configuration_descriptor_imp error");
210  }
211  }
212 
213  memcpy(cmd_frame.payload, frame, frame_len);
214  aem_cmd_read_desc_resp_returned = jdksavdecc_aem_command_read_descriptor_response_read(&aem_cmd_read_desc_resp,
215  frame,
216  ETHER_HDR_SIZE,
217  frame_len);
218 
219  if (aem_cmd_read_desc_resp_returned < 0)
220  {
221  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "aem_cmd_read_desc_res_read error");
222  return -1;
223  }
224 
225  msg_type = aem_cmd_read_desc_resp.aem_header.aecpdu_header.header.message_type;
226  status = aem_cmd_read_desc_resp.aem_header.aecpdu_header.header.status;
227  u_field = aem_cmd_read_desc_resp.aem_header.command_type >> 15 & 0x01; // u_field = the msb of the uint16_t command_type
228  desc_type = jdksavdecc_uint16_get(frame, ETHER_HDR_SIZE + JDKSAVDECC_AEM_COMMAND_READ_DESCRIPTOR_RESPONSE_OFFSET_DESCRIPTOR);
229 
230  aecp_controller_state_machine_ref->update_inflight_for_rcvd_resp(notification_id, msg_type, u_field, &cmd_frame);
231 
232  bool store_descriptor = false;
233  if (status == avdecc_lib::AEM_STATUS_SUCCESS)
234  {
235  switch (desc_type)
236  {
237  case JDKSAVDECC_DESCRIPTOR_ENTITY:
238  store_descriptor = true;
239  break;
240 
241  case JDKSAVDECC_DESCRIPTOR_CONFIGURATION:
242  if (entity_desc_vec.size() == 1 && entity_desc_vec.at(current_entity_desc)->config_desc_count() == 0)
243  {
244  store_descriptor = true;
245  }
246  break;
247 
248  default:
249  if (entity_desc_vec.size() == 1 && entity_desc_vec.at(current_entity_desc)->config_desc_count() >= 1)
250  {
251  store_descriptor = true;
252  }
253  break;
254  }
255  }
256 
257  if (store_descriptor)
258  {
259  try
260  {
261  switch (desc_type)
262  {
263  case JDKSAVDECC_DESCRIPTOR_ENTITY:
264  if (entity_desc_vec.size() == 0)
265  {
266  entity_desc_vec.push_back(new entity_descriptor_imp(this, frame, read_desc_offset, frame_len));
267  current_config_desc = entity_desc_vec.at(current_entity_desc)->current_configuration();
268  }
269  break;
270 
271  case JDKSAVDECC_DESCRIPTOR_CONFIGURATION:
272  entity_desc_vec.at(current_entity_desc)->store_config_desc(this, frame, read_desc_offset, frame_len);
273  break;
274 
275  case JDKSAVDECC_DESCRIPTOR_AUDIO_UNIT:
276  if (config_desc_imp_ref != nullptr)
277  config_desc_imp_ref->store_audio_unit_desc(this, frame, read_desc_offset, frame_len);
278  break;
279 
280  case JDKSAVDECC_DESCRIPTOR_STREAM_INPUT:
281  if (config_desc_imp_ref != nullptr)
282  config_desc_imp_ref->store_stream_input_desc(this, frame, read_desc_offset, frame_len);
283  break;
284 
285  case JDKSAVDECC_DESCRIPTOR_STREAM_OUTPUT:
286  if (config_desc_imp_ref != nullptr)
287  config_desc_imp_ref->store_stream_output_desc(this, frame, read_desc_offset, frame_len);
288  break;
289 
290  case JDKSAVDECC_DESCRIPTOR_JACK_INPUT:
291  if (config_desc_imp_ref != nullptr)
292  config_desc_imp_ref->store_jack_input_desc(this, frame, read_desc_offset, frame_len);
293  break;
294 
295  case JDKSAVDECC_DESCRIPTOR_JACK_OUTPUT:
296  if (config_desc_imp_ref != nullptr)
297  config_desc_imp_ref->store_jack_output_desc(this, frame, read_desc_offset, frame_len);
298  break;
299 
300  case JDKSAVDECC_DESCRIPTOR_AVB_INTERFACE:
301  if (config_desc_imp_ref != nullptr)
302  config_desc_imp_ref->store_avb_interface_desc(this, frame, read_desc_offset, frame_len);
303  break;
304 
305  case JDKSAVDECC_DESCRIPTOR_CLOCK_SOURCE:
306  if (config_desc_imp_ref != nullptr)
307  config_desc_imp_ref->store_clock_source_desc(this, frame, read_desc_offset, frame_len);
308  break;
309 
310  case JDKSAVDECC_DESCRIPTOR_MEMORY_OBJECT:
311  if (config_desc_imp_ref != nullptr)
312  config_desc_imp_ref->store_memory_object_desc(this, frame, read_desc_offset, frame_len);
313  break;
314 
315  case JDKSAVDECC_DESCRIPTOR_LOCALE:
316  if (config_desc_imp_ref != nullptr)
317  config_desc_imp_ref->store_locale_desc(this, frame, read_desc_offset, frame_len);
318  break;
319 
320  case JDKSAVDECC_DESCRIPTOR_STRINGS:
321  if (config_desc_imp_ref != nullptr)
322  config_desc_imp_ref->store_strings_desc(this, frame, read_desc_offset, frame_len);
323  break;
324 
325  case JDKSAVDECC_DESCRIPTOR_STREAM_PORT_INPUT:
326  if (config_desc_imp_ref != nullptr)
327  config_desc_imp_ref->store_stream_port_input_desc(this, frame, read_desc_offset, frame_len);
328  break;
329 
330  case JDKSAVDECC_DESCRIPTOR_STREAM_PORT_OUTPUT:
331  if (config_desc_imp_ref != nullptr)
332  config_desc_imp_ref->store_stream_port_output_desc(this, frame, read_desc_offset, frame_len);
333  break;
334 
335  case JDKSAVDECC_DESCRIPTOR_AUDIO_CLUSTER:
336  if (config_desc_imp_ref != nullptr)
337  config_desc_imp_ref->store_audio_cluster_desc(this, frame, read_desc_offset, frame_len);
338  break;
339 
340  case JDKSAVDECC_DESCRIPTOR_AUDIO_MAP:
341  if (config_desc_imp_ref != nullptr)
342  config_desc_imp_ref->store_audio_map_desc(this, frame, read_desc_offset, frame_len);
343  break;
344 
345  case JDKSAVDECC_DESCRIPTOR_CLOCK_DOMAIN:
346  if (config_desc_imp_ref != nullptr)
347  config_desc_imp_ref->store_clock_domain_desc(this, frame, read_desc_offset, frame_len);
348  break;
349 
350  case JDKSAVDECC_DESCRIPTOR_CONTROL:
351  if (config_desc_imp_ref != nullptr)
352  config_desc_imp_ref->store_control_desc(this, frame, read_desc_offset, frame_len);
353  break;
354 
355  case JDKSAVDECC_DESCRIPTOR_EXTERNAL_PORT_INPUT:
356  if (config_desc_imp_ref != nullptr)
357  config_desc_imp_ref->store_external_port_input_desc(this, frame, read_desc_offset, frame_len);
358  break;
359 
360  case JDKSAVDECC_DESCRIPTOR_EXTERNAL_PORT_OUTPUT:
361  if (config_desc_imp_ref != nullptr)
362  config_desc_imp_ref->store_external_port_output_desc(this, frame, read_desc_offset, frame_len);
363  break;
364 
365  default:
366  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "Descriptor %s is not yet implemented in avdecc-lib.", utility::aem_desc_value_to_name(desc_type));
367  break;
368  }
369  }
370  catch (const avdecc_read_descriptor_error & ia)
371  {
372  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "0x%llx, catch %s", entity_id(), ia.what());
373  }
374 
375  configuration_descriptor * cd = NULL;
376 
377  if ((entity_desc_vec.size() >= 1) && (entity_desc_vec.at(current_entity_desc)->config_desc_count() >= 1))
378  {
379  cd = entity_desc_vec.at(current_entity_desc)->get_config_desc_by_index(current_config_desc);
380  }
381  background_read_deduce_next(cd, desc_type, (void *)frame, read_desc_offset);
382  }
383  background_read_update_inflight(desc_type, (void *)frame, read_desc_offset);
384  background_read_submit_pending();
385 
386  if ((entity_desc_vec.size() >= 1) && (entity_desc_vec.at(current_entity_desc)->config_desc_count() >= 1))
387  {
388  if (m_backbround_read_inflight.empty() && m_backbround_read_pending.empty())
389  {
390  notification_imp_ref->post_notification_msg(END_STATION_READ_COMPLETED, end_station_entity_id, 0, 0, 0, 0, NULL);
391  }
392  }
393 
394  return 0;
395 }

445 void end_station_imp::background_read_submit_pending(void)
446 {
447  // if there are no pending inflight, but the background read list is not
448  // empty submit the next set of read operations
449  if (m_backbround_read_inflight.empty() && !m_backbround_read_pending.empty())
450  {
451  background_read_request * b_first = m_backbround_read_pending.front();
452  m_backbround_read_pending.pop_front();
453  log_imp_ref->post_log_msg(LOGGING_LEVEL_DEBUG, "Background read of %s index %d", utility::aem_desc_value_to_name(b_first->m_type), b_first->m_index);
454  read_desc_init(b_first->m_type, b_first->m_index);
455  b_first->m_timer.start(750); // 750 ms timeout (1722.1 timeout is 250ms)
456  m_backbround_read_inflight.push_back(b_first);
457 
458  if (!m_backbround_read_pending.empty())
459  {
460  background_read_request * b_next = m_backbround_read_pending.front();
461  while (b_next->m_type == b_first->m_type)
462  {
463  m_backbround_read_pending.pop_front();
464  log_imp_ref->post_log_msg(LOGGING_LEVEL_DEBUG, "Background read of %s index %d", utility::aem_desc_value_to_name(b_next->m_type), b_next->m_index);
465  read_desc_init(b_next->m_type, b_next->m_index);
466  b_next->m_timer.start(750); // 750 ms timeout (1722.1 timeout is 250ms)
467  m_backbround_read_inflight.push_back(b_next);
468  if (m_backbround_read_pending.empty())
469  {
470  break;
471  }
472  else
473  {
474  b_next = m_backbround_read_pending.front();
475  }
476  }
477  }
478  }
