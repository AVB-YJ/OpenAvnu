84 // Returns 1 if the queue was not empty before adding the new command,
85 // 0 if the queue was empty before adding the new command,
86 // or -1 if an error occurred.
87 static int addCommandToQueue(openavb_aecp_AEMCommandResponse_t *command)
88 {
89  int returnVal;
90 
91  if (!s_commandQueue) { return -1; }
92  if (!command) { return -1; }
93 
94  AECP_QUEUE_LOCK();
95  // Determine if the queue has something in it.
96  returnVal = (openavbListFirst(s_commandQueue) != NULL ? 1 : 0);
97 
98  // Add the command to the end of the linked list.
99  if (openavbListAdd(s_commandQueue, command) == NULL) {
100  returnVal = -1;
101  }
102  AECP_QUEUE_UNLOCK();
103  return (returnVal);
104 }

138  pCommand->headers.status = OPENAVB_AEM_COMMAND_STATUS_NOT_SUPPORTED;
139  }
140  else {
141  openavb_avdecc_entity_model_t *pAem = openavbAemGetModel();
142  if (!pAem) {
143  AVB_TRACE_EXIT(AVB_TRACE_AECP);
144  return;
145  }
146 
147  AEM_LOCK();
148  if (pCommand->entityModelPdu.command_data.acquireEntityCmd.flags & OPENAVB_AEM_ACQUIRE_ENTITY_COMMAND_FLAG_RELEASE) {
149  // AVDECC_TODO - need to add mutex for the entity model
150  AVB_LOG_DEBUG("Entity Released");
151  pAem->entityAcquired = FALSE;
152  }
153  else {
154  if (pAem->entityAcquired) {
155  if (memcmp(pAem->acquiredControllerId, pCommand->commonPdu.controller_entity_id, sizeof(pAem->acquiredControllerId))) {
156  // AVDECC_TODO - Per 7.4.1 ACQUIRE_ENTITY Command. Need to add the CONTROLLER_AVAILABLE command and response to ensure the controller is still alive.
157  // For now just respond acquired by a different controller.
158  pCommand->headers.status = OPENAVB_AEM_COMMAND_STATUS_ENTITY_ACQUIRED;
159  memcpy(pCommand->entityModelPdu.command_data.acquireEntityRsp.owner_id, pAem->acquiredControllerId, sizeof(pCommand->commonPdu.controller_entity_id));
160  }
161  else {
162  // Entity was already acquired by this controller, so indicate a successful acquisition.
163  pCommand->headers.status = OPENAVB_AEM_COMMAND_STATUS_SUCCESS;
164  }
165  }
166  else {
167  pAem->entityAcquired = TRUE;
168  memcpy(pAem->acquiredControllerId, pCommand->commonPdu.controller_entity_id, sizeof(pAem->acquiredControllerId));
169  memcpy(pCommand->entityModelPdu.command_data.acquireEntityRsp.owner_id, pCommand->commonPdu.controller_entity_id, sizeof(pCommand->commonPdu.controller_entity_id));
170  AVB_LOGF_DEBUG("Entity Acquired by " ENTITYID_FORMAT, pCommand->entityModelPdu.command_data.acquireEntityRsp.owner_id);
171  }
172  }
173  AEM_UNLOCK();
174  }
175 
176  AVB_TRACE_EXIT(AVB_TRACE_AECP);
177 }

198 // Returns TRUE is current rcvdCommand controlID matches the AEM acquire or lock controller id
199 bool processCommandCheckRestriction_CorrectController()
200 {
201  AVB_TRACE_ENTRY(AVB_TRACE_AECP);
202  bool bResult = TRUE;
203 
204  openavb_aecp_AEMCommandResponse_t *pCommand = openavbAecpSMEntityModelEntityVars.rcvdCommand;
205  if (!pCommand) {
206  AVB_LOG_ERROR("processCommandCheckRestriction_CorrectController called without command");
207  AVB_TRACE_EXIT(AVB_TRACE_AECP);
208  return bResult;
209  }
210 
211  openavb_avdecc_entity_model_t *pAem = openavbAemGetModel();
212  if (pAem) {
213  AEM_LOCK();
214  if (pAem->entityAcquired) {
215  if (memcmp(pAem->acquiredControllerId, pCommand->commonPdu.controller_entity_id, sizeof(pAem->acquiredControllerId)) == 0) {
216  bResult = TRUE;
217  }
218  else {
219  AVB_LOGF_DEBUG("Access denied to " ENTITYID_FORMAT " as " ENTITYID_FORMAT " already acquired it", pCommand->commonPdu.controller_entity_id, pAem->acquiredControllerId);
220  bResult = FALSE;
221  }
222  }
223  else if (pAem->entityLocked) {
224  if (memcmp(pAem->lockedControllerId, pCommand->commonPdu.controller_entity_id, sizeof(pAem->lockedControllerId)) == 0) {
225  bResult = TRUE;
226  }
227  else {
228  AVB_LOGF_DEBUG("Access denied to " ENTITYID_FORMAT " as " ENTITYID_FORMAT " already locked it", pCommand->commonPdu.controller_entity_id, pAem->lockedControllerId);
229  bResult = FALSE;
230  }
231  }
232  AEM_UNLOCK();
233  }
234 
235  AVB_TRACE_EXIT(AVB_TRACE_AECP);
236  return bResult;
237 }

276 
277  // Set to Not Implemented. Will be overridden by commands that are implemented.
278  pCommand->headers.status = OPENAVB_AEM_COMMAND_STATUS_NOT_IMPLEMENTED;
279 
280  switch (pCommand->entityModelPdu.command_type) {
281  case OPENAVB_AEM_COMMAND_CODE_ACQUIRE_ENTITY:
282  // Implemented in the acquireEntity() function. Should never get here.
283  break;
284  case OPENAVB_AEM_COMMAND_CODE_LOCK_ENTITY:
285  // Implemented in the lockEntity() function. Should never get here.
286  break;
287  case OPENAVB_AEM_COMMAND_CODE_ENTITY_AVAILABLE:
288  pCommand->headers.status = OPENAVB_AEM_COMMAND_STATUS_SUCCESS;
289  break;
290  case OPENAVB_AEM_COMMAND_CODE_CONTROLLER_AVAILABLE:
291  break;

295  openavb_aecp_response_data_read_descriptor_t *pRsp = &pCommand->entityModelPdu.command_data.readDescriptorRsp;
296  U16 configuration_index = pCmd->configuration_index;
297  U16 descriptor_type = pCmd->descriptor_type;
298  U16 descriptor_index = pCmd->descriptor_index;
299 
300  openavbRC rc = openavbAemSerializeDescriptor(configuration_index, descriptor_type, descriptor_index,
301  sizeof(pRsp->descriptor_data), pRsp->descriptor_data, &pRsp->descriptor_length);
302  if (IS_OPENAVB_FAILURE(rc)) {
303  U8 *pDst = pRsp->descriptor_data;
304 
305  // Send a basic response back to the controller.
306  pCommand->headers.status = OPENAVB_AEM_COMMAND_STATUS_NO_SUCH_DESCRIPTOR;
307  pRsp->configuration_index = configuration_index;
308  pRsp->reserved = 0;
309  OCT_D2BHTONS(pDst, descriptor_type);
310  OCT_D2BHTONS(pDst, descriptor_index);
311  pRsp->descriptor_length = pDst - pRsp->descriptor_data;
312  }
313  else {
314  pCommand->headers.status = OPENAVB_AEM_COMMAND_STATUS_SUCCESS;
315  }
316  }
317  break;
318  case OPENAVB_AEM_COMMAND_CODE_WRITE_DESCRIPTOR:
319  break;
320  case OPENAVB_AEM_COMMAND_CODE_SET_CONFIGURATION:
321  break;
322  case OPENAVB_AEM_COMMAND_CODE_GET_CONFIGURATION:
323  break;

328 
329  if (processCommandCheckRestriction_CorrectController()) {

331  if (pCmd->descriptor_type == OPENAVB_AEM_DESCRIPTOR_STREAM_INPUT) {

1157

1159  AECP_SM_LOCK();

1166 
1167  // Wait for a change in state
1168  while (state == OPENAVB_AECP_SM_ENTITY_MODEL_ENTITY_STATE_WAITING && bRunning) {
1169  AECP_SM_UNLOCK();
1170  SEM_ERR_T(err);

1172  AECP_SM_LOCK();
1173 
1174  if (SEM_IS_ERR_NONE(err)) {
1175  if (openavbAecpSMEntityModelEntityVars.doTerminate) {
1176  bRunning = FALSE;
1177  }

1179  state = OPENAVB_AECP_SM_ENTITY_MODEL_ENTITY_STATE_UNSOLICITED_RESPONSE;

1182  state = OPENAVB_AECP_SM_ENTITY_MODEL_ENTITY_STATE_RECEIVED_COMMAND;

1193

1199  if (openavbAecpSMEntityModelEntityVars.rcvdCommand == NULL) {

1204  openavbAecpSMGlobalVars.myEntityID,

1206  // Not intended for us.

1218  // State machine defines just returning the request command. Doing that in the processCommand function for consistency.

1220  }
1221  else {
1222  processCommand();
1223  }
1224 
1225  openavbAecpMessageSend(openavbAecpSMEntityModelEntityVars.rcvdCommand);
1226  free(openavbAecpSMEntityModelEntityVars.rcvdCommand);
1227  }

1229  state = OPENAVB_AECP_SM_ENTITY_MODEL_ENTITY_STATE_WAITING;
1230  break;
1231 
1232  default:
1233  AVB_LOG_ERROR("State: Unknown");
1234  bRunning = FALSE; // Unexpected
1235  break;
1236 
1237  }
1238  }
1239  AECP_SM_UNLOCK();
1240 
1241  AVB_TRACE_EXIT(AVB_TRACE_AECP);
