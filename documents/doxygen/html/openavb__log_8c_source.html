86 void avbLogRTRender(log_queue_item_t *pLogItem)
87 {
88  if (logRTQueue) {
89  pLogItem->msg[0] = 0x00;
90  bool bMore = TRUE;
91  while (bMore) {
92  openavb_queue_elem_t elem = openavbQueueTailLock(logRTQueue);
93  if (elem) {
94  log_rt_queue_item_t *pLogRTItem = (log_rt_queue_item_t *)openavbQueueData(elem);
95 
96  switch (pLogRTItem->dataType) {
97  case LOG_RT_DATATYPE_CONST_STR:
98  strcat((char *)pLogItem->msg, pLogRTItem->pFormat);
99  break;
100  case LOG_RT_DATATYPE_NOW_TS:
101  snprintf(rt_msg, LOG_RT_MSG_LEN, "[%lu:%09lu] ", pLogRTItem->data.nowTS.tv_sec, pLogRTItem->data.nowTS.tv_nsec);
102  strcat((char *)pLogItem->msg, rt_msg);
103  break;
104  case LOG_RT_DATATYPE_U16:
105  snprintf(rt_msg, LOG_RT_MSG_LEN, pLogRTItem->pFormat, pLogRTItem->data.unsignedShortVar);
106  strcat((char *)pLogItem->msg, rt_msg);
107  break;
108  case LOG_RT_DATATYPE_S16:
109  snprintf(rt_msg, LOG_RT_MSG_LEN, pLogRTItem->pFormat, pLogRTItem->data.signedShortVar);
110  strcat((char *)pLogItem->msg, rt_msg);
111  break;
112  case LOG_RT_DATATYPE_U32:
113  snprintf(rt_msg, LOG_RT_MSG_LEN, pLogRTItem->pFormat, pLogRTItem->data.unsignedLongVar);
114  strcat((char *)pLogItem->msg, rt_msg);
115  break;
116  case LOG_RT_DATATYPE_S32:
117  snprintf(rt_msg, LOG_RT_MSG_LEN, pLogRTItem->pFormat, pLogRTItem->data.signedLongVar);
118  strcat((char *)pLogItem->msg, rt_msg);
119  break;
120  case LOG_RT_DATATYPE_U64:
121  snprintf(rt_msg, LOG_RT_MSG_LEN, pLogRTItem->pFormat, pLogRTItem->data.unsignedLongLongVar);
122  strcat((char *)pLogItem->msg, rt_msg);
123  break;
124  case LOG_RT_DATATYPE_S64:
125  snprintf(rt_msg, LOG_RT_MSG_LEN, pLogRTItem->pFormat, pLogRTItem->data.signedLongLongVar);
126  strcat((char *)pLogItem->msg, rt_msg);
127  break;
128  case LOG_RT_DATATYPE_FLOAT:
129  snprintf(rt_msg, LOG_RT_MSG_LEN, pLogRTItem->pFormat, pLogRTItem->data.floatVar);
130  strcat((char *)pLogItem->msg, rt_msg);
131  break;
132  default:
133  break;
134  }
135 
136  if (pLogRTItem->bEnd) {
137  if (OPENAVB_TCAL_LOG_EXTRA_NEWLINE)
138  strcat((char *)pLogItem->msg, "\n");
139  bMore = FALSE;
140  }
141  openavbQueueTailPull(logRTQueue);
142  }
143 
144  }
145  }

148 extern U32 DLL_EXPORT avbLogGetMsg(U8 *pBuf, U32 bufSize)
149 {
150  U32 dataLen = 0;
151  if (logQueue) {
152  openavb_queue_elem_t elem = openavbQueueTailLock(logQueue);
153  if (elem) {
154  log_queue_item_t *pLogItem = (log_queue_item_t *)openavbQueueData(elem);
155 
156  if (pLogItem->bRT)
157  avbLogRTRender(pLogItem);
158 
159  dataLen = strlen((const char *)pLogItem->msg);
160  if (dataLen <= bufSize)
161  memcpy(pBuf, (U8 *)pLogItem->msg, dataLen);
162  else
163  memcpy(pBuf, (U8 *)pLogItem->msg, bufSize);
164  openavbQueueTailPull(logQueue);
165  return dataLen;
166  }
167  }
168  return dataLen;
169 }

171 void *loggingThreadFn(void *pv)
172 {
173  do {
174  SLEEP_MSEC(LOG_QUEUE_SLEEP_MSEC);
175 
176  bool more = TRUE;
177  bool flush = FALSE;
178 
179  while (more) {
180  more = FALSE;
181  openavb_queue_elem_t elem = openavbQueueTailLock(logQueue);
182  if (elem) {
183  log_queue_item_t *pLogItem = (log_queue_item_t *)openavbQueueData(elem);
184 
185  if (pLogItem->bRT)
186  avbLogRTRender(pLogItem);
187 
188  fputs((const char *)pLogItem->msg, logOutputFd);
189  openavbQueueTailPull(logQueue);
190  more = TRUE;
191  flush = TRUE;
192  }
193  }
194  if (flush)
195  fflush(logOutputFd);
196  } while (loggingThreadRunning);
197 
198  return NULL;
199 }

246 extern void DLL_EXPORT avbLogFn(
247  int level,
248  const char *tag,
249  const char *company,
250  const char *component,
251  const char *path,
252  int line,
253  const char *fmt,
254  ...)
255 {
256  if (level <= AVB_LOG_LEVEL) {
257  va_list args;
258  va_start(args, fmt);
259 
260  LOG_LOCK();
261 
262  vsprintf(msg, fmt, args);
263 
264  if (OPENAVB_LOG_FILE_INFO && path) {
265  char* file = strrchr(path, '/');
266  if (!file)
267  file = strrchr(path, '\\');
268  if (file)
269  file += 1;
270  else
271  file = (char*)path;
272  snprintf(file_msg, LOG_FILE_LEN, " %s:%d", file, line);
273  }
274  if (OPENAVB_LOG_PROC_INFO) {
275  snprintf(proc_msg, LOG_PROC_LEN, " P:%5.5d", GET_PID());
276  }
277  if (OPENAVB_LOG_THREAD_INFO) {
278  snprintf(thread_msg, LOG_THREAD_LEN, " T:%lu", THREAD_SELF());
279  }
280  if (OPENAVB_LOG_TIME_INFO) {
281  time_t tNow = time(NULL);
282  struct tm tmNow;
283  localtime_r(&tNow, &tmNow);
284 
285  snprintf(time_msg, LOG_TIME_LEN, "%2.2d:%2.2d:%2.2d", tmNow.tm_hour, tmNow.tm_min, tmNow.tm_sec);
286  }
287  if (OPENAVB_LOG_TIMESTAMP_INFO) {
288  struct timespec nowTS;
289  CLOCK_GETTIME(OPENAVB_CLOCK_REALTIME, &nowTS);
290 
291  snprintf(timestamp_msg, LOG_TIMESTAMP_LEN, "%lu:%09lu", nowTS.tv_sec, nowTS.tv_nsec);
292  }
293 
294  // using sprintf and puts allows using static buffers rather than heap.
295  if (OPENAVB_TCAL_LOG_EXTRA_NEWLINE)
296  /* S32 full_msg_len = */ snprintf(full_msg, LOG_FULL_MSG_LEN, "[%s%s%s%s %s %s%s] %s: %s\n", time_msg, timestamp_msg, proc_msg, thread_msg, company, component, file_msg, tag, msg);
297  else
298  /* S32 full_msg_len = */ snprintf(full_msg, LOG_FULL_MSG_LEN, "[%s%s%s%s %s %s%s] %s: %s", time_msg, timestamp_msg, proc_msg, thread_msg, company, component, file_msg, tag, msg);
299 
300  if (!OPENAVB_LOG_FROM_THREAD && !OPENAVB_LOG_PULL_MODE) {
301  fputs(full_msg, logOutputFd);
302  fflush(logOutputFd);
303  }
304  else {
305  if (logQueue) {
306  openavb_queue_elem_t elem = openavbQueueHeadLock(logQueue);
307  if (elem) {
308  log_queue_item_t *pLogItem = (log_queue_item_t *)openavbQueueData(elem);
309  pLogItem->bRT = FALSE;
310  strncpy((char *)pLogItem->msg, full_msg, LOG_QUEUE_MSG_LEN);
311  openavbQueueHeadPush(logQueue);
312  }
313  }
314  }
315 
316  va_end(args);
317 
318  LOG_UNLOCK();
319  }
320 }

322 extern void DLL_EXPORT avbLogRT(int level, bool bBegin, bool bItem, bool bEnd, char *pFormat, log_rt_datatype_t dataType, void *pVar)
323 {
324  if (logRTQueue) {
325  if (bBegin) {
326  LOG_LOCK();
327 
328  openavb_queue_elem_t elem = openavbQueueHeadLock(logRTQueue);
329  if (elem) {
330  log_rt_queue_item_t *pLogRTItem = (log_rt_queue_item_t *)openavbQueueData(elem);
331  pLogRTItem->bEnd = FALSE;
332  pLogRTItem->pFormat = NULL;
333  pLogRTItem->dataType = LOG_RT_DATATYPE_NOW_TS;
334  CLOCK_GETTIME(OPENAVB_CLOCK_REALTIME, &pLogRTItem->data.nowTS);
335  openavbQueueHeadPush(logRTQueue);
336  }
337  }
338 
339  if (bItem) {
340  openavb_queue_elem_t elem = openavbQueueHeadLock(logRTQueue);
341  if (elem) {
342  log_rt_queue_item_t *pLogRTItem = (log_rt_queue_item_t *)openavbQueueData(elem);
343  if (bEnd)
344  pLogRTItem->bEnd = TRUE;
345  else
346  pLogRTItem->bEnd = FALSE;
347  pLogRTItem->pFormat = pFormat;
348  pLogRTItem->dataType = dataType;
349 
350  switch (pLogRTItem->dataType) {
351  case LOG_RT_DATATYPE_CONST_STR:
352  break;
353  case LOG_RT_DATATYPE_U16:
354  pLogRTItem->data.unsignedLongVar = *(U16 *)pVar;
355  break;
356  case LOG_RT_DATATYPE_S16:
357  pLogRTItem->data.signedLongVar = *(S16 *)pVar;
358  break;
359  case LOG_RT_DATATYPE_U32:
360  pLogRTItem->data.unsignedLongVar = *(U32 *)pVar;
361  break;
362  case LOG_RT_DATATYPE_S32:
363  pLogRTItem->data.signedLongVar = *(S32 *)pVar;
364  break;
365  case LOG_RT_DATATYPE_U64:
366  pLogRTItem->data.unsignedLongLongVar = *(U64 *)pVar;
367  break;
368  case LOG_RT_DATATYPE_S64:
369  pLogRTItem->data.signedLongLongVar = *(S64 *)pVar;
370  break;
371  case LOG_RT_DATATYPE_FLOAT:
372  pLogRTItem->data.floatVar = *(float *)pVar;
373  break;
374  default:
375  break;
376  }
377  openavbQueueHeadPush(logRTQueue);
378  }
379  }
380 
381  if (!bItem && bEnd) {
382  openavb_queue_elem_t elem = openavbQueueHeadLock(logRTQueue);
383  if (elem) {
384  log_rt_queue_item_t *pLogRTItem = (log_rt_queue_item_t *)openavbQueueData(elem);
385  pLogRTItem->bEnd = TRUE;
386  pLogRTItem->pFormat = NULL;
387  pLogRTItem->dataType = LOG_RT_DATATYPE_NONE;
388  openavbQueueHeadPush(logRTQueue);
389  }
390  }
391 
392  if (bEnd) {
393  if (logQueue) {
394  openavb_queue_elem_t elem = openavbQueueHeadLock(logQueue);
395  if (elem) {
396  log_queue_item_t *pLogItem = (log_queue_item_t *)openavbQueueData(elem);
397  pLogItem->bRT = TRUE;
398  if (OPENAVB_LOG_FROM_THREAD) {
399  openavbQueueHeadPush(logQueue);
400  } else {
401  avbLogRTRender(pLogItem);
402  fputs((const char *)pLogItem->msg, logOutputFd);
403  fflush(logOutputFd);
404  openavbQueueHeadUnlock(logQueue);
405  }
406  }
407  }
408 
409  LOG_UNLOCK();
410  }
411  }
412 }

414 extern void DLL_EXPORT avbLogBuffer(
415  int level,
416  const U8 *pData,
417  int dataLen,
418  int lineLen,
419  const char *company,
420  const char *component,
421  const char *path,
422  int line)
423 {
424  char szDataLine[ 400 ];
425  char *pszOut;
426  int i, j;
427 
428  if (level > AVB_LOG_LEVEL) { return; }
429 
430  for (i = 0; i < dataLen; i += lineLen) {
431  /* Create the hexadecimal output for the buffer. */
432  pszOut = szDataLine;
433  *pszOut++ = '\t';
434  for (j = i; j < i + lineLen; ++j) {
435  if (j < dataLen) {
436  sprintf(pszOut, "%02x ", pData[j]);
437  } else {
438  strcpy(pszOut, " ");
439  }
440  pszOut += 3;
441  }
442 
443  *pszOut++ = ' ';
444  *pszOut++ = ' ';
445 
446  /* Append the ASCII equivalent of each character. */
447  for (j = i; j < dataLen && j < i + lineLen; ++j) {
448  if (pData[j] >= 0x20 && pData[j] < 0x7f) {
449  *pszOut++ = (char) pData[j];
450  } else {
451  *pszOut++ = '.';
452  }
453  }
454 
455  /* Display this line of text. */
456  *pszOut = '\0';
457  avbLogFn(level, "BUFFER", company, component, path, line, "%s", szDataLine);
458  }
459 }
