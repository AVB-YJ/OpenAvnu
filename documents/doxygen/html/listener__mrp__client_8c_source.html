62 int send_msg(char *data, int data_len, struct mrp_listener_ctx *ctx)
63 {
64  struct sockaddr_in addr;
65 
66  if (ctx->control_socket == -1)
67  return -1;
68  if (data == NULL)
69  return -1;
70 
71  memset(&addr, 0, sizeof(addr));
72  addr.sin_family = AF_INET;
73  addr.sin_port = htons(MRPD_PORT_DEFAULT);
74  addr.sin_addr.s_addr = inet_addr("127.0.0.1");
75  inet_aton("127.0.0.1", &addr.sin_addr);
76  return sendto(ctx->control_socket, data, data_len, 0,
77  (struct sockaddr*)&addr, (socklen_t)sizeof(addr));
78 }

80 int msg_process(char *buf, int buflen, struct mrp_listener_ctx *ctx)
81 {
82  uint32_t id;
83  int j, l=0;
84  unsigned int vid;
85  unsigned int priority;
86 
87  fprintf(stderr, "Msg: %s\n", buf);
88 
89  if (strncmp(buf, "SNE T:", 6) == 0 || strncmp(buf, "SJO T:", 6) == 0)
90  {
91  l = 6; /* skip "Sxx T:" */
92  while ((l < buflen) && ('S' != buf[l++]));
93  if (l == buflen)
94  return -1;
95  l++;
96  for(j = 0; j < 8 ; l+=2, j++)
97  {
98  sscanf(&buf[l],"%02x",&id);
99  ctx->stream_id[j] = (unsigned char)id;
100  }
101  l+=3;
102  for(j = 0; j < 6 ; l+=2, j++)
103  {
104  sscanf(&buf[l],"%02x",&id);
105  ctx->dst_mac[j] = (unsigned char)id;
106  }
107  ctx->talker = 1;
108  }
109 
110  if (strncmp(buf, "SJO D:", 6) == 0)
111  {
112  l=8;
113  sscanf(&(buf[l]), "%d", &id);
114  l=l+4;
115  sscanf(&(buf[l]), "%d", &priority);
116  l=l+4;
117  sscanf(&(buf[l]), "%x", &vid);
118 
119  if (id == 6)
120  {
121  ctx->domain_class_a_id = id;
122  ctx->domain_class_a_priority = priority;
123  ctx->domain_class_a_vid = vid;
124  ctx->domain_a_valid = 1;
125  }
126  else
127  {
128  ctx->domain_class_b_id = id;
129  ctx->domain_class_b_priority = priority;
130  ctx->domain_class_b_vid = vid;
131  ctx->domain_b_valid = 1;
132  }
133  l+=4;
134 
135  }
136  return 0;
137 }

169 void *mrp_monitor_thread(void *arg)
170 {
171  char *msgbuf;
172  struct sockaddr_in client_addr;
173  struct msghdr msg;
174  struct iovec iov;
175  int bytes = 0;
176  struct pollfd fds;
177  int rc;
178  struct mrp_listener_ctx *ctx = (struct mrp_listener_ctx*) arg;
179 
180  msgbuf = (char *)malloc(MAX_MRPD_CMDSZ);
181  if (NULL == msgbuf)
182  return NULL;
183  while (!ctx->halt_tx) {
184  fds.fd = ctx->control_socket;
185  fds.events = POLLIN;
186  fds.revents = 0;
187  rc = poll(&fds, 1, 100);
188  if (rc < 0) {
189  free(msgbuf);
190  pthread_exit(NULL);
191  }
192  if (rc == 0)
193  continue;
194  if ((fds.revents & POLLIN) == 0) {
195  free(msgbuf);
196  pthread_exit(NULL);
197  }
198  memset(&msg, 0, sizeof(msg));
199  memset(&client_addr, 0, sizeof(client_addr));
200  memset(msgbuf, 0, MAX_MRPD_CMDSZ);
201  iov.iov_len = MAX_MRPD_CMDSZ;
202  iov.iov_base = msgbuf;
203  msg.msg_name = &client_addr;
204  msg.msg_namelen = sizeof(client_addr);
205  msg.msg_iov = &iov;
206  msg.msg_iovlen = 1;
207  bytes = recvmsg(ctx->control_socket, &msg, 0);
208  if (bytes < 0)
209  continue;
210  msg_process(msgbuf, bytes, ctx);
211  }
212  free(msgbuf);
213  pthread_exit(NULL);
214 }

244 int mrp_get_domain(struct mrp_listener_ctx *ctx, struct mrp_domain_attr *class_a, struct mrp_domain_attr *class_b)
245 {
246  char *msgbuf;
247  int ret;
248 
249  /* we may not get a notification if we are joining late,
250  * so query for what is already there ...
251  */
252  msgbuf = malloc(1500);
253  if (NULL == msgbuf)
254  return -1;
255  memset(msgbuf, 0, 1500);
256  sprintf(msgbuf, "S??");
257  ret = send_msg(msgbuf, 1500, ctx);
258  free(msgbuf);
259  if (ret != 1500)
260  return -1;
261  while (!ctx->halt_tx && (ctx->domain_a_valid == 0) && (ctx->domain_b_valid == 0))
262  usleep(20000);
263  class_a->id = 0;
264  class_a->priority = 0;
265  class_a->vid = 0;
266  class_b->id = 0;
267  class_b->priority = 0;
268  class_b->vid = 0;
269  if (ctx->domain_a_valid) {
270  class_a->id = ctx->domain_class_a_id;
271  class_a->priority = ctx->domain_class_a_priority;
272  class_a->vid = ctx->domain_class_a_vid;
273  }
274  if (ctx->domain_b_valid) {
275  class_b->id = ctx->domain_class_b_id;
276  class_b->priority = ctx->domain_class_b_priority;
277  class_b->vid = ctx->domain_class_b_vid;
278  }
279  return 0;
280 }
