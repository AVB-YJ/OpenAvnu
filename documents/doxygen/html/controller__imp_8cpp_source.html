54 /*
55 * The end_stations class is added here so that in the rare case that an endpoint is added by the background discovery
56 * thread, causing the vector of end_stations to be re-allocated, a foreground process can still obtain a handle
57 * to an end-station and send commands. Note that end_stations are never deleted by this library, even if they go
58 * off-line, so end_station classes and their pointers will remain valid.
59 */
60 class end_stations
61 {
62 private:
63  size_t m_count;
64  std::mutex locker;
65  std::vector<end_station_imp *> end_station_vec; // Store a list of End Station objects
66 public:
67  end_stations()
68  {
69  m_count = 0;
70  };
71  ~end_stations()
72  {
73  for (uint32_t i = 0; i < end_station_vec.size(); i++)
74  delete end_station_vec.at(i);
75  };
76  /* use local m_count to avoid mutex operations */
77  const size_t size(void)
78  {
79  return m_count;
80  };
81  end_station_imp * at(size_t i)
82  {
83  end_station_imp * ep;
84  std::lock_guard<std::mutex> guard(locker);
85  ep = end_station_vec.at(i);
86  return ep;
87  };
88  void push_back(end_station_imp * ep)
89  {
90  locker.lock();
91  end_station_vec.push_back(ep);
92  m_count++;
93  locker.unlock();
94  };
95 };

302 {
303  uint64_t end_station_entity_id;
304  uint32_t disconnected_end_station_index;
305  if (aecp_controller_state_machine_ref)
306  aecp_controller_state_machine_ref->tick();
307  if (acmp_controller_state_machine_ref)

321  {
322  end_station_array->at(i)->background_read_update_timeouts();
323  end_station_array->at(i)->background_read_submit_pending();

327 int controller_imp::find_in_end_station(struct jdksavdecc_eui64 & other_entity_id, bool isUnsolicited, const uint8_t * frame)
328 {
329  struct jdksavdecc_eui64 other_controller_id = jdksavdecc_acmpdu_get_controller_entity_id(frame, ETHER_HDR_SIZE);
330 
331  for (uint32_t i = 0; i < end_station_array->size(); i++)
332  {
333  struct jdksavdecc_eui64 end_entity_id = end_station_array->at(i)->get_adp()->get_entity_entity_id();
334  struct jdksavdecc_eui64 this_controller_id = end_station_array->at(i)->get_adp()->get_controller_entity_id();
335  bool entity_id_match = false;
336  bool controller_id_match = false;
337 
338  if (jdksavdecc_eui64_compare(&end_entity_id, &other_entity_id) == 0)
339  {
340  entity_id_match = true;
341  }
342 
343  //Do not try to find the controller_id if it is an unsolicited response
344  if (isUnsolicited && entity_id_match)
345  {
346  return i;
347  }
348  else
349  {
350  if (jdksavdecc_eui64_compare(&other_controller_id, &this_controller_id) == 0 ||
351  jdksavdecc_eui64_compare(&other_controller_id, &end_entity_id) == 0)
352  {
353  controller_id_match = true;
354  }
355 
356  if (entity_id_match && controller_id_match)
357  {
358  return i;
359  }
360  }
361  }
362 
363  return -1;
364 }

366 void controller_imp::rx_packet_event(void *& notification_id,
367  bool & is_notification_id_valid,
368  const uint8_t * frame,
369  size_t frame_len,
370  int & status,
371  uint16_t & operation_id,
372  bool & is_operation_id_valid)
373 {
374  uint64_t dest_mac_addr;
375  utility::convert_eui48_to_uint64(frame, dest_mac_addr);
376  is_operation_id_valid = false;
377 
378  if ((dest_mac_addr == net_interface_ref->mac_addr()) || (dest_mac_addr & UINT64_C(0x010000000000))) // Process if the packet dest is our MAC address or a multicast address
379  {
380  uint8_t subtype = jdksavdecc_common_control_header_get_subtype(frame, ETHER_HDR_SIZE);
381 
382  switch (subtype)
383  {
384  case JDKSAVDECC_SUBTYPE_ADP:
385  {
386  end_station_imp * end_station = NULL;
387  bool found_adp_in_end_station = false;
388 
389  jdksavdecc_adpdu adpdu;
390  memset(&adpdu, 0, sizeof(adpdu));
391  jdksavdecc_adpdu_read(&adpdu, frame, ETHER_HDR_SIZE, frame_len);
392 
393  status = AVDECC_LIB_STATUS_INVALID;
394  is_notification_id_valid = false;
395 
396  if ((adpdu.entity_capabilities & JDKSAVDECC_ADP_ENTITY_CAPABILITY_GENERAL_CONTROLLER_IGNORE) ||
397  (adpdu.entity_capabilities & JDKSAVDECC_ADP_ENTITY_CAPABILITY_ENTITY_NOT_READY))
398  {
399  // The entity indicates that we should not enumerate it
400  break;
401  }
402 
403  if ((m_entity_capabilities_flags & adpdu.entity_capabilities) != m_entity_capabilities_flags ||
404  (m_talker_capabilities_flags & adpdu.talker_capabilities) != m_talker_capabilities_flags ||
405  (m_listener_capabilities_flags & adpdu.listener_capabilities) != m_listener_capabilities_flags)
406  {
407  //The entity has been filtered out by the user set exclusion flags
408  break;
409  }
410 
415  for (uint32_t i = 0; i < end_station_array->size(); i++)
416  {
417  struct jdksavdecc_eui64 end_entity_id = end_station_array->at(i)->get_adp()->get_entity_entity_id();
418  if (jdksavdecc_eui64_compare(&end_entity_id, &adpdu.header.entity_id) == 0)
419  {
420  found_adp_in_end_station = true;
421  end_station = end_station_array->at(i);
422  }
423  }
424 
425  if (jdksavdecc_eui64_convert_to_uint64(&adpdu.header.entity_id) != 0)
426  {
427  if (!found_adp_in_end_station)
428  {
429  if (adp_discovery_state_machine_ref)
430  adp_discovery_state_machine_ref->state_avail(frame, frame_len);
431  end_station_array->push_back(new end_station_imp(frame, frame_len));
432  end_station_array->at(end_station_array->size() - 1)->set_connected();
433  }
434  else
435  {
436  if ((adpdu.available_index < end_station->get_adp()->get_available_index()) ||
437  (jdksavdecc_eui64_convert_to_uint64(&adpdu.entity_model_id) != end_station->get_adp()->get_entity_model_id()))
438  {
439  log_imp_ref->post_log_msg(LOGGING_LEVEL_DEBUG, "Re-enumerating end station with entity_id %ull", end_station->entity_id());
440  end_station->end_station_reenumerate();
441  }
442 
443  end_station->get_adp()->proc_adpdu(frame, frame_len);
444 
445  if (end_station->get_connection_status() == 'D')
446  {
447  end_station->set_connected();
448  if (adp_discovery_state_machine_ref)
449  adp_discovery_state_machine_ref->state_avail(frame, frame_len);
450  }
451  else
452  {
453  if (adp_discovery_state_machine_ref)
454  adp_discovery_state_machine_ref->state_avail(frame, frame_len);
455  }
456  }
457  }
458  else if (adpdu.header.message_type != JDKSAVDECC_ADP_MESSAGE_TYPE_ENTITY_DISCOVER)
459  {
460  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "Invalid ADP packet with an entity ID of 0.");
461  }
462  }
463  break;
464 
465  case JDKSAVDECC_SUBTYPE_AECP:
466  {
467  int found_end_station_index = -1;
468  uint32_t msg_type = jdksavdecc_common_control_header_get_control_data(frame, ETHER_HDR_SIZE);
469  struct jdksavdecc_eui64 entity_entity_id = jdksavdecc_common_control_header_get_stream_id(frame, ETHER_HDR_SIZE);
470  uint16_t cmd_type = jdksavdecc_aecpdu_aem_get_command_type(frame, ETHER_HDR_SIZE);
471  bool isUnsolicited = cmd_type >> 15 & 0x01;
472 
473  /* check dest mac address is ours */
474  if (dest_mac_addr == net_interface_ref->mac_addr())
475  {
476  if (msg_type == JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_COMMAND &&
477  cmd_type == JDKSAVDECC_AEM_COMMAND_CONTROLLER_AVAILABLE)
478  {
479  send_controller_avail_response(frame, frame_len);
480  }
481  else
482  {
486  found_end_station_index = find_in_end_station(entity_entity_id, isUnsolicited, frame);
487  if (found_end_station_index >= 0)
488  {
489  switch (msg_type)
490  {
491 
492  case JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_RESPONSE:
493  {
494  end_station_array->at(found_end_station_index)->proc_rcvd_aem_resp(notification_id, frame, frame_len, status, operation_id, is_operation_id_valid);
495 
496  is_notification_id_valid = true;
497  break;
498  }
499  case JDKSAVDECC_AECP_MESSAGE_TYPE_ADDRESS_ACCESS_RESPONSE:
500  {
501  end_station_array->at(found_end_station_index)->proc_rcvd_aecp_aa_resp(notification_id, frame, frame_len, status);
502 
503  is_notification_id_valid = true;
504  break;
505  }
506  }
507  }
508  else
509  {
510  status = AVDECC_LIB_STATUS_INVALID;
511  }
512  }
513  }
514  }
515  break;
516 
517  case JDKSAVDECC_SUBTYPE_ACMP:
518  {
519  int found_end_station_index = -1;
520  bool found_acmp_in_end_station = false;
521  struct jdksavdecc_eui64 entity_entity_id;
522  uint32_t msg_type = jdksavdecc_common_control_header_get_control_data(frame, ETHER_HDR_SIZE);
523 
524  if ((msg_type == JDKSAVDECC_ACMP_MESSAGE_TYPE_GET_TX_STATE_RESPONSE) ||
525  (msg_type == JDKSAVDECC_ACMP_MESSAGE_TYPE_GET_TX_CONNECTION_RESPONSE))
526  {
527  entity_entity_id = jdksavdecc_acmpdu_get_talker_entity_id(frame, ETHER_HDR_SIZE);
528  }
529  else if ((msg_type == JDKSAVDECC_ACMP_MESSAGE_TYPE_CONNECT_RX_RESPONSE) ||
530  (msg_type == JDKSAVDECC_ACMP_MESSAGE_TYPE_DISCONNECT_RX_RESPONSE) ||
531  (msg_type == JDKSAVDECC_ACMP_MESSAGE_TYPE_GET_RX_STATE_RESPONSE))
532  {
533  entity_entity_id = jdksavdecc_acmpdu_get_listener_entity_id(frame, ETHER_HDR_SIZE);
534  }
535 
536  found_end_station_index = find_in_end_station(entity_entity_id, false, frame);
537  if (found_end_station_index >= 0)
538  found_acmp_in_end_station = true;
539 
540  if (found_acmp_in_end_station)
541  {
542  end_station_array->at(found_end_station_index)->proc_rcvd_acmp_resp(msg_type, notification_id, frame, frame_len, status);
543  is_notification_id_valid = true;
544  }
545  else
546  {
547  log_imp_ref->post_log_msg(LOGGING_LEVEL_DEBUG, "Wait for correct ACMP response packet.");
548  status = AVDECC_LIB_STATUS_INVALID;
549  }
550  }
551  break;
552 
553  default:
554  break;
555  }
556  }
557 }

559 void controller_imp::tx_packet_event(void * notification_id, uint32_t notification_flag, uint8_t * frame, size_t frame_len)
560 {
561  uint8_t subtype = jdksavdecc_common_control_header_get_subtype(frame, ETHER_HDR_SIZE);
562  struct jdksavdecc_frame packet_frame;
563 
564  packet_frame.length = (uint16_t)frame_len;
565  assert(frame_len <= sizeof(packet_frame.payload));
566  memcpy(packet_frame.payload, frame, frame_len);
567 
568  if (subtype == JDKSAVDECC_SUBTYPE_AECP)
569  {
570  if (aecp_controller_state_machine_ref)
571  {
572  aecp_controller_state_machine_ref->state_send_cmd(notification_id, notification_flag, &packet_frame);
573  memcpy(frame, packet_frame.payload, frame_len); // Get the updated frame with sequence id
574  }
575  }
576  else if (subtype == JDKSAVDECC_SUBTYPE_ACMP)
577  {
578  if (acmp_controller_state_machine_ref)
579  {
580  acmp_controller_state_machine_ref->state_command(notification_id, notification_flag, &packet_frame);
581  memcpy(frame, packet_frame.payload, frame_len); // Get the updated frame with sequence id
582  }
583  }
584  else
585  {
586  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "Invalid Subtype: %x", subtype);
587  }

628 {

630  struct jdksavdecc_eui48 src_address;
631  struct jdksavdecc_eui48 temp_address;
632  int send_frame_returned = 0;
633  size_t pos = 0;
634  uint8_t * tx_frame = new uint8_t[frame_len];
635 
636  if (tx_frame == NULL)
637  {
638  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "unable to allocate response frame");

641 
642  memcpy(tx_frame, frame, frame_len);
643 
644  dest_address = jdksavdecc_eui48_get(tx_frame, pos + 0);
645  src_address = jdksavdecc_eui48_get(tx_frame, pos + 6);
646 
647  // swap MAC addresses
648  temp_address = src_address;
649  src_address = dest_address;
650  dest_address = temp_address;
651 
652  jdksavdecc_eui48_set(dest_address, tx_frame, pos + 0);
653  jdksavdecc_eui48_set(src_address, tx_frame, pos + 6);
654 
655  //set message type
656  jdksavdecc_common_control_header_set_control_data(JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_RESPONSE,
657  tx_frame,
658  pos + ETHER_HDR_SIZE);
659 
660  //send packet
661  send_frame_returned = net_interface_ref->send_frame(tx_frame, frame_len);
662 
663  if (send_frame_returned < 0)
664  {
665  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "netif_send_frame error");
666  assert(send_frame_returned >= 0);
667  }
668
