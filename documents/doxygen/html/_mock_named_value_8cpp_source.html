254 bool MockNamedValue::equals(const MockNamedValue& p) const
255 {
256  if((type_ == "long int") && (p.type_ == "int"))
257  return value_.longIntValue_ == p.value_.intValue_;
258  else if((type_ == "int") && (p.type_ == "long int"))
259  return value_.intValue_ == p.value_.longIntValue_;
260  else if((type_ == "unsigned int") && (p.type_ == "int"))
261  return (long)value_.unsignedIntValue_ == (long)p.value_.intValue_;
262  else if((type_ == "int") && (p.type_ == "unsigned int"))
263  return (long)value_.intValue_ == (long)p.value_.unsignedIntValue_;
264  else if((type_ == "unsigned long int") && (p.type_ == "int"))
265  return value_.unsignedLongIntValue_ == (unsigned long)p.value_.intValue_;
266  else if((type_ == "int") && (p.type_ == "unsigned long int"))
267  return (unsigned long)value_.intValue_ == p.value_.unsignedLongIntValue_;
268  else if((type_ == "unsigned int") && (p.type_ == "long int"))
269  return (long int)value_.unsignedIntValue_ == p.value_.longIntValue_;
270  else if((type_ == "long int") && (p.type_ == "unsigned int"))
271  return value_.longIntValue_ == (long int)p.value_.unsignedIntValue_;
272  else if((type_ == "unsigned int") && (p.type_ == "unsigned long int"))
273  return value_.unsignedIntValue_ == p.value_.unsignedLongIntValue_;
274  else if((type_ == "unsigned long int") && (p.type_ == "unsigned int"))
275  return value_.unsignedLongIntValue_ == p.value_.unsignedIntValue_;
276  else if((type_ == "long int") && (p.type_ == "unsigned long int"))
277  return (value_.longIntValue_ >= 0) && (value_.longIntValue_ == (long) p.value_.unsignedLongIntValue_);
278  else if((type_ == "unsigned long int") && (p.type_ == "long int"))
279  return (p.value_.longIntValue_ >= 0) && ((long)value_.unsignedLongIntValue_ == p.value_.longIntValue_);
280 
281  if (type_ != p.type_) return false;
282 
283  if (type_ == "bool")
284  return value_.boolValue_ == p.value_.boolValue_;
285  else if (type_ == "int")
286  return value_.intValue_ == p.value_.intValue_;
287  else if (type_ == "unsigned int")
288  return value_.unsignedIntValue_ == p.value_.unsignedIntValue_;
289  else if (type_ == "long int")
290  return value_.longIntValue_ == p.value_.longIntValue_;
291  else if (type_ == "unsigned long int")
292  return value_.unsignedLongIntValue_ == p.value_.unsignedLongIntValue_;
293  else if (type_ == "const char*")
294  return SimpleString(value_.stringValue_) == SimpleString(p.value_.stringValue_);
295  else if (type_ == "void*")
296  return value_.pointerValue_ == p.value_.pointerValue_;
297  else if (type_ == "const void*")
298  return value_.constPointerValue_ == p.value_.constPointerValue_;
299  else if (type_ == "void (*)()")
300  return value_.functionPointerValue_ == p.value_.functionPointerValue_;
301  else if (type_ == "double")
302  return (doubles_equal(value_.doubleValue_, p.value_.doubleValue_, 0.005));
303  else if (type_ == "const unsigned char*")
304  {
305  if (size_ != p.size_) {
306  return false;
307  }
308  return SimpleString::MemCmp(value_.memoryBufferValue_, p.value_.memoryBufferValue_, size_) == 0;
309  }
310 
311  if (comparator_)
312  return comparator_->isEqual(value_.objectPointerValue_, p.value_.objectPointerValue_);
313 
314  return false;
315 }

327 SimpleString MockNamedValue::toString() const
328 {
329  if (type_ == "bool")
330  return StringFrom(value_.boolValue_);
331  else if (type_ == "int")
332  return StringFrom(value_.intValue_) + " " + BracketsFormattedHexStringFrom(value_.intValue_);
333  else if (type_ == "unsigned int")
334  return StringFrom(value_.unsignedIntValue_) + " " + BracketsFormattedHexStringFrom(value_.unsignedIntValue_);
335  else if (type_ == "long int")
336  return StringFrom(value_.longIntValue_) + " " + BracketsFormattedHexStringFrom(value_.longIntValue_);
337  else if (type_ == "unsigned long int")
338  return StringFrom(value_.unsignedLongIntValue_) + " " + BracketsFormattedHexStringFrom(value_.unsignedLongIntValue_);
339  else if (type_ == "const char*")
340  return value_.stringValue_;
341  else if (type_ == "void*")
342  return StringFrom(value_.pointerValue_);
343  else if (type_ == "void (*)()")
344  return StringFrom(value_.functionPointerValue_);
345  else if (type_ == "const void*")
346  return StringFrom(value_.constPointerValue_);
347  else if (type_ == "double")
348  return StringFrom(value_.doubleValue_);
349  else if (type_ == "const unsigned char*")
350  return StringFromBinaryWithSizeOrNull(value_.memoryBufferValue_, size_);
351 
352  if (comparator_)
353  return comparator_->valueToString(value_.objectPointerValue_);
354 
355  return StringFromFormat("No comparator found for type: \"%s\"", type_.asCharString());
356
