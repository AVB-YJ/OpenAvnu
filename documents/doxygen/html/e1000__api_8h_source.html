104 /*
105  * TBI_ACCEPT macro definition:
106  *
107  * This macro requires:
108  * a = a pointer to struct e1000_hw
109  * status = the 8 bit status field of the Rx descriptor with EOP set
110  * errors = the 8 bit error field of the Rx descriptor with EOP set
111  * length = the sum of all the length fields of the Rx descriptors that
112  * make up the current frame
113  * last_byte = the last byte of the frame DMAed by the hardware
114  * min_frame_size = the minimum frame length we want to accept.
115  * max_frame_size = the maximum frame length we want to accept.
116  *
117  * This macro is a conditional that should be used in the interrupt
118  * handler's Rx processing routine when RxErrors have been detected.
119  *
120  * Typical use:
121  * ...
122  * if (TBI_ACCEPT) {
123  * accept_frame = true;
124  * e1000_tbi_adjust_stats(adapter, MacAddress);
125  * frame_length--;
126  * } else {
127  * accept_frame = false;
128  * }
129  * ...
130  */
131 
132 /* The carrier extension symbol, as received by the NIC. */
133 #define CARRIER_EXTENSION 0x0F
134 
135 #define TBI_ACCEPT(a, status, errors, length, last_byte, \
136  min_frame_size, max_frame_size) \
137  (e1000_tbi_sbp_enabled_82543(a) && \
138  (((errors) & E1000_RXD_ERR_FRAME_ERR_MASK) == E1000_RXD_ERR_CE) && \
139  ((last_byte) == CARRIER_EXTENSION) && \
140  (((status) & E1000_RXD_STAT_VP) ? \
141  (((length) > ((min_frame_size) - VLAN_TAG_SIZE)) && \
142  ((length) <= ((max_frame_size) + 1))) : \
143  (((length) > (min_frame_size)) && \
144  ((length) <= ((max_frame_size) + VLAN_TAG_SIZE + 1)))))
145 
146 #ifndef E1000_MAX
147 #define E1000_MAX(a, b) ((a) > (b) ? (a) : (b))
148 #endif
149 #ifndef E1000_DIVIDE_ROUND_UP
150 #define E1000_DIVIDE_ROUND_UP(a, b) (((a) + (b) - 1) / (b)) /* ceil(a/b) */
151 #endif
152 #endif /* _E1000_API_H_ */
