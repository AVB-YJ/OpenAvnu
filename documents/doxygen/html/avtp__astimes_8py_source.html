76 def pkt_avtp(pkt, fout, pkt_count, info):
77 
78  n = 0
79  avtp = pkt[AVTP]
80  if info['seq']['init'] == False:
81  info['seq']['last'] = avtp.sequence
82  info['seq']['init'] = True
83  else:
84  if avtp.sequence < info['seq']['last']:
85  if avtp.sequence != 0:
86  print "Sequence wrap error at packet number %d" % (pkt_count)
87  else:
88  if avtp.sequence - info['seq']['last'] != 1:
89  print "Sequence error at packet number %d (MAC %s)" % (pkt_count, info['this_mac'])
90  info['seq']['last'] = avtp.sequence
91 
92  if avtp.flags == 0x81:
93  if info['ts_count'] == 0:
94  info['ts_accum'] = avtp.ptpTimestamp
95  else:
96  if avtp.ptpTimestamp > info['prev_pkt_ts']:
97  ts_delta = avtp.ptpTimestamp - info['prev_pkt_ts']
98  else:
99  ts_delta = avtp.ptpTimestamp + 0x100000000 - info['prev_pkt_ts']
100  info['ts_accum'] = info['ts_accum'] + ts_delta
101  fout.write("%d, %d\n" % (info['ts_count'], info['ts_accum']))
102  n = 1
103  info['prev_pkt_ts'] = avtp.ptpTimestamp
104  info['ts_count'] = info['ts_count'] + 1
105  return n
