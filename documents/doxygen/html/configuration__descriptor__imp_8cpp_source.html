221 void configuration_descriptor_imp::update_desc_database(descriptor_base_imp * desc, const uint8_t * frame, ssize_t pos, size_t frame_len)
222 {
223  uint16_t desc_type = desc->descriptor_type();
224  uint16_t desc_index = desc->descriptor_index();
225 
226  if (m_all_desc[desc_type].size() <= desc_index)
227  m_all_desc[desc_type].resize(desc_index + 1);
228  if (m_all_desc[desc_type][desc_index])
229  {
230  // exists
231  desc->replace_desc_frame(frame, pos, frame_len);
232  delete desc;
233  }
234  else
235  {
236  // does not exist
237  m_all_desc[desc_type][desc_index] = desc;
238  }
239 }
