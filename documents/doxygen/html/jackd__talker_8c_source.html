138 static void* packetizer_thread(void *arg) {
139  struct igb_packet *cleaned_packets;
140  struct mrp_talker_ctx *ctx = (struct mrp_talker_ctx *)arg;
141  six1883_sample *sample;
142  unsigned total_samples = 0;
143  int err;
144  int i;
145  (void) arg; /* unused */
146 
147  const size_t bytes_to_read = CHANNELS * SAMPLES_PER_FRAME *
148  SAMPLE_SIZE;
149  jack_default_audio_sample_t* framebuf = malloc (bytes_to_read);
150  extern jack_ringbuffer_t* ringbuffer;
151  extern pthread_mutex_t threadLock;
152  extern pthread_cond_t dataReady;
153 
154  pthread_setcanceltype (PTHREAD_CANCEL_ASYNCHRONOUS, NULL);
155  pthread_mutex_lock(&threadLock);
156 
157  while (ctx->listeners && !ctx->halt_tx) {
158  pthread_cond_wait(&dataReady, &threadLock);
159 
160  while ((jack_ringbuffer_read_space(ringbuffer) >= bytes_to_read)) {
161 
162  glob_tmp_packet = glob_free_packets;
163  if (NULL == glob_tmp_packet)
164  goto cleanup;
165  glob_header1722 =
166  (seventeen22_header *) (((char *)glob_tmp_packet->vaddr) + 18);
167  glob_header61883 = (six1883_header *) (glob_header1722 + 1);
168  glob_free_packets = glob_tmp_packet->next;
169 
170  /* unfortuntely unless this thread is at rtprio
171  * you get pre-empted between fetching the time
172  * and programming the packet and get a late packet
173  */
174  glob_tmp_packet->attime = glob_last_time + PACKET_IPG;
175  glob_last_time += PACKET_IPG;
176 
177  jack_ringbuffer_read (ringbuffer, (char*)&framebuf[0], bytes_to_read);
178 
179  glob_header1722->seq_number = glob_seqnum++;
180  if (glob_seqnum % 4 == 0)
181  glob_header1722->timestamp_valid = 0;
182 
183  else
184  glob_header1722->timestamp_valid = 1;
185 
186  glob_time_stamp = htonl(glob_time_stamp);
187  glob_header1722->timestamp = glob_time_stamp;
188  glob_time_stamp = ntohl(glob_time_stamp);
189  glob_time_stamp += PACKET_IPG;
190  glob_header61883->data_block_continuity = total_samples;
191  total_samples += SAMPLES_PER_FRAME*CHANNELS;
192  sample =
193  (six1883_sample *) (((char *)glob_tmp_packet->vaddr) +
194  (18 + sizeof(seventeen22_header) +
195  sizeof(six1883_header)));
196 
197  for (i = 0; i < SAMPLES_PER_FRAME * CHANNELS; ++i) {
198  uint32_t tmp = htonl(MAX_SAMPLE_VALUE * framebuf[i]);
199  sample[i].label = 0x40;
200  memcpy(&(sample[i].value), &(tmp),
201  sizeof(sample[i].value));
202  }
203 
204  err = igb_xmit(&glob_igb_dev, 0, glob_tmp_packet);
205 
206  if (!err) {
207  continue;
208  }
209 
210  if (ENOSPC == err) {
211 
212  /* put back for now */
213  glob_tmp_packet->next = glob_free_packets;
214  glob_free_packets = glob_tmp_packet;
215  }
216 
217 cleanup: igb_clean(&glob_igb_dev, &cleaned_packets);
218  i = 0;
219  while (cleaned_packets) {
220  i++;
221  glob_tmp_packet = cleaned_packets;
222  cleaned_packets = cleaned_packets->next;
223  glob_tmp_packet->next = glob_free_packets;
224  glob_free_packets = glob_tmp_packet;
225  }
226  }
227  }
228  return NULL;
229 }

311  }

315  printf("failed to open interface\n");

317  }
318

321  printf("failed calling msp_get_domain()\n");
322  return EXIT_FAILURE;

324  printf("detected domain Class A PRIO=%d VID=%04x...\n",class_a->priority,
325  class_a->vid);
326

328  if (rc) {
329  printf("mrp_register_domain failed\n");
330  return EXIT_FAILURE;
331  }
332  rc = mrp_join_vlan(class_a, ctx);
333  if (rc) {
334  printf("mrp_join_vlan failed\n");
335  return EXIT_FAILURE;
336  }
337 
338  igb_set_class_bandwidth(&glob_igb_dev, PACKET_IPG / 125000, 0, PKT_SZ - 22,
339  0);
340 
341  memset(glob_stream_id, 0, sizeof(glob_stream_id));
342  memcpy(glob_stream_id, glob_station_addr, sizeof(glob_station_addr));
343 
344  a_packet.dmatime = a_packet.attime = a_packet.flags = 0;
345  a_packet.map.paddr = a_page.dma_paddr;
346  a_packet.map.mmap_size = a_page.mmap_size;
347  a_packet.offset = 0;
348  a_packet.vaddr = a_page.dma_vaddr + a_packet.offset;
349  a_packet.len = PKT_SZ;
350  glob_free_packets = NULL;
351  glob_seqnum = 0;
352 
353  /* divide the dma page into buffers for packets */
354  for (i = 1; i < ((a_page.mmap_size) / PKT_SZ); i++) {
355  glob_tmp_packet = malloc(sizeof(struct igb_packet));
356  if (NULL == glob_tmp_packet) {
357  printf("failed to allocate igb_packet memory!\n");
358  return errno;

361  glob_tmp_packet->offset = (i * PKT_SZ);
362  glob_tmp_packet->vaddr += glob_tmp_packet->offset;
363  glob_tmp_packet->next = glob_free_packets;
364  memset(glob_tmp_packet->vaddr, 0, PKT_SZ); /* MAC header at least */
365  memcpy(glob_tmp_packet->vaddr, glob_dest_addr, sizeof(glob_dest_addr));
366  memcpy(glob_tmp_packet->vaddr + 6, glob_station_addr,
367  sizeof(glob_station_addr));
368 
369  /* Q-tag */
370  ((char *)glob_tmp_packet->vaddr)[12] = 0x81;
371  ((char *)glob_tmp_packet->vaddr)[13] = 0x00;
372  ((char *)glob_tmp_packet->vaddr)[14] =
373  ((ctx->domain_class_a_priority << 13 | ctx->domain_class_a_vid)) >> 8;
374  ((char *)glob_tmp_packet->vaddr)[15] =
375  ((ctx->domain_class_a_priority << 13 | ctx->domain_class_a_vid)) & 0xFF;
376  ((char *)glob_tmp_packet->vaddr)[16] = 0x22; /* 1722 eth type */
377  ((char *)glob_tmp_packet->vaddr)[17] = 0xF0;
378

381  (seventeen22_header *) (((char *)glob_tmp_packet->vaddr) + 18);
382  glob_header1722->cd_indicator = 0;
383  glob_header1722->subtype = 0;
384  glob_header1722->sid_valid = 1;
385  glob_header1722->version = 0;
386  glob_header1722->reset = 0;
387  glob_header1722->reserved0 = 0;
388  glob_header1722->gateway_valid = 0;
389  glob_header1722->reserved1 = 0;

392  memcpy(&(glob_header1722->stream_id), glob_station_addr,
393  sizeof(glob_station_addr));
394  glob_header1722->length = htons(32);
395  glob_header61883 = (six1883_header *) (glob_header1722 + 1);
396  glob_header61883->format_tag = 1;
397  glob_header61883->packet_channel = 0x1F;
398  glob_header61883->packet_tcode = 0xA;
399  glob_header61883->app_control = 0x0;
400  glob_header61883->reserved0 = 0;
401  glob_header61883->source_id = 0x3F;
402  glob_header61883->data_block_size = 1;
403  glob_header61883->fraction_number = 0;
404  glob_header61883->quadlet_padding_count = 0;
405  glob_header61883->source_packet_header = 0;
406  glob_header61883->reserved1 = 0;
407  glob_header61883->eoh = 0x2;
408  glob_header61883->format_id = 0x10;
409  glob_header61883->format_dependent_field = 0x02;
410  glob_header61883->syt = 0xFFFF;
411  glob_tmp_packet->len =
412  18 + sizeof(seventeen22_header) + sizeof(six1883_header) +
413  (SAMPLES_PER_FRAME * CHANNELS * sizeof(six1883_sample));
414  glob_free_packets = glob_tmp_packet;
415  }
416 
417  /*
418  * subtract 16 bytes for the MAC header/Q-tag - pktsz is limited to the
419  * data payload of the ethernet frame .
420  *
421  * IPG is scaled to the Class (A) observation interval of packets per 125 usec
422  */
423 
424  _jackclient = init_jack(ctx);
425 
426  fprintf(stderr, "advertising stream ...\n");
427  rc = mrp_advertise_stream(glob_stream_id, glob_dest_addr, PKT_SZ - 16,
428  PACKET_IPG / 125000, 3900, ctx);
429  if (rc) {
430  printf("mrp_advertise_stream failed\n");
431  return EXIT_FAILURE;
432  }
433 
434  fprintf(stderr, "awaiting a listener ...\n");
435  rc = mrp_await_listener(glob_stream_id,ctx);
436  if (rc) {
437  printf("mrp_await_listener failed\n");
438  return EXIT_FAILURE;
439  }
440  ctx->listeners = 1;
441  printf("got a listener ...\n");
442  ctx->halt_tx = 0;
443 
444  if(-1 == gptpinit(&igb_shm_fd, &igb_mmap)) {
445  return EXIT_FAILURE;
446  }
447 
448  if (-1 == gptpscaling(igb_mmap, &td)) {
449  return EXIT_FAILURE;
450  }
451 
452  if( igb_get_wallclock( &glob_igb_dev, &now_local, NULL ) != 0 ) {
453  fprintf( stderr, "Failed to get wallclock time\n" );

457  delta_local = (unsigned)(now_local - td.local_time);
458  delta_8021as = (unsigned)(td.ml_freqoffset * delta_local);
459  now_8021as = update_8021as + delta_8021as;
460 
461  glob_last_time = now_local + XMIT_DELAY;
462  glob_time_stamp = now_8021as + RENDER_DELAY;
463 
464  rc = nice(-20);
465

470

476

478  PACKET_IPG / 125000, 3900, ctx);
479  if (rc)
480  printf("mrp_unadvertise_stream failed\n");
481 
482  igb_set_class_bandwidth(&glob_igb_dev, 0, 0, 0, 0); /* disable Qav */
483 
484  rc = mrp_disconnect(ctx);
485  if (rc)
486  printf("mrp_disconnect failed\n");

490  free(class_b);

493  err = igb_detach(&glob_igb_dev);
494 
495  pthread_exit(NULL);
496 
497  return EXIT_SUCCESS;
498 }
