47 * | |
48 * | |
49 * - -
50 * | |
51 * |stream id |

53 * | |

55 * --+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+--
56 * | | |

61 * -+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-

154 void openavbMapNullCfgCB(media_q_t *pMediaQ, const char *name, const char *value)
155 {
156  AVB_TRACE_ENTRY(AVB_TRACE_MAP);
157 
158  if (pMediaQ) {
159  pvt_data_t *pPvtData = pMediaQ->pPvtMapInfo;
160  if (!pPvtData) {
161  AVB_LOG_ERROR("Private mapping module data not allocated.");
162  return;
163  }
164 
165  if (strcmp(name, "map_nv_item_count") == 0) {
166  char *pEnd;
167  pPvtData->itemCount = strtol(value, &pEnd, 10);
168  }
169  else if (strcmp(name, "map_nv_tx_rate") == 0
170  || strcmp(name, "map_nv_tx_interval") == 0) {
171  char *pEnd;
172  pPvtData->txInterval = strtol(value, &pEnd, 10);
173  }
174  }
175 
176  AVB_TRACE_EXIT(AVB_TRACE_MAP);
177 }

244 // This talker callback will be called for each AVB observation interval.
245 tx_cb_ret_t openavbMapNullTxCB(media_q_t *pMediaQ, U8 *pData, U32 *dataLen)
246 {
247  AVB_TRACE_ENTRY(AVB_TRACE_MAP_DETAIL);
248  if (pMediaQ && pData && dataLen) {
249  U8 *pHdr = pData;
250  pvt_data_t *pPvtData = pMediaQ->pPvtMapInfo;
251  if (!pPvtData) {
252  AVB_LOG_ERROR("Private mapping module data not allocated.");
253  return TX_CB_RET_PACKET_NOT_READY;
254  }
255 
256  media_q_item_t *pMediaQItem = openavbMediaQTailLock(pMediaQ, TRUE);
257  if (pMediaQItem) {
258  // PTP walltime already set in the interface module. Just add the max transit time.
259  openavbAvtpTimeAddUSec(pMediaQItem->pAvtpTime, pPvtData->maxTransitUsec);
260 
261  // Set timestamp valid flag
262  if (openavbAvtpTimeTimestampIsValid(pMediaQItem->pAvtpTime))
263  pHdr[HIDX_AVTP_HIDE7_TV1] |= 0x01; // Set
264  else {
265  pHdr[HIDX_AVTP_HIDE7_TV1] &= ~0x01; // Clear
266  }
267 
268  // Set timestamp uncertain flag
269  if (openavbAvtpTimeTimestampIsUncertain(pMediaQItem->pAvtpTime))
270  pHdr[HIDX_AVTP_HIDE7_TU1] |= 0x01; // Set
271  else pHdr[HIDX_AVTP_HIDE7_TU1] &= ~0x01; // Clear
272 
273  *(U32 *)(&pHdr[HIDX_AVTP_TIMESPAMP32]) = htonl(openavbAvtpTimeGetAvtpTimestamp(pMediaQItem->pAvtpTime));
274  pHdr[HIDX_OPENAVB_FORMAT8] = MAP_NULL_OPENAVB_FORMAT;
275  pHdr[HIDX_OPENAVB_RESERVEDA8] = 0x00;
276  pHdr[HIDX_OPENAVB_RESERVEDB8] = 0x00;
277  pHdr[HIDX_OPENAVB_RESERVEDC8] = 0x00;
278  *(U32 *)(&pHdr[HIDX_OPENAVB_FORMAT_SPEC32]) = 0x00000000;
279 
280  *dataLen = TOTAL_HEADER_SIZE; // No data
281  openavbMediaQTailPull(pMediaQ);
282  AVB_TRACE_LINE(AVB_TRACE_MAP_LINE);
283  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
284  return TX_CB_RET_PACKET_READY;
285  }
286  else {
287  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
288  return TX_CB_RET_PACKET_NOT_READY; // Media queue empty
289  }
290  }
291  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
292  return TX_CB_RET_PACKET_NOT_READY;
293 }

303 // This callback occurs when running as a listener and data is available.
304 bool openavbMapNullRxCB(media_q_t *pMediaQ, U8 *pData, U32 dataLen)
305 {
306  AVB_TRACE_ENTRY(AVB_TRACE_MAP_DETAIL);
307  if (pMediaQ && pData) {
308  U8 *pHdr = pData;
309 
310  media_q_item_t *pMediaQItem = openavbMediaQHeadLock(pMediaQ);
311  if (pMediaQItem) {
312  // Get the timestamp and place it in the media queue item.
313  U32 timestamp = ntohl(*(U32 *)(&pHdr[HIDX_AVTP_TIMESPAMP32]));
314  openavbAvtpTimeSetToTimestamp(pMediaQItem->pAvtpTime, timestamp);
315 
316  // Set timestamp valid and timestamp uncertain flags
317  openavbAvtpTimeSetTimestampValid(pMediaQItem->pAvtpTime, (pHdr[HIDX_AVTP_HIDE7_TV1] & 0x01) ? TRUE : FALSE);
318  openavbAvtpTimeSetTimestampUncertain(pMediaQItem->pAvtpTime, (pHdr[HIDX_AVTP_HIDE7_TU1] & 0x01) ? TRUE : FALSE);
319 
320  pMediaQItem->dataLen = 0; // Regardless of what is sent nullify the data to the interface.
321  openavbMediaQHeadPush(pMediaQ);
322  AVB_TRACE_LINE(AVB_TRACE_MAP_LINE);
323  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
324  return TRUE;
325  }
326  else {
327  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
328  return FALSE; // Media queue full
329  }
330  }
331  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
332  return FALSE;
333 }

349 // Initialization entry point into the mapping module. Will need to be included in the .ini file.
350 extern DLL_EXPORT bool openavbMapNullInitialize(media_q_t *pMediaQ, openavb_map_cb_t *pMapCB, U32 inMaxTransitUsec)
351 {
352  AVB_TRACE_ENTRY(AVB_TRACE_MAP);
353 
354  if (pMediaQ) {
355  pMediaQ->pMediaQDataFormat = strdup(MapNullMediaQDataFormat);
356  pMediaQ->pPvtMapInfo = calloc(1, sizeof(pvt_data_t)); // Memory freed by the media queue when the media queue is destroyed.
357 
358  if (!pMediaQ->pMediaQDataFormat || !pMediaQ->pPvtMapInfo) {
359  AVB_LOG_ERROR("Unable to allocate memory for mapping module.");
360  return FALSE;
361  }
362 
363  pvt_data_t *pPvtData = pMediaQ->pPvtMapInfo;
364 
365  pMapCB->map_cfg_cb = openavbMapNullCfgCB;
366  pMapCB->map_subtype_cb = openavbMapNullSubtypeCB;
367  pMapCB->map_avtp_version_cb = openavbMapNullAvtpVersionCB;
368  pMapCB->map_max_data_size_cb = openavbMapNullMaxDataSizeCB;
369  pMapCB->map_transmit_interval_cb = openavbMapNullTransmitIntervalCB;
370  pMapCB->map_gen_init_cb = openavbMapNullGenInitCB;
371  pMapCB->map_tx_init_cb = openavbMapNullTxInitCB;
372  pMapCB->map_tx_cb = openavbMapNullTxCB;
373  pMapCB->map_rx_init_cb = openavbMapNullRxInitCB;
374  pMapCB->map_rx_cb = openavbMapNullRxCB;
375  pMapCB->map_end_cb = openavbMapNullEndCB;
376  pMapCB->map_gen_end_cb = openavbMapNullGenEndCB;
377 
378  pPvtData->itemCount = 20;
379  pPvtData->txInterval = 0;
380  pPvtData->maxTransitUsec = inMaxTransitUsec;
381 
382  openavbMediaQSetMaxLatency(pMediaQ, inMaxTransitUsec);
383  }
384 
385  AVB_TRACE_EXIT(AVB_TRACE_MAP);
386  return TRUE;
387 }
