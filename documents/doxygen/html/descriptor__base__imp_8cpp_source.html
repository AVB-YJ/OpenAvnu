154 int descriptor_base_imp::default_send_acquire_entity_cmd(descriptor_base_imp * desc_base_imp_ref, void * notification_id, uint32_t acquire_entity_flag)
155 {
156  struct jdksavdecc_frame cmd_frame;
157  struct jdksavdecc_aem_command_acquire_entity aem_cmd_acquire_entity;
158  ssize_t aem_cmd_acquire_entity_returned;
159 
160  /***************************************** AECP Common Data *********************************************/
161  aem_cmd_acquire_entity.aem_header.aecpdu_header.controller_entity_id = base_end_station_imp_ref->get_adp()->get_controller_entity_id();
162  // Fill aem_cmd_acquire_entity.sequence_id in AEM Controller State Machine
163  aem_cmd_acquire_entity.aem_header.command_type = JDKSAVDECC_AEM_COMMAND_ACQUIRE_ENTITY;
164 
165  /************************* AECP Message Specific Data **************************/
166  aem_cmd_acquire_entity.aem_acquire_flags = acquire_entity_flag;
167  jdksavdecc_eui64_init(&aem_cmd_acquire_entity.owner_entity_id);
168  aem_cmd_acquire_entity.descriptor_type = desc_base_imp_ref->descriptor_type();
169  aem_cmd_acquire_entity.descriptor_index = desc_base_imp_ref->descriptor_index();
170 
171  /**************************** Fill frame payload with AECP data and send the frame **********************/
172  aecp_controller_state_machine_ref->ether_frame_init(base_end_station_imp_ref->mac(), &cmd_frame,
173  ETHER_HDR_SIZE + JDKSAVDECC_AEM_COMMAND_ACQUIRE_ENTITY_COMMAND_LEN);
174  aem_cmd_acquire_entity_returned = jdksavdecc_aem_command_acquire_entity_write(&aem_cmd_acquire_entity,
175  cmd_frame.payload,
176  ETHER_HDR_SIZE,
177  sizeof(cmd_frame.payload));
178 
179  if (aem_cmd_acquire_entity_returned < 0)
180  {
181  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "aem_cmd_acquire_entity_write error\n");
182  assert(aem_cmd_acquire_entity_returned >= 0);
183  return -1;
184  }
185 
186  aecp_controller_state_machine_ref->common_hdr_init(JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_COMMAND,
187  &cmd_frame,
188  base_end_station_imp_ref->entity_id(),
189  JDKSAVDECC_AEM_COMMAND_ACQUIRE_ENTITY_COMMAND_LEN -
190  JDKSAVDECC_COMMON_CONTROL_HEADER_LEN);
191  system_queue_tx(notification_id, CMD_WITH_NOTIFICATION, cmd_frame.payload, cmd_frame.length);
192 
193  return 0;
194 }
