44 void audio_unit_descriptor_response_imp::sampling_rates_init(const uint8_t * frame)
45 {
46  uint16_t offset = ETHER_HDR_SIZE + JDKSAVDECC_AEM_COMMAND_READ_DESCRIPTOR_RESPONSE_LEN + sampling_rates_offset();
47  uint32_t sampling_rate = 0;
48 
49  for (uint32_t i = 0; i < sampling_rates_count(); i++)
50  {
51  sampling_rate = jdksavdecc_uint32_get(frame, offset);
52  sample_rates_vec.push_back(sampling_rate);
53  offset += 0x4;
54  }
55 }
