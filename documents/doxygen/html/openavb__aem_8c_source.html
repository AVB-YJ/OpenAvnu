106 openavbRC openavbAemAddDescriptorToConfiguration(U16 descriptorType, U16 configIdx)
107 {
108  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
109  int i1;
110 
111  // Make sure Entity Model is created
112  if (!openavbAemCheckModel(FALSE)) {
113  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVBAVDECC_RC_ENTITY_MODEL_MISSING), AVB_TRACE_AEM);
114  }
115 
116  if (configIdx == OPENAVB_AEM_DESCRIPTOR_INVALID) {
117  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVBAVDECC_RC_INVALID_CONFIG_IDX), AVB_TRACE_AEM);
118  }
119  openavb_aem_configuration_t *pConfiguration = openavbArrayDataIdx(pAemEntityModel->aemConfigurations, configIdx);
120  if (!pConfiguration) {
121  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVBAVDECC_RC_INVALID_CONFIG_IDX), AVB_TRACE_AEM);
122  }
123 
124  openavb_aem_descriptor_configuration_t *pConfig = pConfiguration->pDescriptorConfiguration;
125  if (!pConfig) {
126  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVBAVDECC_RC_INVALID_CONFIG_IDX), AVB_TRACE_AEM);
127  }
128 
129  // Check if the new descriptor type is in the configuration array
130  for (i1 = 0; i1 < pConfig->descriptor_counts_count; i1++) {
131  if (pConfig->descriptor_counts[i1].descriptor_type == descriptorType) {
132  // Found this descriptor type so just increase the count.
133  pConfig->descriptor_counts[i1].count++;
134  AVB_RC_TRACE_RET(OPENAVB_AVDECC_SUCCESS, AVB_TRACE_AEM);
135  }
136  }
137 
138  // Update counts
139  pConfig->descriptor_counts[pConfig->descriptor_counts_count].descriptor_type = descriptorType;
140  pConfig->descriptor_counts[pConfig->descriptor_counts_count].count++;
141  pConfig->descriptor_counts_count++;
142 
143  AVB_RC_TRACE_RET(OPENAVB_AVDECC_SUCCESS, AVB_TRACE_AEM);
144 }

153 openavb_array_t openavbAemGetDescriptorArray(U16 configIdx, U16 descriptorType)
154 {
155  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
156 
157  openavb_array_t retDescriptors = NULL;
158 
159  // Make sure Entity Model is created
160  if (!openavbAemCheckModel(TRUE)) {
161  AVB_TRACE_EXIT(AVB_TRACE_AEM);
162  return NULL;
163  }
164 
165  if (descriptorType == OPENAVB_AEM_DESCRIPTOR_ENTITY) {
166  // There is always only a single Entity Descriptor and therefore there isn't an array. NULL will be returned.

175  retDescriptors = pConfig->descriptorsArray[descriptorType];
176  }
177  }
178 
179  // Search the alternate location for non-top-level descriptors.
180  if (!retDescriptors) {
181  retDescriptors = pAemEntityModel->aemNonTopLevelDescriptorsArray[descriptorType];
182  }
183  }
184 
185  AVB_TRACE_EXIT(AVB_TRACE_AEM);
186  return retDescriptors;

189 void *openavbAemFindDescriptor(U16 configIdx, U16 descriptorType, U16 descriptorIdx)
190 {
191  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
192 
193  void *retDescriptor = NULL;
194 
195  // Make sure Entity Model is created
196  if (!openavbAemCheckModel(TRUE)) {
197  AVB_TRACE_EXIT(AVB_TRACE_AEM);
198  return NULL;
199  }
200 
201  if (descriptorType == OPENAVB_AEM_DESCRIPTOR_ENTITY) {
202  if (pAemEntityModel->pDescriptorEntity &&
203  pAemEntityModel->pDescriptorEntity->descriptor_index == descriptorIdx) {
204  retDescriptor = pAemEntityModel->pDescriptorEntity;
205  }
206  }
207  else if (descriptorType == OPENAVB_AEM_DESCRIPTOR_CONFIGURATION) {
208  if (configIdx != OPENAVB_AEM_DESCRIPTOR_INVALID) {
209  openavb_aem_configuration_t *pConfig = openavbArrayDataIdx(pAemEntityModel->aemConfigurations, configIdx);
210  if (pConfig &&
211  pConfig->pDescriptorConfiguration &&
212  pConfig->pDescriptorConfiguration->descriptor_index == descriptorIdx) {
213  retDescriptor = pConfig->pDescriptorConfiguration;
214  }
215  }
216  }
217  else {
218  openavb_array_t descriptors = NULL;
219  if (configIdx != OPENAVB_AEM_DESCRIPTOR_INVALID) {
220  openavb_aem_configuration_t *pConfig = openavbArrayDataIdx(pAemEntityModel->aemConfigurations, configIdx);
221  if (pConfig) {
222  descriptors = pConfig->descriptorsArray[descriptorType];
223  if (descriptors) {
224  retDescriptor = openavbArrayDataIdx(descriptors, descriptorIdx);
225  }
226  }
227  }
228 
229  // Search the alternate location for non-top-level descriptors.
230  if (!retDescriptor) {
231  descriptors = pAemEntityModel->aemNonTopLevelDescriptorsArray[descriptorType];
232  if (descriptors) {
233  retDescriptor = openavbArrayDataIdx(descriptors, descriptorIdx);
234  }
235  }
236  }
237 
238  AVB_TRACE_EXIT(AVB_TRACE_AEM);
239  return retDescriptor;
240 }

330 extern DLL_EXPORT bool openavbAemAddDescriptor(void *pDescriptor, U16 configIdx, U16 *pResultIdx)
331 {
332  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
333 
334  // Make sure Entity Model is created
335  if (!openavbAemCheckModel(TRUE)) {
336  AVB_TRACE_EXIT(AVB_TRACE_AEM);
337  return FALSE;
338  }
339 
340  if (!pDescriptor || !pResultIdx) {
341  AVB_RC_LOG(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_INVALID_ARGUMENT);
342  AVB_TRACE_EXIT(AVB_TRACE_AEM);
343  return FALSE;
344  }
345 
346  openavb_aem_descriptor_common_t *pDescriptorCommon = pDescriptor;
347 
348  if (pDescriptorCommon->descriptor_type == OPENAVB_AEM_DESCRIPTOR_ENTITY) {
349  // Entity Descriptors aren't handled in this function.
350  AVB_RC_LOG(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_INVALID_ARGUMENT);
351  AVB_TRACE_EXIT(AVB_TRACE_AEM);
352  return FALSE;
353  }
354  if (pDescriptorCommon->descriptor_type == OPENAVB_AEM_DESCRIPTOR_CONFIGURATION) {
355  // Configuration descriptors have special processing
356  bool result = openavbAemAddDescriptorConfiguration(pDescriptor, pResultIdx);
357  AVB_TRACE_EXIT(AVB_TRACE_AEM);
358  return result;
359  }
360 
361  openavb_array_t descriptors = NULL;
362  if (pDescriptorCommon->descriptorPvtPtr->bTopLevel) {
363  openavb_aem_configuration_t *pConfig = openavbArrayDataIdx(pAemEntityModel->aemConfigurations, configIdx);
364  if (!pConfig) {
365  AVB_RC_LOG(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_INVALID_ARGUMENT);
366  AVB_LOGF_ERROR("Invalid configuration index %u", configIdx);
367  AVB_TRACE_EXIT(AVB_TRACE_AEM);
368  return FALSE;
369  }
370  descriptors = pConfig->descriptorsArray[pDescriptorCommon->descriptor_type];
371 
372  if (!descriptors) {
373  // Create array for this type of descriptor in this configuration
374  AVB_LOGF_DEBUG("Created description array for configuration %u, type 0x%04x", configIdx, pDescriptorCommon->descriptor_type);
375  descriptors = openavbArrayNewArray(pDescriptorCommon->descriptorPvtPtr->size);
376  pConfig->descriptorsArray[pDescriptorCommon->descriptor_type] = descriptors;
377  }
378  } else {
379  descriptors = pAemEntityModel->aemNonTopLevelDescriptorsArray[pDescriptorCommon->descriptor_type];
380 
381  if (!descriptors) {
382  // Create array for this type of descriptor in this configuration
383  AVB_LOGF_DEBUG("Created non-top-level description array for type 0x%04x", pDescriptorCommon->descriptor_type);
384  descriptors = openavbArrayNewArray(pDescriptorCommon->descriptorPvtPtr->size);
385  pAemEntityModel->aemNonTopLevelDescriptorsArray[pDescriptorCommon->descriptor_type] = descriptors;
386  }
387  }
388 
389  if (descriptors) {
390  if (openavbArrayFindData(descriptors, pDescriptor) >= 0) {
391  AVB_LOGF_ERROR("Attempted to add same descriptor (type 0x%04x) twice", pDescriptorCommon->descriptor_type);
392  AVB_TRACE_EXIT(AVB_TRACE_AEM);
393  return FALSE;
394  }
395 
396  openavb_array_elem_t elem = openavbArrayAdd(descriptors, pDescriptor);
397  if (elem) {
398  *pResultIdx = openavbArrayGetIdx(elem);
399  pDescriptorCommon->descriptor_index = *pResultIdx;
400  if (pDescriptorCommon->descriptorPvtPtr->bTopLevel) {
401  if (!IS_OPENAVB_SUCCESS(openavbAemAddDescriptorToConfiguration(pDescriptorCommon->descriptor_type, configIdx))) {
402  AVB_TRACE_EXIT(AVB_TRACE_AEM);
403  return FALSE;
404  }
405  }
406  AVB_TRACE_EXIT(AVB_TRACE_AEM);
407  return TRUE;
408  }
409  }
410 
411  AVB_TRACE_EXIT(AVB_TRACE_AEM);
412  return FALSE;
413 }

427 {

429  U16 descriptorIdx = 0;
