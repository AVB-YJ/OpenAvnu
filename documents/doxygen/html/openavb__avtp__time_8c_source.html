163 // This function will take a AVTP timestamp and set the local AvtpTime values for it.
164 // Commonly this is used in the RX mapping callbacks, to take the AVTP timestamp received
165 // on the listener and place it into the AvtpTime so that the media queue and determine
166 // the correct time to release the media queue item for presentation.
167 void openavbAvtpTimeSetToTimestamp(avtp_time_t *pAvtpTime, U32 timestamp)
168 {
169  if (pAvtpTime) {
170  timespec_t tmNow;
171  if (CLOCK_GETTIME(OPENAVB_CLOCK_WALLTIME, &tmNow)) {
172  U64 nsNow = x_getNsTime(&tmNow);
173  U32 tsNow = x_getTimestamp(nsNow);
174 
175  U32 delta;
176  if (tsNow < timestamp) {
177  delta = timestamp - tsNow;
178  }
179  else if (tsNow > timestamp) {
180  delta = timestamp + (0x100000000ULL - tsNow);
181  }
182  else {
183  delta = 0;
184  }
185 
186  if (delta < OPENAVB_AVTP_TIME_MAX_TS_DIFF) {
187  // Normal case, timestamp is upcoming
188  pAvtpTime->timeNsec = nsNow + delta;
189  }
190  else {
191  // Timestamp is past
192  pAvtpTime->timeNsec = nsNow - (0x100000000ULL - delta);
193  }
194 
195  pAvtpTime->bTimestampValid = TRUE;
196  pAvtpTime->bTimestampUncertain = FALSE;
197  }
198  else {
199  pAvtpTime->bTimestampValid = FALSE;
200  pAvtpTime->bTimestampUncertain = TRUE;
201  }
202  }
203  else {
204  AVB_RC_LOG(AVB_RC(OPENAVB_AVTP_TIME_FAILURE | OPENAVBAVTPTIME_RC_INVALID_PTP_TIME));
205  }
206 }

315 // Check if the AvtpTime is past now. If something goes wrong return true to allow
316 // data to continue to flow.
317 bool openavbAvtpTimeIsPast(avtp_time_t *pAvtpTime)
318 {
319  if (pAvtpTime) {
320  timespec_t tmNow;
321 
322  if (!pAvtpTime->bTimestampValid || pAvtpTime->bTimestampUncertain) {
323  return TRUE; // If timestamp can't be trusted assume time is past.
324  }
325 
326  if (CLOCK_GETTIME(OPENAVB_CLOCK_WALLTIME, &tmNow)) {
327  U64 nsNow = x_getNsTime(&tmNow);
328 
329  if (nsNow >= pAvtpTime->timeNsec) {
330  return TRUE; // Normal timestamp time reached.
331  }
332 
333  if (nsNow + pAvtpTime->maxLatencyNsec < pAvtpTime->timeNsec) {
334  IF_LOG_INTERVAL(100) {
335  AVB_LOGF_INFO("Timestamp out of range: Now:%" PRIu64 " TSTime%" PRIu64 " MaxLatency:%" PRIu64 "ns Delta:%" PRIu64 "ns", nsNow, pAvtpTime->timeNsec, pAvtpTime->maxLatencyNsec, pAvtpTime->timeNsec - nsNow);
336  }
337  return TRUE;
338  }
339 
340  return FALSE;
341  }
342  else {
343  return TRUE; // If timestamp can't be retrieved assume time is past to keep data flowing.
344  }
345  }
346  else {
347  AVB_RC_LOG(AVB_RC(OPENAVB_AVTP_TIME_FAILURE | OPENAVBAVTPTIME_RC_INVALID_PTP_TIME));
348  return TRUE;
349  }
350 }

381 bool openavbAvtpTimeUsecTill(avtp_time_t *pAvtpTime, U32 *pUsecTill)
382 {
383  if (pAvtpTime) {
384  if (pAvtpTime->bTimestampValid && !pAvtpTime->bTimestampUncertain) {
385 
386  timespec_t tmNow;
387 
388  if (CLOCK_GETTIME(OPENAVB_CLOCK_WALLTIME, &tmNow)) {
389  U64 nsNow = x_getNsTime(&tmNow);
390 
391  if (pAvtpTime->timeNsec >= nsNow) {
392  U32 usecTill = (pAvtpTime->timeNsec - nsNow) / NANOSECONDS_PER_USEC;
393 
394  if (usecTill <= MICROSECONDS_PER_SECOND * 5) {
395  *pUsecTill = usecTill;
396  return TRUE;
397  }
398  else {
399  return FALSE;
400  }
401  }
402  else {
403  *pUsecTill = 0;
404  return TRUE;
405  }
406  }
407  }
408 
409  // When the timestamp is invalid or uncertain assume timestamp is now.
410  *pUsecTill = 0;
411  return TRUE;
412  }
413  else {
414  AVB_RC_LOG(AVB_RC(OPENAVB_AVTP_TIME_FAILURE | OPENAVBAVTPTIME_RC_INVALID_PTP_TIME));
415  return FALSE;
416  }
417 }
