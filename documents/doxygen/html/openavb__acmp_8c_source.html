63 openavbRC openavbAcmpStart()
64 {
65  AVB_TRACE_ENTRY(AVB_TRACE_ACMP);
66 
67  openavb_avdecc_entity_model_t *pAem = openavbAemGetModel();
68  if (!pAem) {
69  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVBAVDECC_RC_ENTITY_MODEL_MISSING), AVB_TRACE_ACMP);
70  }
71 
72  {
73  MUTEX_ATTR_HANDLE(mta);
74  MUTEX_ATTR_INIT(mta);
75  MUTEX_ATTR_SET_TYPE(mta, MUTEX_ATTR_TYPE_DEFAULT);
76  MUTEX_ATTR_SET_NAME(mta, "openavbAcmpMutex");
77  MUTEX_CREATE_ERR();
78  MUTEX_CREATE(openavbAcmpMutex, mta);
79  MUTEX_LOG_ERR("Could not create/initialize 'openavbAcmpMutex' mutex");
80  }
81 
82  ACMP_LOCK();
83  {
84  // Populate global ACMP vars
85  openavb_acmp_control_header_t *pHeader = &openavbAcmpSMGlobalVars.controlHeader;
86 
87  pHeader->cd = 1;
88  pHeader->subtype = OPENAVB_ACMP_AVTP_SUBTYPE;
89  pHeader->sv = 0;
90  pHeader->version = 0;
91  //pHeader->message_type = 0; // Set later depending on message type
92  //pHeader->status = 0; // Set later depending on message
93  pHeader->control_data_length = 44;
94  //pHeader->stream_id; // Set later depending on stream
95  }
96 
97  {
98  memcpy(openavbAcmpSMGlobalVars.my_id, pAem->pDescriptorEntity->entity_id, sizeof(openavbAcmpSMGlobalVars.my_id));
99  }
100  ACMP_UNLOCK();
101 
102  openavbRC rc = openavbAcmpMessageHandlerStart();
103  if (IS_OPENAVB_FAILURE(rc)) {
104  openavbAcmpStop();
105  AVB_RC_TRACE_RET(rc, AVB_TRACE_ACMP);
106  }
107 
108  if (gAvdeccCfg.bListener) {
109  if (!openavbAcmpSMListenerStart()) {
110  AVB_RC_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVBAVDECC_RC_GENERIC), AVB_TRACE_ACMP);
111  }
112  bListenerStarted = TRUE;
113  }
114 
115  if (gAvdeccCfg.bTalker) {
116  if (!openavbAcmpSMTalkerStart()) {
117  AVB_RC_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVBAVDECC_RC_GENERIC), AVB_TRACE_ACMP);
118  }
119  bTalkerStarted = TRUE;
120  }
121 
122  if (!openavbAcmpSMControllerStart()) {
123  AVB_RC_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVBAVDECC_RC_GENERIC), AVB_TRACE_ACMP);
124  }
125 
126  AVB_RC_TRACE_RET(OPENAVB_AVDECC_SUCCESS, AVB_TRACE_ACMP);
127 }

129 void openavbAcmpStop()
130 {
131  AVB_TRACE_ENTRY(AVB_TRACE_ACMP);
132 
133  openavbAcmpSMControllerStop();
134 
135  if (bTalkerStarted) {
136  openavbAcmpSMTalkerStop();
137  }
138 
139  if (bListenerStarted) {
140  openavbAcmpSMListenerStop();
141  }
142 
143  openavbAcmpMessageHandlerStop();
144 
145  AVB_TRACE_EXIT(AVB_TRACE_ACMP);
146 }
