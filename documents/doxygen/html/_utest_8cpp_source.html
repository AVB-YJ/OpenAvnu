33 bool doubles_equal(double d1, double d2, double threshold)
34 {
35  if (PlatformSpecificIsNan(d1) || PlatformSpecificIsNan(d2) || PlatformSpecificIsNan(threshold))
36  return false;
37 
38  if (PlatformSpecificIsInf(d1) && PlatformSpecificIsInf(d2))
39  {
40  return true;
41  }
42 
43  return PlatformSpecificFabs(d1 - d2) <= threshold;
44 }

178 {
179  hasFailed_ = false;
180  HelperTestRunInfo runInfo(this, plugin, &result);
181  if (isRunInSeperateProcess())

197 void UtestShell::runOneTestInCurrentProcess(TestPlugin* plugin, TestResult& result)
198 {
199  plugin->runAllPreTestAction(*this, result);
200 
201  //save test context, so that test class can be tested
202  UtestShell* savedTest = UtestShell::getCurrent();
203  TestResult* savedResult = UtestShell::getTestResult();
204 
205  result.countRun();
206  UtestShell::setTestResult(&result);
207  UtestShell::setCurrentTest(this);
208 
209  Utest* testToRun = createTest();
210  testToRun->run();
211  destroyTest(testToRun);
212 
213  UtestShell::setCurrentTest(savedTest);
214  UtestShell::setTestResult(savedResult);
215 
216  plugin->runAllPostTestAction(*this, result);
217 }
