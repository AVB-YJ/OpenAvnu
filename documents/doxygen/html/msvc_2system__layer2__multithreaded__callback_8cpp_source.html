96 int system_layer2_multithreaded_callback::queue_tx_frame(void * notification_id, uint32_t notification_flag, uint8_t * frame, size_t frame_len)
97 {
98  struct poll_thread_data thread_data;
99 
100  assert(frame_len < 2048);
101  thread_data.frame = new uint8_t[2048];
102  if (!thread_data.frame)
103  {
104  exit(EXIT_FAILURE);
105  }
106  thread_data.frame_len = frame_len;
107  memcpy(thread_data.frame, frame, frame_len);
108  thread_data.notification_id = notification_id;
109  thread_data.notification_flag = notification_flag;
110  poll_tx.tx_queue->queue_push(&thread_data);
111 
112  //Check for conditions that cause wait for completion.
113  if (wait_mgr->primed_state() &&
114  wait_mgr->match_id(notification_id) &&
115  (notification_flag == CMD_WITH_NOTIFICATION))
116  {
117  int status = 0;
118 
119  status = wait_mgr->set_active_state();
120  assert(status == 0);
121  WaitForSingleObject(waiting_sem, INFINITE);
122  resp_status_for_cmd = wait_mgr->get_completion_status();
123  status = wait_mgr->set_idle_state();
124  assert(status == 0);
125  }
126  return 0;
127 }

145 int system_layer2_multithreaded_callback::proc_wpcap_thread_callback()
146 {
147  int status;
148  struct poll_thread_data thread_data;
149  const uint8_t * frame;
150  uint16_t length;
151 
152  while (WaitForSingleObject(poll_rx.queue_thread.kill_sem, 0))
153  {
154  status = netif_obj_in_system->capture_frame(&frame, &length);
155 
156  if (status > 0)
157  {
158  if (length > 2048)
159  {
160  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "wpcap returned packet larger than 1600 bytes");
161  continue;
162  }
163  thread_data.frame_len = length;
164  thread_data.frame = new uint8_t[2048];
165  if (!thread_data.frame)
166  {
167  exit(EXIT_FAILURE);
168  }
169  memcpy(thread_data.frame, frame, thread_data.frame_len);
170  poll_rx.rx_queue->queue_push(&thread_data);
171  }
172  else
173  {
174  if (!SetEvent(poll_rx.timeout_event))
175  {
176  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "SetEvent pkt_event_wpcap_timeout failed");
177  exit(EXIT_FAILURE);
178  }
179  }
180  }
181 
182  return 0;
183 }

217 int system_layer2_multithreaded_callback::init_wpcap_thread()
218 {
219  poll_rx.rx_queue = new system_message_queue(sizeof(struct poll_thread_data));
220  poll_rx.queue_thread.kill_sem = CreateSemaphore(NULL, 0, 32767, NULL);
221  poll_rx.timeout_event = CreateEvent(NULL, FALSE, FALSE, NULL);
222  poll_events_array[WPCAP_TIMEOUT] = poll_rx.timeout_event;
223  poll_events_array[WPCAP_RX_PACKET] = poll_rx.rx_queue->queue_data_available_object();
224  poll_rx.queue_thread.handle = CreateThread(NULL, // Default security descriptor
225  0, // Default stack size
226  proc_wpcap_thread, // Point to the start address of the thread
227  this, // Data to be passed to the thread
228  0, // Flag controlling the creation of the thread
229  &poll_rx.queue_thread.id // Thread identifier
230  );
231 
232  if (poll_rx.queue_thread.handle == NULL)
233  {
234  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "Error creating the wpcap thread");
235  exit(EXIT_FAILURE);
236  }
237 
238  poll_tx.tx_queue = new system_message_queue(sizeof(struct poll_thread_data));
239  poll_tx.queue_thread.kill_sem = CreateSemaphore(NULL, 0, 32767, NULL);
240  poll_tx.timeout_event = CreateEvent(NULL, FALSE, FALSE, NULL);
241  poll_events_array[WPCAP_TX_PACKET] = poll_tx.tx_queue->queue_data_available_object();
242 
243  poll_events_array[KILL_ALL] = CreateEvent(NULL, FALSE, FALSE, NULL);
244 
245  waiting_sem = CreateSemaphore(NULL, 0, 32767, NULL);
246 
247  return 0;

270 int system_layer2_multithreaded_callback::poll_single()
271 {
272  struct poll_thread_data thread_data;
273  DWORD dwEvent;
274  DWORD poll_count = sizeof(poll_events_array) / sizeof(HANDLE);
275  int status = 0;
276 
277  //adp_discovery_state_machine_ref->state_discover(NULL); // Send ENTITY_DISCOVER message
278 
279  dwEvent = WaitForMultipleObjects(poll_count, poll_events_array, FALSE, INFINITE);
280 
281  switch (dwEvent)
282  {
283  case WAIT_OBJECT_0 + WPCAP_TIMEOUT:
284  break;
285 
286  case WAIT_OBJECT_0 + WPCAP_RX_PACKET:
287  {
288  poll_rx.rx_queue->queue_pop_nowait(&thread_data);
289 
290  bool is_notification_id_valid = false;
291  int rx_status = -1;
292  uint16_t operation_id = 0;
293  bool is_operation_id_valid = false;
294 
295  controller_obj_in_system->rx_packet_event(thread_data.notification_id,
296  is_notification_id_valid,
297  thread_data.frame,
298  thread_data.frame_len,
299  rx_status,
300  operation_id,
301  is_operation_id_valid);
302 
303  if (
304  wait_mgr->active_state() &&
305  is_notification_id_valid &&
306  wait_mgr->match_id(thread_data.notification_id) &&
307  !controller_obj_in_system->is_inflight_cmd_with_notification_id(wait_mgr->get_notify_id()) &&
308  !controller_obj_in_system->is_active_operation_with_notification_id(wait_mgr->get_notify_id()))
309  {
310  int status = wait_mgr->set_completion_status(rx_status);
311  assert(status == 0);
312  ReleaseSemaphore(waiting_sem, 1, NULL);
313  }
314  delete[] thread_data.frame;
315  }
316  break;
317 
318  case WAIT_OBJECT_0 + WPCAP_TX_PACKET:
319  poll_tx.tx_queue->queue_pop_nowait(&thread_data);
320 
321  controller_obj_in_system->tx_packet_event(thread_data.notification_id,
322  thread_data.notification_flag,
323  thread_data.frame,
324  thread_data.frame_len);
325  delete[] thread_data.frame;
326  break;
327 
328  case WAIT_OBJECT_0 + KILL_ALL: // Exit or kill event
329  status = -1;
330  break;
331  }
332 
333  if (tick_timer.timeout()) // Check tick timeout
334  {
335  bool notification_id_incomplete = false;
336 
337  if (wait_mgr->active_state())
338  {
339  if (controller_obj_in_system->is_inflight_cmd_with_notification_id(wait_mgr->get_notify_id()) ||
340  controller_obj_in_system->is_active_operation_with_notification_id(wait_mgr->get_notify_id()))
341  notification_id_incomplete = true;
342  }
343 
344  // If waiting on a command to complete and the command was "inflight" prior to calling the
345  // timer tick update (ie notification_id_incomplete == true) AND after calling the timer tick
346  // update the command is no longer "inflight", timeout processing has removed it, so signal the
347  // waiting app thread.
348  controller_obj_in_system->time_tick_event();
349 
350  bool is_timeout_for_waiting_notify_id = wait_mgr->active_state() && notification_id_incomplete &&
351  !controller_obj_in_system->is_inflight_cmd_with_notification_id(wait_mgr->get_notify_id()) &&
352  !controller_obj_in_system->is_active_operation_with_notification_id(wait_mgr->get_notify_id());
353  if (is_timeout_for_waiting_notify_id)
354  {
355  int status = wait_mgr->set_completion_status(AVDECC_LIB_STATUS_TICK_TIMEOUT);
356  assert(status == 0);
357  ReleaseSemaphore(waiting_sem, 1, NULL);
358  }
359 
360  tick_timer.start(NETIF_READ_TIMEOUT_MS);
361  }
362 
363  return status;
364 }

366 int STDCALL system_layer2_multithreaded_callback::process_close()
367 {
368 
369  /**************** Send kill events to threads ****************/
370  ReleaseSemaphore(poll_rx.queue_thread.kill_sem, 1, NULL);
371  ReleaseSemaphore(poll_thread.kill_sem, 1, NULL);
372  SetEvent(poll_events_array[KILL_ALL]);
373 
374  while (WaitForSingleObject(poll_rx.queue_thread.handle, 0) != WAIT_OBJECT_0)
375  {
376  Sleep(100);
377  }
378  while (WaitForSingleObject(poll_thread.handle, 0) != WAIT_OBJECT_0)
379  {
380  Sleep(100);
381  }
382 
383  CloseHandle(poll_rx.queue_thread.kill_sem);
384  CloseHandle(poll_thread.kill_sem);
385 
386  return 0;
387 }
