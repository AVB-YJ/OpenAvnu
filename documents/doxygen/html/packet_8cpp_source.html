83 packet_error_t openInterfaceByAddr( struct packet_handle *handle, packet_addr_t *addr, int timeout ) {
84  packet_error_t ret = PACKET_NO_ERROR;
85  char name[WINPCAP_INTERFACENAME_LENGTH+1] = "\0";
86 
87  PIP_ADAPTER_ADDRESSES pAdapterAddress;
88  IP_ADAPTER_ADDRESSES AdapterAddress[32]; // Allocate information for up to 32 NICs
89  DWORD dwBufLen = sizeof(AdapterAddress); // Save memory size of buffer
90 
91  DWORD dwStatus = GetAdaptersAddresses( AF_UNSPEC, 0, NULL, AdapterAddress, &dwBufLen);
92 
93  if( dwStatus != ERROR_SUCCESS ) {
94  ret = PACKET_IFLOOKUP_ERROR;
95  goto fnexit;
96  }
97 
98  for( pAdapterAddress = AdapterAddress; pAdapterAddress != NULL; pAdapterAddress = pAdapterAddress->Next ) {
99  if( pAdapterAddress->PhysicalAddressLength == ETHER_ADDR_OCTETS &&
100  memcmp( pAdapterAddress->PhysicalAddress, addr->addr, ETHER_ADDR_OCTETS ) == 0 ) {
101  break;
102  }
103  }
104 
105  if( pAdapterAddress != NULL ) {
106  strcpy_s( name, WINPCAP_INTERFACENAMEPREFIX );
107  strncpy_s( name+WINPCAP_INTERFACENAMEPREFIX_LENGTH, WINPCAP_INTERFACENAMESUFFIX_LENGTH+1, pAdapterAddress->AdapterName, WINPCAP_INTERFACENAMESUFFIX_LENGTH );
108  printf( "Opening: %s\n", name );
109  handle->iface = pcap_open( name, MAX_FRAME_SIZE, PCAP_OPENFLAG_MAX_RESPONSIVENESS | PCAP_OPENFLAG_PROMISCUOUS | PCAP_OPENFLAG_NOCAPTURE_LOCAL,
110  timeout, NULL, handle->errbuf );
111  if( handle->iface == NULL ) {
112  ret = PACKET_IFLOOKUP_ERROR;
113  goto fnexit;
114  }
115  handle->iface_addr = *addr;
116  } else {
117  ret = PACKET_IFNOTFOUND_ERROR;
118  goto fnexit;
119  }
120 
121 fnexit:

123 }

131  packet_error_t ret = PACKET_NO_ERROR;
132  uint16_t ethertype_no = PLAT_htons( ethertype );
133  uint8_t *payload_ptr = payload;
134 
135  if( length < PACKET_HDR_LENGTH ) {
136  ret = PACKET_BADBUFFER_ERROR;
137  }
138 
139  // Build Header
140  memcpy( payload_ptr, addr->addr, ETHER_ADDR_OCTETS ); payload_ptr+= ETHER_ADDR_OCTETS;
141  memcpy( payload_ptr, handle->iface_addr.addr, ETHER_ADDR_OCTETS ); payload_ptr+= ETHER_ADDR_OCTETS;
142  memcpy( payload_ptr, &ethertype_no, sizeof( ethertype_no ));
143 
144  if( pcap_sendpacket( handle->iface, payload, (int) length+PACKET_HDR_LENGTH ) != 0 ) {
145  ret = PACKET_XMIT_ERROR;
146  goto fnexit;
147  }
148 
149 fnexit:

154  packet_error_t ret = PACKET_NO_ERROR;

156 
157  sprintf_s( filter_expression+strlen(filter_expression), 31-strlen(filter_expression), "%hx", ethertype );
158  if( pcap_compile( handle->iface, &handle->filter, filter_expression, 1, 0 ) == -1 ) {
159  ret = PACKET_BIND_ERROR;

164  goto fnexit;
165  }

173 // Call to recvFrame must be thread-safe. However call to pcap_next_ex() isn't because of somewhat undefined memory management semantics.
174 // Wrap call to pcap library with mutex
175 packet_error_t recvFrame( struct packet_handle *handle, packet_addr_t *addr, uint8_t *payload, size_t &length ) {
176  packet_error_t ret = PACKET_NO_ERROR;
177  struct pcap_pkthdr *hdr_r;
178  u_char *data;
179 
180  int pcap_result;
181  DWORD wait_result;
182 
183  wait_result = WaitForSingleObject( handle->capture_lock, 1000 );
184  if( wait_result != WAIT_OBJECT_0 ) {
185  ret = PACKET_GETMUTEX_ERROR;
186  goto fnexit;
187  }
188 
189  pcap_result = pcap_next_ex( handle->iface, &hdr_r, (const u_char **) &data );
190  if( pcap_result == 0 ) {
191  ret = PACKET_RECVTIMEOUT_ERROR;
192  } else if( pcap_result < 0 ) {
193  ret = PACKET_RECVFAILED_ERROR;
194  } else {
195  length = hdr_r->len-PACKET_HDR_LENGTH >= length ? length : hdr_r->len-PACKET_HDR_LENGTH;
196  memcpy( payload, data+PACKET_HDR_LENGTH, length );
197  memcpy( addr->addr, data+ETHER_ADDR_OCTETS, ETHER_ADDR_OCTETS );
198  }
199 
200  if( !ReleaseMutex( handle->capture_lock )) {
201  ret = PACKET_RLSMUTEX_ERROR;
202  goto fnexit;
203  }
204 
205 fnexit:
206  return ret;
207 }
