63 // Check if packet size is SIZE by looking for the first N sync bytes
64 int check_size(FILE* pFile, int size, int first)
65 {
66  int i; char b;
67  assert(pFile);
68  assert(size >= BASE_TS_PACKET_SIZE);
69  assert(first >= 0);
70 
71  for (i = 0; i < CHECK_COUNT; i++) {
72  fseek(pFile, first + i * size, 0);
73  b = fgetc(pFile);
74  if (b != SYNC_BYTE)
75  return 0;
76  }
77 
78  fprintf(stderr, "transport packet size is %d\n", size);
79  return size;
80 }
