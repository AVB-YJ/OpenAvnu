63 // Each configuration name value pair for this mapping will result in this callback being called.
64 void openavbIntfNullCfgCB(media_q_t *pMediaQ, const char *name, const char *value)
65 {
66  AVB_TRACE_ENTRY(AVB_TRACE_INTF);
67 
68  if (pMediaQ) {
69  char *pEnd;
70  long tmp;
71 
72  pvt_data_t *pPvtData = pMediaQ->pPvtIntfInfo;
73  if (!pPvtData) {
74  AVB_LOG_ERROR("Private interface module data not allocated.");
75  return;
76  }
77 
78  if (strcmp(name, "intf_nv_ignore_timestamp") == 0) {
79  tmp = strtol(value, &pEnd, 10);
80  if (*pEnd == '\0' && tmp == 1) {
81  pPvtData->ignoreTimestamp = (tmp == 1);
82  }
83  }
84  }
85 
86  AVB_TRACE_EXIT(AVB_TRACE_INTF);
87 }

103 // This callback will be called for each AVB transmit interval. Commonly this will be
104 // 4000 or 8000 times per second.
105 bool openavbIntfNullTxCB(media_q_t *pMediaQ)
106 {
107  AVB_TRACE_ENTRY(AVB_TRACE_INTF_DETAIL);
108  if (pMediaQ) {
109  //pvt_data_t *pPvtData = pMediaQ->pPvtIntfInfo;
110 
111  media_q_item_t *pMediaQItem = openavbMediaQHeadLock(pMediaQ);
112  if (pMediaQItem) {
113  openavbAvtpTimeSetToWallTime(pMediaQItem->pAvtpTime);
114 
115  // Set 1 byte
116  *(U8 *)(pMediaQItem->pPubData) = 0xff;
117  pMediaQItem->dataLen = 1;
118 
119  openavbMediaQHeadPush(pMediaQ);
120  AVB_TRACE_EXIT(AVB_TRACE_INTF_DETAIL);
121  return TRUE;
122  }
123  else {
124  AVB_TRACE_EXIT(AVB_TRACE_INTF_DETAIL);
125  return FALSE; // Media queue full
126  }
127  }
128  AVB_TRACE_EXIT(AVB_TRACE_INTF_DETAIL);
129  return FALSE;
130 }

140 // This callback is called when acting as a listener.
141 bool openavbIntfNullRxCB(media_q_t *pMediaQ)
142 {
143  AVB_TRACE_ENTRY(AVB_TRACE_INTF_DETAIL);
144  if (pMediaQ) {
145  bool moreItems = TRUE;
146  pvt_data_t *pPvtData = pMediaQ->pPvtIntfInfo;
147  if (!pPvtData) {
148  AVB_LOG_ERROR("Private interface module data not allocated.");
149  return FALSE;
150  }
151 
152  while (moreItems) {
153  media_q_item_t *pMediaQItem = openavbMediaQTailLock(pMediaQ, pPvtData->ignoreTimestamp);
154  if (pMediaQItem) {
155  openavbMediaQTailPull(pMediaQ);
156  }
157  else {
158  moreItems = FALSE;
159  }
160  }
161  }
162  AVB_TRACE_EXIT(AVB_TRACE_INTF_DETAIL);
163  return FALSE;
164 }

180 // Main initialization entry point into the interface module
181 extern DLL_EXPORT bool openavbIntfNullInitialize(media_q_t *pMediaQ, openavb_intf_cb_t *pIntfCB)
182 {
183  AVB_TRACE_ENTRY(AVB_TRACE_INTF);
184 
185  if (pMediaQ) {
186  pMediaQ->pPvtIntfInfo = calloc(1, sizeof(pvt_data_t)); // Memory freed by the media queue when the media queue is destroyed.
187 
188  if (!pMediaQ->pPvtIntfInfo) {
189  AVB_LOG_ERROR("Unable to allocate memory for AVTP interface module.");
190  return FALSE;
191  }
192 
193  pvt_data_t *pPvtData = pMediaQ->pPvtIntfInfo;
194 
195  pIntfCB->intf_cfg_cb = openavbIntfNullCfgCB;
196  pIntfCB->intf_gen_init_cb = openavbIntfNullGenInitCB;
197  pIntfCB->intf_tx_init_cb = openavbIntfNullTxInitCB;
198  pIntfCB->intf_tx_cb = openavbIntfNullTxCB;
199  pIntfCB->intf_rx_init_cb = openavbIntfNullRxInitCB;
200  pIntfCB->intf_rx_cb = openavbIntfNullRxCB;
201  pIntfCB->intf_end_cb = openavbIntfNullEndCB;
202  pIntfCB->intf_gen_end_cb = openavbIntfNullGenEndCB;
203 
204  pPvtData->ignoreTimestamp = FALSE;
205  }
206 
207  AVB_TRACE_EXIT(AVB_TRACE_INTF);
208  return TRUE;
209 }
