35 /* In the hardware, registers are laid out either singly, in arrays
36  * spaced 0x100 bytes apart, or in contiguous tables. We assume
37  * most tests take place on arrays or single registers (handled
38  * as a single-element array) and special-case the tables.
39  * Table tests are always pattern tests.
40  *
41  * We also make provision for some required setup steps by specifying
42  * registers to be written without any read-back testing.
43  */
44

46 #define SET_READ_TEST 2
47 #define WRITE_NO_TEST 3
48 #define TABLE32_TEST 4
49 #define TABLE64_TEST_LO 5
