48 bool LinuxTimestamperGeneric::HWTimestamper_adjclockphase( int64_t phase_adjust ) {
49  struct timex tx;
50  LinuxNetworkInterfaceList::iterator iface_iter;
51  bool ret = true;
52  LinuxTimerFactory factory;
53  OSTimer *timer = factory.createTimer();
54 
55  /* Walk list of interfaces disabling them all */
56  iface_iter = iface_list.begin();
57  for
58  ( iface_iter = iface_list.begin(); iface_iter != iface_list.end();
59  ++iface_iter ) {
60  (*iface_iter)->disable_rx_queue();
61  }
62 
63  rxTimestampList.clear();
64 
65  /* Wait 180 ms - This is plenty of time for any time sync frames
66  to clear the queue */
67  timer->sleep(180000);
68 
69  ++version;
70 
71  tx.modes = ADJ_SETOFFSET | ADJ_NANO;
72  if( phase_adjust >= 0 ) {
73  tx.time.tv_sec = phase_adjust / 1000000000LL;
74  tx.time.tv_usec = phase_adjust % 1000000000LL;
75  } else {
76  tx.time.tv_sec = (phase_adjust / 1000000000LL)-1;
77  tx.time.tv_usec = (phase_adjust % 1000000000LL)+1000000000;
78  }
79 
80  if( !Adjust( &tx )) {
81  ret = false;
82  }
83 
84  // Walk list of interfaces re-enabling them
85  iface_iter = iface_list.begin();
86  for( iface_iter = iface_list.begin(); iface_iter != iface_list.end();
87  ++iface_iter ) {
88  (*iface_iter)->clear_reenable_rx_queue();
89  }
90 
91  delete timer;
92  return ret;
93 }
