50 static char * number(char * buf, char * end, long long num, int base, int size, int precision, int type)
51 {
52  char c,sign,tmp[66];
53  const char *digits;
54  const char small_digits[] = "0123456789abcdefghijklmnopqrstuvwxyz";
55  const char large_digits[] = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ";
56  int i;
57 
58  digits = (type & _kc_LARGE) ? large_digits : small_digits;
59  if (type & _kc_LEFT)
60  type &= ~_kc_ZEROPAD;
61  if (base < 2 || base > 36)
62  return 0;
63  c = (type & _kc_ZEROPAD) ? '0' : ' ';
64  sign = 0;
65  if (type & _kc_SIGN) {
66  if (num < 0) {
67  sign = '-';
68  num = -num;
69  size--;
70  } else if (type & _kc_PLUS) {
71  sign = '+';
72  size--;
73  } else if (type & _kc_SPACE) {
74  sign = ' ';
75  size--;
76  }
77  }
78  if (type & _kc_SPECIAL) {
79  if (base == 16)
80  size -= 2;
81  else if (base == 8)
82  size--;
83  }
84  i = 0;
85  if (num == 0)
86  tmp[i++]='0';
87  else while (num != 0)
88  tmp[i++] = digits[do_div(num,base)];
89  if (i > precision)
90  precision = i;
91  size -= precision;
92  if (!(type&(_kc_ZEROPAD+_kc_LEFT))) {
93  while(size-->0) {
94  if (buf <= end)
95  *buf = ' ';
96  ++buf;
97  }
98  }
99  if (sign) {
100  if (buf <= end)
101  *buf = sign;
102  ++buf;
103  }
104  if (type & _kc_SPECIAL) {
105  if (base==8) {
106  if (buf <= end)
107  *buf = '0';
108  ++buf;
109  } else if (base==16) {
110  if (buf <= end)
111  *buf = '0';
112  ++buf;
113  if (buf <= end)
114  *buf = digits[33];
115  ++buf;
116  }
117  }
118  if (!(type & _kc_LEFT)) {
119  while (size-- > 0) {
120  if (buf <= end)
121  *buf = c;
122  ++buf;
123  }
124  }
125  while (i < precision--) {
126  if (buf <= end)
127  *buf = '0';
128  ++buf;
129  }
130  while (i-- > 0) {
131  if (buf <= end)
132  *buf = tmp[i];
133  ++buf;
134  }
135  while (size-- > 0) {
136  if (buf <= end)
137  *buf = ' ';
138  ++buf;
139  }
140  return buf;
141 }

143 int _kc_vsnprintf(char *buf, size_t size, const char *fmt, va_list args)
144 {
145  int len;
146  unsigned long long num;
147  int i, base;
148  char *str, *end, c;
149  const char *s;
150 
151  int flags; /* flags to number() */
152 
153  int field_width; /* width of output field */
154  int precision; /* min. # of digits for integers; max
155  number of chars for from string */
156  int qualifier; /* 'h', 'l', or 'L' for integer fields */
157  /* 'z' support added 23/7/1999 S.H. */
158  /* 'z' changed to 'Z' --davidm 1/25/99 */
159 
160  str = buf;
161  end = buf + size - 1;
162 
163  if (end < buf - 1) {
164  end = ((void *) -1);
165  size = end - buf + 1;
166  }
167 
168  for (; *fmt ; ++fmt) {
169  if (*fmt != '%') {
170  if (str <= end)
171  *str = *fmt;
172  ++str;
173  continue;
174  }
175 
176  /* process flags */
177  flags = 0;
178  repeat:
179  ++fmt; /* this also skips first '%' */
180  switch (*fmt) {
181  case '-': flags |= _kc_LEFT; goto repeat;
182  case '+': flags |= _kc_PLUS; goto repeat;
183  case ' ': flags |= _kc_SPACE; goto repeat;
184  case '#': flags |= _kc_SPECIAL; goto repeat;
185  case '0': flags |= _kc_ZEROPAD; goto repeat;
186  }
187 
188  /* get field width */
189  field_width = -1;
190  if (isdigit(*fmt))
191  field_width = skip_atoi(&fmt);
192  else if (*fmt == '*') {
193  ++fmt;
194  /* it's the next argument */
195  field_width = va_arg(args, int);
196  if (field_width < 0) {
197  field_width = -field_width;
198  flags |= _kc_LEFT;
199  }
200  }
201 
202  /* get the precision */
203  precision = -1;
204  if (*fmt == '.') {
205  ++fmt;
206  if (isdigit(*fmt))
207  precision = skip_atoi(&fmt);
208  else if (*fmt == '*') {
209  ++fmt;
210  /* it's the next argument */
211  precision = va_arg(args, int);
212  }
213  if (precision < 0)
214  precision = 0;
215  }
216 
217  /* get the conversion qualifier */
218  qualifier = -1;
219  if (*fmt == 'h' || *fmt == 'l' || *fmt == 'L' || *fmt =='Z') {
220  qualifier = *fmt;
221  ++fmt;
222  }
223 
224  /* default base */
225  base = 10;
226 
227  switch (*fmt) {
228  case 'c':
229  if (!(flags & _kc_LEFT)) {
230  while (--field_width > 0) {
231  if (str <= end)
232  *str = ' ';
233  ++str;
234  }
235  }
236  c = (unsigned char) va_arg(args, int);
237  if (str <= end)
238  *str = c;
239  ++str;
240  while (--field_width > 0) {
241  if (str <= end)
242  *str = ' ';
243  ++str;
244  }
245  continue;
246 
247  case 's':
248  s = va_arg(args, char *);
249  if (!s)
250  s = "<NULL>";
251 
252  len = strnlen(s, precision);
253 
254  if (!(flags & _kc_LEFT)) {
255  while (len < field_width--) {
256  if (str <= end)
257  *str = ' ';
258  ++str;
259  }
260  }
261  for (i = 0; i < len; ++i) {
262  if (str <= end)
263  *str = *s;
264  ++str; ++s;
265  }
266  while (len < field_width--) {
267  if (str <= end)
268  *str = ' ';
269  ++str;
270  }
271  continue;
272 
273  case 'p':
274  if (field_width == -1) {
275  field_width = 2*sizeof(void *);
276  flags |= _kc_ZEROPAD;
277  }
278  str = number(str, end,
279  (unsigned long) va_arg(args, void *),
280  16, field_width, precision, flags);
281  continue;
282 
283 
284  case 'n':
285  /* FIXME:
286  * What does C99 say about the overflow case here? */
287  if (qualifier == 'l') {
288  long * ip = va_arg(args, long *);
289  *ip = (str - buf);
290  } else if (qualifier == 'Z') {
291  size_t * ip = va_arg(args, size_t *);
292  *ip = (str - buf);
293  } else {
294  int * ip = va_arg(args, int *);
295  *ip = (str - buf);
296  }
297  continue;
298 
299  case '%':
300  if (str <= end)
301  *str = '%';
302  ++str;
303  continue;
304 
305  /* integer number formats - set up the flags and "break" */
306  case 'o':
307  base = 8;
308  break;
309 
310  case 'X':
311  flags |= _kc_LARGE;
312  case 'x':
313  base = 16;
314  break;
315 
316  case 'd':
317  case 'i':
318  flags |= _kc_SIGN;
319  case 'u':
320  break;
321 
322  default:
323  if (str <= end)
324  *str = '%';
325  ++str;
326  if (*fmt) {
327  if (str <= end)
328  *str = *fmt;
329  ++str;
330  } else {
331  --fmt;
332  }
333  continue;
334  }
335  if (qualifier == 'L')
336  num = va_arg(args, long long);
337  else if (qualifier == 'l') {
338  num = va_arg(args, unsigned long);
339  if (flags & _kc_SIGN)
340  num = (signed long) num;
341  } else if (qualifier == 'Z') {
342  num = va_arg(args, size_t);
343  } else if (qualifier == 'h') {
344  num = (unsigned short) va_arg(args, int);
345  if (flags & _kc_SIGN)
346  num = (signed short) num;
347  } else {
348  num = va_arg(args, unsigned int);
349  if (flags & _kc_SIGN)
350  num = (signed int) num;
351  }
352  str = number(str, end, num, base,
353  field_width, precision, flags);
354  }
355  if (str <= end)
356  *str = '\0';
357  else if (size > 0)
358  /* don't write out a null byte if the buf size is zero */
359  *end = '\0';
360  /* the trailing null byte doesn't count towards the total
361  * ++str;
362  */
363  return str-buf;
364 }

550 unsigned long find_next_bit(const unsigned long *addr, unsigned long size,
551  unsigned long offset)
552 {
553  const unsigned long *p = addr + BITOP_WORD(offset);
554  unsigned long result = offset & ~(BITS_PER_LONG-1);
555  unsigned long tmp;
556 
557  if (offset >= size)
558  return size;
559  size -= result;
560  offset %= BITS_PER_LONG;
561  if (offset) {
562  tmp = *(p++);
563  tmp &= (~0UL << offset);
564  if (size < BITS_PER_LONG)
565  goto found_first;
566  if (tmp)
567  goto found_middle;
568  size -= BITS_PER_LONG;
569  result += BITS_PER_LONG;
570  }
571  while (size & ~(BITS_PER_LONG-1)) {
572  if ((tmp = *(p++)))
573  goto found_middle;
574  result += BITS_PER_LONG;
575  size -= BITS_PER_LONG;
576  }
577  if (!size)
578  return result;
579  tmp = *p;
580 
581 found_first:
582  tmp &= (~0UL >> (BITS_PER_LONG - size));
583  if (tmp == 0UL) /* Are any bits set? */
584  return result + size; /* Nope. */
585 found_middle:
586  return result + ffs(tmp);
587 }

603 uint32_t __attribute__((weak)) _kc__div64_32(uint64_t *n, uint32_t base)
604 {
605  uint64_t rem = *n;
606  uint64_t b = base;
607  uint64_t res, d = 1;
608  uint32_t high = rem >> 32;
609 
610  /* Reduce the thing a bit first */
611  res = 0;
612  if (high >= base) {
613  high /= base;
614  res = (uint64_t) high << 32;
615  rem -= (uint64_t) (high*base) << 32;
616  }
617 
618  while ((int64_t)b > 0 && b < rem) {
619  b = b+b;
620  d = d+d;
621  }
622 
623  do {
624  if (rem >= b) {
625  rem -= b;
626  res += d;
627  }
628  b >>= 1;
629  d >>= 1;
630  } while (d);
631 
632  *n = res;
633  return rem;
634 }
