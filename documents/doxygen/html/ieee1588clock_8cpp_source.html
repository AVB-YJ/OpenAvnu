271 FrequencyRatio IEEE1588Clock::calcLocalSystemClockRateDifference( Timestamp local_time, Timestamp system_time ) {
272  unsigned long long inter_system_time;
273  unsigned long long inter_local_time;
274  FrequencyRatio ppt_offset;
275 
276  GPTP_LOG_DEBUG( "Calculated local to system clock rate difference" );
277 
278  if( !_local_system_freq_offset_init ) {
279  _prev_system_time = system_time;
280  _prev_local_time = local_time;
281 
282  _local_system_freq_offset_init = true;
283 
284  return 1.0;
285  }
286 
287  inter_system_time =
288  TIMESTAMP_TO_NS(system_time) - TIMESTAMP_TO_NS(_prev_system_time);
289  inter_local_time =
290  TIMESTAMP_TO_NS(local_time) - TIMESTAMP_TO_NS(_prev_local_time);
291 
292  if( inter_system_time != 0 ) {
293  ppt_offset = ((FrequencyRatio)inter_local_time)/inter_system_time;
294  } else {
295  ppt_offset = 1.0;
296  }
297 
298  _prev_system_time = system_time;
299  _prev_local_time = local_time;
300 
301  return ppt_offset;
302 }

306 FrequencyRatio IEEE1588Clock::calcMasterLocalClockRateDifference( Timestamp master_time, Timestamp sync_time ) {
307  unsigned long long inter_sync_time;
308  unsigned long long inter_master_time;
309  FrequencyRatio ppt_offset;
310 
311  GPTP_LOG_DEBUG( "Calculated master to local clock rate difference" );
312 
313  if( !_master_local_freq_offset_init ) {
314  _prev_sync_time = sync_time;
315  _prev_master_time = master_time;
316 
317  _master_local_freq_offset_init = true;
318 
319  return 1.0;
320  }
321 
322  inter_sync_time =
323  TIMESTAMP_TO_NS(sync_time) - TIMESTAMP_TO_NS(_prev_sync_time);
324 
325  uint64_t master_time_ns = TIMESTAMP_TO_NS(master_time);
326  uint64_t prev_master_time_ns = TIMESTAMP_TO_NS(_prev_master_time);
327 
328  inter_master_time = master_time_ns - prev_master_time_ns;
329 
330  if( inter_sync_time != 0 ) {
331  ppt_offset = ((FrequencyRatio)inter_master_time)/inter_sync_time;
332  } else {
333  ppt_offset = 1.0;
334  }
335 
336  if( master_time_ns < prev_master_time_ns ) {
337  GPTP_LOG_ERROR("Negative time jump detected - inter_master_time: %lld, inter_sync_time: %lld, incorrect ppt_offset: %Lf",
338  inter_master_time, inter_sync_time, ppt_offset);
339  _master_local_freq_offset_init = false;
340 
341  return NEGATIVE_TIME_JUMP;
342  }
343 
344  _prev_sync_time = sync_time;
345  _prev_master_time = master_time;
346 
347  return ppt_offset;
348 }

350 void IEEE1588Clock::setMasterOffset
351 ( CommonPort *port, int64_t master_local_offset,
352  Timestamp local_time, FrequencyRatio master_local_freq_offset,
353  int64_t local_system_offset, Timestamp system_time,
354  FrequencyRatio local_system_freq_offset, unsigned sync_count,
355  unsigned pdelay_count, PortState port_state, bool asCapable )
356 {
357  _master_local_freq_offset = master_local_freq_offset;
358  _local_system_freq_offset = local_system_freq_offset;
359 
360  if (port->getTestMode()) {
361  GPTP_LOG_STATUS("Clock offset:%lld Clock rate ratio:%Lf Sync Count:%u PDelay Count:%u",
362  master_local_offset, master_local_freq_offset, sync_count, pdelay_count);
363  }
364 
365  if( ipc != NULL ) {
366  uint8_t grandmaster_id[PTP_CLOCK_IDENTITY_LENGTH];
367  uint8_t clock_id[PTP_CLOCK_IDENTITY_LENGTH];
368  PortIdentity port_identity;
369  uint16_t port_number;
370 
371  grandmaster_clock_identity.getIdentityString(grandmaster_id);
372  clock_identity.getIdentityString(clock_id);
373  port->getPortIdentity(port_identity);
374  port_identity.getPortNumber(&port_number);
375 
376  ipc->update(
377  master_local_offset, local_system_offset, master_local_freq_offset,
378  local_system_freq_offset, TIMESTAMP_TO_NS(local_time),
379  sync_count, pdelay_count, port_state, asCapable);
380 
381  ipc->update_grandmaster(
382  grandmaster_id, domain_number);
383 
384  ipc->update_network_interface(
385  clock_id, priority1,
386  clock_quality.cq_class, clock_quality.offsetScaledLogVariance,
387  clock_quality.clockAccuracy,
388  priority2, domain_number,
389  port->getSyncInterval(),
390  port->getAnnounceInterval(),
391  0, // TODO: Was port->getPDelayInterval() before refactoring. What do we do now?
392  port_number);
393  }
394 
395  if( master_local_offset == 0 && master_local_freq_offset == 1.0 ) {
396  return;
397  }
398 
399  if( _syntonize ) {
400  if( _new_syntonization_set_point || _phase_error_violation > PHASE_ERROR_MAX_COUNT ) {
401  _new_syntonization_set_point = false;
402  _phase_error_violation = 0;
403  /* Make sure that there are no transmit operations
404  in progress */
405  getTxLockAll();
406  if (port->getTestMode()) {
407  GPTP_LOG_STATUS("Adjust clock phase offset:%lld", -master_local_offset);
408  }
409  port->adjustClockPhase( -master_local_offset );
410  _master_local_freq_offset_init = false;
411  restartPDelayAll();
412  putTxLockAll();
413  master_local_offset = 0;
414  }
415 
416  // Adjust for frequency offset
417  long double phase_error = (long double) -master_local_offset;
418  if( fabsl(phase_error) > PHASE_ERROR_THRESHOLD ) {
419  ++_phase_error_violation;
420  } else {
421  _phase_error_violation = 0;
422 
423  float syncPerSec = (float)(1.0 / pow((float)2, port->getSyncInterval()));
424  _ppm += (float) ((INTEGRAL * syncPerSec * phase_error) + PROPORTIONAL*((master_local_freq_offset-1.0)*1000000));
425 
426  GPTP_LOG_DEBUG("phase_error = %Lf, ppm = %f", phase_error, _ppm );
427  }
428 
429  if( _ppm < LOWER_FREQ_LIMIT ) _ppm = LOWER_FREQ_LIMIT;
430  if( _ppm > UPPER_FREQ_LIMIT ) _ppm = UPPER_FREQ_LIMIT;
431  if ( port->getTestMode() ) {
432  GPTP_LOG_STATUS("Adjust clock rate ppm:%f", _ppm);
433  }
434  if( !port->adjustClockRate( _ppm ) ) {
435  GPTP_LOG_ERROR( "Failed to adjust clock rate" );
436  }
437  }
438 
439  return;
440 }
