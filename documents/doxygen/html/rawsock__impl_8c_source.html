99 // Pre-set the ethernet header information that will be used on TX frames
100 bool baseRawsockTxSetHdr(void *pvRawsock, hdr_info_t *pHdr)
101 {
102  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK_DETAIL);
103  base_rawsock_t *rawsock = (base_rawsock_t*)pvRawsock;
104 
105  if (!VALID_TX_RAWSOCK(rawsock)) {
106  AVB_LOG_ERROR("Setting TX header; invalid argument");
107  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
108  return FALSE;
109  }
110 
111  // source address
112  if (pHdr->shost) {
113  memcpy(&(rawsock->ethHdr.notag.shost), pHdr->shost, ETH_ALEN);
114  }
115  // destination address
116  if (pHdr->dhost) {
117  memcpy(&(rawsock->ethHdr.notag.dhost), pHdr->dhost, ETH_ALEN);
118  }
119 
120  // VLAN tag?
121  if (!pHdr->vlan) {
122  // No, set ethertype in normal location
123  rawsock->ethHdr.notag.ethertype = htons(rawsock->ethertype);
124  // and set ethernet header length
125  rawsock->ethHdrLen = sizeof(eth_hdr_t);
126  }
127  else {
128  // Add VLAN tag
129  AVB_LOGF_DEBUG("VLAN=%d pcp=%d vid=%d", pHdr->vlan_vid, pHdr->vlan_pcp, pHdr->vlan_vid);
130 
131  // Build bitfield with vlan_pcp and vlan_vid.
132  // I think CFI bit is alway 0
133  u_int16_t bits = 0;
134  bits |= (pHdr->vlan_pcp << 13) & 0xE000;
135  bits |= pHdr->vlan_vid & 0x0FFF;
136 
137  // Create VLAN tag
138  rawsock->ethHdr.tagged.vlan.tpip = htons(ETHERTYPE_VLAN);
139  rawsock->ethHdr.tagged.vlan.bits = htons(bits);
140  rawsock->ethHdr.tagged.ethertype = htons(rawsock->ethertype);
141  // and set ethernet header length
142  rawsock->ethHdrLen = sizeof(eth_vlan_hdr_t);
143  }
144 
145  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
146  return TRUE;
147 }

185 // Parse the ethernet frame header. Returns length of header, or -1 for failure
186 int baseRawsockRxParseHdr(void *pvRawsock, U8 *pBuffer, hdr_info_t *pInfo)
187 {
188  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK_DETAIL);
189 
190  memset(pInfo, 0, sizeof(hdr_info_t));
191 
192  eth_hdr_t *eth_hdr = (eth_hdr_t*)pBuffer;
193  pInfo->dhost = eth_hdr->dhost;
194  pInfo->shost = eth_hdr->shost;
195  pInfo->ethertype = ntohs(eth_hdr->ethertype);
196  pInfo->ts.tv_sec = 0;
197  pInfo->ts.tv_nsec = 0;
198  int hdrLen = sizeof(eth_hdr_t);
199 
200  if (pInfo->ethertype == ETHERTYPE_8021Q) {
201  U16 vlan_bits = ntohs(*(U16*)(pBuffer + hdrLen));
202  pInfo->vlan = TRUE;
203  pInfo->vlan_vid = vlan_bits & 0x0FFF;
204  pInfo->vlan_pcp = (vlan_bits >> 13) & 0x0007;
205  hdrLen += 2;
206  pInfo->ethertype = ntohs(*(U16 *)(pBuffer + hdrLen));
207  hdrLen += 2;
208  }
209 
210  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
211  return hdrLen;
212 }
