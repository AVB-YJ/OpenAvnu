62 void log::post_log_msg(int32_t level, const char * fmt, ...)
63 {
64  if (level <= log_level)
65  {
66  va_list arglist;
67  uint32_t index;
68 
69  if ((write_index - read_index) > LOG_BUF_COUNT)
70  {
71  missed_log_event_cnt++;
72  return;
73  }
74  index = InterlockedExchangeAdd(&write_index, 1);
75  va_start(arglist, fmt);
76  vsprintf_s(log_buf[index % LOG_BUF_COUNT].msg, sizeof(log_buf[0].msg), fmt, arglist); // Write to log_buf using write_index
77  va_end(arglist);
78  log_buf[index % LOG_BUF_COUNT].level = level;
79  log_buf[index % LOG_BUF_COUNT].time_stamp_ms = 0;
80 
81  post_log_event();
82  }
83 }
