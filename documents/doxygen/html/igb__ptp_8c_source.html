42 /*
43  * The 82580 timesync updates the system timer every 8ns by 8ns,
44  * and this update value cannot be reprogrammed.
45  *
46  * Neither the 82576 nor the 82580 offer registers wide enough to hold
47  * nanoseconds time values for very long. For the 82580, SYSTIM always
48  * counts nanoseconds, but the upper 24 bits are not availible. The
49  * frequency is adjusted by changing the 32 bit fractional nanoseconds
50  * register, TIMINCA.
51  *
52  * For the 82576, the SYSTIM register time unit is affect by the
53  * choice of the 24 bit TININCA:IV (incvalue) field. Five bits of this
54  * field are needed to provide the nominal 16 nanosecond period,
55  * leaving 19 bits for fractional nanoseconds.
56  *
57  * We scale the NIC clock cycle by a large factor so that relatively
58  * small clock corrections can be added or subtracted at each clock
59  * tick. The drawbacks of a large factor are a) that the clock
60  * register overflows more quickly (not such a big deal) and b) that
61  * the increment per tick has to fit into 24 bits. As a result we
62  * need to use a shift of 19 so we can fit a value of 16 into the
63  * TIMINCA register.
64  *
65  *
66  * SYSTIMH SYSTIML
67  * +--------------+ +---+---+------+
68  * 82576 | 32 | | 8 | 5 | 19 |
69  * +--------------+ +---+---+------+
70  * \________ 45 bits _______/ fract
71  *
72  * +----------+---+ +--------------+
73  * 82580 | 24 | 8 | | 32 |
74  * +----------+---+ +--------------+
75  * reserved \______ 40 bits _____/
76  *
77  *
78  * The 45 bit 82576 SYSTIM overflows every
79  * 2^45 * 10^-9 / 3600 = 9.77 hours.
80  *
81  * The 40 bit 82580 SYSTIM overflows every
82  * 2^40 * 10^-9 / 60 = 18.3 minutes.
83  */
84

132  val |= lo;
133 
134  return val;

189 static void igb_ptp_systim_to_hwtstamp(struct igb_adapter *adapter,
190  struct skb_shared_hwtstamps *hwtstamps,
191  u64 systim)
192 {
193  unsigned long flags;
194  u64 ns;
195 
196  switch (adapter->hw.mac.type) {
197  case e1000_82576:
198  case e1000_82580:
199  case e1000_i350:
200  case e1000_i354:
201  spin_lock_irqsave(&adapter->tmreg_lock, flags);
202 
203  ns = timecounter_cyc2time(&adapter->tc, systim);
204 
205  spin_unlock_irqrestore(&adapter->tmreg_lock, flags);
206 
207  memset(hwtstamps, 0, sizeof(*hwtstamps));
208  hwtstamps->hwtstamp = ns_to_ktime(ns);
209  break;
210  case e1000_i210:
211  case e1000_i211:
212  memset(hwtstamps, 0, sizeof(*hwtstamps));
213  /* Upper 32 bits contain s, lower 32 bits contain ns. */
214  hwtstamps->hwtstamp = ktime_set(systim >> 32,
215  systim & 0xFFFFFFFF);
216  break;
217  default:
218  break;
219  }
220 }

226 static int igb_ptp_adjfreq_82576(struct ptp_clock_info *ptp, s32 ppb)
227 {
228  struct igb_adapter *igb = container_of(ptp, struct igb_adapter,
229  ptp_caps);
230  struct e1000_hw *hw = &igb->hw;
231  int neg_adj = 0;
232  u64 rate;
233  u32 incvalue;
234 
235  if (ppb < 0) {
236  neg_adj = 1;
237  ppb = -ppb;
238  }
239  rate = ppb;
240  rate <<= 14;
241  rate = div_u64(rate, 1953125);
242 
243  incvalue = 16 << IGB_82576_TSYNC_SHIFT;
244 
245  if (neg_adj)
246  incvalue -= rate;
247  else
248  incvalue += rate;
249 
250  E1000_WRITE_REG(hw, E1000_TIMINCA, INCPERIOD_82576
251  | (incvalue & INCVALUE_82576_MASK));
252 
253  return 0;
254 }

256 static int igb_ptp_adjfreq_82580(struct ptp_clock_info *ptp, s32 ppb)
257 {
258  struct igb_adapter *igb = container_of(ptp, struct igb_adapter,
259  ptp_caps);
260  struct e1000_hw *hw = &igb->hw;
261  int neg_adj = 0;
262  u64 rate;
263  u32 inca;
264 
265  if (ppb < 0) {
266  neg_adj = 1;
267  ppb = -ppb;
268  }
269  rate = ppb;
270  rate <<= 26;
271  rate = div_u64(rate, 1953125);
272 
273  /* At 2.5G speeds, the TIMINCA register on I354 updates the clock 2.5x
274  * as quickly. Account for this by dividing the adjustment by 2.5.
275  */
276  if (hw->mac.type == e1000_i354) {
277  u32 status = E1000_READ_REG(hw, E1000_STATUS);
278 
279  if ((status & E1000_STATUS_2P5_SKU) &&
280  !(status & E1000_STATUS_2P5_SKU_OVER)) {
281  rate <<= 1;
282  rate = div_u64(rate, 5);
283  }
284  }
285 
286  inca = rate & INCVALUE_MASK;
287  if (neg_adj)
288  inca |= ISGN;
289 
290  E1000_WRITE_REG(hw, E1000_TIMINCA, inca);
291 
292  return 0;
293 }

488  AUX1_SEL_SDP0, AUX1_SEL_SDP1, AUX1_SEL_SDP2, AUX1_SEL_SDP3,

578 static int igb_ptp_feature_enable_i210(struct ptp_clock_info *ptp,
579  struct ptp_clock_request *rq, int on)
580 {
581  struct igb_adapter *igb =
582  container_of(ptp, struct igb_adapter, ptp_caps);
583  struct e1000_hw *hw = &igb->hw;
584  unsigned long flags;
585  u32 tsim;
586 #ifdef HAVE_PTP_1588_CLOCK_PINS
587  u32 tsauxc, tsauxc_mask, tsim_mask, trgttiml, trgttimh, freqout;
588  struct timespec64 ts;
589  int use_freq = 0, pin = -1;
590  s64 ns;
591 #endif /* HAVE_PTP_1588_CLOCK_PINS */
592 
593  switch (rq->type) {
594 #ifdef HAVE_PTP_1588_CLOCK_PINS
595  case PTP_CLK_REQ_EXTTS:
596  if (on) {
597  pin = ptp_find_pin(igb->ptp_clock, PTP_PF_EXTTS,
598  rq->extts.index);
599  if (pin < 0)
600  return -EBUSY;
601  }
602  if (rq->extts.index == 1) {
603  tsauxc_mask = TSAUXC_EN_TS1;
604  tsim_mask = TSINTR_AUTT1;
605  } else {
606  tsauxc_mask = TSAUXC_EN_TS0;
607  tsim_mask = TSINTR_AUTT0;
608  }
609  spin_lock_irqsave(&igb->tmreg_lock, flags);
610  tsauxc = E1000_READ_REG(hw, E1000_TSAUXC);
611  tsim = E1000_READ_REG(hw, E1000_TSIM);
612  if (on) {
613  igb_pin_extts(igb, rq->extts.index, pin);
614  tsauxc |= tsauxc_mask;
615  tsim |= tsim_mask;
616  } else {
617  tsauxc &= ~tsauxc_mask;
618  tsim &= ~tsim_mask;
619  }
620  E1000_WRITE_REG(hw, E1000_TSAUXC, tsauxc);
621  E1000_WRITE_REG(hw, E1000_TSIM, tsim);
622  spin_unlock_irqrestore(&igb->tmreg_lock, flags);
623  return 0;
624 
625  case PTP_CLK_REQ_PEROUT:
626  if (on) {
627  pin = ptp_find_pin(igb->ptp_clock, PTP_PF_PEROUT,
628  rq->perout.index);

646  tsim_mask = 0;

650  }
651  trgttiml = E1000_TRGTTIML1;
652  trgttimh = E1000_TRGTTIMH1;
653  freqout = E1000_FREQOUT1;
654  } else {

658  } else {
659  tsauxc_mask = TSAUXC_EN_TT0;

662  trgttiml = E1000_TRGTTIML0;
663  trgttimh = E1000_TRGTTIMH0;

665  }

667  tsauxc = E1000_READ_REG(hw, E1000_TSAUXC);
668  tsim = E1000_READ_REG(hw, E1000_TSIM);
669  if (rq->perout.index == 1) {
670  tsauxc &= ~(TSAUXC_EN_TT1 | TSAUXC_EN_CLK1 | TSAUXC_ST1);
671  tsim &= ~TSINTR_TT1;
672  } else {
673  tsauxc &= ~(TSAUXC_EN_TT0 | TSAUXC_EN_CLK0 | TSAUXC_ST0);
674  tsim &= ~TSINTR_TT0;
675  }
676  if (on) {
677  int i = rq->perout.index;
678  igb_pin_perout(igb, i, pin, use_freq);
679  igb->perout[i].start.tv_sec = rq->perout.start.sec;
680  igb->perout[i].start.tv_nsec = rq->perout.start.nsec;
681  igb->perout[i].period.tv_sec = ts.tv_sec;
682  igb->perout[i].period.tv_nsec = ts.tv_nsec;
683  E1000_WRITE_REG(hw, trgttimh, rq->perout.start.sec);
684  E1000_WRITE_REG(hw, trgttiml, rq->perout.start.nsec);
685  if (use_freq)
686  E1000_WRITE_REG(hw, freqout, ns);
687  tsauxc |= tsauxc_mask;
688  tsim |= tsim_mask;
689  }
690  E1000_WRITE_REG(hw, E1000_TSAUXC, tsauxc);
691  E1000_WRITE_REG(hw, E1000_TSIM, tsim);
692  spin_unlock_irqrestore(&igb->tmreg_lock, flags);
693  return 0;
694 #endif /* HAVE_PTP_1588_CLOCK_PINS */
695 
696  case PTP_CLK_REQ_PPS:
697  spin_lock_irqsave(&igb->tmreg_lock, flags);
698  tsim = E1000_READ_REG(hw, E1000_TSIM);
699  if (on)
700  tsim |= TSINTR_SYS_WRAP;
701  else
702  tsim &= ~TSINTR_SYS_WRAP;
703  E1000_WRITE_REG(hw, E1000_TSIM, tsim);
704  spin_unlock_irqrestore(&igb->tmreg_lock, flags);
705  return 0;
706 
707 #ifndef HAVE_PTP_1588_CLOCK_PINS
708  default:
709  break;
710 #endif /* HAVE_PTP_1588_CLOCK_PINS */
711  }
712 
713  return -EOPNOTSUPP;
714 }

953 static int igb_ptp_set_timestamp_mode(struct igb_adapter *adapter,
954  struct hwtstamp_config *config)
955 {
956  struct e1000_hw *hw = &adapter->hw;
957  u32 tsync_tx_ctl = E1000_TSYNCTXCTL_ENABLED;
958  u32 tsync_rx_ctl = E1000_TSYNCRXCTL_ENABLED;
959  u32 tsync_rx_cfg = 0;
960  bool is_l4 = false;
961  bool is_l2 = false;
962  u32 regval;
963 
964  /* reserved for future extensions */
965  if (config->flags)
966  return -EINVAL;
967 
968  switch (config->tx_type) {
969  case HWTSTAMP_TX_OFF:
970  tsync_tx_ctl = 0;
971  case HWTSTAMP_TX_ON:
972  break;
973  default:
974  return -ERANGE;
975  }
976 
977  switch (config->rx_filter) {
978  case HWTSTAMP_FILTER_NONE:
979  tsync_rx_ctl = 0;
980  break;
981  case HWTSTAMP_FILTER_PTP_V1_L4_SYNC:
982  tsync_rx_ctl |= E1000_TSYNCRXCTL_TYPE_L4_V1;
983  tsync_rx_cfg = E1000_TSYNCRXCFG_PTP_V1_SYNC_MESSAGE;
984  is_l4 = true;
985  break;
986  case HWTSTAMP_FILTER_PTP_V1_L4_DELAY_REQ:
987  tsync_rx_ctl |= E1000_TSYNCRXCTL_TYPE_L4_V1;
988  tsync_rx_cfg = E1000_TSYNCRXCFG_PTP_V1_DELAY_REQ_MESSAGE;
989  is_l4 = true;
990  break;
991  case HWTSTAMP_FILTER_PTP_V2_EVENT:
992  case HWTSTAMP_FILTER_PTP_V2_L2_EVENT:
993  case HWTSTAMP_FILTER_PTP_V2_L4_EVENT:
994  case HWTSTAMP_FILTER_PTP_V2_SYNC:
995  case HWTSTAMP_FILTER_PTP_V2_L2_SYNC:
996  case HWTSTAMP_FILTER_PTP_V2_L4_SYNC:
997  case HWTSTAMP_FILTER_PTP_V2_DELAY_REQ:
998  case HWTSTAMP_FILTER_PTP_V2_L2_DELAY_REQ:
999  case HWTSTAMP_FILTER_PTP_V2_L4_DELAY_REQ:
1000  tsync_rx_ctl |= E1000_TSYNCRXCTL_TYPE_EVENT_V2;
1001  config->rx_filter = HWTSTAMP_FILTER_PTP_V2_EVENT;
1002  is_l2 = true;
1003  is_l4 = true;
1004  break;
1005  case HWTSTAMP_FILTER_PTP_V1_L4_EVENT:
1006  case HWTSTAMP_FILTER_ALL:
1007  /*
1008  * 82576 cannot timestamp all packets, which it needs to do to
1009  * support both V1 Sync and Delay_Req messages
1010  */
1011  if (hw->mac.type != e1000_82576) {
1012  tsync_rx_ctl |= E1000_TSYNCRXCTL_TYPE_ALL;
1013  config->rx_filter = HWTSTAMP_FILTER_ALL;
1014  break;
1015  }
1016  /* fall through */
1017  default:
1018  config->rx_filter = HWTSTAMP_FILTER_NONE;
1019  return -ERANGE;
1020  }
1021 
1022  if (hw->mac.type == e1000_82575) {
1023  if (tsync_rx_ctl | tsync_tx_ctl)
1024  return -EINVAL;
1025  return 0;
1026  }
1027 
1028  /*
1029  * Per-packet timestamping only works if all packets are
1030  * timestamped, so enable timestamping in all packets as
1031  * long as one rx filter was configured.
1032  */
1033  if ((hw->mac.type >= e1000_82580) && tsync_rx_ctl) {
1034  tsync_rx_ctl = E1000_TSYNCRXCTL_ENABLED;
1035  tsync_rx_ctl |= E1000_TSYNCRXCTL_TYPE_ALL;
1036  config->rx_filter = HWTSTAMP_FILTER_ALL;
1037  is_l2 = true;
1038  is_l4 = true;
1039 
1040  if ((hw->mac.type == e1000_i210) ||
1041  (hw->mac.type == e1000_i211)) {
1042  regval = E1000_READ_REG(hw, E1000_RXPBS);
1043  regval |= E1000_RXPBS_CFG_TS_EN;
1044  E1000_WRITE_REG(hw, E1000_RXPBS, regval);
1045  }
1046  }
1047 
1048  /* enable/disable TX */
1049  regval = E1000_READ_REG(hw, E1000_TSYNCTXCTL);
1050  regval &= ~E1000_TSYNCTXCTL_ENABLED;
1051  regval |= tsync_tx_ctl;
1052  E1000_WRITE_REG(hw, E1000_TSYNCTXCTL, regval);
1053 
1054  /* enable/disable RX */
1055  regval = E1000_READ_REG(hw, E1000_TSYNCRXCTL);
1056  regval &= ~(E1000_TSYNCRXCTL_ENABLED | E1000_TSYNCRXCTL_TYPE_MASK);
1057  regval |= tsync_rx_ctl;
1058  E1000_WRITE_REG(hw, E1000_TSYNCRXCTL, regval);
1059 
1060  /* define which PTP packets are time stamped */
1061  E1000_WRITE_REG(hw, E1000_TSYNCRXCFG, tsync_rx_cfg);
1062 
1063  /* define ethertype filter for timestamped packets */
1064  if (is_l2)
1065  E1000_WRITE_REG(hw, E1000_ETQF(3),
1066  (E1000_ETQF_FILTER_ENABLE | /* enable filter */
1067  E1000_ETQF_1588 | /* enable timestamping */
1068  ETH_P_1588)); /* 1588 eth protocol type */
1069  else
1070  E1000_WRITE_REG(hw, E1000_ETQF(3), 0);
1071 
1072  /* L4 Queue Filter[3]: filter by destination port and protocol */
1073  if (is_l4) {
1074  u32 ftqf = (IPPROTO_UDP /* UDP */
1075  | E1000_FTQF_VF_BP /* VF not compared */
1076  | E1000_FTQF_1588_TIME_STAMP /* Enable Timestamp */
1077  | E1000_FTQF_MASK); /* mask all inputs */
1078  ftqf &= ~E1000_FTQF_MASK_PROTO_BP; /* enable protocol check */
1079 
1080  E1000_WRITE_REG(hw, E1000_IMIR(3), htons(PTP_EV_PORT));
1081  E1000_WRITE_REG(hw, E1000_IMIREXT(3),
1082  (E1000_IMIREXT_SIZE_BP | E1000_IMIREXT_CTRL_BP));
1083  if (hw->mac.type == e1000_82576) {
1084  /* enable source port check */
1085  E1000_WRITE_REG(hw, E1000_SPQF(3), htons(PTP_EV_PORT));
1086  ftqf &= ~E1000_FTQF_MASK_SOURCE_PORT_BP;
1087  }
1088  E1000_WRITE_REG(hw, E1000_FTQF(3), ftqf);
1089  } else {
1090  E1000_WRITE_REG(hw, E1000_FTQF(3), E1000_FTQF_MASK);
1091  }
1092  E1000_WRITE_FLUSH(hw);
1093 
1094  /* clear TX/RX time stamp registers, just to be sure */
1095  regval = E1000_READ_REG(hw, E1000_TXSTMPL);
1096  regval = E1000_READ_REG(hw, E1000_TXSTMPH);
1097  regval = E1000_READ_REG(hw, E1000_RXSTMPL);
1098  regval = E1000_READ_REG(hw, E1000_RXSTMPH);
1099 
1100  return 0;
1101 }
