120 int STDCALL net_interface_imp::select_interface_by_num(uint32_t interface_num)
121 {
122  uint32_t index;
123  IP_ADAPTER_INFO * AdapterInfo;
124  IP_ADAPTER_INFO * Current;
125  ULONG AIS;
126  DWORD status;
127  int timeout_ms = NETIF_READ_TIMEOUT_MS;
128 
129  if (interface_num < 1 || interface_num > total_devs)
130  {
131  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "Interface number out of range.");
132  pcap_freealldevs(all_devs); // Free the device list
133  exit(EXIT_FAILURE);
134  }
135 
136  for (dev = all_devs, index = 0; index < interface_num - 1; dev = dev->next, index++)
137  ; // Jump to the selected adapter
138 
139  /************************************************************** Open the device ****************************************************************/
140  if ((pcap_interface = pcap_open_live(dev->name, // Name of the device
141  65536, // Portion of the packet to capture
142  // 65536 guarantees that the whole packet will be captured on all the link layers
143  PCAP_OPENFLAG_PROMISCUOUS, // In promiscuous mode, all packets including packets of other hosts are captured
144  timeout_ms, // Read timeout in ms
145  err_buf // Error buffer
146  )) == NULL)
147  {
148  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "Unable to open the adapter. %s is not supported by WinPcap.", dev->name);
149  pcap_freealldevs(all_devs); // Free the device list
150  exit(EXIT_FAILURE);
151  }
152 
153  /****************************** Lookup IP address ***************************/
154  AdapterInfo = (IP_ADAPTER_INFO *)calloc(total_devs, sizeof(IP_ADAPTER_INFO));
155  AIS = sizeof(IP_ADAPTER_INFO) * total_devs;
156 
157  if (GetAdaptersInfo(AdapterInfo, &AIS) == ERROR_BUFFER_OVERFLOW)
158  {
159  free(AdapterInfo);
160  AdapterInfo = (IP_ADAPTER_INFO *)calloc(1, AIS);
161 
162  if (AdapterInfo == NULL)
163  {
164  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "Allocating memory needed to call GetAdaptersinfo.", dev->name);
165  exit(EXIT_FAILURE);
166  }
167 
168  status = GetAdaptersInfo(AdapterInfo, &AIS);
169 
170  if (status != ERROR_SUCCESS)
171  {
172  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "GetAdaptersInfo call in netif_win32_pcap.c failed.", dev->name);
173  free(AdapterInfo);
174  exit(EXIT_FAILURE);
175  }
176  }
177 
178  for (Current = AdapterInfo; Current != NULL; Current = Current->Next)
179  {
180  if (strstr(dev->name, Current->AdapterName) != 0)
181  {
182  uint32_t my_ip;
183  ULONG len;
184  uint8_t tmp[16];
185 
186  my_ip = inet_addr(Current->IpAddressList.IpAddress.String);
187  len = sizeof(tmp);
188  SendARP(my_ip, INADDR_ANY, tmp, &len);
189  utility::convert_eui48_to_uint64(&tmp[0], mac);
190  }
191  }
192 
193  uint16_t ether_type[1];
194  ether_type[0] = JDKSAVDECC_AVTP_ETHERTYPE;
195  set_capture_ether_type(ether_type, 1); // Set the filter
196 
197  free(AdapterInfo);
198  return 0;
199 }

201 int net_interface_imp::set_capture_ether_type(uint16_t * ether_type, uint32_t count)
202 {
203  struct bpf_program fcode;
204  char ether_type_string[512];
205  char ether_type_single[64];
206  const unsigned char * ether_packet = NULL;
207  struct pcap_pkthdr pcap_header;
208 
209  ether_type_string[0] = 0;
210 
211  for (uint32_t i = 0; i < count; i++)
212  {
213  sprintf(ether_type_single, "ether proto 0x%04x", ether_type[i]);
214  strcat(ether_type_string, ether_type_single);
215 
216  if ((i + 1) < count)
217  {
218  strcat(ether_type_string, " or ");
219  }
220  }
221 
222  /************************************** Compile a filter **************************************/
223  if (pcap_compile(pcap_interface, &fcode, ether_type_string, 1, 0) < 0)
224  {
225  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "Unable to compile the packet filter.");
226  pcap_freealldevs(all_devs); // Free the device list
227  return -1;
228  }
229 
230  /********************************** Set the filter *********************************/
231  if (pcap_setfilter(pcap_interface, &fcode) < 0)
232  {
233  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "Error setting the filter.");
234  pcap_freealldevs(all_devs); // Free the device list
235  return -1;
236  }
237 
238  /*********** Flush any packets that might be present **********/
239  for (uint32_t index_j = 0; index_j < 1; index_j++)
240  {
241  ether_packet = pcap_next(pcap_interface, &pcap_header);
242  }
243 
244  return 0;
245 }

248 {
249  struct pcap_pkthdr * header;

251 
252  *frame_len = 0;
253  error = pcap_next_ex(pcap_interface, &header, frame);
254 
255  if (error > 0)
256  {
257  cmd_frame = *frame;
258  *frame_len = (uint16_t)header->len;
259  return 1;
260  }
261 
262  return -2; // Timeout
