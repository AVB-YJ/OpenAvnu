223 s32 e1000_setup_init_funcs(struct e1000_hw *hw, bool init_device)
224 {
225  s32 ret_val;
226 
227  /* Can't do much good without knowing the MAC type. */
228  ret_val = e1000_set_mac_type(hw);
229  if (ret_val) {
230  DEBUGOUT("ERROR: MAC type could not be set properly.\n");
231  goto out;
232  }
233 
234  if (!hw->hw_addr) {
235  DEBUGOUT("ERROR: Registers not mapped\n");
236  ret_val = -E1000_ERR_CONFIG;
237  goto out;
238  }
239 
240  /*
241  * Init function pointers to generic implementations. We do this first
242  * allowing a driver module to override it afterward.
243  */
244  e1000_init_mac_ops_generic(hw);
245  e1000_init_phy_ops_generic(hw);
246  e1000_init_nvm_ops_generic(hw);
247  e1000_init_mbx_ops_generic(hw);
248 
249  /*
250  * Set up the init function pointers. These are functions within the
251  * adapter family file that sets up function pointers for the rest of
252  * the functions in that family.
253  */
254  switch (hw->mac.type) {
255  case e1000_82575:
256  case e1000_82576:
257  case e1000_82580:
258  case e1000_i350:
259  case e1000_i354:
260  e1000_init_function_pointers_82575(hw);
261  break;
262  case e1000_i210:
263  case e1000_i211:
264  e1000_init_function_pointers_i210(hw);
265  break;
266  default:
267  DEBUGOUT("Hardware not supported\n");
268  ret_val = -E1000_ERR_CONFIG;
269  break;
270  }
271 
272  /*
273  * Initialize the rest of the function pointers. These require some
274  * register reads/writes in some cases.
275  */
276  if (!(ret_val) && init_device) {
277  ret_val = e1000_init_mac_params(hw);
278  if (ret_val)
279  goto out;
280 
281  ret_val = e1000_init_nvm_params(hw);
282  if (ret_val)
283  goto out;
284 
285  ret_val = e1000_init_phy_params(hw);
286  if (ret_val)
287  goto out;
288 
289  ret_val = e1000_init_mbx_params(hw);
290  if (ret_val)
291  goto out;
292  }
293 
294 out:
295  return ret_val;
296 }
