147 TEST(MockCheckedActualCall, MockIgnoredActualCallWorksAsItShould)
148 {
149  MockIgnoredActualCall actual;
150  actual.withName("func");
151  actual.withCallOrder(1);
152 
153  CHECK(false == actual.returnBoolValue());
154  CHECK(true == actual.returnBoolValueOrDefault(true));
155  CHECK(false == actual.returnBoolValueOrDefault(false));
156  CHECK(0 == actual.returnUnsignedLongIntValue());
157  CHECK(0 == actual.returnIntValue());
158  CHECK(1ul == actual.returnUnsignedLongIntValueOrDefault(1ul));
159  CHECK(1 == actual.returnIntValueOrDefault(1));
160  CHECK(0 == actual.returnLongIntValue());
161  CHECK(1l == actual.returnLongIntValueOrDefault(1l));
162  CHECK(0 == actual.returnUnsignedIntValue());
163  CHECK(1u == actual.returnUnsignedIntValueOrDefault(1u));
164  DOUBLES_EQUAL(0.0, actual.returnDoubleValue(), 0.0);
165  DOUBLES_EQUAL(1.5, actual.returnDoubleValueOrDefault(1.5), 0.0);
166  STRCMP_EQUAL("bla", actual.returnStringValueOrDefault("bla"));
167  STRCMP_EQUAL("", actual.returnStringValue());
168  CHECK(0 == actual.returnPointerValue());
169  CHECK((void*) 0x2 == actual.returnPointerValueOrDefault((void*) 0x2));
170  CHECK(0 == actual.returnConstPointerValue());
171  CHECK((const void*) 0x2 == actual.returnConstPointerValueOrDefault((const void*) 0x2));
172  CHECK(0 == actual.returnFunctionPointerValue());
173  CHECK((void(*)()) 1 == actual.returnFunctionPointerValueOrDefault((void(*)()) 0x1));
174  CHECK_FALSE(actual.hasReturnValue());
175  CHECK(actual.returnValue().equals(MockNamedValue("")));
176 }

178 TEST(MockCheckedActualCall, remainderOfMockActualCallTraceWorksAsItShould)
179 {
180  int value;
181  const int const_value = 1;
182  const unsigned char mem_buffer[] = { 0xFE, 0x15 };
183  void (*function_value)() = (void (*)())0xDEAD;
184  MockActualCallTrace actual;
185  actual.withName("func");
186  actual.withCallOrder(1);
187  actual.onObject(&value);
188 
189  actual.withBoolParameter("bool", true);
190  actual.withUnsignedIntParameter("unsigned_int", (unsigned int) 1);
191  actual.withUnsignedLongIntParameter("unsigned_long", (unsigned long)1);
192  actual.withLongIntParameter("long_int", (long int) 1);
193  actual.withPointerParameter("pointer", &value);
194  actual.withConstPointerParameter("const_pointer", &const_value);
195  actual.withFunctionPointerParameter("function_pointer", function_value);
196  actual.withMemoryBufferParameter("mem_buffer", mem_buffer, sizeof(mem_buffer));
197  actual.withParameterOfType("int", "named_type", &const_value);
198 
199  SimpleString expectedString("\nFunction name:func");
200  expectedString += " withCallOrder:1";
201  expectedString += " onObject:0x";
202  expectedString += HexStringFrom(&value);
203  expectedString += " bool:true";
204  expectedString += " unsigned_int:1 (0x1)";
205  expectedString += " unsigned_long:1 (0x1)";
206  expectedString += " long_int:1 (0x1)";
207  expectedString += " pointer:0x";
208  expectedString += HexStringFrom(&value);
209  expectedString += " const_pointer:0x";
210  expectedString += HexStringFrom(&const_value);
211  expectedString += " function_pointer:0x";
212  expectedString += HexStringFrom(function_value);
213  expectedString += " mem_buffer:Size = 2 | HexContents = FE 15";
214  expectedString += " int named_type:0x";
215  expectedString += HexStringFrom(&const_value);
216  STRCMP_EQUAL(expectedString.asCharString(), actual.getTraceOutput());
217 
218  CHECK_FALSE(actual.hasReturnValue());
219  CHECK(actual.returnValue().equals(MockNamedValue("")));
220  CHECK(false == actual.returnBoolValue());
221  CHECK(false == actual.returnBoolValueOrDefault(true));
222  CHECK(0 == actual.returnLongIntValue());
223  CHECK(0 == actual.returnUnsignedLongIntValue());
224  CHECK(0 == actual.returnIntValue());
225  CHECK(0 == actual.returnUnsignedLongIntValueOrDefault(1ul));
226  CHECK(0 == actual.returnIntValueOrDefault(1));
227  CHECK(0 == actual.returnLongIntValue());
228  CHECK(0 == actual.returnLongIntValueOrDefault(1l));
229  CHECK(0 == actual.returnUnsignedIntValue());
230  CHECK(0 == actual.returnUnsignedIntValueOrDefault(1u));
231  DOUBLES_EQUAL(0.0, actual.returnDoubleValue(), 0.0);
232  DOUBLES_EQUAL(0.0, actual.returnDoubleValueOrDefault(1.0), 0.0);
233  STRCMP_EQUAL("", actual.returnStringValueOrDefault("bla"));
234  STRCMP_EQUAL("", actual.returnStringValue());
235  CHECK(0 == actual.returnPointerValue());
236  CHECK(0 == actual.returnPointerValueOrDefault((void*) 0x0));
237  CHECK(0 == actual.returnConstPointerValue());
238  CHECK(0 == actual.returnConstPointerValueOrDefault((const void*) 0x0));
239  CHECK(0 == actual.returnFunctionPointerValue());
240  CHECK(0 == actual.returnFunctionPointerValueOrDefault((void (*)()) 0x0));
241 }
