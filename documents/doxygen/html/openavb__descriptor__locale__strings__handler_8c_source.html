89 extern DLL_EXPORT openavb_aem_descriptor_locale_strings_handler_group_t * openavbAemDescriptorLocaleStringsHandlerGet_group(
90  openavb_aem_descriptor_locale_strings_handler_t *pDescriptor,
91  const char *aLocaleIdentifier,
92  bool bCanCreate)
93 {
94  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
95 
96  if (!pDescriptor || !aLocaleIdentifier) {
97  AVB_RC_LOG(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_INVALID_ARGUMENT));
98  AVB_TRACE_EXIT(AVB_TRACE_AEM);
99  return NULL;
100  }
101 
102  // Do we have a matching item?
103  openavb_aem_descriptor_locale_strings_handler_group_t *pTest = pDescriptor->pFirstGroup;
104  while (pTest) {
105  if (strncmp(aLocaleIdentifier, (const char *) pTest->pLocale->locale_identifier, OPENAVB_AEM_STRLEN_MAX) == 0) {
106  AVB_TRACE_EXIT(AVB_TRACE_AEM);
107  return pTest;
108  }
109  pTest = pTest->pNext;
110  }
111 
112  // Can we allocate a new language structure?
113  if (bCanCreate) {
114  // Allocate a new group.
115  openavb_aem_descriptor_locale_strings_handler_group_t *pNew = malloc(sizeof(openavb_aem_descriptor_locale_strings_handler_group_t));
116  if (!pNew) {
117  AVB_TRACE_EXIT(AVB_TRACE_AEM);
118  return NULL;
119  }
120  pNew->pLocale = openavbAemDescriptorLocaleNew();
121  pNew->pStrings = openavbAemDescriptorStringsNew();
122  if (!pNew->pLocale || !pNew->pStrings) {
123  free(pNew->pLocale);
124  free(pNew->pStrings);
125  free(pNew);
126  AVB_TRACE_EXIT(AVB_TRACE_AEM);
127  return NULL;
128  }
129 
130  openavbAemDescriptorLocaleSet_locale_identifier(pNew->pLocale, aLocaleIdentifier);
131 
132  // Add the new group to the start of the linked list.
133  pNew->pNext = pDescriptor->pFirstGroup;
134  pDescriptor->pFirstGroup = pNew;
135 
136  AVB_TRACE_EXIT(AVB_TRACE_AEM);
137  return pNew;
138  }
139 
140  AVB_TRACE_EXIT(AVB_TRACE_AEM);
141  return NULL;
142 }

144 extern DLL_EXPORT const char * openavbAemDescriptorLocaleStringsHandlerGet_local_string(
145  openavb_aem_descriptor_locale_strings_handler_t *pDescriptor,
146  const char *aLocaleIdentifier,
147  U8 uStringIndex)
148 {
149  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
150 
151  if (!pDescriptor || !aLocaleIdentifier || uStringIndex >= OPENAVB_AEM_NUM_DESCRIPTOR_STRINGS) {
152  AVB_RC_LOG(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_INVALID_ARGUMENT));
153  AVB_TRACE_EXIT(AVB_TRACE_AEM);
154  return NULL;
155  }
156 
157  // Get the group structure to get this string from.
158  openavb_aem_descriptor_locale_strings_handler_group_t *pGroup = openavbAemDescriptorLocaleStringsHandlerGet_group(pDescriptor, aLocaleIdentifier, FALSE);
159  if (!pGroup) {
160  AVB_TRACE_EXIT(AVB_TRACE_AEM);
161  return NULL;
162  }
163 
164  const char * ret = openavbAemDescriptorStringsGet_string(pGroup->pStrings, uStringIndex);
165  AVB_TRACE_EXIT(AVB_TRACE_AEM);
166  return ret;
167 }

169 extern DLL_EXPORT bool openavbAemDescriptorLocaleStringsHandlerSet_local_string(
170  openavb_aem_descriptor_locale_strings_handler_t *pDescriptor,
171  const char *aLocaleIdentifier,
172  const char *aString,
173  U8 uStringIndex)
174 {
175  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
176 
177  if (!pDescriptor || !aLocaleIdentifier || !aString || uStringIndex >= OPENAVB_AEM_NUM_DESCRIPTOR_STRINGS) {
178  AVB_RC_LOG(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_INVALID_ARGUMENT));
179  AVB_TRACE_EXIT(AVB_TRACE_AEM);
180  return FALSE;
181  }
182 
183  // Get the group structure to add this string to. Create it if needed.
184  openavb_aem_descriptor_locale_strings_handler_group_t *pGroup = openavbAemDescriptorLocaleStringsHandlerGet_group(pDescriptor, aLocaleIdentifier, TRUE);
185  if (!pGroup) {
186  AVB_TRACE_EXIT(AVB_TRACE_AEM);
187  return FALSE;
188  }
189 
190  bool ret = openavbAemDescriptorStringsSet_string(pGroup->pStrings, aString, uStringIndex);
191  if (ret) {
192  openavbAemDescriptorLocaleSet_number_of_strings(pGroup->pLocale, 1);
193  openavbAemDescriptorLocaleSet_base_strings(pGroup->pLocale, 0);
194  }
195 
196  AVB_TRACE_EXIT(AVB_TRACE_AEM);
197  return ret;
198 }

200 extern DLL_EXPORT bool openavbAemDescriptorLocaleStringsHandlerAddToConfiguration(
201  openavb_aem_descriptor_locale_strings_handler_t *pDescriptor, U16 nConfigIdx)
202 {
203  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
204 
205  openavb_aem_descriptor_locale_strings_handler_group_t *pCurrent = pDescriptor->pFirstGroup;
206  while (pCurrent) {
207  U16 nResultIdx;
208 
209  // Add the locale to the configuration.
210  if (!openavbAemAddDescriptor(pCurrent->pLocale, nConfigIdx, &nResultIdx)) {
211  AVB_LOG_ERROR("Locale Add Descriptor Failure");
212  AVB_TRACE_EXIT(AVB_TRACE_AEM);
213  return false;
214  }
215 
216  // Add the strings descriptor, if it hasn't already been added.
217  if (openavbAemGetDescriptorIndex(OPENAVB_AEM_DESCRIPTOR_INVALID, pCurrent->pStrings) == OPENAVB_AEM_DESCRIPTOR_INVALID) {
218  if (!openavbAemAddDescriptor(pCurrent->pStrings, OPENAVB_AEM_DESCRIPTOR_INVALID /* Not configuration-specific */, &nResultIdx)) {
219  AVB_LOG_ERROR("Strings Add Descriptor Failure");
220  AVB_TRACE_EXIT(AVB_TRACE_AEM);
221  return false;
222  }
223  }
224 
225  pCurrent = pCurrent->pNext;
226  }
227 
228  AVB_TRACE_EXIT(AVB_TRACE_AEM);
229  return true;
230 }
