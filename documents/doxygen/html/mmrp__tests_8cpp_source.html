122  for (i = 0; i < 6; i++)
123  a_ref.attribute.macaddr[i] = i + 1; /* matches defn of STREAM_DA */

130  CHECK(attrib != NULL);
131 
132  /*
133  * Generate a LVA event.

135  * the attribute is encoded into a PDU.
136  */
137  mmrp_event(MRP_EVENT_LVATIMER, NULL);
138 
139  /* verify that all tx flags are zero by scanning the attribute list */
140  attrib = MMRP_db->attrib_list;

142  {
143  tx_flag_count += attrib->applicant.tx;
144  attrib = attrib->next;

146  CHECK(mrpd_send_packet_count() > 0);
