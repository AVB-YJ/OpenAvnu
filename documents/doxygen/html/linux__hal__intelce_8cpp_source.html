55 int LinuxTimestamperIntelCE::ce_timestamp_common
56 ( PortIdentity *identity, uint16_t sequenceId, Timestamp &timestamp,
57  unsigned &clock_value, bool tx ) {
58  uint64_t timestamp_s;
59  uint16_t captured_sequence;
60  uint16_t port_no;
61  PortIdentity captured_id;
62  ismd_sysclk_msg_t message;
63  ismd_result_t result;
64  int ret = GPTP_EC_EAGAIN;
65 
66  if( tx ) {
67  result = ismd_sysclk_get_tx_time( &timestamp_s, &message );
68  if( timestamp_s == last_tx_time ) {
69  result = ISMD_ERROR_NO_DATA_AVAILABLE;
70  } else {
71  last_tx_time = timestamp_s;
72  }
73  } else {
74  struct timespec ts;
75  clock_gettime( CLOCK_REALTIME, &ts );
76  result = ismd_sysclk_get_rx_time( &timestamp_s, &message );
77  if( timestamp_s == last_rx_time ) {
78  result = ISMD_ERROR_NO_DATA_AVAILABLE;
79  } else {
80  last_rx_time = timestamp_s;
81  }
82  }
83 
84  if( result == ISMD_ERROR_NO_DATA_AVAILABLE ) {
85  goto fail;
86  } else if( result != ISMD_SUCCESS ) {
87  ret = GPTP_EC_FAILURE;
88  goto fail;
89  }
90 
91  timestamp_s = scale_clock( timestamp_s );
92 
93  captured_id.setClockIdentity( ClockIdentity(message.msgid) );
94  port_no =
95  PLAT_ntohs(*((uint16_t *)(message.msgid+PTP_CLOCK_IDENTITY_LENGTH)));
96  captured_id.setPortNumber( &port_no );
97  captured_sequence = PLAT_ntohs( *((uint16_t *)message.msgseq ));
98 
99  if( captured_sequence != sequenceId || captured_id != *identity ) {
100  uint16_t cap_port_no;
101  uint16_t id_port_no;
102  captured_id.getPortNumber(&cap_port_no);
103  identity->getPortNumber(&id_port_no);
104  goto fail;
105  }
106 
107  ret = GPTP_EC_SUCCESS;
108 
109  if( tx ) {
110  timestamp_s += TX_PHY_TIME;
111  } else {
112  timestamp_s -= RX_PHY_TIME;
113  }
114 
115  timestamp.set64( timestamp_s );
116  timestamp._version = version;
117  clock_value = (unsigned) timestamp_s;
118 
119  fail:
120  return ret;
121 }

188 net_result LinuxNetworkInterface::nrecv
189 ( LinkLayerAddress *addr, uint8_t *payload, size_t &length ) {
190  fd_set readfds;
191  int err;
192  struct sockaddr_ll remote;
193  net_result ret = net_succeed;
194  bool got_net_lock;
195 
196  struct timeval timeout = { 0, 16000 }; // 16 ms
197 
198  if( !net_lock.lock( &got_net_lock ) || !got_net_lock ) {
199  GPTP_LOG_ERROR( "A Failed to lock mutex" );
200  return net_fatal;
201  }
202 
203  FD_ZERO( &readfds );
204  FD_SET( sd_event, &readfds );
205 
206  err = select( sd_event+1, &readfds, NULL, NULL, &timeout );
207  if( err == 0 ) {
208  ret = net_trfail;
209  goto done;
210  } else if( err == -1 ) {
211  if( err == EINTR ) {
212  // Caught signal
213  GPTP_LOG_ERROR( "select() recv signal" );
214  ret = net_trfail;
215  goto done;
216  } else {
217  GPTP_LOG_ERROR( "select() failed" );
218  ret = net_fatal;
219  goto done;
220  }
221  } else if( !FD_ISSET( sd_event, &readfds )) {
222  ret = net_trfail;
223  goto done;
224  }
225 
226  err = recv( sd_event, payload, length, 0 );
227  if( err < 0 ) {
228  GPTP_LOG_ERROR( "recvmsg() failed: %s", strerror(errno) );
229  ret = net_fatal;
230  goto done;
231  }
232  *addr = LinkLayerAddress( remote.sll_addr );
233 
234  length = err;
235 
236  done:
237  if( !net_lock.unlock()) {
238  GPTP_LOG_ERROR( "A Failed to unlock, %d", err );
239  return net_fatal;
240  }
241
