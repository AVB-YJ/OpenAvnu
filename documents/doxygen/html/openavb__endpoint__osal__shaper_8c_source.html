99 /* Local function to interact with the SHAPER daemon. */
100 static void* shaperThread(void *arg)
101 {
102  struct addrinfo hints, *ai, *p;
103  int ret;
104 
105  fd_set master, read_fds;
106  int fdmax;
107 
108  char recvbuffer[200];
109  int recvbytes;
110 
111  AVB_LOG_DEBUG("Shaper Thread Starting");
112 
113  /* Create a localhost socket. */
114  memset(&hints, 0, sizeof hints);
115  hints.ai_family = AF_UNSPEC;
116  hints.ai_socktype = SOCK_STREAM;
117  hints.ai_flags = 0;
118  if ((ret = getaddrinfo("localhost", shaperDaemonPort, &hints, &ai)) != 0) {
119  AVB_LOGF_ERROR("getaddrinfo failure %s", gai_strerror(ret));
120  shaperState = SHAPER_STATE_ERROR;
121  return NULL;
122  }
123 
124  for(p = ai; p != NULL; p = p->ai_next) {
125  socketfd = socket(p->ai_family, p->ai_socktype, p->ai_protocol);
126  if (socketfd == -1) {
127  continue;
128  }
129  ret = connect(socketfd, p->ai_addr, p->ai_addrlen);
130  if (ret == -1) {
131  close(socketfd);
132  socketfd = -1;
133  continue;
134  } else {
135  break;
136  }
137  }
138 
139  freeaddrinfo(ai);
140 
141  if (p == NULL) {
142  AVB_LOGF_ERROR("Shaper: Unable to connect to the daemon, error %d (%s)", errno, strerror(errno));
143  shaperState = SHAPER_STATE_ERROR;
144  return NULL;
145  }
146 
147  if (fcntl(socketfd, F_SETFL, O_NONBLOCK) < 0)
148  {
149  AVB_LOG_ERROR("Shaper: Could not set the socket to non-blocking");
150  close(socketfd);
151  socketfd = -1;
152  shaperState = SHAPER_STATE_ERROR;
153  return NULL;
154  }
155 
156  FD_ZERO(&read_fds);
157  FD_ZERO(&master);
158  FD_SET(STDIN_FILENO, &master);
159  FD_SET(socketfd, &master);
160  fdmax = socketfd;
161 
162 
163  /*
164  * Main event loop
165  */
166 
167  shaperState = SHAPER_STATE_CONNECTED;
168  AVB_LOG_INFO("Shaper daemon available");
169 
170  while (shaperRunning || shaperState == SHAPER_STATE_ENABLED)
171  {
172  if (!shaperRunning && shaperState == SHAPER_STATE_ENABLED)
173  {
174  // TODO: Tell the SHAPER daemon to stop shaping.
175  break;
176  }
177 
178  /* Wait for something to happen. */
179  struct timeval tv_timeout = { 1, 0 };
180  read_fds = master;
181  ret = select(fdmax+1, &read_fds, NULL, NULL, &tv_timeout);
182  if (ret < 0)
183  {
184  AVB_LOGF_ERROR("Shaper: select() error %d (%s)", errno, strerror(errno));
185  shaperState = SHAPER_STATE_ERROR;
186  break;
187  }
188 
189  /* Handle any responses received. */
190  if (FD_ISSET(socketfd, &read_fds))
191  {
192  while ((recvbytes = recv(socketfd, recvbuffer, sizeof(recvbuffer) - 1, 0)) > 0)
193  {
194  /* Log the response data */
195  recvbuffer[recvbytes] = '\0';
196  if (strncmp(recvbuffer, "ERROR:", 6) == 0) {
197  AVB_LOGF_ERROR("Shaper Response: %s", recvbuffer + 7);
198  }
199  else if (strncmp(recvbuffer, "WARNING:", 8) == 0) {
200  AVB_LOGF_WARNING("Shaper Response: %s", recvbuffer + 9);
201  }
202  else if (strncmp(recvbuffer, "DEBUG:", 6) == 0) {
203  AVB_LOGF_DEBUG("Shaper Response: %s", recvbuffer + 7);
204  }
205  else {
206  AVB_LOGF_DEBUG("Shaper Response: %s", recvbuffer);
207  }
208  }
209  if (recvbytes == 0)
210  {
211  /* The SHAPER daemon closed the connection. Assume it shut down, and we should too. */
212  // AVDECC_TODO: Should we try to reconnect?
213  AVB_LOG_ERROR("Shaper daemon exited.");
214  shaperState = SHAPER_STATE_ERROR;
215  break;
216  }
217  if (recvbytes < 0 && errno != EWOULDBLOCK)
218  {
219  /* Something went wrong. Abort! */
220  AVB_LOGF_ERROR("Shaper: Error %d reading from network socket (%s)", errno, strerror(errno));
221  shaperState = SHAPER_STATE_ERROR;
222  break;
223  }
224  }
225  }
226 
227  if (shaperState < SHAPER_STATE_NOT_AVAILABLE) {
228  shaperState = SHAPER_STATE_NOT_AVAILABLE;
229  }
230 
231  close(socketfd);
232  socketfd = -1;
233 
234  AVB_LOG_DEBUG("Shaper Thread Done");
235  return NULL;
236 }

238 bool openavbShaperInitialize(const char *ifname, unsigned int shaperPort)
239 {
240  AVB_TRACE_ENTRY(AVB_TRACE_SHAPER);
241 
242  MUTEX_ATTR_HANDLE(mta);
243  MUTEX_ATTR_INIT(mta);
244  MUTEX_ATTR_SET_TYPE(mta, MUTEX_ATTR_TYPE_DEFAULT);
245  MUTEX_ATTR_SET_NAME(mta, "shaperMutex");
246  MUTEX_CREATE_ERR();
247  MUTEX_CREATE(shaperMutex, mta);
248  MUTEX_LOG_ERR("Could not create/initialize 'shaperMutex' mutex");
249 
250  memset(shaperReservationList, 0, sizeof(shaperReservationList));
251 
252  if (shaperPort == 0) {
253  shaperState = SHAPER_STATE_NOT_AVAILABLE;
254  }
255  else {
256  shaperState = SHAPER_STATE_UNKNOWN;
257  sprintf(shaperDaemonPort, "%u", shaperPort);
258 
259  // Save the interface-only name to pass to the daemon later.
260  const char *ifonly = strchr(ifname, ':');
261  if (ifonly) {
262  ifonly++; // Go past the colon
263  } else {
264  ifonly = ifname; // No colon in interface name
265  }
266  strncpy(interfaceOnly, ifonly, sizeof(interfaceOnly));
267  interfaceOnly[sizeof(interfaceOnly) - 1] = '\0';
268 
269  shaperRunning = TRUE;
270  int err = pthread_create(&shaperThreadHandle, NULL, shaperThread, NULL);
271  if (err) {
272  shaperRunning = FALSE;
273  shaperState = SHAPER_STATE_ERROR;
274  AVB_LOGF_ERROR("Failed to start SHAPER thread: %s", strerror(err));
275  }
276  }
277 
278  AVB_TRACE_EXIT(AVB_TRACE_SHAPER);
279  return true;
280 }

299 bool openavbShaperDaemonAvailable(void)
300 {
301  AVB_TRACE_ENTRY(AVB_TRACE_SHAPER);
302 
303  if (!shaperRunning) {
304  AVB_TRACE_EXIT(AVB_TRACE_SHAPER);
305  return FALSE;
306  }
307 
308  // Wait for the daemon to connect.
309  if (shaperState == SHAPER_STATE_UNKNOWN) {
310  SLEEP_MSEC(50);
311  }
312 
313  if (shaperState > SHAPER_STATE_UNKNOWN && shaperState < SHAPER_STATE_NOT_AVAILABLE) {
314  AVB_TRACE_EXIT(AVB_TRACE_SHAPER);
315  return TRUE;
316  }
317 
318  if (shaperState != SHAPER_STATE_NOT_AVAILABLE) {
319  AVB_LOG_WARNING("Shaper Daemon not available.");
320  }
321  AVB_TRACE_EXIT(AVB_TRACE_SHAPER);
322  return FALSE;
323 }

325 void* openavbShaperHandle(SRClassIdx_t sr_class, int measurement_interval_usec, int max_frame_size_bytes, int max_frames_per_interval, const unsigned char * stream_da)
326 {
327  // If the daemon is not available, abort!
328  if (socketfd == -1 || !openavbShaperDaemonAvailable())
329  {
330  AVB_LOG_ERROR("Shaper reservation attempted with no daemon available");
331  return NULL;
332  }
333 
334  // Find the first available index.
335  int i;
336  for (i = 0; i < MAX_SHAPER_RESERVATIONS; ++i)
337  {
338  if (!(shaperReservationList[i].in_use))
339  {
340  break;
341  }
342  }
343  if (i >= MAX_SHAPER_RESERVATIONS)
344  {
345  AVB_LOG_ERROR("No Shaper reservations are available");
346  return NULL;
347  }
348 
349  // Fill in the information.
350  shaperReservationList[i].in_use = TRUE;
351  shaperReservationList[i].sr_class = sr_class;
352  shaperReservationList[i].measurement_interval = measurement_interval_usec;
353  shaperReservationList[i].max_frame_size = max_frame_size_bytes;
354  shaperReservationList[i].max_frames_per_interval = max_frames_per_interval;
355  memcpy(shaperReservationList[i].stream_da, stream_da, ETH_ALEN);
356 
357  // Send the information to the Shaper daemon.
358  // Reserving Bandwidth Example: -ri eth2 -c A -s 125 -b 74 -f 1 -a ff:ff:ff:ff:ff:11\n
359  char szCommand[200];
360  sprintf(szCommand, "-ri %s -c %c -s %u -b %u -f %u -a %02x:%02x:%02x:%02x:%02x:%02x",
361  interfaceOnly,
362  ( shaperReservationList[i].sr_class == SR_CLASS_A ? 'A' : 'B' ),
363  shaperReservationList[i].measurement_interval,
364  shaperReservationList[i].max_frame_size,
365  shaperReservationList[i].max_frames_per_interval,
366  shaperReservationList[i].stream_da[0],
367  shaperReservationList[i].stream_da[1],
368  shaperReservationList[i].stream_da[2],
369  shaperReservationList[i].stream_da[3],
370  shaperReservationList[i].stream_da[4],
371  shaperReservationList[i].stream_da[5]);
372  AVB_LOGF_DEBUG("Sending Shaper command: %s", szCommand);
373  strcat(szCommand, "\n");
374  if (send(socketfd, szCommand, strlen(szCommand), 0) < 0)
375  {
376  /* Something went wrong. Abort! */
377  AVB_LOGF_ERROR("Shaper: Error %d writing to network socket (%s)", errno, strerror(errno));
378  shaperState = SHAPER_STATE_ERROR;
379  return NULL;
380  }
381 
382  shaperState = SHAPER_STATE_ENABLED;
383 
384  // TODO: Verify that the command was successful.
385 
386  return (void *)(&(shaperReservationList[i]));
387 }

404  shaperReservationList[i].stream_da[2],

407  shaperReservationList[i].stream_da[5]);
408

411  AVB_LOGF_DEBUG("Sending Shaper command: %s", szCommand);

415  /* Something went wrong. */
416  AVB_LOGF_ERROR("Shaper: Error %d writing to network socket (%s)", errno, strerror(errno));
417  shaperState = SHAPER_STATE_ERROR;
418  }
419  else {
420  // TODO: Verify that the command was successful.
421  }
422  }
423  break;
424  }
425  }
