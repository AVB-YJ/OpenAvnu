32 /*
33 * MODULE SUMMARY : Pipe mapping module.
34 *
35 * The Pipe mapping is an AVB "experimental" mapping format. It will
36 * pass throught any data from interface modules unchanged.
37 * This can be useful for development, testing or custom solutions.
38 *
39 *----------------------------------------------------------------*
40 *
41 * HEADERS
42 *
43 * -+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-
44 * |C| |S| |M| |G|T| | |T|
45 * |D|subtype |V|vers |R|R|V|V|sequence number|reserved |U|
46 * --+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+--
47 * | |
48 * | |
49 * - -
50 * | |
51 * |stream id |
52 * --+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+--
53 * | |
54 * |AVTP timestamp |
55 * --+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+--
56 * | | | |
57 * | Vendor_eui_1 |
58 * --+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+--
59 * | |
60 * | Data Length | Vendor_eui_2 |
61 * -+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-|-+-+-+-+-+-+-+-
62 *
63 * CD : Standard AVTP
64 * Subtype : Vendor specific 0x6f (Introduced in 1722A(D)
65 * SV : Standard AVTP
66 * Ver` : Standard AVTP
67 * MR : Standard AVTP
68 * R : Standard AVTP
69 * GV : Standard AVTP
70 * TV : Standard AVTP
71 * Sequence number : Standard AVTP
72 * Reserved : Standard AVTP
73 * TU : Standard AVTP
74 * Stream ID : Standard AVTP
75 * AVTP timestamp : Standard AVTP
76 * Vendor_eui_1 : Vendor specific (includes OPENAVB format : Pipe Mapping 0x01)
77 * Data Length : Length of the data payload
78 * Vendor_eui_2 : Vendor specific
79 *
80 */

162 // Each configuration name value pair for this mapping will result in this callback being called.
163 void openavbMapPipeCfgCB(media_q_t *pMediaQ, const char *name, const char *value)
164 {
165  AVB_TRACE_ENTRY(AVB_TRACE_MAP);
166  if (pMediaQ) {
167  pvt_data_t *pPvtData = pMediaQ->pPvtMapInfo;
168  if (!pPvtData) {
169  AVB_LOG_ERROR("Private mapping module data not allocated.");
170  return;
171  }
172 
173  if (strcmp(name, "map_nv_item_count") == 0) {
174  char *pEnd;
175  pPvtData->itemCount = strtol(value, &pEnd, 10);
176  }
177  else if (strcmp(name, "map_nv_tx_rate") == 0
178  || strcmp(name, "map_nv_tx_interval") == 0) {
179  char *pEnd;
180  pPvtData->txInterval = strtol(value, &pEnd, 10);
181  }
182  else if (strcmp(name, "map_nv_max_payload_size") == 0) {
183  char *pEnd;
184  pPvtData->maxPayloadSize = strtol(value, &pEnd, 10);
185  pPvtData->maxDataSize = (pPvtData->maxPayloadSize + TOTAL_HEADER_SIZE);
186  pPvtData->itemSize = pPvtData->maxPayloadSize;
187  }
188  else if (strcmp(name, "map_nv_push_header") == 0) {
189  char *pEnd;
190  long tmp;
191  tmp = strtol(value, &pEnd, 10);
192  if (*pEnd == '\0' && tmp == 1) {
193  pPvtData->push_header = (tmp == 1);
194  }
195  }
196  else if (strcmp(name, "map_nv_pull_header") == 0) {
197  char *pEnd;
198  long tmp;
199  tmp = strtol(value, &pEnd, 10);
200  if (*pEnd == '\0' && tmp == 1) {
201  pPvtData->pull_header = (tmp == 1);
202  }
203  }
204  }
205 
206  AVB_TRACE_EXIT(AVB_TRACE_MAP);
207 }

282 // This talker callback will be called for each AVB observation interval.
283 tx_cb_ret_t openavbMapPipeTxCB(media_q_t *pMediaQ, U8 *pData, U32 *dataLen)
284 {
285  AVB_TRACE_ENTRY(AVB_TRACE_MAP_DETAIL);
286  if (pMediaQ && pData && dataLen) {
287  U8 *pHdr = pData;
288  U8 *pPayload = pData + TOTAL_HEADER_SIZE;
289  pvt_data_t *pPvtData = pMediaQ->pPvtMapInfo;
290  if (!pPvtData) {
291  AVB_LOG_ERROR("Private mapping module data not allocated.");
292  return TX_CB_RET_PACKET_NOT_READY;
293  }
294 
295  media_q_item_t *pMediaQItem = openavbMediaQTailLock(pMediaQ, TRUE);
296 
297  if (pMediaQItem) {
298  if (pMediaQItem->dataLen > 0) {
299  if (pMediaQItem->dataLen > pPvtData->maxDataSize) {
300  AVB_LOGF_ERROR("Media queue data item size too large. Reported size: %d Max Size: %d", pMediaQItem->dataLen, pPvtData->maxDataSize);
301  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
302  openavbMediaQTailPull(pMediaQ);
303  return TX_CB_RET_PACKET_NOT_READY;
304  }
305 
306  // PTP walltime already set in the interface module. Just add the max transit time.
307  openavbAvtpTimeAddUSec(pMediaQItem->pAvtpTime, pPvtData->maxTransitUsec);
308 
309  // Set timestamp valid flag
310  if (openavbAvtpTimeTimestampIsValid(pMediaQItem->pAvtpTime))
311  pHdr[HIDX_AVTP_HIDE7_TV1] |= 0x01; // Set
312  else {
313  pHdr[HIDX_AVTP_HIDE7_TV1] &= ~0x01; // Clear
314  }
315 
316  // Set timestamp uncertain flag
317  if (openavbAvtpTimeTimestampIsUncertain(pMediaQItem->pAvtpTime))
318  pHdr[HIDX_AVTP_HIDE7_TU1] |= 0x01; // Set
319  else pHdr[HIDX_AVTP_HIDE7_TU1] &= ~0x01; // Clear
320 
321  *(U32 *)(&pHdr[HIDX_AVTP_TIMESPAMP32]) = htonl(openavbAvtpTimeGetAvtpTimestamp(pMediaQItem->pAvtpTime));
322  *(U32 *)(&pHdr[HIDX_OPENAVB_FORMAT8]) = 0x00000000;
323  pHdr[HIDX_OPENAVB_FORMAT8] = MAP_PIPE_OPENAVB_FORMAT;
324  // for alignment
325  U16 payloadLen = htons(pMediaQItem->dataLen);;
326  memcpy(&pHdr[HIDX_AVTP_DATALEN16], &payloadLen, sizeof(U16));
327 
328  *(U16 *)(&pHdr[HIDX_VENDOR2_EUI16]) = 0x0000;
329 
330  if (pPvtData->pull_header) {
331  memcpy(pData, pMediaQItem->pPubData, pMediaQItem->dataLen);
332  *dataLen = pMediaQItem->dataLen;
333  }
334  else {
335  memcpy(pPayload, pMediaQItem->pPubData, pMediaQItem->dataLen);
336  *dataLen = pMediaQItem->dataLen + TOTAL_HEADER_SIZE;
337  }
338 
339  openavbMediaQTailPull(pMediaQ);
340  AVB_TRACE_LINE(AVB_TRACE_MAP_LINE);
341  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
342  return TX_CB_RET_PACKET_READY;
343  }
344  else {
345  openavbMediaQTailPull(pMediaQ);
346  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
347  return TX_CB_RET_PACKET_NOT_READY; // No payload
348  }
349  }
350  else {
351  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
352  return TX_CB_RET_PACKET_NOT_READY; // Media queue empty
353  }
354  }
355  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
356  return TX_CB_RET_PACKET_NOT_READY;
357 }

367 // This callback occurs when running as a listener and data is available.
368 bool openavbMapPipeRxCB(media_q_t *pMediaQ, U8 *pData, U32 dataLen)
369 {
370  AVB_TRACE_ENTRY(AVB_TRACE_MAP_DETAIL);
371  if (pMediaQ && pData) {
372  const U8 *pHdr = pData;
373  U8 *pPayload = pData + TOTAL_HEADER_SIZE;
374  pvt_data_t *pPvtData = pMediaQ->pPvtMapInfo;
375  if (!pPvtData) {
376  AVB_LOG_ERROR("Private mapping module data not allocated.");
377  return FALSE;
378  }
379 
380  media_q_item_t *pMediaQItem = openavbMediaQHeadLock(pMediaQ);
381  if (pMediaQItem) {
382  U32 timestamp = ntohl(*(U32 *)(&pHdr[HIDX_AVTP_TIMESPAMP32]));
383  openavbAvtpTimeSetToTimestamp(pMediaQItem->pAvtpTime, timestamp);
384 
385  // Set timestamp valid and timestamp uncertain flags
386  openavbAvtpTimeSetTimestampValid(pMediaQItem->pAvtpTime, (pHdr[HIDX_AVTP_HIDE7_TV1] & 0x01) ? TRUE : FALSE);
387  openavbAvtpTimeSetTimestampUncertain(pMediaQItem->pAvtpTime, (pHdr[HIDX_AVTP_HIDE7_TU1] & 0x01) ? TRUE : FALSE);
388 
389  if (pPvtData->push_header) {
390  if (pMediaQItem->itemSize >= dataLen) {
391  memcpy(pMediaQItem->pPubData, pData, dataLen);
392  pMediaQItem->dataLen = dataLen;
393  }
394  else {
395  AVB_LOG_ERROR("Data to large for media queue.");
396  pMediaQItem->dataLen = 0;
397  }
398  }
399  else {
400  // for alignment
401  U16 payloadLen;
402  memcpy(&payloadLen, &pHdr[HIDX_AVTP_DATALEN16], sizeof(U16));
403  payloadLen = ntohs(payloadLen);
404 
405  if (pMediaQItem->itemSize >= payloadLen) {
406  memcpy(pMediaQItem->pPubData, pPayload, payloadLen);
407  pMediaQItem->dataLen = payloadLen;
408  }
409  else {
410  AVB_LOG_ERROR("Data to large for media queue.");
411  pMediaQItem->dataLen = 0;
412  }
413  }
414 
415  openavbMediaQHeadPush(pMediaQ);
416  AVB_TRACE_LINE(AVB_TRACE_MAP_LINE);
417  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
418  return TRUE;
419  }
420  else {
421  IF_LOG_INTERVAL(1000) AVB_LOG_INFO("Media queue full");
422  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
423  return FALSE; // Media queue full
424  }
425  }
426  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
427  return FALSE;
428 }

444 // Initialization entry point into the mapping module. Will need to be included in the .ini file.
445 extern DLL_EXPORT bool openavbMapPipeInitialize(media_q_t *pMediaQ, openavb_map_cb_t *pMapCB, U32 inMaxTransitUsec)
446 {
447  AVB_TRACE_ENTRY(AVB_TRACE_MAP);
448 
449  if (pMediaQ) {
450  pMediaQ->pMediaQDataFormat = strdup(MapPipeMediaQDataFormat);
451  pMediaQ->pPvtMapInfo = calloc(1, sizeof(pvt_data_t)); // Memory freed by the media queue when the media queue is destroyed.
452 
453  if (!pMediaQ->pMediaQDataFormat || !pMediaQ->pPvtMapInfo) {
454  AVB_LOG_ERROR("Unable to allocate memory for mapping module.");
455  return FALSE;
456  }
457 
458  pvt_data_t *pPvtData = pMediaQ->pPvtMapInfo;
459 
460  pMapCB->map_cfg_cb = openavbMapPipeCfgCB;
461  pMapCB->map_subtype_cb = openavbMapPipeSubtypeCB;
462  pMapCB->map_avtp_version_cb = openavbMapPipeAvtpVersionCB;
463  pMapCB->map_max_data_size_cb = openavbMapPipeMaxDataSizeCB;
464  pMapCB->map_transmit_interval_cb = openavbMapPipeTransmitIntervalCB;
465  pMapCB->map_gen_init_cb = openavbMapPipeGenInitCB;
466  pMapCB->map_tx_init_cb = openavbMapPipeTxInitCB;
467  pMapCB->map_tx_cb = openavbMapPipeTxCB;
468  pMapCB->map_rx_init_cb = openavbMapPipeRxInitCB;
469  pMapCB->map_rx_cb = openavbMapPipeRxCB;
470  pMapCB->map_end_cb = openavbMapPipeEndCB;
471  pMapCB->map_gen_end_cb = openavbMapPipeGenEndCB;
472 
473  pPvtData->itemCount = 20;
474  pPvtData->txInterval = 0;
475  pPvtData->push_header = FALSE;
476  pPvtData->pull_header = FALSE;
477  pPvtData->maxTransitUsec = inMaxTransitUsec;
478 
479  pPvtData->maxPayloadSize = 1024;
480  pPvtData->maxDataSize = (pPvtData->maxPayloadSize + TOTAL_HEADER_SIZE);
481  pPvtData->itemSize = pPvtData->maxDataSize;
482 
483  openavbMediaQSetMaxLatency(pMediaQ, inMaxTransitUsec);
484  }
485 
486  AVB_TRACE_EXIT(AVB_TRACE_MAP);
487  return TRUE;
488 }
