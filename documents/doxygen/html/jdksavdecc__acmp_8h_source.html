599 static inline ssize_t jdksavdecc_acmpdu_read( struct jdksavdecc_acmpdu *p, void const *base, ssize_t pos, size_t len )
600 {
601  ssize_t r = jdksavdecc_validate_range( pos, len, JDKSAVDECC_ACMPDU_LEN );
602  if ( r >= 0 )
603  {
604  jdksavdecc_acmpdu_common_control_header_read( &p->header, base, pos, len );
605  p->controller_entity_id = jdksavdecc_acmpdu_get_controller_entity_id( base, pos );
606  p->talker_entity_id = jdksavdecc_acmpdu_get_talker_entity_id( base, pos );
607  p->listener_entity_id = jdksavdecc_acmpdu_get_listener_entity_id( base, pos );
608  p->talker_unique_id = jdksavdecc_acmpdu_get_talker_unique_id( base, pos );
609  p->listener_unique_id = jdksavdecc_acmpdu_get_listener_unique_id( base, pos );
610  p->stream_dest_mac = jdksavdecc_acmpdu_get_stream_dest_mac( base, pos );
611  p->connection_count = jdksavdecc_acmpdu_get_connection_count( base, pos );
612  p->sequence_id = jdksavdecc_acmpdu_get_sequence_id( base, pos );
613  p->flags = jdksavdecc_acmpdu_get_flags( base, pos );
614  p->stream_vlan_id = jdksavdecc_acmpdu_get_stream_vlan_id( base, pos );
615  p->reserved = jdksavdecc_acmpdu_get_reserved( base, pos );
616  }
617  return r;
618 }
