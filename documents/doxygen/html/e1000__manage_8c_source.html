35 u8 e1000_calculate_checksum(u8 *buffer, u32 length)
36 {
37  u32 i;
38  u8 sum = 0;
39 
40  DEBUGFUNC("e1000_calculate_checksum");
41 
42  if (!buffer)
43  return 0;
44 
45  for (i = 0; i < length; i++)
46  sum += buffer[i];
47 
48  return (u8) (0 - sum);
49 }

61 s32 e1000_mng_enable_host_if_generic(struct e1000_hw *hw)
62 {
63  u32 hicr;
64  u8 i;
65 
66  DEBUGFUNC("e1000_mng_enable_host_if_generic");
67 
68  if (!hw->mac.arc_subsystem_valid) {
69  DEBUGOUT("ARC subsystem not valid.\n");
70  return -E1000_ERR_HOST_INTERFACE_COMMAND;
71  }
72 
73  /* Check that the host interface is enabled. */
74  hicr = E1000_READ_REG(hw, E1000_HICR);
75  if (!(hicr & E1000_HICR_EN)) {
76  DEBUGOUT("E1000_HOST_EN bit disabled.\n");
77  return -E1000_ERR_HOST_INTERFACE_COMMAND;
78  }
79  /* check the previous command is completed */
80  for (i = 0; i < E1000_MNG_DHCP_COMMAND_TIMEOUT; i++) {
81  hicr = E1000_READ_REG(hw, E1000_HICR);
82  if (!(hicr & E1000_HICR_C))
83  break;
84  msec_delay_irq(1);
85  }
86 
87  if (i == E1000_MNG_DHCP_COMMAND_TIMEOUT) {
88  DEBUGOUT("Previous command timeout failed .\n");
89  return -E1000_ERR_HOST_INTERFACE_COMMAND;
90  }
91 
92  return E1000_SUCCESS;
93 }

120 bool e1000_enable_tx_pkt_filtering_generic(struct e1000_hw *hw)
121 {
122  struct e1000_host_mng_dhcp_cookie *hdr = &hw->mng_cookie;
123  u32 *buffer = (u32 *)&hw->mng_cookie;
124  u32 offset;
125  s32 ret_val, hdr_csum, csum;
126  u8 i, len;
127 
128  DEBUGFUNC("e1000_enable_tx_pkt_filtering_generic");
129 
130  hw->mac.tx_pkt_filtering = true;
131 
132  /* No manageability, no filtering */
133  if (!hw->mac.ops.check_mng_mode(hw)) {
134  hw->mac.tx_pkt_filtering = false;
135  return hw->mac.tx_pkt_filtering;
136  }
137 
138  /* If we can't read from the host interface for whatever
139  * reason, disable filtering.
140  */
141  ret_val = e1000_mng_enable_host_if_generic(hw);
142  if (ret_val != E1000_SUCCESS) {
143  hw->mac.tx_pkt_filtering = false;
144  return hw->mac.tx_pkt_filtering;
145  }
146 
147  /* Read in the header. Length and offset are in dwords. */
148  len = E1000_MNG_DHCP_COOKIE_LENGTH >> 2;
149  offset = E1000_MNG_DHCP_COOKIE_OFFSET >> 2;
150  for (i = 0; i < len; i++)
151  *(buffer + i) = E1000_READ_REG_ARRAY_DWORD(hw, E1000_HOST_IF,
152  offset + i);
153  hdr_csum = hdr->checksum;
154  hdr->checksum = 0;
155  csum = e1000_calculate_checksum((u8 *)hdr,
156  E1000_MNG_DHCP_COOKIE_LENGTH);
157  /* If either the checksums or signature don't match, then
158  * the cookie area isn't considered valid, in which case we
159  * take the safe route of assuming Tx filtering is enabled.
160  */
161  if ((hdr_csum != csum) || (hdr->signature != E1000_IAMT_SIGNATURE)) {
162  hw->mac.tx_pkt_filtering = true;
163  return hw->mac.tx_pkt_filtering;
164  }
165 
166  /* Cookie area is valid, make the final check for filtering. */
167  if (!(hdr->status & E1000_MNG_DHCP_COOKIE_STATUS_PARSING))
168  hw->mac.tx_pkt_filtering = false;
169 
170  return hw->mac.tx_pkt_filtering;
171 }

180 s32 e1000_mng_write_cmd_header_generic(struct e1000_hw *hw,
181  struct e1000_host_mng_command_header *hdr)
182 {
183  u16 i, length = sizeof(struct e1000_host_mng_command_header);
184 
185  DEBUGFUNC("e1000_mng_write_cmd_header_generic");
186 
187  /* Write the whole command header structure with new checksum. */
188 
189  hdr->checksum = e1000_calculate_checksum((u8 *)hdr, length);
190 
191  length >>= 2;
192  /* Write the relevant command block into the ram area. */
193  for (i = 0; i < length; i++) {
194  E1000_WRITE_REG_ARRAY_DWORD(hw, E1000_HOST_IF, i,
195  *((u32 *) hdr + i));
196  E1000_WRITE_FLUSH(hw);
197  }
198 
199  return E1000_SUCCESS;

214 s32 e1000_mng_host_if_write_generic(struct e1000_hw *hw, u8 *buffer,
215  u16 length, u16 offset, u8 *sum)
216 {
217  u8 *tmp;
218  u8 *bufptr = buffer;
219  u32 data = 0;
220  u16 remaining, i, j, prev_bytes;
221 
222  DEBUGFUNC("e1000_mng_host_if_write_generic");
223 
224  /* sum = only sum of the data and it is not checksum */
225 
226  if (length == 0 || offset + length > E1000_HI_MAX_MNG_DATA_LENGTH)
227  return -E1000_ERR_PARAM;
228 
229  tmp = (u8 *)&data;
230  prev_bytes = offset & 0x3;
231  offset >>= 2;
232 
233  if (prev_bytes) {
234  data = E1000_READ_REG_ARRAY_DWORD(hw, E1000_HOST_IF, offset);
235  for (j = prev_bytes; j < sizeof(u32); j++) {
236  *(tmp + j) = *bufptr++;
237  *sum += *(tmp + j);
238  }
239  E1000_WRITE_REG_ARRAY_DWORD(hw, E1000_HOST_IF, offset, data);
240  length -= j - prev_bytes;
241  offset++;
242  }
243 
244  remaining = length & 0x3;
245  length -= remaining;
246 
247  /* Calculate length in DWORDs */
248  length >>= 2;
249 
250  /* The device driver writes the relevant command block into the
251  * ram area.
252  */
253  for (i = 0; i < length; i++) {
254  for (j = 0; j < sizeof(u32); j++) {
255  *(tmp + j) = *bufptr++;
256  *sum += *(tmp + j);
257  }
258 
259  E1000_WRITE_REG_ARRAY_DWORD(hw, E1000_HOST_IF, offset + i,
260  data);
261  }
262  if (remaining) {
263  for (j = 0; j < sizeof(u32); j++) {
264  if (j < remaining)
265  *(tmp + j) = *bufptr++;
266  else
267  *(tmp + j) = 0;
268 
269  *sum += *(tmp + j);
270  }
271  E1000_WRITE_REG_ARRAY_DWORD(hw, E1000_HOST_IF, offset + i,
272  data);
273  }
274 
275  return E1000_SUCCESS;
276 }

286 s32 e1000_mng_write_dhcp_info_generic(struct e1000_hw *hw, u8 *buffer,
287  u16 length)
288 {
289  struct e1000_host_mng_command_header hdr;
290  s32 ret_val;
291  u32 hicr;
292 
293  DEBUGFUNC("e1000_mng_write_dhcp_info_generic");
294 
295  hdr.command_id = E1000_MNG_DHCP_TX_PAYLOAD_CMD;
296  hdr.command_length = length;
297  hdr.reserved1 = 0;
298  hdr.reserved2 = 0;
299  hdr.checksum = 0;
300 
301  /* Enable the host interface */
302  ret_val = e1000_mng_enable_host_if_generic(hw);
303  if (ret_val)
304  return ret_val;
305

307  ret_val = e1000_mng_host_if_write_generic(hw, buffer, length,
308  sizeof(hdr), &(hdr.checksum));
309  if (ret_val)
310  return ret_val;
311

313  ret_val = e1000_mng_write_cmd_header_generic(hw, &hdr);
314  if (ret_val)
315  return ret_val;
316 
317  /* Tell the ARC a new command is pending. */
318  hicr = E1000_READ_REG(hw, E1000_HICR);
319  E1000_WRITE_REG(hw, E1000_HICR, hicr | E1000_HICR_C);
320 
321  return E1000_SUCCESS;

331 bool e1000_enable_mng_pass_thru(struct e1000_hw *hw)
332 {
333  u32 manc;
334  u32 fwsm, factps;
335 
336  DEBUGFUNC("e1000_enable_mng_pass_thru");
337 
338  if (!hw->mac.asf_firmware_present)
339  return false;
340 
341  manc = E1000_READ_REG(hw, E1000_MANC);
342 
343  if (!(manc & E1000_MANC_RCV_TCO_EN))
344  return false;
345 
346  if (hw->mac.has_fwsm) {
347  fwsm = E1000_READ_REG(hw, E1000_FWSM);
348  factps = E1000_READ_REG(hw, E1000_FACTPS);
349 
350  if (!(factps & E1000_FACTPS_MNGCG) &&
351  ((fwsm & E1000_FWSM_MODE_MASK) ==
352  (e1000_mng_mode_pt << E1000_FWSM_MODE_SHIFT)))
353  return true;
354  } else if ((manc & E1000_MANC_SMBUS_EN) &&
355  !(manc & E1000_MANC_ASF_EN)) {
356  return true;
357  }
358 
359  return false;

371 s32 e1000_host_interface_command(struct e1000_hw *hw, u8 *buffer, u32 length)
372 {
373  u32 hicr, i;
374 
375  DEBUGFUNC("e1000_host_interface_command");
376 
377  if (!(hw->mac.arc_subsystem_valid)) {
378  DEBUGOUT("Hardware doesn't support host interface command.\n");
379  return E1000_SUCCESS;
380  }
381 
382  if (!hw->mac.asf_firmware_present) {
383  DEBUGOUT("Firmware is not present.\n");
384  return E1000_SUCCESS;
385  }
386 
387  if (length == 0 || length & 0x3 ||
388  length > E1000_HI_MAX_BLOCK_BYTE_LENGTH) {
389  DEBUGOUT("Buffer length failure.\n");
390  return -E1000_ERR_HOST_INTERFACE_COMMAND;
391  }
392 
393  /* Check that the host interface is enabled. */
394  hicr = E1000_READ_REG(hw, E1000_HICR);
395  if (!(hicr & E1000_HICR_EN)) {
396  DEBUGOUT("E1000_HOST_EN bit disabled.\n");
397  return -E1000_ERR_HOST_INTERFACE_COMMAND;
398  }
399 
400  /* Calculate length in DWORDs */
401  length >>= 2;
402 
403  /* The device driver writes the relevant command block
404  * into the ram area.
405  */
406  for (i = 0; i < length; i++)
407  E1000_WRITE_REG_ARRAY_DWORD(hw, E1000_HOST_IF, i,
408  *((u32 *)buffer + i));
409 
410  /* Setting this bit tells the ARC that a new command is pending. */
411  E1000_WRITE_REG(hw, E1000_HICR, hicr | E1000_HICR_C);
412 
413  for (i = 0; i < E1000_HI_COMMAND_TIMEOUT; i++) {
414  hicr = E1000_READ_REG(hw, E1000_HICR);
415  if (!(hicr & E1000_HICR_C))
416  break;
417  msec_delay(1);
418  }
419 
420  /* Check command successful completion. */
421  if (i == E1000_HI_COMMAND_TIMEOUT ||
422  (!(E1000_READ_REG(hw, E1000_HICR) & E1000_HICR_SV))) {
423  DEBUGOUT("Command has failed with no status valid.\n");
424  return -E1000_ERR_HOST_INTERFACE_COMMAND;
425  }
426 
427  for (i = 0; i < length; i++)
428  *((u32 *)buffer + i) = E1000_READ_REG_ARRAY_DWORD(hw,
429  E1000_HOST_IF,
430  i);
431 
432  return E1000_SUCCESS;
433 }
444 s32 e1000_load_firmware(struct e1000_hw *hw, u8 *buffer, u32 length)
445 {
446  u32 hicr, hibba, fwsm, icr, i;
447 
448  DEBUGFUNC("e1000_load_firmware");
449 
450  if (hw->mac.type < e1000_i210) {
451  DEBUGOUT("Hardware doesn't support loading FW by the driver\n");
452  return -E1000_ERR_CONFIG;
453  }
454 
455  /* Check that the host interface is enabled. */
456  hicr = E1000_READ_REG(hw, E1000_HICR);
457  if (!(hicr & E1000_HICR_EN)) {
458  DEBUGOUT("E1000_HOST_EN bit disabled.\n");
459  return -E1000_ERR_CONFIG;
460  }
461  if (!(hicr & E1000_HICR_MEMORY_BASE_EN)) {
462  DEBUGOUT("E1000_HICR_MEMORY_BASE_EN bit disabled.\n");
463  return -E1000_ERR_CONFIG;
464  }
465 
466  if (length == 0 || length & 0x3 || length > E1000_HI_FW_MAX_LENGTH) {
467  DEBUGOUT("Buffer length failure.\n");
468  return -E1000_ERR_INVALID_ARGUMENT;
469  }
470 
471  /* Clear notification from ROM-FW by reading ICR register */
472  icr = E1000_READ_REG(hw, E1000_ICR_V2);
473 
474  /* Reset ROM-FW */
475  hicr = E1000_READ_REG(hw, E1000_HICR);
476  hicr |= E1000_HICR_FW_RESET_ENABLE;
477  E1000_WRITE_REG(hw, E1000_HICR, hicr);
478  hicr |= E1000_HICR_FW_RESET;
479  E1000_WRITE_REG(hw, E1000_HICR, hicr);
480  E1000_WRITE_FLUSH(hw);
481 
482  /* Wait till MAC notifies about its readiness after ROM-FW reset */
483  for (i = 0; i < (E1000_HI_COMMAND_TIMEOUT * 2); i++) {
484  icr = E1000_READ_REG(hw, E1000_ICR_V2);
485  if (icr & E1000_ICR_MNG)
486  break;
487  msec_delay(1);
488  }
489 
490  /* Check for timeout */
491  if (i == E1000_HI_COMMAND_TIMEOUT) {
492  DEBUGOUT("FW reset failed.\n");
493  return -E1000_ERR_HOST_INTERFACE_COMMAND;
494  }
495 
496  /* Wait till MAC is ready to accept new FW code */
497  for (i = 0; i < E1000_HI_COMMAND_TIMEOUT; i++) {
498  fwsm = E1000_READ_REG(hw, E1000_FWSM);
499  if ((fwsm & E1000_FWSM_FW_VALID) &&
500  ((fwsm & E1000_FWSM_MODE_MASK) >> E1000_FWSM_MODE_SHIFT ==
501  E1000_FWSM_HI_EN_ONLY_MODE))
502  break;
503  msec_delay(1);
504  }
505 
506  /* Check for timeout */
507  if (i == E1000_HI_COMMAND_TIMEOUT) {
508  DEBUGOUT("FW reset failed.\n");
509  return -E1000_ERR_HOST_INTERFACE_COMMAND;
510  }
511 
512  /* Calculate length in DWORDs */
513  length >>= 2;
514 
515  /* The device driver writes the relevant FW code block
516  * into the ram area in DWORDs via 1kB ram addressing window.
517  */
518  for (i = 0; i < length; i++) {
519  if (!(i % E1000_HI_FW_BLOCK_DWORD_LENGTH)) {
520  /* Point to correct 1kB ram window */
521  hibba = E1000_HI_FW_BASE_ADDRESS +
522  ((E1000_HI_FW_BLOCK_DWORD_LENGTH << 2) *
523  (i / E1000_HI_FW_BLOCK_DWORD_LENGTH));
524 
525  E1000_WRITE_REG(hw, E1000_HIBBA, hibba);
526  }
527 
528  E1000_WRITE_REG_ARRAY_DWORD(hw, E1000_HOST_IF,
529  i % E1000_HI_FW_BLOCK_DWORD_LENGTH,
530  *((u32 *)buffer + i));
531  }
532 
533  /* Setting this bit tells the ARC that a new FW is ready to execute. */
534  hicr = E1000_READ_REG(hw, E1000_HICR);
535  E1000_WRITE_REG(hw, E1000_HICR, hicr | E1000_HICR_C);
536 
537  for (i = 0; i < E1000_HI_COMMAND_TIMEOUT; i++) {
538  hicr = E1000_READ_REG(hw, E1000_HICR);
539  if (!(hicr & E1000_HICR_C))
540  break;
541  msec_delay(1);
542  }
543 
544  /* Check for successful FW start. */
545  if (i == E1000_HI_COMMAND_TIMEOUT) {
546  DEBUGOUT("New FW did not start within timeout period.\n");
547  return -E1000_ERR_HOST_INTERFACE_COMMAND;
548  }
549 
550  return E1000_SUCCESS;
551 }
