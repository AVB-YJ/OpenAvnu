54 template <typename T>
55 struct wrap : public T
56 {
57  wrap() { memset( this, 0, sizeof( *this ) ); }
58 
59  wrap( T const &other )
60  {
61  if ( &other != this )
62  {
63  memcpy( this, &other, sizeof( *this ) );
64  }
65  }
66 
67  void reset() { memset( this, 0, sizeof( *this ) ); }
68 
69  T *get() { return this; }
70 
71  T const *get() const { return this; }
72 
73  T const &operator=( const T &other )
74  {
75  if ( &other != this )
76  {
77  memcpy( this, &other, sizeof( *this ) );
78  }
79  return *this;
80  }
81 
82  bool operator==( T const &other ) { return memcmp( this, &other, sizeof( *this ) ) == 0; }
83 
84  bool operator!=( T const &other ) { return memcmp( this, &other, sizeof( *this ) ) != 0; }
85 
86  bool operator<( T const &other ) { return memcmp( this, &other, sizeof( *this ) ) < 0; }
87 
88  bool operator<=( T const &other ) { return memcmp( this, &other, sizeof( *this ) ) <= 0; }
89 
90  bool operator>=( T const &other ) { return memcmp( this, &other, sizeof( *this ) ) >= 0; }
91 
92  bool operator>( T const &other ) { return memcmp( this, &other, sizeof( *this ) ) > 0; }
