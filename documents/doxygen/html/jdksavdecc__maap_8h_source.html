304 static inline ssize_t jdksavdecc_maap_read( struct jdksavdecc_maap *p, void const *base, ssize_t pos, size_t len )
305 {
306  ssize_t r = jdksavdecc_validate_range( pos, len, JDKSAVDECC_MAAP_LEN );
307  if ( r >= 0 )
308  {
309  jdksavdecc_maap_common_control_header_read( &p->header, base, pos, len );
310  p->requested_start_address = jdksavdecc_maap_get_requested_start_address( base, pos );
311  p->requested_count = jdksavdecc_maap_get_requested_count( base, pos );
312  p->conflict_start_address = jdksavdecc_maap_get_conflict_start_address( base, pos );
313  p->conflict_count = jdksavdecc_maap_get_conflict_count( base, pos );
314  }
315  return r;
316 }
