71 void timespec_add_usec(struct timespec *t, unsigned long us)
72 {
73  t->tv_nsec += us * NANOSECONDS_PER_USEC;
74  t->tv_sec += t->tv_nsec / NANOSECONDS_PER_SECOND;
75  t->tv_nsec = t->tv_nsec % NANOSECONDS_PER_SECOND;
76 }
77 
78 void timespec_sub_usec(struct timespec *t, unsigned long us)
79 {
80  t->tv_nsec -= us * NANOSECONDS_PER_USEC;
81  t->tv_sec += t->tv_nsec / NANOSECONDS_PER_SECOND;
82  t->tv_nsec = t->tv_nsec % NANOSECONDS_PER_SECOND;
83  if (t->tv_nsec < 0) {
84  t->tv_sec--;
85  t->tv_nsec = NANOSECONDS_PER_SECOND + t->tv_nsec;
86  }
87 }

89 unsigned long timespec_usec_diff(struct timespec *t1, struct timespec *t2)
90 {
91  return (t1->tv_sec - t2->tv_sec) * MICROSECONDS_PER_SECOND
92  + (t1->tv_nsec - t2->tv_nsec) / NANOSECONDS_PER_USEC;
93 }

95 int timespec_cmp(struct timespec *a, struct timespec *b)
96 {
97  if (a->tv_sec > b->tv_sec)
98  return 1;
99  else if (a->tv_sec < b->tv_sec)
100  return -1;
101  else {
102  if (a->tv_nsec > b->tv_nsec)
103  return 1;
104  else if (a->tv_nsec < b->tv_nsec)
105  return -1;
106  }
107  return 0;
108 }

403 EXTERN_DLL_EXPORT bool openavbTLConfigure(tl_handle_t handle, openavb_tl_cfg_t *pCfgIn, openavb_tl_cfg_name_value_t *pNVCfg)
404 {
405  AVB_TRACE_ENTRY(AVB_TRACE_TL);
406 
407  tl_state_t *pTLState = (tl_state_t *)handle;
408 
409  if (!pTLState) {
410  AVB_LOG_ERROR("Invalid handle.");
411  AVB_TRACE_EXIT(AVB_TRACE_TL);
412  return FALSE;
413  }
414 
415  // Create the mediaQ
416  pTLState->pMediaQ = openavbMediaQCreate();
417  if (!pTLState->pMediaQ) {
418  AVB_LOG_ERROR("Unable to create media queue");
419  AVB_TRACE_EXIT(AVB_TRACE_TL);
420  return FALSE;
421  }
422 
423  // CORE_TODO: It's not safe to simply copy the openavb_tl_cfg_t since there are embedded pointers in the cfg_mac_t member.
424  // Those pointers need to be updated after a copy. Longer term the cfg_mac_t should be changed to not contain the mac
425  // member to remedy this issue and avoid further bugs.
426  memcpy(&pTLState->cfg, pCfgIn, sizeof(openavb_tl_cfg_t));
427  pTLState->cfg.dest_addr.mac = &pTLState->cfg.dest_addr.buffer;
428  pTLState->cfg.stream_addr.mac = &pTLState->cfg.stream_addr.buffer;
429 
430  openavb_tl_cfg_t *pCfg = &pTLState->cfg;
431 
432  if (!((pCfg->role == AVB_ROLE_TALKER) || (pCfg->role == AVB_ROLE_LISTENER))) {
433  AVB_LOG_ERROR("Talker - Listener Config Error: invalid role");
434  return FALSE;
435  }
436 
437  if ((pCfg->role == AVB_ROLE_TALKER) && (pCfg->max_interval_frames == 0)) {
438  AVB_LOG_ERROR("Talker - Listener Config Error: talker role requires 'max_interval_frames'");
439  return FALSE;
440  }
441 
442  openavbMediaQSetMaxStaleTail(pTLState->pMediaQ, pCfg->max_stale);
443 
444  if (!openavbTLOpenLinkLibsOsal(pTLState)) {
445  AVB_LOG_ERROR("Failed to open mapping / interface library");
446  return FALSE;
447  }
448 
449  if (pCfg->pMapInitFn && pCfg->pMapInitFn(pTLState->pMediaQ, &pCfg->map_cb, pCfg->max_transit_usec)) {
450  checkMapCallbacks(&pTLState->cfg);
451  }
452  else {
453  AVB_LOG_ERROR("Mapping initialize function error.");
454  return FALSE;
455  }
456 
457  if (pCfg->pIntfInitFn && pCfg->pIntfInitFn(pTLState->pMediaQ, &pCfg->intf_cb)) {
458  checkIntfCallbacks(&pTLState->cfg);
459  }
460  else {
461  AVB_LOG_ERROR("Interface initialize function error.");
462  return FALSE;
463  }
464 
465  // Submit configuration values to mapping and interface modules
466  int i;
467  for (i = 0; i < pNVCfg->nLibCfgItems; i++) {
468  if (MATCH_LEFT(pNVCfg->libCfgNames[i], "intf_nv_", 8)) {
469  if (pCfg->intf_cb.intf_cfg_cb) {
470  pCfg->intf_cb.intf_cfg_cb(pTLState->pMediaQ, pNVCfg->libCfgNames[i], pNVCfg->libCfgValues[i]);
471  }
472  else {
473  AVB_LOGF_ERROR("No interface module cfg function; ignoring %s", pNVCfg->libCfgNames[i]);
474  }
475  }
476  else if (MATCH_LEFT(pNVCfg->libCfgNames[i], "map_nv_", 7)) {
477  if (pCfg->map_cb.map_cfg_cb) {
478  pCfg->map_cb.map_cfg_cb(pTLState->pMediaQ, pNVCfg->libCfgNames[i], pNVCfg->libCfgValues[i]);
479  }
480  else {
481  AVB_LOGF_ERROR("No mapping module cfg function; ignoring %s", pNVCfg->libCfgNames[i]);
482  }
483  }
484  else {
485  assert(0);
486  }
487  } // for loop ends
488 
489  pTLState->cfg.map_cb.map_gen_init_cb(pTLState->pMediaQ);
490  pTLState->cfg.intf_cb.intf_gen_init_cb(pTLState->pMediaQ);
491 
492  // Initialize the AVDECC support for this Talker/Listener.
493  pTLState->bAvdeccMsgRunning = TRUE;
494  THREAD_CREATE_AVDECC_MSG();
495 
496  return TRUE;
497 }

499 EXTERN_DLL_EXPORT bool openavbTLRun(tl_handle_t handle)
500 {
501  bool retVal = FALSE;
502 
503  AVB_TRACE_ENTRY(AVB_TRACE_TL);
504 
505  tl_state_t *pTLState = (tl_state_t *)handle;
506 
507  do {
508  if (!pTLState) {
509  AVB_LOG_ERROR("Invalid handle.");
510  break;
511  }
512 
513  pTLState->bRunning = TRUE;
514  pTLState->bPaused = FALSE;
515  if (pTLState->cfg.role == AVB_ROLE_TALKER) {
516  THREAD_CREATE_TALKER();
517 
518  if (pTLState->cfg.thread_rt_priority != 0) { THREAD_SET_RT_PRIORITY(pTLState->TLThread, pTLState->cfg.thread_rt_priority); }
519  if (pTLState->cfg.thread_affinity != 0xFFFFFFFF) { THREAD_PIN(pTLState->TLThread, pTLState->cfg.thread_affinity); }
520  }
521  else if (pTLState->cfg.role == AVB_ROLE_LISTENER) {
522  THREAD_CREATE_LISTENER();
523 
524  if (pTLState->cfg.thread_rt_priority != 0) { THREAD_SET_RT_PRIORITY(pTLState->TLThread, pTLState->cfg.thread_rt_priority); }
525  if (pTLState->cfg.thread_affinity != 0xFFFFFFFF) { THREAD_PIN(pTLState->TLThread, pTLState->cfg.thread_affinity); }
526  }
527 
528  retVal = TRUE;
529 
530  } while (0);
531 
532 
533 
534  AVB_TRACE_EXIT(AVB_TRACE_TL);
535  return retVal;
536 }

564 EXTERN_DLL_EXPORT bool openavbTLClose(tl_handle_t handle)
565 {
566  AVB_TRACE_ENTRY(AVB_TRACE_TL);
567 
568  tl_state_t *pTLState = (tl_state_t *)handle;
