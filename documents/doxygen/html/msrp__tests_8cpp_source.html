279  * is registered in the MSRP database.
280  */
281 TEST(MsrpTestGroup, RegisterTalkerAdv)
282 {
283  struct msrp_attribute a_ref;
284  struct msrp_attribute *a_msrp = NULL;
285  int err_index = 0;
286  int parse_status = 0;
287  char id_substring[] = "S" PARSE_ASSIGN STREAM_ID;
288  char cmd_string[] = ST_PLUS_PLUS;
289 
290  CHECK(MSRP_db != NULL);
291 
292  /* here we fill in a_ref struct with target values */
293  eui64_write(a_ref.attribute.talk_listen.StreamID, 0xDEADBEEFBADFCA11ull);
294  a_ref.type = MSRP_TALKER_ADV_TYPE;
295 
296  /* use string interface to get MSRP to create TalkerAdv attrib in it's database */
297  msrp_recv_cmd(cmd_string, sizeof(cmd_string), &client);
298  CHECK(msrp_tests_cmd_ok(test_state.ctl_msg_data));
299 
300  /* lookup the created attrib */

302  CHECK(a_msrp != NULL);
303 
304  /* check for NEW event */
