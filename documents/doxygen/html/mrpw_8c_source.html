143 struct netif *netif_open(int timeout_ms)
144 {
145  int inum = 0;
146  int i = 0, total_interfaces = 0;
147  struct netif *netif;
148  PIP_ADAPTER_INFO AdapterInfo;
149  PIP_ADAPTER_INFO Current;
150  DWORD status, size;
151 
152  netif = (struct netif *)calloc(1, sizeof(struct netif));
153 
154  /* Retrieve the device list on the local machine */
155  if (pcap_findalldevs(&netif->alldevs, netif->errbuf) == -1) {
156  printf("Error finding interfaces\n");
157  goto error_return;
158  }
159 
160  /* Print the list */
161  for (netif->d = netif->alldevs; netif->d; netif->d = netif->d->next) {
162  /*printf("%d. %s", ++i, d->name); */
163  printf("%d", ++i);
164  if (netif->d->description)
165  printf(" (%-70s)\n", netif->d->description);
166  else
167  printf(" (No description available)\n");
168  }
169  total_interfaces = i;
170 
171  if (i == 0) {
172  printf
173  ("\nNo interfaces found! Make sure WinPcap is installed.\n");
174  goto error_return;
175  }
176 
177  printf("Enter the interface number (1-%d):", i);
178  scanf_s("%d", &inum);
179 
180  if (inum < 1 || inum > i) {
181  printf("\nInterface number out of range.\n");
182  /* Free the device list */
183  pcap_freealldevs(netif->alldevs);
184  goto error_return;
185  }
186 
187  /* Jump to the selected adapter */
188  for (netif->d = netif->alldevs, i = 0; i < inum - 1;
189  netif->d = netif->d->next, i++) ;
190 
191  /* Open the device */
192  if ((netif->pcap_interface = pcap_open_live(netif->d->name, // name of the device
193  65536, // portion of the packet to capture
194  // 65536 guarantees that the whole packet will be captured on all the link layers
195  1, // promiscuous mode
196  timeout_ms, // read timeout in ms
197  netif->errbuf // error buffer
198  )) == NULL) {
199  fprintf(stderr,
200  "\nUnable to open the adapter. %s is not supported by WinPcap\n",
201  netif->d->name);
202  /* Free the device list */
203  pcap_freealldevs(netif->alldevs);
204  goto error_return;
205  }
206 
207  /* lookup ip address */
208 
209  /* first, call to lookup the size */
210  status = GetAdaptersInfo(NULL, &size);
211  if (status != ERROR_BUFFER_OVERFLOW)
212  goto error_return;
213  AdapterInfo = (PIP_ADAPTER_INFO)malloc(size);
214 
215  status = GetAdaptersInfo(AdapterInfo, &size);
216  if (status != ERROR_SUCCESS) {
217  fprintf(stderr,
218  "\nError, GetAdaptersInfo() call in netif_win32_pcap.c failed\n");
219  free(AdapterInfo);
220  goto error_return;
221  }
222 
223  for (Current = AdapterInfo; Current != NULL; Current = Current->Next) {
224  if (strstr(netif->d->name, Current->AdapterName) != 0) {
225  memcpy(netif->mac, &Current->Address, sizeof(netif->mac));
226  }
227  }
228  free(AdapterInfo);
229  return netif;
230 
231  error_return:
232  free(netif);
233  return NULL;
234 }

337 unsigned int clock_monotonic_in_ms(void)
338 {
339  LARGE_INTEGER count;
340  QueryPerformanceCounter(&count);

342  QueryPerformanceFrequency(&clock_monotonic_freq);
343  clock_monotonic_in_ms_init = 1;
344  }
345  return (unsigned int)((double)count.QuadPart * 1000.0 /
346  (double)clock_monotonic_freq.QuadPart);

396 int mrpd_timer_timeout(HTIMER timerfd)
397 {
398  struct wtimer *t = (struct wtimer *)timerfd;
399  if (t->running && !t->elapsed) {
400  unsigned int elapsed_ms;
401  unsigned int current_time = clock_monotonic_in_ms();
402  elapsed_ms = current_time - t->start_time;
403  if (elapsed_ms > t->count) {
404  t->elapsed = TRUE;
405  }
406  }
407  return t->elapsed;
408 }

480 int process_ctl_msg(char *buf, int buflen, struct sockaddr_in *client)
481 {
482 
483  char respbuf[8];
484  /*
485  * Inbound/output commands from/to a client:
486  *
487  * When sent from a client, indicates an operation on the
488  * internal attribute databases. When sent by the daemon to
489  * a client, indicates an event such as a new attribute arrival,
490  * or a leaveall timer to force clients to re-advertise continued
491  * interest in an attribute.
492  *
493  * BYE Client detaches from daemon
494  *
495  * M+? - JOIN_MT a MAC address or service declaration
496  * M++ JOIN_IN a MAC Address (XXX: MMRP doesn't use 'New' though?)
497  * M-- - LV a MAC address or service declaration
498  * V+? - JOIN_MT a VID (VLAN ID)
499  * V++ - JOIN_IN a VID (VLAN ID)
500  * V-- - LV a VID (VLAN ID)
501  * S+? - JOIN_MT a Stream
502  * S++ - JOIN_IN a Stream
503  * S-- - LV a Stream
504  *
505  * Outbound messages
506  * ERC - error, unrecognized command
507  * ERP - error, unrecognized parameter
508  * ERI - error, internal
509  * OK+ - success
510  *
511  * Registrar Commands
512  *
513  * M?? - query MMRP Registrar MAC Address database
514  * V?? - query MVRP Registrar VID database
515  * S?? - query MSRP Registrar database
516  *
517  * Registrar Responses (to ?? commands)
518  *
519  * MIN - Registered
520  * MMT - Registered, Empty
521  * MLV - Registered, Leaving
522  * MNE - New attribute notification
523  * MJO - JOIN attribute notification
524  * MLV - LEAVE attribute notification
525  * VIN - Registered
526  * VMT - Registered, Empty
527  * VLV - Registered, Leaving
528  * SIN - Registered
529  * SMT - Registered, Empty
530  * SLV - Registered, Leaving
531  *
532  */
533 
534  memset(respbuf, 0, sizeof(respbuf));
535 
536 #if LOG_CLIENT_RECV
537  if (logging_enable)
538  printf("CMD:%s from CLNT %d\n", buf, client->sin_port);
539 #endif
540 
541  if (buflen < 3) {
542  printf("buflen = %d!\b", buflen);
543 
544  return -1;
545  }
546 
547  switch (buf[0]) {
548  case 'M':
549  return mmrp_recv_cmd(buf, buflen, client);
550  break;
551  case 'V':
552  return mvrp_recv_cmd(buf, buflen, client);
553  break;
554  case 'S':
555  return msrp_recv_cmd(buf, buflen, client);
556  break;
557  case 'B':
558  mmrp_bye(client);
559  mvrp_bye(client);
560  msrp_bye(client);
561  break;
562  default:
563  printf("unrecognized command %s\n", buf);
564  snprintf(respbuf, sizeof(respbuf) - 1, "ERC %s", buf);
565  mrpd_send_ctl_msg(client, respbuf, sizeof(respbuf));
566  return -1;
567  break;
568  }
569 
570  return 0;
571 }

626 int init_local_ctl(void)
627 {
628  struct sockaddr_in addr;
629  SOCKET sock_fd = INVALID_SOCKET;
630  int rc;
631  u_long iMode = 1;
632  DWORD iResult;
633  int timeout = 100; // ms
634 
635  sock_fd = socket(AF_INET, SOCK_DGRAM, 0);
636  if (sock_fd == INVALID_SOCKET)
637  goto out;
638 
639  iResult = setsockopt(sock_fd, SOL_SOCKET, SO_RCVTIMEO,
640  (void *)&timeout, sizeof(timeout));
641  if (iResult != NO_ERROR)
642  goto out;
643  iResult = setsockopt(sock_fd, SOL_SOCKET, SO_SNDTIMEO,
644  (void *)&timeout, sizeof(timeout));
645  if (iResult != NO_ERROR)
646  goto out;
647 
648  memset(&addr, 0, sizeof(addr));
649  addr.sin_family = AF_INET;
650  addr.sin_port = htons(mrpd_port);
651  addr.sin_addr.s_addr = inet_addr("127.0.0.1");
652 
653  rc = bind(sock_fd, (struct sockaddr *)&addr, sizeof(addr));
654 
655  if (rc == SOCKET_ERROR)
656  goto out;
657 
658  control_socket = sock_fd;
659 
660  return 0;
661  out:
662  if (sock_fd != INVALID_SOCKET)
663  closesocket(sock_fd);
664 
665  return -1;
666 }

777 int mrpw_run_once(void)
778 {

780  struct ctl_thread_params localhost_pkt;
781  uint16_t length = 0;
782  uint8_t *payload;
783  uint16_t protocol;

787  WaitForMultipleObjects(sizeof(pkt_events) / sizeof(HANDLE),
788  pkt_events,
789  FALSE,
790  100); /* 100ms wait */
791 
792  /* special exit case */
793  if (WAIT_OBJECT_0 + app_event_kill_all == dwEvent)
794  return -1;
795 
796  switch (dwEvent) {
797  case WAIT_TIMEOUT:
798  case WAIT_OBJECT_0 + loop_time_tick:
799  /* timeout - run protocols */
800  if (mmrp_enable) {
801  if (mrpd_timer_timeout(MMRP_db->mrp_db.lva_timer))
802  mmrp_event(MRP_EVENT_LVATIMER, NULL);
803  if (mrpd_timer_timeout(MMRP_db->mrp_db.lv_timer))
804  mmrp_event(MRP_EVENT_LVTIMER, NULL);
805  if (mrpd_timer_timeout(MMRP_db->mrp_db.join_timer))
806  mmrp_event(MRP_EVENT_TX, NULL);
807  }
808 
809  if (mvrp_enable) {
810  if (mrpd_timer_timeout(MVRP_db->mrp_db.lva_timer))
811  mvrp_event(MRP_EVENT_LVATIMER, NULL);
812  if (mrpd_timer_timeout(MVRP_db->mrp_db.lv_timer))
813  mvrp_event(MRP_EVENT_LVTIMER, NULL);
814  if (mrpd_timer_timeout(MVRP_db->mrp_db.join_timer))
815  mvrp_event(MRP_EVENT_TX, NULL);
816  }
817 
818  if (msrp_enable) {
819  if (mrpd_timer_timeout(MSRP_db->mrp_db.lva_timer))
820  msrp_event(MRP_EVENT_LVATIMER, NULL);
821  if (mrpd_timer_timeout(MSRP_db->mrp_db.lv_timer))
822  msrp_event(MRP_EVENT_LVTIMER, NULL);
823  if (mrpd_timer_timeout(MSRP_db->mrp_db.join_timer))
824  msrp_event(MRP_EVENT_TX, NULL);
825  }
826 
827  if (mrpd_timer_timeout(periodic_timer)) {
828  //printf("mrpd_timer_timeout(periodic_timer)\n");
829  if (mmrp_enable)
830  mmrp_event(MRP_EVENT_PERIODIC, NULL);
831  if (mvrp_enable)
832  mvrp_event(MRP_EVENT_PERIODIC, NULL);
833  if (msrp_enable)
834  msrp_event(MRP_EVENT_PERIODIC, NULL);
835  mrpd_timer_restart(periodic_timer);
836  }
837  if (mrpd_timer_timeout(gc_timer)) {
838  mrpd_reclaim();
839  }
840  mrpd_timer_restart(timer_check_tick);
841  break;
842 
843  case WAIT_OBJECT_0 + pkt_event_wpcap:
844  que_pop_nowait(que_wpcap, &wpcap_pkt);
845  proto = &wpcap_pkt.frame[12];
846  protocol = (uint16_t) proto[0] << 8 | (uint16_t) proto[1];
847  payload = proto + 2;
848 
849  last_pdu_buffer = wpcap_pkt.frame;
850  last_pdu_buffer_size = wpcap_pkt.length;
851 
852  switch (protocol) {
853  case MVRP_ETYPE:
854  if (mvrp_enable)
855  mvrp_recv_msg();
856  break;
857 
858  case MMRP_ETYPE:
859  if (mmrp_enable)
860  mmrp_recv_msg();
861  break;
862 
863  case MSRP_ETYPE:
864  if (msrp_enable)
865  msrp_recv_msg();
866  break;
867  }
868  if (mrpd_timer_timeout(&timer_check_tick)) {
869  if (!SetEvent(pkt_events[loop_time_tick])) {
870  printf("SetEvent loop_time_tick failed (%d)\n",
871  GetLastError());
872  exit(-1);
873  }
874  }
875  break;
876 
877  case WAIT_OBJECT_0 + pkt_event_wpcap_timeout:
878  //printf("pkt_event_wpcap_timeout\n");
879  break;
880 
881  case WAIT_OBJECT_0 + pkt_event_localhost:
882  que_pop_nowait(que_localhost, &localhost_pkt);
883  process_ctl_msg(localhost_pkt.msgbuf,
884  localhost_pkt.bytes, (struct sockaddr_in *)
885  &localhost_pkt.client_addr);
886  if (mrpd_timer_timeout(&timer_check_tick)) {
887  if (!SetEvent(pkt_events[loop_time_tick])) {
888  printf("SetEvent loop_time_tick failed (%d)\n",
889  GetLastError());
890  exit(-1);
891  }
892  }
893  break;
894 
895  case WAIT_OBJECT_0 + pkt_event_localhost_timeout:
896  //printf("pkt_event_localhost_timeout\n");
897  break;
898 
899  default:
900  printf("Unknown event %d\n", dwEvent);
901  }
902  return 0;
903 }
