70 int notification_imp::proc_notification_thread_callback()
71 {
72  DWORD dwEvent;
73 
74  while (true)
75  {
76  dwEvent = WaitForMultipleObjects(2, poll_events, FALSE, INFINITE);
77 
78  if (dwEvent == (WAIT_OBJECT_0 + NOTIFICATION_EVENT))
79  {
80  if ((write_index - read_index) > 0)
81  {
82  notification_callback(user_obj,
83  notification_buf[read_index % NOTIFICATION_BUF_COUNT].notification_type,
84  notification_buf[read_index % NOTIFICATION_BUF_COUNT].entity_id,
85  notification_buf[read_index % NOTIFICATION_BUF_COUNT].cmd_type,
86  notification_buf[read_index % NOTIFICATION_BUF_COUNT].desc_type,
87  notification_buf[read_index % NOTIFICATION_BUF_COUNT].desc_index,
88  notification_buf[read_index % NOTIFICATION_BUF_COUNT].cmd_status,
89  notification_buf[read_index % NOTIFICATION_BUF_COUNT].notification_id); // Call callback function
90  read_index++;
91  }
92  }
93  else
94  {
95  SetEvent(poll_events[KILL_EVENT]);
96  break;
97  }
98  }
99 
100  return 0;
101 }
