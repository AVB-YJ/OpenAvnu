71 maap_log_queue_t maapLogQueueNewQueue(uint32_t elemSize, uint32_t queueSize)
72 {
73  maap_log_queue_t retQueue;
74 
75  if (elemSize < 1 || queueSize < 1)
76  return NULL;
77 
78  retQueue = calloc(1, sizeof(struct maap_log_queue));
79  if (retQueue) {
80  retQueue->elemArray = calloc(queueSize, sizeof(struct maap_log_queue_elem));
81  if (retQueue->elemArray) {
82  uint32_t i1;
83  for (i1 = 0; i1 < queueSize; i1++) {
84  retQueue->elemArray[i1].data = calloc(1, elemSize);
85  if (!retQueue->elemArray[i1].data) {
86  maapLogQueueDeleteQueue(retQueue);
87  return NULL;
88  }
89  }
90  }
91  else {
92  maapLogQueueDeleteQueue(retQueue);
93  return NULL;
94  }
95 
96  retQueue->elemSize = elemSize;
97  retQueue->queueSize = queueSize;
98  retQueue->head = 0;
99  retQueue->tail = 0;
100  }
101 
102  return retQueue;
103 }
