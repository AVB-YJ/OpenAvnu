86 /***********************************************
87  * Signal handler - used to respond to signals.
88  * Allows graceful cleanup.
89  */
90 static void openavbTLSigHandler(int signal)
91 {
92  AVB_TRACE_ENTRY(AVB_TRACE_HOST);
93 
94  if (signal == SIGINT || signal == SIGTERM) {
95  if (bRunning) {
96  AVB_LOG_INFO("Host shutting down");
97  bRunning = FALSE;
98  }
99  else {
100  // Force shutdown
101  exit(2);
102  }
103  }
104  else if (signal == SIGUSR1) {
105  AVB_LOG_DEBUG("Waking up streaming thread");
106  }
107  else {
108  AVB_LOG_ERROR("Unexpected signal");
109  }
110 
111  AVB_TRACE_EXIT(AVB_TRACE_HOST);
112 }
