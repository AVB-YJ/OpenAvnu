567 void MemoryLeakWarningPlugin::postTestAction(UtestShell& test, TestResult& result)
568 {
569  memLeakDetector_->stopChecking();
570  int leaks = memLeakDetector_->totalMemoryLeaks(mem_leak_period_checking);
571 
572  if (!ignoreAllWarnings_ && expectedLeaks_ != leaks && failureCount_ == result.getFailureCount()) {
573  TestFailure f(&test, memLeakDetector_->report(mem_leak_period_checking));
574  result.addFailure(f);
575  }
576  memLeakDetector_->markCheckingPeriodLeaksAsNonCheckingPeriod();
577  ignoreAllWarnings_ = false;
578  expectedLeaks_ = 0;
