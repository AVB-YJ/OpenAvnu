279 int LinuxTimestamperGeneric::HWTimestamper_txtimestamp
280 ( PortIdentity *identity, PTPMessageId messageId, Timestamp &timestamp,
281  unsigned &clock_value, bool last )
282 {
283  int err;
284  int ret = GPTP_EC_EAGAIN;
285  struct msghdr msg;
286  struct cmsghdr *cmsg;
287  struct sockaddr_ll remote;
288  struct iovec sgentry;
289  PTPMessageId reflectedMessageId;
290  uint8_t reflected_bytes[ETHER_HDR_LEN + PTP_COMMON_HDR_LENGTH];
291  uint8_t *gptpCommonHeader;
292  uint16_t sequenceId;
293  struct {
294  struct cmsghdr cm;
295  char control[256];
296  } control;
297 
298  if( sd == -1 ) return -1;
299  memset( &msg, 0, sizeof( msg ));
300 
301  msg.msg_iov = &sgentry;
302  msg.msg_iovlen = 1;
303 
304  sgentry.iov_base = reflected_bytes;
305  sgentry.iov_len = sizeof(reflected_bytes);
306 
307  gptpCommonHeader = reflected_bytes + ETHER_HDR_LEN;
308 
309  memset( &remote, 0, sizeof(remote));
310  msg.msg_name = (caddr_t) &remote;
311  msg.msg_namelen = sizeof( remote );
312  msg.msg_control = &control;
313  msg.msg_controllen = sizeof(control);
314 
315  err = recvmsg( sd, &msg, MSG_ERRQUEUE );
316  if( err == -1 ) {
317  if( errno == EAGAIN ) {
318  ret = GPTP_EC_EAGAIN;
319  goto done;
320  }
321  else {
322  ret = GPTP_EC_FAILURE;
323  goto done;
324  }
325  }
326  sequenceId = PLAT_ntohs(*((uint16_t*)(PTP_COMMON_HDR_SEQUENCE_ID(gptpCommonHeader))));
327  reflectedMessageId.setSequenceId(sequenceId);
328  reflectedMessageId.setMessageType((MessageType)(*PTP_COMMON_HDR_TRANSSPEC_MSGTYPE(gptpCommonHeader) & 0xF));
329  if (messageId != reflectedMessageId) {
330  GPTP_LOG_WARNING("Timestamp discarded due to wrong message id");
331  ret = GPTP_EC_EAGAIN;
332  goto done;
333  }
334 
335  // Retrieve the timestamp
336  cmsg = CMSG_FIRSTHDR(&msg);
337  while( cmsg != NULL ) {
338  if( cmsg->cmsg_level == SOL_SOCKET &&
339  cmsg->cmsg_type == SO_TIMESTAMPING ) {
340  struct timespec *ts_device, *ts_system;
341  Timestamp device, system;
342  ts_system = ((struct timespec *) CMSG_DATA(cmsg)) + 1;
343  system = tsToTimestamp( ts_system );
344  ts_device = ts_system + 1; device = tsToTimestamp( ts_device );
345  system._version = version;
346  device._version = version;
347  timestamp = device;
348  ret = 0;
349  break;
350  }
351  cmsg = CMSG_NXTHDR(&msg,cmsg);
352  }
353 
354  if( ret != 0 ) {
355  GPTP_LOG_ERROR("Received a error message, but didn't find a valid timestamp");
356  }
357 
358  done:
359  if( ret == 0 || last ) {
360  net_lock->unlock();
361  }
362 
363  return ret;
364 }

414 /* Return *a - *b */
415 static inline ptp_clock_time pct_diff
416 ( struct ptp_clock_time *a, struct ptp_clock_time *b ) {
417  ptp_clock_time result;
418  if( a->nsec >= b->nsec ) {
419  result.nsec = a->nsec - b->nsec;
420  } else {
421  --a->sec;
422  result.nsec = (MAX_NSEC - b->nsec) + a->nsec;
423  }
424  result.sec = a->sec - b->sec;
425 
426  return result;
427 }

434 static inline Timestamp pctTimestamp( struct ptp_clock_time *t ) {
435  Timestamp result;
436 
437  result.seconds_ls = t->sec & 0xFFFFFFFF;
438  result.seconds_ms = t->sec >> sizeof(result.seconds_ls)*8;
439  result.nanoseconds = t->nsec;

441  return result;

445 bool LinuxTimestamperGeneric::HWTimestamper_gettime
446 ( Timestamp *system_time, Timestamp *device_time, uint32_t *local_clock,
447  uint32_t *nominal_clock_rate ) const
448 {
449  if( phc_fd == -1 )
450  return false;
451 
452 #ifdef PTP_HW_CROSSTSTAMP
453  if( precise_timestamp_enabled )
454  {
455  struct ptp_sys_offset_precise offset;
456  memset( &offset, 0, sizeof(offset));
457  if( ioctl( phc_fd, PTP_SYS_OFFSET_PRECISE, &offset ) == 0 )
458  {
459  *device_time = pctTimestamp( &offset.device );
460  *system_time = pctTimestamp( &offset.sys_realtime );
461 
462  return true;
463  }
464  }
465 #endif
466 
467  {
468  unsigned i;
469  struct ptp_clock_time *pct;
470  struct ptp_clock_time *system_time_l = NULL, *device_time_l = NULL;
471  int64_t interval = LLONG_MAX;
472  struct ptp_sys_offset offset;
473 
474  memset( &offset, 0, sizeof(offset));
475  offset.n_samples = PTP_MAX_SAMPLES;
476  if( ioctl( phc_fd, PTP_SYS_OFFSET, &offset ) == -1 )
477  return false;
478 
479  pct = &offset.ts[0];
480  for( i = 0; i < offset.n_samples; ++i ) {
481  int64_t interval_t;
482  interval_t = pctns(pct_diff( pct+2*i+2, pct+2*i ));
483  if( interval_t < interval ) {
484  system_time_l = pct+2*i;
485  device_time_l = pct+2*i+1;
486  interval = interval_t;
487  }
488  }
489 
490  if (device_time_l)
491  *device_time = pctTimestamp( device_time_l );
492  if (system_time_l)
493  *system_time = pctTimestamp( system_time_l );
494  }
495 
496  return true;
497 }
