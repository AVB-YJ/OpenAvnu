130 void JUnitTestOutput::printCurrentTestStarted(const UtestShell& test)
131 {

135

193 void JUnitTestOutput::writeTestCases()
194 {
195  JUnitTestCaseResultNode* cur = impl_->results_.head_;
196 
197  while (cur) {
198  SimpleString buf = StringFromFormat(
199  "<testcase classname=\"%s%s%s\" name=\"%s\" assertions=\"%d\" time=\"%d.%03d\" file=\"%s\" line=\"%d\">\n",
200  impl_->package_.asCharString(),
201  impl_->package_.isEmpty() == true ? "" : ".",
202  impl_->results_.group_.asCharString(),
203  cur->name_.asCharString(),
204  cur->checkCount_ - impl_->results_.totalCheckCount_,
205  (int) (cur->execTime_ / 1000), (int)(cur->execTime_ % 1000),
206  cur->file_.asCharString(),
207  cur->lineNumber_);
208  writeToFile(buf.asCharString());
209 
210  impl_->results_.totalCheckCount_ = cur->checkCount_;
211 
212  if (cur->failure_) {
213  writeFailure(cur);
214  }
215  else if (cur->ignored_) {
216  writeToFile("<skipped />\n");
217  }
218  writeToFile("</testcase>\n");
219  cur = cur->next_;
220  }
221 }

223 void JUnitTestOutput::writeFailure(JUnitTestCaseResultNode* node)
224 {
225  SimpleString message = node->failure_->getMessage().asCharString();
226  message.replace('"', '\'');
227  message.replace('<', '[');
228  message.replace('>', ']');
229  message.replace("&", "&amp;");
230  message.replace("\n", "{newline}");
231  SimpleString buf = StringFromFormat(
232  "<failure message=\"%s:%d: %s\" type=\"AssertionFailedError\">\n",
233  node->failure_->getFileName().asCharString(),
234  node->failure_->getFailureLineNumber(), message.asCharString());
235  writeToFile(buf.asCharString());
236  writeToFile("</failure>\n");
237 }
