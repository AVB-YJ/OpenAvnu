78 s32 e1000_acquire_swfw_sync_i210(struct e1000_hw *hw, u16 mask)
79 {
80  u32 swfw_sync;
81  u32 swmask = mask;
82  u32 fwmask = mask << 16;
83  s32 ret_val = E1000_SUCCESS;
84  s32 i = 0, timeout = 200; /* FIXME: find real value to use here */
85 
86  DEBUGFUNC("e1000_acquire_swfw_sync_i210");
87 
88  while (i < timeout) {
89  if (e1000_get_hw_semaphore_i210(hw)) {
90  ret_val = -E1000_ERR_SWFW_SYNC;
91  goto out;
92  }
93 
94  swfw_sync = E1000_READ_REG(hw, E1000_SW_FW_SYNC);
95  if (!(swfw_sync & (fwmask | swmask)))
96  break;
97 
98  /*
99  * Firmware currently using resource (fwmask)
100  * or other software thread using resource (swmask)
101  */
102  e1000_put_hw_semaphore_generic(hw);
103  msec_delay_irq(5);
104  i++;
105  }
106 
107  if (i == timeout) {
108  DEBUGOUT("Driver can't access resource, SW_FW_SYNC timeout.\n");
109  ret_val = -E1000_ERR_SWFW_SYNC;
110  goto out;
111  }
112 
113  swfw_sync |= swmask;
114  E1000_WRITE_REG(hw, E1000_SW_FW_SYNC, swfw_sync);
115 
116  e1000_put_hw_semaphore_generic(hw);
117 
118 out:
119  return ret_val;
120 }

131 {

133 
134  DEBUGFUNC("e1000_release_swfw_sync_i210");
135

152 static s32 e1000_get_hw_semaphore_i210(struct e1000_hw *hw)
153 {
154  u32 swsm;
155  s32 timeout = hw->nvm.word_size + 1;
156  s32 i = 0;
157 
158  DEBUGFUNC("e1000_get_hw_semaphore_i210");
159 
160  /* Get the SW semaphore */
161  while (i < timeout) {
162  swsm = E1000_READ_REG(hw, E1000_SWSM);
163  if (!(swsm & E1000_SWSM_SMBI))
164  break;
165 
166  usec_delay(50);
167  i++;
168  }
169 
170  if (i == timeout) {
171  /* In rare circumstances, the SW semaphore may already be held
172  * unintentionally. Clear the semaphore once before giving up.
173  */
174  if (hw->dev_spec._82575.clear_semaphore_once) {
175  hw->dev_spec._82575.clear_semaphore_once = false;
176  e1000_put_hw_semaphore_generic(hw);
177  for (i = 0; i < timeout; i++) {
178  swsm = E1000_READ_REG(hw, E1000_SWSM);
179  if (!(swsm & E1000_SWSM_SMBI))
180  break;
181 
182  usec_delay(50);
183  }
184  }
185 
186  /* If we do not have the semaphore here, we have to give up. */
187  if (i == timeout) {
188  DEBUGOUT("Driver can't access device - SMBI bit is set.\n");
189  return -E1000_ERR_NVM;
190  }
191  }
192 
193  /* Get the FW semaphore. */
194  for (i = 0; i < timeout; i++) {
195  swsm = E1000_READ_REG(hw, E1000_SWSM);
196  E1000_WRITE_REG(hw, E1000_SWSM, swsm | E1000_SWSM_SWESMBI);
197 
198  /* Semaphore acquired if bit latched */
199  if (E1000_READ_REG(hw, E1000_SWSM) & E1000_SWSM_SWESMBI)
200  break;
201 
202  usec_delay(50);
203  }
204 
205  if (i == timeout) {
206  /* Release semaphores */
207  e1000_put_hw_semaphore_generic(hw);
208  DEBUGOUT("Driver can't access the NVM\n");
209  return -E1000_ERR_NVM;
210  }
211 
212  return E1000_SUCCESS;
213 }

225 s32 e1000_read_nvm_srrd_i210(struct e1000_hw *hw, u16 offset, u16 words,
226  u16 *data)
227 {
228  s32 status = E1000_SUCCESS;
229  u16 i, count;
230 
231  DEBUGFUNC("e1000_read_nvm_srrd_i210");
232 
233  /* We cannot hold synchronization semaphores for too long,
234  * because of forceful takeover procedure. However it is more efficient
235  * to read in bursts than synchronizing access for each word. */
236  for (i = 0; i < words; i += E1000_EERD_EEWR_MAX_COUNT) {
237  count = (words - i) / E1000_EERD_EEWR_MAX_COUNT > 0 ?
238  E1000_EERD_EEWR_MAX_COUNT : (words - i);
239  if (hw->nvm.ops.acquire(hw) == E1000_SUCCESS) {
240  status = e1000_read_nvm_eerd(hw, offset, count,
241  data + i);
242  hw->nvm.ops.release(hw);
243  } else {
244  status = E1000_ERR_SWFW_SYNC;
245  }
246 
247  if (status != E1000_SUCCESS)
248  break;
249  }
250 
251  return status;
252 }

311 static s32 e1000_write_nvm_srwr(struct e1000_hw *hw, u16 offset, u16 words,
312  u16 *data)
313 {
314  struct e1000_nvm_info *nvm = &hw->nvm;
315  u32 i, k, eewr = 0;
316  u32 attempts = 100000;
317  s32 ret_val = E1000_SUCCESS;
318 
319  DEBUGFUNC("e1000_write_nvm_srwr");
320 
321  /*
322  * A check for invalid values: offset too large, too many words,
323  * too many words for the offset, and not enough words.
324  */
325  if ((offset >= nvm->word_size) || (words > (nvm->word_size - offset)) ||
326  (words == 0)) {
327  DEBUGOUT("nvm parameter(s) out of bounds\n");
328  ret_val = -E1000_ERR_NVM;
329  goto out;
330  }
331 
332  for (i = 0; i < words; i++) {
333  eewr = ((offset+i) << E1000_NVM_RW_ADDR_SHIFT) |
334  (data[i] << E1000_NVM_RW_REG_DATA) |
335  E1000_NVM_RW_REG_START;
336 
337  E1000_WRITE_REG(hw, E1000_SRWR, eewr);
338 
339  for (k = 0; k < attempts; k++) {
340  if (E1000_NVM_RW_REG_DONE &
341  E1000_READ_REG(hw, E1000_SRWR)) {
342  ret_val = E1000_SUCCESS;
343  break;
344  }
345  usec_delay(5);
346  }
347 
348  if (ret_val != E1000_SUCCESS) {
349  DEBUGOUT("Shadow RAM write EEWR timed out\n");
350  break;
351  }
352  }
353 
354 out:
355  return ret_val;
356 }

366 static s32 e1000_read_invm_word_i210(struct e1000_hw *hw, u8 address, u16 *data)
367 {
368  s32 status = -E1000_ERR_INVM_VALUE_NOT_FOUND;
369  u32 invm_dword;
370  u16 i;
371  u8 record_type, word_address;
372 
373  DEBUGFUNC("e1000_read_invm_word_i210");
374 
375  for (i = 0; i < E1000_INVM_SIZE; i++) {
376  invm_dword = E1000_READ_REG(hw, E1000_INVM_DATA_REG(i));
377  /* Get record type */
378  record_type = INVM_DWORD_TO_RECORD_TYPE(invm_dword);
379  if (record_type == E1000_INVM_UNINITIALIZED_STRUCTURE)
380  break;
381  if (record_type == E1000_INVM_CSR_AUTOLOAD_STRUCTURE)
382  i += E1000_INVM_CSR_AUTOLOAD_DATA_SIZE_IN_DWORDS;
383  if (record_type == E1000_INVM_RSA_KEY_SHA256_STRUCTURE)
384  i += E1000_INVM_RSA_KEY_SHA256_DATA_SIZE_IN_DWORDS;
385  if (record_type == E1000_INVM_WORD_AUTOLOAD_STRUCTURE) {
386  word_address = INVM_DWORD_TO_WORD_ADDRESS(invm_dword);
387  if (word_address == address) {
388  *data = INVM_DWORD_TO_WORD_DATA(invm_dword);
389  DEBUGOUT2("Read INVM Word 0x%02x = %x",
390  address, *data);
391  status = E1000_SUCCESS;
392  break;
393  }
394  }
395  }
396  if (status != E1000_SUCCESS)
397  DEBUGOUT1("Requested word 0x%02x not found in OTP\n", address);
398  return status;
399 }

408 static s32 e1000_read_invm_i210(struct e1000_hw *hw, u16 offset,
409  u16 E1000_UNUSEDARG words, u16 *data)
410 {
411  s32 ret_val = E1000_SUCCESS;
412 
413  DEBUGFUNC("e1000_read_invm_i210");
414 
415  /* Only the MAC addr is required to be present in the iNVM */
416  switch (offset) {
417  case NVM_MAC_ADDR:
418  ret_val = e1000_read_invm_word_i210(hw, (u8)offset, &data[0]);
419  ret_val |= e1000_read_invm_word_i210(hw, (u8)offset+1,
420  &data[1]);
421  ret_val |= e1000_read_invm_word_i210(hw, (u8)offset+2,
422  &data[2]);
423  if (ret_val != E1000_SUCCESS)
424  DEBUGOUT("MAC Addr not found in iNVM\n");
425  break;
426  case NVM_INIT_CTRL_2:
427  ret_val = e1000_read_invm_word_i210(hw, (u8)offset, data);
428  if (ret_val != E1000_SUCCESS) {
429  *data = NVM_INIT_CTRL_2_DEFAULT_I211;
430  ret_val = E1000_SUCCESS;
431  }
432  break;
433  case NVM_INIT_CTRL_4:
434  ret_val = e1000_read_invm_word_i210(hw, (u8)offset, data);
435  if (ret_val != E1000_SUCCESS) {
436  *data = NVM_INIT_CTRL_4_DEFAULT_I211;
437  ret_val = E1000_SUCCESS;
438  }
439  break;
440  case NVM_LED_1_CFG:
441  ret_val = e1000_read_invm_word_i210(hw, (u8)offset, data);
442  if (ret_val != E1000_SUCCESS) {
443  *data = NVM_LED_1_CFG_DEFAULT_I211;
444  ret_val = E1000_SUCCESS;
445  }
446  break;
447  case NVM_LED_0_2_CFG:
448  ret_val = e1000_read_invm_word_i210(hw, (u8)offset, data);
449  if (ret_val != E1000_SUCCESS) {
450  *data = NVM_LED_0_2_CFG_DEFAULT_I211;
451  ret_val = E1000_SUCCESS;
452  }
453  break;
454  case NVM_ID_LED_SETTINGS:
455  ret_val = e1000_read_invm_word_i210(hw, (u8)offset, data);
456  if (ret_val != E1000_SUCCESS) {
457  *data = ID_LED_RESERVED_FFFF;
458  ret_val = E1000_SUCCESS;
459  }
460  break;
461  case NVM_SUB_DEV_ID:
462  *data = hw->subsystem_device_id;
463  break;
464  case NVM_SUB_VEN_ID:
465  *data = hw->subsystem_vendor_id;
466  break;
467  case NVM_DEV_ID:
468  *data = hw->device_id;
469  break;
470  case NVM_VEN_ID:
471  *data = hw->vendor_id;
472  break;
473  default:
474  DEBUGOUT1("NVM word 0x%02x is not mapped.\n", offset);
475  *data = NVM_RESERVED_WORD;
476  break;
477  }
478  return ret_val;
479 }

488 s32 e1000_read_invm_version(struct e1000_hw *hw,
489  struct e1000_fw_version *invm_ver)
490 {
491  u32 *record = NULL;
492  u32 *next_record = NULL;
493  u32 i = 0;
494  u32 invm_dword = 0;
495  u32 invm_blocks = E1000_INVM_SIZE - (E1000_INVM_ULT_BYTES_SIZE /
496  E1000_INVM_RECORD_SIZE_IN_BYTES);
497  u32 buffer[E1000_INVM_SIZE];
498  s32 status = -E1000_ERR_INVM_VALUE_NOT_FOUND;
499  u16 version = 0;
500 
501  DEBUGFUNC("e1000_read_invm_version");
502 
503  /* Read iNVM memory */
504  for (i = 0; i < E1000_INVM_SIZE; i++) {
505  invm_dword = E1000_READ_REG(hw, E1000_INVM_DATA_REG(i));
506  buffer[i] = invm_dword;
507  }
508 
509  /* Read version number */
510  for (i = 1; i < invm_blocks; i++) {
511  record = &buffer[invm_blocks - i];
512  next_record = &buffer[invm_blocks - i + 1];
513 
514  /* Check if we have first version location used */
515  if ((i == 1) && ((*record & E1000_INVM_VER_FIELD_ONE) == 0)) {
516  version = 0;
517  status = E1000_SUCCESS;
518  break;
519  }
520  /* Check if we have second version location used */
521  else if ((i == 1) &&
522  ((*record & E1000_INVM_VER_FIELD_TWO) == 0)) {
523  version = (*record & E1000_INVM_VER_FIELD_ONE) >> 3;
524  status = E1000_SUCCESS;
525  break;
526  }
527  /*
528  * Check if we have odd version location
529  * used and it is the last one used
530  */
531  else if ((((*record & E1000_INVM_VER_FIELD_ONE) == 0) &&
532  ((*record & 0x3) == 0)) || (((*record & 0x3) != 0) &&
533  (i != 1))) {
534  version = (*next_record & E1000_INVM_VER_FIELD_TWO)
535  >> 13;
536  status = E1000_SUCCESS;
537  break;
538  }
539  /*
540  * Check if we have even version location
541  * used and it is the last one used
542  */
543  else if (((*record & E1000_INVM_VER_FIELD_TWO) == 0) &&
544  ((*record & 0x3) == 0)) {
545  version = (*record & E1000_INVM_VER_FIELD_ONE) >> 3;
546  status = E1000_SUCCESS;
547  break;
548  }
549  }
550 
551  if (status == E1000_SUCCESS) {
552  invm_ver->invm_major = (version & E1000_INVM_MAJOR_MASK)
553  >> E1000_INVM_MAJOR_SHIFT;
554  invm_ver->invm_minor = version & E1000_INVM_MINOR_MASK;
555  }
556  /* Read Image Type */
557  for (i = 1; i < invm_blocks; i++) {
558  record = &buffer[invm_blocks - i];
559  next_record = &buffer[invm_blocks - i + 1];
560 
561  /* Check if we have image type in first location used */
562  if ((i == 1) && ((*record & E1000_INVM_IMGTYPE_FIELD) == 0)) {
563  invm_ver->invm_img_type = 0;
564  status = E1000_SUCCESS;
565  break;
566  }
567  /* Check if we have image type in first location used */
568  else if ((((*record & 0x3) == 0) &&
569  ((*record & E1000_INVM_IMGTYPE_FIELD) == 0)) ||
570  ((((*record & 0x3) != 0) && (i != 1)))) {
571  invm_ver->invm_img_type =
572  (*next_record & E1000_INVM_IMGTYPE_FIELD) >> 23;
573  status = E1000_SUCCESS;
574  break;
575  }
576  }
577  return status;
578 }

587 s32 e1000_validate_nvm_checksum_i210(struct e1000_hw *hw)
588 {
589  s32 status = E1000_SUCCESS;
590  s32 (*read_op_ptr)(struct e1000_hw *, u16, u16, u16 *);
591 
592  DEBUGFUNC("e1000_validate_nvm_checksum_i210");
593 
594  if (hw->nvm.ops.acquire(hw) == E1000_SUCCESS) {
595 
596  /*
597  * Replace the read function with semaphore grabbing with
598  * the one that skips this for a while.
599  * We have semaphore taken already here.
600  */
601  read_op_ptr = hw->nvm.ops.read;
602  hw->nvm.ops.read = e1000_read_nvm_eerd;
603 
604  status = e1000_validate_nvm_checksum_generic(hw);
605 
606  /* Revert original read operation. */
607  hw->nvm.ops.read = read_op_ptr;
608 
609  hw->nvm.ops.release(hw);
610  } else {
611  status = E1000_ERR_SWFW_SYNC;
612  }
613 
614  return status;
615 }

626 s32 e1000_update_nvm_checksum_i210(struct e1000_hw *hw)
627 {
628  s32 ret_val;
629  u16 checksum = 0;
630  u16 i, nvm_data;
631 
632  DEBUGFUNC("e1000_update_nvm_checksum_i210");
633 
634  /*
635  * Read the first word from the EEPROM. If this times out or fails, do
636  * not continue or we could be in for a very long wait while every
637  * EEPROM read fails
638  */
639  ret_val = e1000_read_nvm_eerd(hw, 0, 1, &nvm_data);
640  if (ret_val != E1000_SUCCESS) {
641  DEBUGOUT("EEPROM read failed\n");
642  goto out;
643  }
644 
645  if (hw->nvm.ops.acquire(hw) == E1000_SUCCESS) {
646  /*
647  * Do not use hw->nvm.ops.write, hw->nvm.ops.read
648  * because we do not want to take the synchronization
649  * semaphores twice here.
650  */
651 
652  for (i = 0; i < NVM_CHECKSUM_REG; i++) {
653  ret_val = e1000_read_nvm_eerd(hw, i, 1, &nvm_data);
654  if (ret_val) {
655  hw->nvm.ops.release(hw);
656  DEBUGOUT("NVM Read Error while updating checksum.\n");
657  goto out;
658  }
659  checksum += nvm_data;
660  }
661  checksum = (u16) NVM_SUM - checksum;
662  ret_val = e1000_write_nvm_srwr(hw, NVM_CHECKSUM_REG, 1,
663  &checksum);
664  if (ret_val != E1000_SUCCESS) {
665  hw->nvm.ops.release(hw);
666  DEBUGOUT("NVM Write Error while updating checksum.\n");
667  goto out;
668  }
669 
670  hw->nvm.ops.release(hw);
671 
672  ret_val = e1000_update_flash_i210(hw);
673  } else {
674  ret_val = E1000_ERR_SWFW_SYNC;
675  }
676 out:
677  return ret_val;
678 }

736 s32 e1000_pool_flash_update_done_i210(struct e1000_hw *hw)
737 {
738  s32 ret_val = -E1000_ERR_NVM;
739  u32 i, reg;
740 
741  DEBUGFUNC("e1000_pool_flash_update_done_i210");
742 
743  for (i = 0; i < E1000_FLUDONE_ATTEMPTS; i++) {
744  reg = E1000_READ_REG(hw, E1000_EECD);
745  if (reg & E1000_EECD_FLUDONE_I210) {
746  ret_val = E1000_SUCCESS;

915 static s32 e1000_pll_workaround_i210(struct e1000_hw *hw)
916 {
917  s32 ret_val;
918  u32 wuc, mdicnfg, ctrl, ctrl_ext, reg_val;
919  u16 nvm_word, phy_word, pci_word, tmp_nvm;
920  int i;
921 
922  /* Get and set needed register values */
923  wuc = E1000_READ_REG(hw, E1000_WUC);
924  mdicnfg = E1000_READ_REG(hw, E1000_MDICNFG);
925  reg_val = mdicnfg & ~E1000_MDICNFG_EXT_MDIO;
926  E1000_WRITE_REG(hw, E1000_MDICNFG, reg_val);
927 
928  /* Get data from NVM, or set default */
929  ret_val = e1000_read_invm_word_i210(hw, E1000_INVM_AUTOLOAD,
930  &nvm_word);
931  if (ret_val != E1000_SUCCESS)
932  nvm_word = E1000_INVM_DEFAULT_AL;
933  tmp_nvm = nvm_word | E1000_INVM_PLL_WO_VAL;
934  for (i = 0; i < E1000_MAX_PLL_TRIES; i++) {
935  /* check current state directly from internal PHY */
936  e1000_read_phy_reg_gs40g(hw, (E1000_PHY_PLL_FREQ_PAGE |
937  E1000_PHY_PLL_FREQ_REG), &phy_word);
938  if ((phy_word & E1000_PHY_PLL_UNCONF)
939  != E1000_PHY_PLL_UNCONF) {
940  ret_val = E1000_SUCCESS;
941  break;
942  } else {
943  ret_val = -E1000_ERR_PHY;
944  }
945  /* directly reset the internal PHY */
946  ctrl = E1000_READ_REG(hw, E1000_CTRL);
947  E1000_WRITE_REG(hw, E1000_CTRL, ctrl|E1000_CTRL_PHY_RST);
948 
949  ctrl_ext = E1000_READ_REG(hw, E1000_CTRL_EXT);
950  ctrl_ext |= (E1000_CTRL_EXT_PHYPDEN | E1000_CTRL_EXT_SDLPE);
951  E1000_WRITE_REG(hw, E1000_CTRL_EXT, ctrl_ext);
952 
953  E1000_WRITE_REG(hw, E1000_WUC, 0);
954  reg_val = (E1000_INVM_AUTOLOAD << 4) | (tmp_nvm << 16);
955  E1000_WRITE_REG(hw, E1000_EEARBC_I210, reg_val);
956 
957  e1000_read_pci_cfg(hw, E1000_PCI_PMCSR, &pci_word);
958  pci_word |= E1000_PCI_PMCSR_D3;
959  e1000_write_pci_cfg(hw, E1000_PCI_PMCSR, &pci_word);
960  msec_delay(1);
961  pci_word &= ~E1000_PCI_PMCSR_D3;
962  e1000_write_pci_cfg(hw, E1000_PCI_PMCSR, &pci_word);
963  reg_val = (E1000_INVM_AUTOLOAD << 4) | (nvm_word << 16);
964  E1000_WRITE_REG(hw, E1000_EEARBC_I210, reg_val);
965 
966  /* restore WUC register */
967  E1000_WRITE_REG(hw, E1000_WUC, wuc);
968  }
969  /* restore MDICNFG setting */
970  E1000_WRITE_REG(hw, E1000_MDICNFG, mdicnfg);
971  return ret_val;
972 }
