63 class WindowsNPipeMessage {
64  protected:
65  DWORD sz;
66  NPIPE_MSG_TYPE type;
68  OVERLAPPED ol_read;
69  DWORD ol_read_req;
70  public:
75  void init() {
76  sz = sizeof( WindowsNPipeMessage );
77  }
83  bool write( HANDLE pipe ) {
84  DWORD bytes_written;
85  DWORD last_error = ERROR_SUCCESS;
86  if( sz == 0 ) return false;
87  if( WriteFile( pipe, this, sz, &bytes_written, NULL ) == 0 ) {
88  last_error = GetLastError();
89  }
90  if( last_error == ERROR_SUCCESS || last_error == ERROR_PIPE_LISTENING ) {
91  return true;
92  }
93  GPTP_LOG_ERROR( "Failed to write to named pipe: %u", last_error );
94  return false;
95  }
103  long read_ol( HANDLE pipe, long offs, HANDLE event ) {
104  DWORD bytes_read;
105  long sz_l = (long) sz;
106  LPOVERLAPPED lol;
107  if( sz_l - offs < 0 || sz_l == 0 ) return -1;
108  if( sz_l - offs == 0 ) return offs;
109  ol_read_req = sz_l-offs;
110  if( event != NULL ) {
111  memset( &ol_read, 0, sizeof( ol_read ));
112  ol_read.hEvent = event;
113  lol = &ol_read;
114  } else {
115  lol = NULL;
116  }
117  if( ReadFile( pipe, ((char *)this)+offs, ol_read_req, &bytes_read, lol ) == 0 ) {
118  int err = GetLastError();
119  if( err != ERROR_IO_PENDING ) {
120  GPTP_LOG_ERROR( "Failed to read %d bytes from named pipe: %u", ol_read_req, err );
121  }
122  return err == ERROR_IO_PENDING ? 0 : -1;
123  }
124  return (bytes_read == sz_l-offs) ? offs+bytes_read : -1;
125  }
132  long read_ol_complete( HANDLE pipe ) {
133  DWORD bytes_read;
134  if( GetOverlappedResult( pipe, &ol_read, &bytes_read, false ) == 0 ) {
135  return -1;
136  }
137  return bytes_read;
138  }
145  long read( HANDLE pipe, long offs ) {
146  return read_ol( pipe, offs, NULL );
147  }
152  NPIPE_MSG_TYPE getType() { return type; }
153 };
