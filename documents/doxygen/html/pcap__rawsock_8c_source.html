42 static pcap_t* open_pcap_dev(const char* ifname, int frameSize, char* errbuf)
43 {
44  pcap_t* handle = pcap_create(ifname, errbuf);
45  if (handle) {
46  int err;
47  err = pcap_set_snaplen(handle, frameSize);
48  if (err) AVB_LOGF_WARNING("Cannot set snap len %d", err);
49 
50  err = pcap_set_promisc(handle, 1);
51  if (err) AVB_LOGF_WARNING("Cannot set promisc %d", err);
52 
53  err = pcap_set_immediate_mode(handle, 1);
54  if (err) AVB_LOGF_WARNING("Cannot set immediate mode %d", err);
55 
56  // we need timeout (here 100ms) otherwise we could block for ever
57  err = pcap_set_timeout(handle, 100);
58  if (err) AVB_LOGF_WARNING("Cannot set timeout %d", err);
59 
60  err = pcap_set_tstamp_precision(handle, PCAP_TSTAMP_PRECISION_NANO);
61  if (err) AVB_LOGF_WARNING("Cannot set tstamp nano precision %d", err);
62 
63  err = pcap_set_tstamp_type(handle, PCAP_TSTAMP_ADAPTER_UNSYNCED);
64  if (err) AVB_LOGF_WARNING("Cannot set tstamp adapter unsynced %d", err);
65 
66  err = pcap_activate(handle);
67  if (err) AVB_LOGF_WARNING("Cannot activate pcap %d", err);
68  }
69  return handle;
70 }

72 // Open a rawsock for TX or RX
73 void *pcapRawsockOpen(pcap_rawsock_t* rawsock, const char *ifname, bool rx_mode, bool tx_mode, U16 ethertype, U32 frame_size, U32 num_frames)
74 {
75  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK);
76 
77  AVB_LOGF_DEBUG("Open, rx=%d, tx=%d, ethertype=%x size=%d, num=%d", rx_mode, tx_mode, ethertype, frame_size, num_frames);
78 
79  baseRawsockOpen(&rawsock->base, ifname, rx_mode, tx_mode, ethertype, frame_size, num_frames);
80 
81  if (tx_mode) {
82  AVB_LOG_DEBUG("pcap rawsock transmit mode will bypass FQTSS");
83  }
84 
85  rawsock->handle = 0;
86 
87  // Get info about the network device
88  if (!simpleAvbCheckInterface(ifname, &(rawsock->base.ifInfo))) {
89  AVB_LOGF_ERROR("Creating rawsock; bad interface name: %s", ifname);
90  free(rawsock);
91  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
92  return NULL;
93  }
94 
95  // Deal with frame size.
96  if (rawsock->base.frameSize == 0) {
97  // use interface MTU as max frames size, if none specified
98  rawsock->base.frameSize = rawsock->base.ifInfo.mtu + ETH_HLEN + VLAN_HLEN;
99  }
100  else if (rawsock->base.frameSize > rawsock->base.ifInfo.mtu + ETH_HLEN + VLAN_HLEN) {
101  AVB_LOG_ERROR("Creating rawsock; requested frame size exceeds MTU");
102  free(rawsock);
103  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
104  return NULL;
105  }
106 
107  char errbuf[PCAP_ERRBUF_SIZE];
108  rawsock->handle = open_pcap_dev(ifname, rawsock->base.frameSize, errbuf);
109  if (!rawsock->handle) {
110  AVB_LOGF_ERROR("Cannot open device %s: %s", ifname, errbuf);
111  free(rawsock);
112  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
113  return NULL;
114  }
115 
116  // fill virtual functions table
117  rawsock_cb_t *cb = &rawsock->base.cb;
118  cb->close = pcapRawsockClose;
119  cb->getTxFrame = pcapRawsockGetTxFrame;
120  cb->txFrameReady = pcapRawsockTxFrameReady;
121  cb->send = pcapRawsockSend;
122  cb->getRxFrame = pcapRawsockGetRxFrame;
123  cb->rxMulticast = pcapRawsockRxMulticast;
124  cb->rxParseHdr = pcapRawsockRxParseHdr;
125 
126  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
127  return rawsock;
128 }

154 {

157 
158  if (timeNsec) {
159  IF_LOG_INTERVAL(1000) AVB_LOG_WARNING("launch time is unsupported in pcap_rawsock");
160  }
161 
162  if (rawsock) {
163  ret = pcap_sendpacket(rawsock->handle, pBuffer, len);
164  if (ret == -1) {
165  AVB_LOGF_ERROR("pcap_sendpacket failed: %s", pcap_geterr(rawsock->handle));
166  }
167 
168  }
169  return ret == 0;
170 }

180 U8 *pcapRawsockGetRxFrame(void *pvRawsock, U32 timeout, unsigned int *offset, unsigned int *len)
181 {
182  pcap_rawsock_t *rawsock = (pcap_rawsock_t*)pvRawsock;
183 
184  rawsock->rxHeader = 0;
185  const u_char *packet = 0;
186  int ret;
187 
188  if (rawsock) {
189  ret = pcap_next_ex(rawsock->handle, &rawsock->rxHeader, &packet);
190  switch(ret) {
191  case 1:
192  *offset = 0;
193  *len = rawsock->rxHeader->caplen;
194  return (U8*)packet;
195  case -1:
196  AVB_LOGF_ERROR("pcap_next_ex failed: %s", pcap_geterr(rawsock->handle));
197  break;
198  case 0:
199  // timeout;
200  break;
201  case -2:
202  // no packets to be read from savefile
203  // this should not happened
204  break;
205  default:
206  break;
207  }
208  }
209 
210  return NULL;
211 }

213 int pcapRawsockRxParseHdr(void* pvRawsock, U8* pBuffer, hdr_info_t* pInfo)
214 {
215  int hdrLen = baseRawsockRxParseHdr(pvRawsock, pBuffer, pInfo);
216 
217  pcap_rawsock_t *rawsock = (pcap_rawsock_t*)pvRawsock;
218  if (rawsock && rawsock->rxHeader) {
219  pInfo->ts.tv_sec = rawsock->rxHeader->ts.tv_sec;
220  // we requested nanosecond timestamp precision, so probably we don't have to scale here
221  pInfo->ts.tv_nsec = rawsock->rxHeader->ts.tv_usec;
222  }
223  return hdrLen;
224 }

226 // Setup the rawsock to receive multicast packets
227 bool pcapRawsockRxMulticast(void *pvRawsock, bool add_membership, const U8 addr[ETH_ALEN])
228 {
229  pcap_rawsock_t *rawsock = (pcap_rawsock_t*)pvRawsock;
230 
231  struct bpf_program comp_filter_exp;
232  char filter_exp[30];
233 
234  sprintf(filter_exp, "ether dst %02x:%02x:%02x:%02x:%02x:%02x", addr[0], addr[1], addr[2], addr[3], addr[4], addr[5]);
235 
236  AVB_LOGF_DEBUG("%s %d %s", __func__, (int)add_membership, filter_exp);
237 
238  if (pcap_compile(rawsock->handle, &comp_filter_exp, filter_exp, 0, PCAP_NETMASK_UNKNOWN) < 0) {
239  AVB_LOGF_ERROR("Could not parse filter %s: %s", filter_exp, pcap_geterr(rawsock->handle));
240  return false;
241  }
242 
243  if (pcap_setfilter(rawsock->handle, &comp_filter_exp) < 0) {
244  AVB_LOGF_ERROR("Could not install filter %s: %s", filter_exp, pcap_geterr(rawsock->handle));
245  return false;
246  }
247 
248  return true;
249 }
