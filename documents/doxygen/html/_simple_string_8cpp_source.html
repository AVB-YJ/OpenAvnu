248 void SimpleString::replace(const char* to, const char* with)
249 {
250  size_t c = count(to);
251  size_t len = size();
252  size_t tolen = StrLen(to);
253  size_t withlen = StrLen(with);
254 
255  size_t newsize = len + (withlen * c) - (tolen * c) + 1;
256 
257  if (newsize > 1) {
258  char* newbuf = allocStringBuffer(newsize, __FILE__, __LINE__);
259  for (size_t i = 0, j = 0; i < len;) {
260  if (StrNCmp(&buffer_[i], to, tolen) == 0) {
261  StrNCpy(&newbuf[j], with, withlen + 1);
262  j += withlen;
263  i += tolen;
264  }
265  else {
266  newbuf[j] = buffer_[i];
267  j++;
268  i++;
269  }
270  }
271  deallocStringBuffer(buffer_, __FILE__, __LINE__);
272  buffer_ = newbuf;
273  buffer_[newsize - 1] = '\0';
274  }
275  else {
276  deallocStringBuffer(buffer_, __FILE__, __LINE__);
277  buffer_ = getEmptyString();
