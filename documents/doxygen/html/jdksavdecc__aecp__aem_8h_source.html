213 static inline ssize_t jdksavdecc_aecpdu_aem_read( struct jdksavdecc_aecpdu_aem *p, void const *base, ssize_t pos, size_t len )
214 {
215  ssize_t r = jdksavdecc_validate_range( pos, len, JDKSAVDECC_AECPDU_AEM_LEN );
216  if ( r >= 0 )
217  {
218  if ( jdksavdecc_aecpdu_common_read( &p->aecpdu_header, base, pos, len ) >= 0 )
219  {
220  p->command_type = jdksavdecc_aecpdu_aem_get_command_type( base, pos );
221  }
222  }
223  return r;
224 }
