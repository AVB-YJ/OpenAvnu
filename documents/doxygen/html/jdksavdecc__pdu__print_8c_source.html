63 void jdksavdecc_pdu_print( struct jdksavdecc_printer *self, const uint8_t *p, size_t len, int dump_payload )
64 {
65  if ( len > 12 )
66  {
67  int cd = jdksavdecc_common_control_header_get_cd( p, 0 );
68  uint8_t version = (uint8_t)jdksavdecc_common_control_header_get_version( p, 0 );
69 
70  jdksavdecc_printer_print_label( self, "version" );
71  jdksavdecc_printer_print_uint8( self, version );
72  jdksavdecc_printer_print_eol( self );
73 
74  jdksavdecc_printer_print_label( self, "cd and subtype" );
75  jdksavdecc_printer_print_uint16_name( self, jdksavdecc_pdu_print_cd_subtype, p[0] );
76  jdksavdecc_printer_print_eol( self );
77 
78  if ( version == 0 )
79  {
80  jdksavdecc_printer_print_label( self, "sv" );
81  if ( jdksavdecc_common_control_header_get_sv( p, 0 ) )
82  {
83  jdksavdecc_printer_print( self, "valid" );
84  }
85  else
86  {
87  jdksavdecc_printer_print( self, "not valid" );
88  }
89  jdksavdecc_printer_print_eol( self );
90 
91  if ( cd == 0 ) // version 0 stream data
92  {
93  jdksavdecc_printer_print_label( self, "mr" );
94  jdksavdecc_printer_print_uint8( self, (uint8_t)jdksavdecc_common_stream_header_get_mr( p, 0 ) );
95  jdksavdecc_printer_print_eol( self );
96 
97  jdksavdecc_printer_print_label( self, "gv" );
98  jdksavdecc_printer_print_uint8( self, (uint8_t)jdksavdecc_common_stream_header_get_gv( p, 0 ) );
99  jdksavdecc_printer_print_eol( self );
100 
101  jdksavdecc_printer_print_label( self, "tv" );
102  jdksavdecc_printer_print_uint8( self, (uint8_t)jdksavdecc_common_stream_header_get_tv( p, 0 ) );
103  jdksavdecc_printer_print_eol( self );
104 
105  jdksavdecc_printer_print_label( self, "sequence_num" );
106  jdksavdecc_printer_print_uint8( self, (uint8_t)jdksavdecc_common_stream_header_get_sequence_num( p, 0 ) );
107  jdksavdecc_printer_print_eol( self );
108 
109  jdksavdecc_printer_print_label( self, "tu" );
110  jdksavdecc_printer_print_uint8( self, (uint8_t)jdksavdecc_common_stream_header_get_tu( p, 0 ) );
111  jdksavdecc_printer_print_eol( self );
112 
113  jdksavdecc_printer_print_label( self, "stream_id" );
114  jdksavdecc_printer_print_streamid( self, jdksavdecc_common_stream_header_get_stream_id( p, 0 ) );
115  jdksavdecc_printer_print_eol( self );
116 
117  jdksavdecc_printer_print_label( self, "avtp_timestamp" );
118  jdksavdecc_printer_print_uint32( self, jdksavdecc_common_stream_header_get_avtp_timestamp( p, 0 ) );
119  jdksavdecc_printer_print_eol( self );
120 
121  jdksavdecc_printer_print_label( self, "gateway_info" );
122  jdksavdecc_printer_print_uint32( self, jdksavdecc_common_stream_header_get_gateway_info( p, 0 ) );
123  jdksavdecc_printer_print_eol( self );
124 
125  jdksavdecc_printer_print_label( self, "stream_data_length" );
126  jdksavdecc_printer_print_uint16( self, jdksavdecc_common_stream_header_get_stream_data_length( p, 0 ) );
127  jdksavdecc_printer_print_eol( self );
128 
129  jdksavdecc_printer_print_label( self, "protocol_specific_header" );
130  jdksavdecc_printer_print_uint16( self, jdksavdecc_common_stream_header_get_protocol_specific_header( p, 0 ) );
131  jdksavdecc_printer_print_eol( self );
132 
133  if ( dump_payload && len > JDKSAVDECC_COMMON_STREAM_HEADER_LEN )
134  {
135  jdksavdecc_printer_print_label( self, "stream_data_payload" );
136  jdksavdecc_printer_print_block( self, p, len, JDKSAVDECC_COMMON_STREAM_HEADER_LEN, len );
137  }
138  }
139  else // version 0 control data
140  {
141  jdksavdecc_printer_print_label( self, "control_data" );
142  jdksavdecc_printer_print_uint8( self, (uint8_t)jdksavdecc_common_control_header_get_control_data( p, 0 ) );
143  jdksavdecc_printer_print_eol( self );
144 
145  jdksavdecc_printer_print_label( self, "status" );
146  jdksavdecc_printer_print_uint8( self, (uint8_t)jdksavdecc_common_control_header_get_status( p, 0 ) );
147  jdksavdecc_printer_print_eol( self );
148 
149  jdksavdecc_printer_print_label( self, "control_data_length" );
150  jdksavdecc_printer_print_uint16( self,
151  (uint16_t)jdksavdecc_common_control_header_get_control_data_length( p, 0 ) );
152  jdksavdecc_printer_print_eol( self );
153 
154  jdksavdecc_printer_print_label( self, "stream_id" );
155  jdksavdecc_printer_print_streamid( self, jdksavdecc_common_control_header_get_stream_id( p, 0 ) );
156  jdksavdecc_printer_print_eol( self );
157  }
158 
159  if ( dump_payload && len > JDKSAVDECC_COMMON_CONTROL_HEADER_LEN )
160  {
161  jdksavdecc_printer_print_label( self, "payload" );
162  jdksavdecc_printer_print_block( self, p, len, JDKSAVDECC_COMMON_CONTROL_HEADER_LEN, len );
163  }
164  }
165  }
166 }
