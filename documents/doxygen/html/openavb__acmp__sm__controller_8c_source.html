75 static openavb_list_node_t openavbAcmpSMController_findInflightNodeFromCommand(openavb_acmp_ACMPCommandResponse_t *command)
76 {
77  AVB_TRACE_ENTRY(AVB_TRACE_ACMP);

79  openavb_list_node_t node = openavbListIterFirst(openavbAcmpSMControllerVars.inflight);
80  while (node) {
81  openavb_acmp_InflightCommand_t *pInFlightCommand = openavbListData(node);
82  if (pInFlightCommand) {
83  if (memcmp(pInFlightCommand->command.controller_entity_id, command->controller_entity_id, sizeof(pInFlightCommand->command.controller_entity_id)) == 0 &&

85  break;

96 void openavbAcmpSMController_txCommand(U8 messageType, openavb_acmp_ACMPCommandResponse_t *command, bool retry)
97 {
98  AVB_TRACE_ENTRY(AVB_TRACE_ACMP);
99  openavb_list_node_t node = NULL;
100 
101  openavbRC rc = openavbAcmpMessageSend(messageType, command, OPENAVB_ACMP_STATUS_SUCCESS);
102  if (IS_OPENAVB_SUCCESS(rc)) {
103  if (!retry) {
104  node = openavbListNew(openavbAcmpSMControllerVars.inflight, sizeof(openavb_acmp_InflightCommand_t));
105  if (node) {
106  openavb_acmp_InflightCommand_t *pInFlightCommand = openavbListData(node);
107  if (pInFlightCommand) {
108 
109  memcpy(&pInFlightCommand->command, command, sizeof(pInFlightCommand->command));
110  pInFlightCommand->command.message_type = messageType;
111  pInFlightCommand->retried = FALSE;
112  pInFlightCommand->original_sequence_id = command->sequence_id; // AVDECC_TODO - is this correct?
113 
114  CLOCK_GETTIME(OPENAVB_CLOCK_REALTIME, &pInFlightCommand->timer);
115  switch (messageType) {
116  case OPENAVB_ACMP_MESSAGE_TYPE_GET_TX_STATE_RESPONSE:
117  openavbTimeTimespecAddUsec(&pInFlightCommand->timer, OPENAVB_ACMP_COMMAND_TIMEOUT_GET_TX_STATE_COMMAND * MICROSECONDS_PER_MSEC);
118  break;
119  case OPENAVB_ACMP_MESSAGE_TYPE_CONNECT_RX_RESPONSE:
120  openavbTimeTimespecAddUsec(&pInFlightCommand->timer, OPENAVB_ACMP_COMMAND_TIMEOUT_CONNECT_RX_COMMAND * MICROSECONDS_PER_MSEC);
121  break;
122  case OPENAVB_ACMP_MESSAGE_TYPE_DISCONNECT_RX_RESPONSE:
123  openavbTimeTimespecAddUsec(&pInFlightCommand->timer, OPENAVB_ACMP_COMMAND_TIMEOUT_DISCONNECT_RX_COMMAND * MICROSECONDS_PER_MSEC);
124  break;
125  case OPENAVB_ACMP_MESSAGE_TYPE_GET_RX_STATE_RESPONSE:
126  openavbTimeTimespecAddUsec(&pInFlightCommand->timer, OPENAVB_ACMP_COMMAND_TIMEOUT_GET_RX_STATE_COMMAND * MICROSECONDS_PER_MSEC);
127  break;
128  case OPENAVB_ACMP_MESSAGE_TYPE_GET_TX_CONNECTION_RESPONSE:
129  openavbTimeTimespecAddUsec(&pInFlightCommand->timer, OPENAVB_ACMP_COMMAND_TIMEOUT_GET_TX_CONNECTION_COMMAND * MICROSECONDS_PER_MSEC);
130  break;
131  default:
132  AVB_LOGF_ERROR("Unsupported command %u in openavbAcmpSMController_txCommand", messageType);
133  openavbTimeTimespecAddUsec(&pInFlightCommand->timer, OPENAVB_ACMP_COMMAND_TIMEOUT_CONNECT_RX_COMMAND * MICROSECONDS_PER_MSEC);
134  break;
135  }
136  }
137  }
138  }
139  else {
140  // Retry case
141  node = openavbAcmpSMController_findInflightNodeFromCommand(command);
142  if (node) {
143  openavb_acmp_InflightCommand_t *pInFlightCommand = openavbListData(node);
144  if (pInFlightCommand) {
145  pInFlightCommand->retried = TRUE;
146 
147  CLOCK_GETTIME(OPENAVB_CLOCK_REALTIME, &pInFlightCommand->timer);
148  switch (messageType) {
149  case OPENAVB_ACMP_MESSAGE_TYPE_GET_TX_STATE_RESPONSE:
150  openavbTimeTimespecAddUsec(&pInFlightCommand->timer, OPENAVB_ACMP_COMMAND_TIMEOUT_GET_TX_STATE_COMMAND * MICROSECONDS_PER_MSEC);
151  break;
152  case OPENAVB_ACMP_MESSAGE_TYPE_CONNECT_RX_RESPONSE:
153  openavbTimeTimespecAddUsec(&pInFlightCommand->timer, OPENAVB_ACMP_COMMAND_TIMEOUT_CONNECT_RX_COMMAND * MICROSECONDS_PER_MSEC);
154  break;
155  case OPENAVB_ACMP_MESSAGE_TYPE_DISCONNECT_RX_RESPONSE:
156  openavbTimeTimespecAddUsec(&pInFlightCommand->timer, OPENAVB_ACMP_COMMAND_TIMEOUT_DISCONNECT_RX_COMMAND * MICROSECONDS_PER_MSEC);
157  break;
158  case OPENAVB_ACMP_MESSAGE_TYPE_GET_RX_STATE_RESPONSE:
159  openavbTimeTimespecAddUsec(&pInFlightCommand->timer, OPENAVB_ACMP_COMMAND_TIMEOUT_GET_RX_STATE_COMMAND * MICROSECONDS_PER_MSEC);
160  break;
161  case OPENAVB_ACMP_MESSAGE_TYPE_GET_TX_CONNECTION_RESPONSE:
162  openavbTimeTimespecAddUsec(&pInFlightCommand->timer, OPENAVB_ACMP_COMMAND_TIMEOUT_GET_TX_CONNECTION_COMMAND * MICROSECONDS_PER_MSEC);
163  break;
164  default:
165  AVB_LOGF_ERROR("Unsupported command %u in openavbAcmpSMController_txCommand", messageType);
166  openavbTimeTimespecAddUsec(&pInFlightCommand->timer, OPENAVB_ACMP_COMMAND_TIMEOUT_CONNECT_RX_COMMAND * MICROSECONDS_PER_MSEC);
167  break;
168  }
169  }
170  }
171  }
172  }
173  else {
174  // Failed to send command
175  openavbAcmpMessageSend(messageType, command, OPENAVB_ACMP_STATUS_COULD_NOT_SEND_MESSAGE);
176  if (retry) {
177  node = openavbAcmpSMController_findInflightNodeFromCommand(command);
178  if (node) {
179  openavbListDelete(openavbAcmpSMControllerVars.inflight, node);
180  }
181  }
182  }
183 
184  AVB_TRACE_EXIT(AVB_TRACE_ACMP);
185 }

220 void openavbAcmpSMControllerStateMachine()
221 {
222  AVB_TRACE_ENTRY(AVB_TRACE_ACMP);
223 
224  openavb_acmp_InflightCommand_t *pInflightActive = NULL; // The inflight command for the current state
225  openavb_acmp_sm_controller_state_t state = OPENAVB_ACMP_SM_CONTROLLER_STATE_WAITING;
226 
227  // Lock such that the mutex is held unless waiting for a semaphore. Synchronous processing of command responses.
228  ACMP_SM_LOCK();
229  while (bRunning) {
230  switch (state) {
231  case OPENAVB_ACMP_SM_CONTROLLER_STATE_WAITING:
232  AVB_TRACE_LINE(AVB_TRACE_ACMP);
233  AVB_LOG_DEBUG("State: OPENAVB_ACMP_SM_CONTROLLER_STATE_WAITING");
234 
235  openavbAcmpSMControllerVars.rcvdResponse = FALSE;
236 
237  // Calculate timeout for inflight commands
238  // Start is a arbitrary large time out.
239  struct timespec timeout;
240  CLOCK_GETTIME(OPENAVB_CLOCK_REALTIME, &timeout);
241  timeout.tv_sec += 60; // At most will timeout after 60 seconds
242 
243  // Look for soonest inflight command timeout
244  openavb_list_node_t node = openavbListIterFirst(openavbAcmpSMControllerVars.inflight);
245  while (node) {
246  openavb_acmp_InflightCommand_t *pInflight = openavbListData(node);
247  if ((openavbTimeTimespecCmp(&pInflight->timer, &timeout) < 0)) {
248  timeout.tv_sec = pInflight->timer.tv_sec;
249  timeout.tv_nsec = pInflight->timer.tv_nsec;
250  }
251  node = openavbListIterNext(openavbAcmpSMControllerVars.inflight);
252  }
253 
254  ACMP_SM_UNLOCK();
255  U32 timeoutMSec = 0;
256  struct timespec now;
257  CLOCK_GETTIME(OPENAVB_CLOCK_REALTIME, &now);
258  timeoutMSec = openavbTimeUntilMSec(&now, &timeout);
259  SEM_ERR_T(err);
260  SEM_TIMEDWAIT(openavbAcmpSMControllerSemaphore, timeoutMSec, err);
261  ACMP_SM_LOCK();
262 
263  if (!SEM_IS_ERR_NONE(err)) {
264  if (SEM_IS_ERR_TIMEOUT(err)) {
265  struct timespec now;
266  CLOCK_GETTIME(OPENAVB_CLOCK_REALTIME, &now);
267 
268  // Look for a timed out inflight command
269  openavb_list_node_t node = openavbListIterFirst(openavbAcmpSMControllerVars.inflight);
270  while (node) {
271  openavb_acmp_InflightCommand_t *pInflight = openavbListData(node);
272  if ((openavbTimeTimespecCmp(&now, &pInflight->timer) >= 0)) {
273  // Found a timed out command
274  state = OPENAVB_ACMP_SM_CONTROLLER_STATE_TIMEOUT;
275  pInflightActive = pInflight;
276  break;
277  }
278  node = openavbListIterNext(openavbAcmpSMControllerVars.inflight);
279  }
280  }
281  }
282  else {
283  if (openavbAcmpSMControllerVars.doTerminate) {
284  bRunning = FALSE;
285  }
286  else if (openavbAcmpSMControllerVars.rcvdResponse &&
287  memcmp(pRcvdCmdResp->controller_entity_id, openavbAcmpSMGlobalVars.my_id, sizeof(openavbAcmpSMGlobalVars.my_id)) == 0) {
288  // Look for a corresponding inflight command
289  openavb_list_node_t node = openavbListIterFirst(openavbAcmpSMControllerVars.inflight);
290  while (node) {
291  openavb_acmp_InflightCommand_t *pInflight = openavbListData(node);
292  if (pRcvdCmdResp->sequence_id == pInflight->command.sequence_id &&
293  pRcvdCmdResp->message_type == pInflight->command.message_type + 1) {
294  // Found a corresponding command
295  state = OPENAVB_ACMP_SM_CONTROLLER_STATE_RESPONSE;
296  pInflightActive = pInflight;
297  break;
298  }
299  node = openavbListIterNext(openavbAcmpSMControllerVars.inflight);
300  }
301  }
302  break;
303 
304  case OPENAVB_ACMP_SM_CONTROLLER_STATE_TIMEOUT:
305  {
306  AVB_TRACE_LINE(AVB_TRACE_ACMP);
307  AVB_LOG_DEBUG("State: OPENAVB_ACMP_SM_CONTROLLER_STATE_TIMEOUT");
308 
309  if (pInflightActive) {
310  if (pInflightActive->retried) {
311  openavbAcmpSMController_removeInflight(&pInflightActive->command);
312  }
313  else {
314  openavbAcmpSMController_txCommand(pInflightActive->command.message_type, &pInflightActive->command, TRUE);
315  }
316  }
317  pInflightActive = NULL;
318  state = OPENAVB_ACMP_SM_CONTROLLER_STATE_WAITING;
319  }
320  break;
321 
322  case OPENAVB_ACMP_SM_CONTROLLER_STATE_RESPONSE:
323  {
324  AVB_TRACE_LINE(AVB_TRACE_ACMP);
325  AVB_LOG_DEBUG("State: OPENAVB_ACMP_SM_CONTROLLER_STATE_RESPONSE");
326 
327  if (pInflightActive) {
328  openavbAcmpSMController_cancelTimeout(pRcvdCmdResp);
329  openavbAcmpSMController_processResponse(pRcvdCmdResp);
330  openavbAcmpSMController_removeInflight(pRcvdCmdResp);
331  }
332  pInflightActive = NULL;
333  state = OPENAVB_ACMP_SM_CONTROLLER_STATE_WAITING;
334  }
335  break;
336  }
337  }
338  }
339  ACMP_SM_UNLOCK();
340 
341  AVB_TRACE_EXIT(AVB_TRACE_ACMP);
342 }
