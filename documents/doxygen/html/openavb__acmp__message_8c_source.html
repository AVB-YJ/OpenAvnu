108 bool openavbAcmpOpenSocket(const char* ifname, U16 vlanID, U8 vlanPCP)
109 {
110  AVB_TRACE_ENTRY(AVB_TRACE_ACMP);
111 
112  hdr_info_t hdr;
113 
114  rxSock = openavbRawsockOpen(ifname, TRUE, FALSE, ETHERTYPE_AVTP, ACMP_FRAME_LEN, ACMP_NUM_RX_BUFFERS);
115  txSock = openavbRawsockOpen(ifname, FALSE, TRUE, ETHERTYPE_AVTP, ACMP_FRAME_LEN, ACMP_NUM_TX_BUFFERS);
116 
117  if (txSock && rxSock
118  && openavbRawsockGetAddr(txSock, ADDR_PTR(&intfAddr))
119  && ether_aton_r(ACMP_PROTOCOL_ADDR, &acmpAddr)
120  && openavbRawsockRxMulticast(rxSock, TRUE, ADDR_PTR(&acmpAddr)))
121  {
122  if (!openavbRawsockRxAVTPSubtype(rxSock, OPENAVB_ACMP_AVTP_SUBTYPE | 0x80)) {
123  AVB_LOG_DEBUG("RX AVTP Subtype not supported");
124  }
125 
126  memset(&hdr, 0, sizeof(hdr_info_t));
127  hdr.shost = ADDR_PTR(&intfAddr);
128  hdr.dhost = ADDR_PTR(&acmpAddr);
129  hdr.ethertype = ETHERTYPE_AVTP;
130  if (vlanID != 0 || vlanPCP != 0) {
131  hdr.vlan = TRUE;
132  hdr.vlan_pcp = vlanPCP;
133  hdr.vlan_vid = vlanID;
134  AVB_LOGF_DEBUG("VLAN pcp=%d vid=%d", hdr.vlan_pcp, hdr.vlan_vid);
135  }
136  if (!openavbRawsockTxSetHdr(txSock, &hdr)) {
137  AVB_LOG_ERROR("TX socket Header Failure");
138  openavbAcmpCloseSocket();
139  AVB_TRACE_EXIT(AVB_TRACE_ACMP);
140  return false;
141  }
142 
143  AVB_TRACE_EXIT(AVB_TRACE_ACMP);
144  return true;
145  }
146 
147  AVB_LOG_ERROR("Invalid socket");
148  openavbAcmpCloseSocket();
149 
150  AVB_TRACE_EXIT(AVB_TRACE_ACMP);
151  return false;
152 }

154 static void openavbAcmpMessageRxFrameParse(U8* payload, int payload_len, hdr_info_t *hdr)
155 {
156  AVB_TRACE_ENTRY(AVB_TRACE_ACMP);
157 
158  openavb_acmp_control_header_t acmpHeader;
159  openavb_acmp_ACMPCommandResponse_t acmpCommandResponse;
160 
161 #if 0
162  AVB_LOGF_DEBUG("openavbAcmpMessageRxFrameParse packet data (length %d):", payload_len);
163  AVB_LOG_BUFFER(AVB_LOG_LEVEL_DEBUG, payload, payload_len, 16);
164 #endif
165 
166  U8 *pSrc = payload;
167  {
168  openavb_acmp_control_header_t *pDst1 = &acmpHeader;
169  openavb_acmp_ACMPCommandResponse_t *pDst2 = &acmpCommandResponse;
170 
171  memset(&acmpHeader, 0, sizeof(acmpHeader));
172  memset(&acmpCommandResponse, 0, sizeof(acmpCommandResponse));
173 
174  // AVTP Control Header
175  BIT_B2DNTOHB(pDst1->cd, pSrc, 0x80, 7, 0);
176  BIT_B2DNTOHB(pDst1->subtype, pSrc, 0x7f, 0, 1);
177  BIT_B2DNTOHB(pDst1->sv, pSrc, 0x80, 7, 0);
178  BIT_B2DNTOHB(pDst1->version, pSrc, 0x70, 4, 0);
179  BIT_B2DNTOHB(pDst2->message_type, pSrc, 0x0f, 0, 1);
180  BIT_B2DNTOHB(pDst2->status, pSrc, 0xf800, 11, 0);
181  BIT_B2DNTOHS(pDst1->control_data_length, pSrc, 0x07ff, 0, 2);
182  OCT_B2DMEMCP(pDst2->stream_id, pSrc);
183 
184  // ACMP PDU
185  OCT_B2DMEMCP(pDst2->controller_entity_id, pSrc);
186  OCT_B2DMEMCP(pDst2->talker_entity_id, pSrc);
187  OCT_B2DMEMCP(pDst2->listener_entity_id, pSrc);
188  OCT_B2DNTOHS(pDst2->talker_unique_id, pSrc);
189  OCT_B2DNTOHS(pDst2->listener_unique_id, pSrc);
190  OCT_B2DMEMCP(pDst2->stream_dest_mac, pSrc);
191  OCT_B2DNTOHS(pDst2->connection_count, pSrc);
192  OCT_B2DNTOHS(pDst2->sequence_id, pSrc);
193  OCT_B2DNTOHS(pDst2->flags, pSrc);
194  OCT_B2DNTOHS(pDst2->stream_vlan_id, pSrc);
195  }
196 
197  // Update the state machine
198  switch (acmpCommandResponse.message_type) {
199  case OPENAVB_ACMP_MESSAGE_TYPE_CONNECT_TX_COMMAND:
200  openavbAcmpSMTalkerSet_rcvdConnectTXCmd(&acmpCommandResponse);
201  break;
202  case OPENAVB_ACMP_MESSAGE_TYPE_CONNECT_TX_RESPONSE:
203  openavbAcmpSMListenerSet_rcvdConnectTXResp(&acmpCommandResponse);
204  break;
205  case OPENAVB_ACMP_MESSAGE_TYPE_DISCONNECT_TX_COMMAND:
206  openavbAcmpSMTalkerSet_rcvdDisconnectTXCmd(&acmpCommandResponse);
207  break;
208  case OPENAVB_ACMP_MESSAGE_TYPE_DISCONNECT_TX_RESPONSE:
209  openavbAcmpSMListenerSet_rcvdDisconnectTXResp(&acmpCommandResponse);
210  break;
211  case OPENAVB_ACMP_MESSAGE_TYPE_GET_TX_STATE_COMMAND:
212  openavbAcmpSMTalkerSet_rcvdGetTXState(&acmpCommandResponse);
213  break;
214  case OPENAVB_ACMP_MESSAGE_TYPE_GET_TX_STATE_RESPONSE:
215  openavbAcmpSMControllerSet_rcvdResponse(&acmpCommandResponse);
216  break;
217  case OPENAVB_ACMP_MESSAGE_TYPE_CONNECT_RX_COMMAND:
218  openavbAcmpSMListenerSet_rcvdConnectRXCmd(&acmpCommandResponse);
219  break;
220  case OPENAVB_ACMP_MESSAGE_TYPE_CONNECT_RX_RESPONSE:
221  openavbAcmpSMControllerSet_rcvdResponse(&acmpCommandResponse);
222  break;
223  case OPENAVB_ACMP_MESSAGE_TYPE_DISCONNECT_RX_COMMAND:
224  openavbAcmpSMListenerSet_rcvdDisconnectRXCmd(&acmpCommandResponse);
225  break;
226  case OPENAVB_ACMP_MESSAGE_TYPE_DISCONNECT_RX_RESPONSE:
227  openavbAcmpSMControllerSet_rcvdResponse(&acmpCommandResponse);
228  break;
229  case OPENAVB_ACMP_MESSAGE_TYPE_GET_RX_STATE_COMMAND:
230  openavbAcmpSMListenerSet_rcvdGetRXState(&acmpCommandResponse);
231  break;
232  case OPENAVB_ACMP_MESSAGE_TYPE_GET_RX_STATE_RESPONSE:
233  openavbAcmpSMControllerSet_rcvdResponse(&acmpCommandResponse);
234  break;
235  case OPENAVB_ACMP_MESSAGE_TYPE_GET_TX_CONNECTION_COMMAND:
236  openavbAcmpSMTalkerSet_rcvdGetTXConnectionCmd(&acmpCommandResponse);
237  break;
238  case OPENAVB_ACMP_MESSAGE_TYPE_GET_TX_CONNECTION_RESPONSE:
239  openavbAcmpSMControllerSet_rcvdResponse(&acmpCommandResponse);
240  break;
241  }
242 
243  AVB_TRACE_EXIT(AVB_TRACE_ACMP);
244 }

246 static void openavbAcmpMessageRxFrameReceive(U32 timeoutUsec)
247 {
248  AVB_TRACE_ENTRY(AVB_TRACE_ACMP);
249 
250  hdr_info_t hdrInfo;
251  unsigned int offset, len;
252  U8 *pBuf, *pFrame;
253 
254  memset(&hdrInfo, 0, sizeof(hdr_info_t));
255 
256  pBuf = (U8 *)openavbRawsockGetRxFrame(rxSock, timeoutUsec, &offset, &len);
257  if (pBuf) {
258  pFrame = pBuf + offset;
259 
260  offset = openavbRawsockRxParseHdr(rxSock, pBuf, &hdrInfo);
261  {
262 #ifndef UBUNTU
263  if (hdrInfo.ethertype == ETHERTYPE_8021Q) {
264  // Oh! Need to look past the VLAN tag
265  U16 vlan_bits = ntohs(*(U16 *)(pFrame + offset));
266  hdrInfo.vlan = TRUE;
267  hdrInfo.vlan_vid = vlan_bits & 0x0FFF;
268  hdrInfo.vlan_pcp = (vlan_bits >> 13) & 0x0007;
269  offset += 2;
270  hdrInfo.ethertype = ntohs(*(U16 *)(pFrame + offset));
271  offset += 2;
272  }
273 #endif
274 
275  // Make sure that this is an AVTP packet
276  // (Should always be AVTP if it's to our AVTP-specific multicast address)
277  if (hdrInfo.ethertype == ETHERTYPE_AVTP) {
278  // parse the PDU only for ACMP messages
279  if (*(pFrame + offset) == (0x80 | OPENAVB_ACMP_AVTP_SUBTYPE)) {
280  openavbAcmpMessageRxFrameParse(pFrame + offset, len - offset, &hdrInfo);
281  }
282  }
283  else {
284  AVB_LOG_WARNING("Received non-AVTP frame!");
285  AVB_LOGF_DEBUG("Unexpected packet data (length %d):", len);
286  AVB_LOG_BUFFER(AVB_LOG_LEVEL_DEBUG, pFrame, len, 16);
287  }
288  }
289 
290  // Release the frame
291  openavbRawsockRelRxFrame(rxSock, pBuf);
292  }
293 
294  AVB_TRACE_EXIT(AVB_TRACE_ACMP);
295 }

297 void openavbAcmpMessageTxFrame(U8 messageType, openavb_acmp_ACMPCommandResponse_t *pACMPCommandResponse, U8 status)
298 {
299  AVB_TRACE_ENTRY(AVB_TRACE_ACMP);
300 
301  U8 *pBuf;
302  U32 size;
303  unsigned int hdrlen = 0;
304 
305  pBuf = openavbRawsockGetTxFrame(txSock, TRUE, &size);
306 
307  if (!pBuf) {
308  AVB_LOG_ERROR("No TX buffer");
309  AVB_TRACE_EXIT(AVB_TRACE_ACMP);
310  return; // AVDECC_TODO - return error result
311  }
312 
313  if (size < ACMP_FRAME_LEN) {
314  AVB_LOG_ERROR("TX buffer too small");
315  openavbRawsockRelTxFrame(txSock, pBuf);
316  pBuf = NULL;
317  AVB_TRACE_EXIT(AVB_TRACE_ACMP);
318  return; // AVDECC_TODO - return error result
319  }
320 
321  memset(pBuf, 0, ACMP_FRAME_LEN);
322  openavbRawsockTxFillHdr(txSock, pBuf, &hdrlen);
323 
324  ACMP_LOCK();
325  U8 *pDst = pBuf + hdrlen;
326  {
327  openavb_acmp_control_header_t *pSrc1 = &openavbAcmpSMGlobalVars.controlHeader;
328  openavb_acmp_ACMPCommandResponse_t *pSrc2 = pACMPCommandResponse;
329 
330  // AVTP Control Header
331  BIT_D2BHTONB(pDst, pSrc1->cd, 7, 0);
332  BIT_D2BHTONB(pDst, pSrc1->subtype, 0, 1);
333  BIT_D2BHTONB(pDst, pSrc1->sv, 7, 0);
334  BIT_D2BHTONB(pDst, pSrc1->version, 4, 0);
335  BIT_D2BHTONB(pDst, messageType, 0, 1);
336  BIT_D2BHTONS(pDst, status, 11, 0);
337  BIT_D2BHTONS(pDst, pSrc1->control_data_length, 0, 2);
338  OCT_D2BMEMCP(pDst, pSrc2->stream_id);
339 
340  // ACMP PDU
341  OCT_D2BMEMCP(pDst, pSrc2->controller_entity_id);
342  OCT_D2BMEMCP(pDst, pSrc2->talker_entity_id);
343  OCT_D2BMEMCP(pDst, pSrc2->listener_entity_id);
344  OCT_D2BHTONS(pDst, pSrc2->talker_unique_id);
345  OCT_D2BHTONS(pDst, pSrc2->listener_unique_id);
346  OCT_D2BMEMCP(pDst, pSrc2->stream_dest_mac);
347  OCT_D2BHTONS(pDst, pSrc2->connection_count);
348  OCT_D2BHTONS(pDst, pSrc2->sequence_id);
349  OCT_D2BHTONS(pDst, pSrc2->flags);
350  OCT_D2BHTONS(pDst, pSrc2->stream_vlan_id);
351  }
352  ACMP_UNLOCK();
353 
354 #if 0
355  AVB_LOGF_DEBUG("openavbAcmpMessageTxFrame packet data (length %d):", hdrlen + AVTP_HDR_LEN + ACMP_DATA_LEN);
356  AVB_LOG_BUFFER(AVB_LOG_LEVEL_DEBUG, pBuf, hdrlen + AVTP_HDR_LEN + ACMP_DATA_LEN, 16);
357 #endif
358 
359  openavbRawsockTxFrameReady(txSock, pBuf, hdrlen + AVTP_HDR_LEN + ACMP_DATA_LEN, 0);
360  openavbRawsockSend(txSock);
361 
362  AVB_TRACE_EXIT(AVB_TRACE_ACMP);
363 }
