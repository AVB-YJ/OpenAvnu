32 /*
33  * MODULE SUMMARY :
34  *
35  * Stream clients (talkers or listeners) must connect to the central
36  * "avdecc_msg" process to create a reservation for their traffic.
37  *
38  * This code implements the server side of the IPC.
39  *
40  * It provides proxy functions for the avdecc_msg to call. The arguments
41  * for those calls are packed into messages, which are unpacked in the
42  * process and then used to call the real functions.
43  *
44  * Current IPC uses unix sockets. Can change this by creating a new
45  * implementations in openavb_avdecc_msg_client.c and openavb_avdecc_msg_server.c
46  */

77 static bool openavbAvdeccMsgSrvrReceiveFromClient(int avdeccMsgHandle, openavbAvdeccMessage_t *msg)
78 {
79  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC_MSG);
80 
81  if (!msg) {
82  AVB_LOG_ERROR("Receiving message; invalid argument passed");
83  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
84  return FALSE;
85  }
86 
87  bool ret = FALSE;
88  switch (msg->type) {
89  case OPENAVB_AVDECC_MSG_VERSION_REQUEST:
90  AVB_LOG_DEBUG("Message received: OPENAVB_AVDECC_MSG_VERSION_REQUEST");
91  ret = openavbAvdeccMsgSrvrHndlVerRqstFromClient(avdeccMsgHandle);
92  break;
93  case OPENAVB_AVDECC_MSG_CLIENT_INIT_IDENTIFY:
94  AVB_LOG_DEBUG("Message received: OPENAVB_AVDECC_MSG_CLIENT_INIT_IDENTIFY");
95  ret = openavbAvdeccMsgSrvrHndlInitIdentifyFromClient(avdeccMsgHandle,
96  msg->params.clientInitIdentify.friendly_name,
97  msg->params.clientInitIdentify.talker);
98  break;
99  case OPENAVB_AVDECC_MSG_C2S_TALKER_STREAM_ID:
100  AVB_LOG_DEBUG("Message received: OPENAVB_AVDECC_MSG_C2S_TALKER_STREAM_ID");
101  ret = openavbAvdeccMsgSrvrHndlTalkerStreamIDFromClient(avdeccMsgHandle,
102  msg->params.c2sTalkerStreamID.sr_class,
103  msg->params.c2sTalkerStreamID.stream_src_mac,
104  ntohs(msg->params.c2sTalkerStreamID.stream_uid),
105  msg->params.c2sTalkerStreamID.stream_dest_mac,
106  ntohs(msg->params.c2sTalkerStreamID.stream_vlan_id));
107  break;
108  case OPENAVB_AVDECC_MSG_CLIENT_CHANGE_NOTIFICATION:
109  AVB_LOG_DEBUG("Message received: OPENAVB_AVDECC_MSG_CLIENT_CHANGE_NOTIFICATION");
110  ret = openavbAvdeccMsgSrvrHndlChangeNotificationFromClient(avdeccMsgHandle, (openavbAvdeccMsgStateType_t) msg->params.clientChangeNotification.current_state);
111  break;
112  default:
113  AVB_LOG_ERROR("Unexpected message received at server");
114  break;
115  }
116 
117  AVB_LOGF_VERBOSE("Message handled, ret=%d", ret);
118  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
119  return ret;
120 }

147 bool openavbAvdeccMsgSrvrHndlInitIdentifyFromClient(int avdeccMsgHandle, char * friendly_name, U8 talker)
148 {
149  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC_MSG);
150  openavb_tl_data_cfg_t * currentStream;
151 
152  avdecc_msg_state_t *pState = AvdeccMsgStateListGet(avdeccMsgHandle);
153  if (pState) {
154  // The handle was already specified. Something has gone terribly wrong!
155  AVB_LOGF_ERROR("avdeccMsgHandle %d already used", avdeccMsgHandle);
156  AvdeccMsgStateListRemove(pState);
157  free(pState);
158  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
159  return false;
160  }
161 
162  // Make sure the supplied string is nil-terminated.
163  friendly_name[FRIENDLY_NAME_SIZE - 1] = '\0';
164 
165  // Create a structure to hold the client information.
166  pState = (avdecc_msg_state_t *) calloc(1, sizeof(avdecc_msg_state_t));
167  if (!pState) {
168  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
169  return false;
170  }
171  pState->avdeccMsgHandle = avdeccMsgHandle;
172  pState->bTalker = (talker != 0);
173  pState->lastRequestedState = pState->lastReportedState = OPENAVB_AVDECC_MSG_UNKNOWN;
174 
175  // Find the state information matching this item.
176  for (currentStream = streamList; currentStream != NULL; currentStream = currentStream->next) {
177  if (strncmp(currentStream->friendly_name, friendly_name, FRIENDLY_NAME_SIZE) == 0)
178  {
179  break;
180  }
181  }
182  if (!currentStream) {
183  AVB_LOGF_WARNING("Ignoring unexpected client %d, friendly_name: %s",
184  avdeccMsgHandle, friendly_name);
185  free(pState);
186  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
187  return false;
188  }
189 
190  // Keep track of this new state item.
191  if (!AvdeccMsgStateListAdd(pState)) {
192  AVB_LOGF_ERROR("Error saving client identity information %d", avdeccMsgHandle);
193  free(pState);
194  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
195  return false;
196  }
197 
198  // Associate this Listener instance with the stream information.
199  pState->stream = currentStream;
200  currentStream->client = pState;
201 
202  AVB_LOGF_INFO("Client %d Detected, friendly_name: %s",
203  avdeccMsgHandle, friendly_name);
204 
205  // Enable ADP support, now that we have at least one Talker/Listener.
206  openavbAdpHaveTL(true);
207 
208  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
209  return true;
210 }

356 bool openavbAvdeccMsgSrvrHndlChangeNotificationFromClient(int avdeccMsgHandle, openavbAvdeccMsgStateType_t currentState)
357 {
358  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC_MSG);
359 
360  avdecc_msg_state_t *pState = AvdeccMsgStateListGet(avdeccMsgHandle);
361  if (!pState) {
362  AVB_LOGF_ERROR("avdeccMsgHandle %d not valid", avdeccMsgHandle);
363  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
364  return false;
365  }
366 
367  // Save the updated state.
368  if (currentState != pState->lastReportedState) {
369  openavbAvdeccMsgStateType_t previousState = pState->lastReportedState;
370  pState->lastReportedState = currentState;
371  AVB_LOGF_INFO("Notified of client %d state change from %s to %s (Last requested: %s)",
372  avdeccMsgHandle, GetStateString(previousState), GetStateString(currentState), GetStateString(pState->lastRequestedState));
373 
374  // If AVDECC did not yet set the client state, set the client state to the desired state.
375  if (pState->lastRequestedState == OPENAVB_AVDECC_MSG_UNKNOWN) {
376  if (pState->stream->initial_state == TL_INIT_STATE_RUNNING && pState->lastReportedState != OPENAVB_AVDECC_MSG_RUNNING) {
377  // Have the client be running if the user explicitly requested it to be running.
378  openavbAvdeccMsgSrvrChangeRequest(avdeccMsgHandle, OPENAVB_AVDECC_MSG_RUNNING);
379  }
380  else if (pState->stream->initial_state != TL_INIT_STATE_RUNNING && pState->lastReportedState == OPENAVB_AVDECC_MSG_RUNNING) {
381  // Have the client not be running if the user didn't explicitly request it to be running.
382  openavbAvdeccMsgSrvrChangeRequest(avdeccMsgHandle, OPENAVB_AVDECC_MSG_STOPPED);
383  }
384  else if (gAvdeccCfg.bFastConnectSupported &&
385  !(pState->bTalker) &&
386  pState->lastReportedState == OPENAVB_AVDECC_MSG_STOPPED) {
387  // Listener started as not running, and is not configured to start in the running state.
388  // See if we should do a fast connect using the saved state.
389  openavb_tl_data_cfg_t *pCfg = pState->stream;
390  if (pCfg) {
391  U16 flags, talker_unique_id;
392  U8 talker_entity_id[8], controller_entity_id[8];
393  bool bAvailable = openavbAvdeccGetSaveStateInfo(pCfg, &flags, &talker_unique_id, &talker_entity_id, &controller_entity_id);
394  if (bAvailable) {
395  openavbAcmpSMListenerSet_doFastConnect(pCfg, flags, talker_unique_id, talker_entity_id, controller_entity_id);
396  }
397  }
398  }
399  }
400  else if (currentState == OPENAVB_AVDECC_MSG_STOPPED_UNEXPECTEDLY) {
401  if (previousState != OPENAVB_AVDECC_MSG_STOPPED_UNEXPECTEDLY &&
402  pState->lastRequestedState == OPENAVB_AVDECC_MSG_RUNNING &&
403  gAvdeccCfg.bFastConnectSupported) {
404  // The Talker disappeared. Use fast connect to assist with reconnecting if it reappears.
405  openavb_tl_data_cfg_t *pCfg = pState->stream;
406  if (pCfg) {
407  U16 flags, talker_unique_id;
408  U8 talker_entity_id[8], controller_entity_id[8];
409  bool bAvailable = openavbAvdeccGetSaveStateInfo(pCfg, &flags, &talker_unique_id, &talker_entity_id, &controller_entity_id);
410  if (bAvailable) {
411  // Set the timed-out status for the Listener's descriptor.
412  // The next time the Talker advertises itself, openavbAcmpSMListenerSet_talkerTestFastConnect() should try to reconnect.
413  openavb_aem_descriptor_stream_io_t *pDescriptor;
414  U16 listenerUniqueId;
415  for (listenerUniqueId = 0; listenerUniqueId < 0xFFFF; ++listenerUniqueId) {
416  pDescriptor = openavbAemGetDescriptor(openavbAemGetConfigIdx(), OPENAVB_AEM_DESCRIPTOR_STREAM_INPUT, listenerUniqueId);
417  if (pDescriptor && pDescriptor->stream &&
418  strcmp(pDescriptor->stream->friendly_name, pCfg->friendly_name) == 0) {
419  // We found a match.
420  AVB_LOGF_INFO("Listener %s waiting to fast connect to flags=0x%04x, talker_unique_id=0x%04x, talker_entity_id=" ENTITYID_FORMAT ", controller_entity_id=" ENTITYID_FORMAT,
421  pCfg->friendly_name,
422  flags,
423  talker_unique_id,
424  ENTITYID_ARGS(talker_entity_id),
425  ENTITYID_ARGS(controller_entity_id));
426  pDescriptor->fast_connect_status = OPENAVB_FAST_CONNECT_STATUS_TIMED_OUT;
427  memcpy(pDescriptor->fast_connect_talker_entity_id, talker_entity_id, sizeof(pDescriptor->fast_connect_talker_entity_id));
428  CLOCK_GETTIME(OPENAVB_CLOCK_REALTIME, &pDescriptor->fast_connect_start_time);
429  pDescriptor->fast_connect_start_time.tv_sec += 5; // Give the Talker some time to shutdown or stabilize
430  break;
431  }
432  }
433  }
434  }
435  }
436 
437  // Now that we have handled this, treat this state as a normal stop.
438  pState->lastReportedState = OPENAVB_AVDECC_MSG_STOPPED;
439  }
440  }
441 
442  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
443  return true;
444 }
