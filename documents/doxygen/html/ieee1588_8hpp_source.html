286  Timestamp operator+( const Timestamp& o ) {
287  uint32_t nanoseconds;
288  uint32_t seconds_ls;
289  uint16_t seconds_ms;
290  uint8_t version;
291  bool carry;
292 
293  nanoseconds = this->nanoseconds;
294  nanoseconds += o.nanoseconds;
295  carry =
296  nanoseconds < this->nanoseconds ||
297  nanoseconds >= MAX_NANOSECONDS ? true : false;
298  nanoseconds -= carry ? MAX_NANOSECONDS : 0;
299 
300  seconds_ls = this->seconds_ls;
301  seconds_ls += o.seconds_ls;
302  seconds_ls += carry ? 1 : 0;
303  carry = seconds_ls < this->seconds_ls ? true : false;
304 
305  seconds_ms = this->seconds_ms;
306  seconds_ms += o.seconds_ms;
307  seconds_ms += carry ? 1 : 0;
308  carry = seconds_ms < this->seconds_ms ? true : false;
309

320  Timestamp operator-( const Timestamp& o ) {
321  uint32_t nanoseconds;
322  uint32_t seconds_ls;
323  uint16_t seconds_ms;
324  uint8_t version;
325  bool carry, borrow_this;
326  unsigned borrow_total = 0;
327 
328  borrow_this = this->nanoseconds < o.nanoseconds;
329  nanoseconds =
330  ((borrow_this ? MAX_NANOSECONDS : 0) + this->nanoseconds) -
331  o.nanoseconds;
332  carry = nanoseconds > MAX_NANOSECONDS;
333  nanoseconds -= carry ? MAX_NANOSECONDS : 0;
334  borrow_total += borrow_this ? 1 : 0;
335 
336  seconds_ls = carry ? 1 : 0;
337  seconds_ls += this->seconds_ls;
338  borrow_this =
339  borrow_total > seconds_ls ||
340  seconds_ls - borrow_total < o.seconds_ls;
341  seconds_ls =
342  borrow_this ? seconds_ls - o.seconds_ls + (uint32_t)-1 :
343  (seconds_ls - borrow_total) - o.seconds_ls;
344  borrow_total = borrow_this ? borrow_total + 1 : 0;
345 
346  seconds_ms = carry ? 1 : 0;
347  seconds_ms += this->seconds_ms;
348  borrow_this =
349  borrow_total > seconds_ms ||
350  seconds_ms - borrow_total < o.seconds_ms;
351  seconds_ms =
352  borrow_this ? seconds_ms - o.seconds_ms + (uint32_t)-1 :
353  (seconds_ms - borrow_total) - o.seconds_ms;
354  borrow_total = borrow_this ? borrow_total + 1 : 0;
355 
356  version = this->_version == o._version ? this->_version :
357  INVALID_TIMESTAMP_VERSION;
