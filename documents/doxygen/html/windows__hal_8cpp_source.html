47 VOID CALLBACK WindowsTimerQueueHandler( PVOID arg_in, BOOLEAN ignore ) {
48  WindowsTimerQueueHandlerArg *arg = (WindowsTimerQueueHandlerArg *) arg_in;
49  size_t diff;
50 
51  // Remove myself from unexpired timer queue
52  AcquireSRWLockExclusive( &arg->timer_queue->lock );
53  diff = arg->timer_queue->arg_list.size();
54  arg->timer_queue->arg_list.remove( arg );
55  diff -= arg->timer_queue->arg_list.size();
56  ReleaseSRWLockExclusive( &arg->timer_queue->lock );
57 
58  // If the remove was unsuccessful bail because we were
59  // stepped on by 'cancelEvent'
60  if( diff == 0 ) return;
61 
62  arg->func( arg->inner_arg );
63 
64  // Add myself to the expired timer queue
65  AcquireSRWLockExclusive( &arg->queue->retiredTimersLock );
66  arg->queue->retiredTimers.push_front( arg );
67  ReleaseSRWLockExclusive( &arg->queue->retiredTimersLock );
68 }

70 inline uint64_t scale64(uint64_t i, uint32_t m, uint32_t n)
71 {
72  uint64_t tmp, res, rem;
73 
74  rem = i % n;
75  i /= n;
76 
77  res = i * m;
78  tmp = rem * m;
79 
80  tmp /= n;
81 
82  return res + tmp;
83 }

85 void WirelessTimestamperCallback( LPVOID arg )
86 {
87  WirelessTimestamperCallbackArg *larg =
88  (WirelessTimestamperCallbackArg *)arg;
89  WindowsWirelessTimestamper *timestamper =
90  dynamic_cast<WindowsWirelessTimestamper *> (larg->timestamper);
91  WirelessDialog tmp1, tmp2;
92  LinkLayerAddress *peer_addr = NULL;
93 
94  if (timestamper == NULL)
95  {
96  GPTP_LOG_ERROR( "Wrong timestamper type: %p",
97  larg->timestamper );
98  return;
99  }
100 
101  switch( larg->iEvent_type )
102  {
103  default:
104  case TIMINGMSMT_CONFIRM_EVENT:
105  tmp1.action_devclk = larg->event_data.confirm.T1;
106  tmp1.ack_devclk = larg->event_data.confirm.T4;
107  tmp1.dialog_token = (BYTE)larg->event_data.confirm.DialogToken;
108  GPTP_LOG_VERBOSE
109  ( "Got confirm, %hhu(%llu,%llu)", tmp1.dialog_token,
110  tmp1.action_devclk, tmp1.ack_devclk );
111  peer_addr = new LinkLayerAddress
112  ( larg->event_data.confirm.PeerMACAddress );
113  larg->timestamper->
114  timingMeasurementConfirmCB( *peer_addr, &tmp1 );
115 
116  break;
117 
118  case TIMINGMSMT_EVENT:
119  tmp1/*prev*/.action_devclk = larg->event_data.indication.
120  indication.T1;
121  tmp1/*prev*/.ack_devclk = larg->event_data.indication.
122  indication.T4;
123  tmp1/*prev*/.dialog_token = (BYTE)larg->event_data.indication.
124  indication.FollowUpDialogToken;
125  tmp2/*curr*/.action_devclk = larg->event_data.indication.
126  indication.T2;
127  tmp2/*curr*/.ack_devclk = larg->event_data.indication.
128  indication.T3;
129  tmp2/*curr*/.dialog_token = (BYTE)larg->event_data.indication.
130  indication.DialogToken;
131  GPTP_LOG_VERBOSE
132  ("Got indication, %hhu(%llu,%llu) %hhu(%llu,%llu)",
133  tmp1.dialog_token, tmp1.action_devclk,
134  tmp1.ack_devclk, tmp2.dialog_token,
135  tmp2.action_devclk, tmp2.ack_devclk);
136  peer_addr = new LinkLayerAddress(larg->event_data.indication.
137  indication.PeerMACAddress);
138 
139  larg->timestamper->timeMeasurementIndicationCB
140  ( *peer_addr, &tmp2, &tmp1,
141  larg->event_data.indication.
142  indication.PtpSpec.fwup_data,
143  larg->event_data.indication.
144  indication.WiFiVSpecHdr.Length - (sizeof(PTP_SPEC) -
145  1));
146 
147  break;
148 
149  case TIMINGMSMT_CORRELATEDTIME_EVENT:
150  timestamper->system_counter =
151  scale64( larg->event_data.ptm_wa.TSC, NS_PER_SECOND,
152  (uint32_t)timestamper->tsc_hz.QuadPart );
153  timestamper->system_time.set64(timestamper->system_counter);
154  // Scale from TM timescale to nanoseconds
155  larg->event_data.ptm_wa.LocalClk *= 10;
156  timestamper->device_time.set64
157  (larg->event_data.ptm_wa.LocalClk*10);
158 
159  break;
160  }
161 
162  delete peer_addr;
163 }

178 bool WindowsWirelessTimestamper::HWTimestamper_gettime
179 ( Timestamp *system_time,
180  Timestamp * device_time,
181  uint32_t * local_clock,
182  uint32_t * nominal_clock_rate ) const
183 {
184  bool refreshed = adapter->refreshCrossTimestamp();
185  if (refreshed)
186  {
187  // We have a fresh cross-timestamp just use it
188  *system_time = this->system_time;
189  *device_time = this->device_time;
190  } else
191  {
192  // We weren't able to get a fresh timestamp,
193  // extrapolate from the last
194  LARGE_INTEGER tsc_now;
195  QueryPerformanceCounter(&tsc_now);
196  unsigned device_delta = (unsigned)
197  (((long double) (tsc_now.QuadPart - system_counter)) /
198  (((long double)tsc_hz.QuadPart) / 1000000000));
199  device_delta = (unsigned)(device_delta*getPort()->
200  getLocalSystemFreqOffset());
201  system_time->set64((uint64_t)
202  (((long double)tsc_now.QuadPart) /
203  ((long double)tsc_hz.QuadPart /
204  1000000000)));
205  device_time->set64(device_delta);
206  *device_time = *device_time + this->device_time;
207  }
208 
209  return true;
210 }

247 bool WindowsEtherTimestamper::HWTimestamper_init( InterfaceLabel *iface_label, OSNetworkInterface *net_iface ) {
248  char network_card_id[64];
249  LinkLayerAddress *addr = dynamic_cast<LinkLayerAddress *>(iface_label);
250  if( addr == NULL ) return false;
251  PIP_ADAPTER_INFO pAdapterInfo;
252  IP_ADAPTER_INFO AdapterInfo[32]; // Allocate information for up to 32 NICs
253  DWORD dwBufLen = sizeof(AdapterInfo); // Save memory size of buffer
254 
255  DWORD dwStatus = GetAdaptersInfo( AdapterInfo, &dwBufLen );
256  if( dwStatus != ERROR_SUCCESS ) return false;
257 
258  for( pAdapterInfo = AdapterInfo; pAdapterInfo != NULL; pAdapterInfo = pAdapterInfo->Next ) {
259  if( pAdapterInfo->AddressLength == ETHER_ADDR_OCTETS && *addr == LinkLayerAddress( pAdapterInfo->Address )) {
260  break;
261  }
262  }
263 
264  if( pAdapterInfo == NULL ) return false;
265 
266  DeviceClockRateMapping *rate_map = DeviceClockRateMap;
267  while (rate_map->device_desc != NULL)
268  {
269  if (strstr(pAdapterInfo->Description, rate_map->device_desc) != NULL)
270  break;
271  ++rate_map;
272  }
273  if (rate_map->device_desc != NULL) {
274  netclock_hz.QuadPart = rate_map->clock_rate;
275  }
276  else {
277  GPTP_LOG_ERROR("Unable to determine clock rate for interface %s", pAdapterInfo->Description);
278  return false;
279  }
280 
281  GPTP_LOG_INFO( "Adapter UID: %s\n", pAdapterInfo->AdapterName );
282  PLAT_strncpy( network_card_id, NETWORK_CARD_ID_PREFIX, 63 );
283  PLAT_strncpy( network_card_id+strlen(network_card_id), pAdapterInfo->AdapterName, 63-strlen(network_card_id) );
284 
285  miniport = CreateFile( network_card_id,
286  GENERIC_READ | GENERIC_WRITE,
287  FILE_SHARE_READ | FILE_SHARE_WRITE,
288  NULL, OPEN_EXISTING, 0, NULL );
289  if( miniport == INVALID_HANDLE_VALUE ) return false;
290 
291  tsc_hz.QuadPart = getTSCFrequency( true );
292  if( tsc_hz.QuadPart == 0 ) {
293  return false;
294  }
295 
296  return true;
297 }
