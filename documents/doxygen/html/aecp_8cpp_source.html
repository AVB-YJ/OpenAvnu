58  int aecp::ether_frame_init(end_station *end_station, struct jdksavdecc_frame *ether_frame)
59  {
60  /*** Offset to write the field to ***/
61  size_t ether_frame_pos = 0x0;
62  jdksavdecc_frame_init(ether_frame);
63 
64  /************************************************ Ethernet Frame *************************************/
65  ether_frame->ethertype = JDKSAVDECC_AVTP_ETHERTYPE;
66  utility::convert_uint64_to_eui48(net_interface_ref->get_mac(), ether_frame->src_address.value);
67  utility::convert_uint64_to_eui48(end_station->get_mac(), ether_frame->dest_address.value);
68  ether_frame->length = AECP_FRAME_LEN; // Length of AECP packet is 64 bytes
69 
70  /*********************** Fill frame payload with Ethernet frame information ********************/
71  jdksavdecc_frame_write(ether_frame, ether_frame->payload, ether_frame_pos, ETHER_HDR_SIZE);
72 
73  return 0;
74  }
