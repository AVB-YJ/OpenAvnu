32 /*
33 * HEADER SUMMARY :
34 *
35 * Implementation of IEEE 802.1Q
36 * Multiple Stream Reservation Protocol
37 * (limited initial implementation for end stations)
38 *
39 * This file declares the "Public" API.
40 *
41 * THIS IMPLEMENTATION ASSUMES THAT THERE WILL BE NO MORE THAN ONE
42 * LISTENER THREAD FOR THE SAME STREAM ON A SINGLE END STATION.
43 * (More than one talker for the same stream makes no sense at all.)
44 */

58 // MSRP Declaration Type - IEEE 802.1Q-2011 Table 35-1
59 typedef enum openavbSrpAttribType {
60  openavbSrp_AtTyp_None = 0x00, // not per spec - SRP internal use only
61  openavbSrp_AtTyp_TalkerAdvertise = 0x01,
62  openavbSrp_AtTyp_TalkerFailed = 0x02,
63  openavbSrp_AtTyp_Listener = 0x03,
64  openavbSrp_AtTyp_Domain = 0x04,
65 } openavbSrpAttribType_t;

67 // MSRP Listener Declaration Subtype (aka "FourPacked Event") - IEEE 802.1Q-2011 Table 35-3
68 typedef enum openavbSrpLsnrDeclSubtype{
69  openavbSrp_LDSt_None = 0x00, // not per spec - SRP internal use (same as ignore is OK)
70  openavbSrp_LDSt_Ignore = 0x00,
71  openavbSrp_LDSt_Asking_Failed = 0x01,
72  openavbSrp_LDSt_Ready = 0x02,
73  openavbSrp_LDSt_Ready_Failed = 0x03,
74  openavbSrp_LDSt_Stream_Info = 0xFE, // NOT per IEEE spec - for use to inform talker when MAAP allocated.
75  openavbSrp_LDSt_Interest = 0xFF, // NOT per IEEE spec - for srpAttachStream() only
76 } openavbSrpLsnrDeclSubtype_t;

78 // Stream Reservation Failure Codes - IEEE 802.1Q-2011 Table 35-6
79 enum openavbSrpFailureCode {
80  openavbSrp_FC_NoFail = 0, // 0: No failure,
81  // 1: Insufficient bandwidth,
82  // 2: Insufficient Bridge resources,
83  openavbSrp_FC_NoClassBandwidth = 3, // Insufficient bandwidth for Traffic Class,
84  // 4: StreamID in use by another Talker,
85  // 5: Stream destination address already in use,
86  // 6: Stream preempted by higher rank,
87  // 7: Reported latency has changed,
88  openavbSrp_FC_NotCapable = 8, // Egress port is not AVBCapable,
89  // 9: Use a different destination_address,
90  // 10: Out of MSRP resources,
91  // 11: Out of MMRP resources,
92  // 12: Cannot store destination_address,
93  // 13: Requested priority is not an SR Class priority,
94  // 14: MaxFrameSize is too large for media,
95  // 15: maxFanInPorts limit has been reached,
96  // 16: Changes in FirstValue for a registered StreamID,
97  // 17: VLAN is blocked on this egress port (Registration Forbidden),
98  // 18: VLAN tagging is disabled on this egress port (untagged set),
99  // 19: SR class priority mismatch.
100 };

108 // AVTP provided callbacks passed into openavbSrpInitialize:
109 //
110 // Callback for SRP to notify AVTP Talker that a Listener Declaration has been
111 // registered (or de-registered)
112 // - avtpHandle is provided to SRP by AVTP via openavbSrpRegisterStream()
113 // and is passed back to AVTP in this call;
114 // - [listener declaration] subtype is from the listener declaration.
115 typedef openavbRC (*strmAttachCb_t) (void* avtpHandle,
116  openavbSrpLsnrDeclSubtype_t subtype);
117 // Callback for SRP to notify AVTP Listener that a Talker Declaration has been
118 // registered (or de-registered)
119 // - avtpHandle is provided to SRP by AVTP via openavbSrpAttachStream()
120 // and is passed back to AVTP in this call;
121 // - declType, DA, tSpec and latency are from the Talker Declaration;
122 // - SRClassId is derived from the priority attribute of the Talker
123 // Declaration;
124 // - pFailInfo is valid for only if declType is openavbSrp_AtTyp_TalkerFailed.
125 typedef openavbRC (*strmRegCb_t) (void* avtpHandle,
126  openavbSrpAttribType_t declType,
127  U8 DA[],
128  AVBTSpec_t* tSpec,
129  SRClassIdx_t SRClassIdx,
130  U32 latency,
131  openavbSrpFailInfo_t* pFailInfo);

162 // or to send a listener declaration back to the indicated stream's talker.
