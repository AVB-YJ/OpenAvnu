36 int jdksavdecc_util_parse_nibble( uint8_t *result, char d1 )
37 {
38  int r = 0;
39 
40  uint8_t d = 0;
41  if ( d1 >= '0' && d1 <= '9' )
42  {
43  d = d1 - '0';
44  r = 1;
45  }
46  else if ( d1 >= 'A' && d1 <= 'F' )
47  {
48  d = 0xa + d1 - 'A';
49  r = 1;
50  }
51  else if ( d1 >= 'a' && d1 <= 'f' )
52  {
53  d = 0xa + d1 - 'a';
54  r = 1;
55  }
56 
57  if ( r == 1 )
58  {
59  *result = *result << 4;
60  *result |= d;
61  }
62  return r;
63 }
