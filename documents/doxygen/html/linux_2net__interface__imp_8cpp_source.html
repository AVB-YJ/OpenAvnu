87 net_interface_imp::net_interface_imp()
88 {
89  struct ifaddrs *ifaddr, *ifa;
90  int family, s;
91  char host[NI_MAXHOST];
92  char ifname[256];
93 
94  total_devs = 0;
95 
96  ip_hdr_store = new ipheader;
97  udp_hdr_store = new udpheader;
98 
99  if (getifaddrs(&ifaddr) == -1)
100  {
101  perror("getifaddrs");
102  exit(EXIT_FAILURE);
103  }
104 
105  // Walk through linked list, maintaining head pointer so we
106  // can free list later
107 
108  for (ifa = ifaddr; ifa != NULL; ifa = ifa->ifa_next)
109  {
110  if (ifa->ifa_addr == NULL)
111  continue;
112 
113  family = ifa->ifa_addr->sa_family;
114 
115  // Display interface name and family (including symbolic
116  // form of the latter for the common families)
117  if (family == AF_INET)
118  {
119  s = getnameinfo(ifa->ifa_addr,
120  (family == AF_INET) ? sizeof(struct sockaddr_in) : sizeof(struct sockaddr_in6),
121  host, NI_MAXHOST, NULL, 0, NI_NUMERICHOST);
122 
123  snprintf(ifname, sizeof(ifname), "%s, address: <%s>", ifa->ifa_name, host);
124  if (s != 0)
125  {
126  printf("getnameinfo() failed: %s\n", gai_strerror(s));
127  exit(EXIT_FAILURE);
128  }
129  ifnames.push_back(ifname);
130  total_devs++;
131  }
132  }
133 
134  freeifaddrs(ifaddr);
135 }

172 int STDCALL net_interface_imp::select_interface_by_num(uint32_t interface_num)
173 {
174  struct sockaddr_ll sll;
175  struct ifreq if_mac;
176  const char * ifname;
177  char * s;
178 
179  // adjust interface numnber since count starts at 1
180  interface_num--;
181 
182  ifname = ifnames[interface_num].c_str();
183  s = (char *)ifname;
184  while (*s != ',')
185  {
186  s++;
187  }
188  *s = 0;
189  rawsock = socket(PF_PACKET, SOCK_RAW, htons(ETH_P_ALL));
190  if (rawsock == -1)
191  {
192  fprintf(stderr, "Socket open failed! %s\nuse sudo?\n", strerror(errno));
193  exit(EXIT_FAILURE);
194  }
195 
196  ifindex = getifindex(rawsock, ifname);
197 
198  // get the mac address of the eth0 interface
199  memset(&if_mac, 0, sizeof(struct ifreq));
200  strncpy(if_mac.ifr_name, ifname, IFNAMSIZ - 1);
201  if (ioctl(rawsock, SIOCGIFHWADDR, &if_mac) < 0)
202  {
203  perror("SIOCGIFHWADDR");
204  exit(EXIT_FAILURE);
205  }
206 
207  setpromiscuous(rawsock, ifindex);
208 
209  memset(&sll, 0, sizeof(sll));
210  sll.sll_family = AF_PACKET;
211  sll.sll_ifindex = ifindex;
212  sll.sll_protocol = htons(ETH_P_ALL);
213  bind(rawsock, (struct sockaddr *)&sll, sizeof(sll));
214 
215  utility::convert_eui48_to_uint64((uint8_t *)if_mac.ifr_hwaddr.sa_data, mac);
216 
217  uint16_t etypes[1] = {0x22f0};
218  set_capture_ether_type(etypes, 1);
219 
220  return 0;
221 }

223 int net_interface_imp::set_capture_ether_type(uint16_t * ether_type, uint32_t count)
224 {
225  struct sock_fprog Filter;
226 
227  Filter.len = sizeof(BPF_code) / 8;
228  Filter.filter = BPF_code;
229 
230  for (unsigned int i = 0; i < count; i++)
231  {
232  if (ether_type[i] != (uint16_t)ethernet_proto_bpf[i])
233  fprintf(stderr, "NETIF - packet filter mismatch\n");
234  }
235 
236  // attach filter to socket
237  if (setsockopt(rawsock, SOL_SOCKET, SO_ATTACH_FILTER, &Filter, sizeof(Filter)) == -1)
238  {
239  fprintf(stderr, "socket attach filter failed! %s\n", strerror(errno));
240  close(rawsock);
241  exit(EXIT_FAILURE);
242  }
243 
244  return 0;
245 }

264 int net_interface_imp::send_frame(uint8_t * frame, uint16_t mem_buf_len)
265 {
266  int send_result;
267 
268  // target address
269  struct sockaddr_ll socket_address;
270 
271  // prepare sockaddr_ll
272 
273  // RAW communication
274  socket_address.sll_family = PF_PACKET;
275  socket_address.sll_protocol = htons(ethertype);
276 
277  // index of the network device
278  // see full code later how to retrieve it
279  socket_address.sll_ifindex = ifindex;
280 
281  // ARP hardware identifier is ethernet
282  socket_address.sll_hatype = ARPHRD_ETHER;
283 
284  // target is another host
285  socket_address.sll_pkttype = PACKET_OTHERHOST;
286 
287  // address length
288  socket_address.sll_halen = ETH_ALEN;
289  // MAC - begin
290  memcpy(&socket_address.sll_addr[0], &frame[0], 6);
291  // MAC - end
292  socket_address.sll_addr[6] = 0x00; // not used
293  socket_address.sll_addr[7] = 0x00; // not used
294 
295  // send the packet
296  send_result = sendto(rawsock, frame, mem_buf_len, 0,
297  (struct sockaddr *)&socket_address, sizeof(socket_address));
298 
299  return send_result;
300 }
