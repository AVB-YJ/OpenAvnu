46 /* Return the current time, in nanoseconds. */
47 static uint64_t getCurrentTime(void)
48 {
49  /* Using GetTickCount64() is not ideal, as it only provides a 10-16 millisecond resolution.
50  * However, for our purposes, it is sufficient.
51  * If higher precision is needed, QueryPerformanceCounter() is one possibility. */
52  uint64_t nTime = GetTickCount64();
53  nTime *= 1000000; /* Convert from milliseconds to nanoseconds */
54  return nTime;
55 }

57 /* Convert a Time structure to a nanoseconds value */
58 static uint64_t timeToNanoseconds(const Time *t)
59 {
60  uint64_t nTime =
61  ((uint64_t)t->tv_sec * 1000000000ULL) + /* Convert seconds to nanoseconds */
62  ((uint64_t)t->tv_usec * 1000ULL); /* Convert microseconds to nanoseconds */
63  return nTime;
64 }
