64 bool WirelessPort::_processEvent( Event e )
65 {
66  bool ret;
67 
68  switch (e)
69  {
70  default:
71  GPTP_LOG_ERROR
72  ("Unhandled event type in "
73  "WirelessPort::processEvent(), %d", e);
74  ret = false;
75  break;
76 
77  case POWERUP:
78  case INITIALIZE:
79  {
80  port_ready_condition->wait_prelock();
81 
82  if ( !linkOpen(_openPort, (void *)this ))
83  {
84  GPTP_LOG_ERROR( "Error creating port thread" );
85  ret = false;
86  break;
87  }
88 
89  port_ready_condition->wait();
90 
91  ret = true;
92  break;
93  }
94 
95  case SYNC_INTERVAL_TIMEOUT_EXPIRES:
96  {
97  WirelessTimestamper *timestamper =
98  dynamic_cast<WirelessTimestamper *>( _hw_timestamper );
99  PTPMessageFollowUp *follow_up;
100  PortIdentity dest_id;
101  uint16_t seq;
102  uint8_t buffer[128];
103  size_t length;
104 
105  memset(buffer, 0, sizeof( buffer ));
106 
107  if( prev_dialog.dialog_token != 0 )
108  {
109  follow_up = new PTPMessageFollowUp( this );
110 
111  getPortIdentity(dest_id);
112  follow_up->setPortIdentity(&dest_id);
113  follow_up->setSequenceId(prev_dialog.fwup_seq);
114  follow_up->setPreciseOriginTimestamp
115  ( prev_dialog.action );
116  length = follow_up->buildMessage
117  (this, buffer + timestamper->getFwUpOffset());
118 
119  delete follow_up;
120  }
121 
122  seq = getNextSyncSequenceId();
123  ret = timestamper->requestTimingMeasurement
124  ( &peer_addr, seq, &prev_dialog, buffer, (int)length )
125  == net_succeed;
126 
127  ret = true;
128  break;
129  }
130 
131  case ANNOUNCE_RECEIPT_TIMEOUT_EXPIRES:
132  case SYNC_RECEIPT_TIMEOUT_EXPIRES:
133  ret = false;
134  break;
135 
136  case STATE_CHANGE_EVENT:
137  ret = false;
138  break;
139  }
140 
141  return ret;
142 }

144 bool WirelessPort::openPort()
145 {
146  port_ready_condition->signal();
147 
148  while (true)
149  {
150  uint8_t buf[128];
151  LinkLayerAddress remote;
152  net_result rrecv;
153  size_t length = sizeof(buf);
154  uint32_t link_speed;
155 
156  if(( rrecv = recv( &remote, buf, length, link_speed ))
157  == net_succeed )
158  {
159  processMessage
160  ((char *)buf, (int)length, &remote, link_speed );
161  }
162  else if( rrecv == net_fatal )
163  {
164  GPTP_LOG_ERROR( "read from network interface failed" );
165  this->processEvent( FAULT_DETECTED );
166  break;
167  }
168  }
169 
170  return false;
171 }

184 void WirelessPort::processMessage
185 (char *buf, int length, LinkLayerAddress *remote, uint32_t link_speed)
186 {
187  GPTP_LOG_VERBOSE( "Processing network buffer" );
188 
189  PTPMessageCommon *msg =
190  buildPTPMessage( buf, (int)length, remote, this );
191 
192  if( msg == NULL )
193  {
194  GPTP_LOG_ERROR( "Discarding invalid message" );
195  return;
196  }
197  GPTP_LOG_VERBOSE( "Processing message" );
198 
199  if( !msg->isEvent( ))
200  {
201  msg->processMessage( this );
202  } else
203  {
204  GPTP_LOG_ERROR( "Received event message on port "
205  "incapable of processing" );
206  msg->PTPMessageCommon::processMessage( this );
207  }
208 
209  if (msg->garbage())
210  delete msg;
211 }
