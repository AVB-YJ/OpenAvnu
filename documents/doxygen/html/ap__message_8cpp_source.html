37 // Octet based data 2 buffer macros
38 #define OCT_D2BMEMCP(d, s) memcpy(d, s, sizeof(s)); d += sizeof(s);
39 #define OCT_D2BBUFCP(d, s, len) memcpy(d, s, len); d += len;
40 #define OCT_D2BHTONB(d, s) *(U8 *)(d) = s; d += sizeof(s);

62 void APMessageTestStatus::sendPort( EtherPort * port )
63 {
64  static uint16_t sequenceId = 0;
65 
66  uint8_t buf_t[256];
67  uint8_t *buf_ptr = buf_t + port->getPayloadOffset();
68  uint16_t tmp16;
69  uint32_t tmp32;
70  uint64_t tmp64;
71 
72  memset(buf_t, 0, 256);
73 
74  // Create packet in buf
75  messageLength = AP_TEST_STATUS_LENGTH;
76 
77  BIT_D2BHTONB(buf_ptr + AP_TEST_STATUS_AVTP_SUBTYPE(AP_TEST_STATUS_OFFSET), 0xfb, 0);
78  BIT_D2BHTONB(buf_ptr + AP_TEST_STATUS_AVTP_VERSION_CONTROL(AP_TEST_STATUS_OFFSET), 0x1, 0);
79  BIT_D2BHTONS(buf_ptr + AP_TEST_STATUS_AVTP_STATUS_LENGTH(AP_TEST_STATUS_OFFSET), 148, 0);
80 
81  port->getLocalAddr()->toOctetArray(buf_ptr + AP_TEST_STATUS_TARGET_ENTITY_ID(AP_TEST_STATUS_OFFSET));
82 
83  tmp16 = PLAT_htons(sequenceId++);
84  memcpy(buf_ptr + AP_TEST_STATUS_SEQUENCE_ID(AP_TEST_STATUS_OFFSET), &tmp16, sizeof(tmp16));
85 
86  BIT_D2BHTONS(buf_ptr + AP_TEST_STATUS_COMMAND_TYPE(AP_TEST_STATUS_OFFSET), 1, 15);
87  BIT_D2BHTONS(buf_ptr + AP_TEST_STATUS_COMMAND_TYPE(AP_TEST_STATUS_OFFSET), 0x29, 0);
88 
89  tmp16 = PLAT_htons(0x09);
90  memcpy(buf_ptr + AP_TEST_STATUS_DESCRIPTOR_TYPE(AP_TEST_STATUS_OFFSET), &tmp16, sizeof(tmp16));
91  tmp16 = PLAT_htons(0x00);
92  memcpy(buf_ptr + AP_TEST_STATUS_DESCRIPTOR_INDEX(AP_TEST_STATUS_OFFSET), &tmp16, sizeof(tmp16));
93 
94  tmp32 = PLAT_htonl(0x07000023);
95  memcpy(buf_ptr + AP_TEST_STATUS_COUNTERS_VALID(AP_TEST_STATUS_OFFSET), &tmp32, sizeof(tmp32));
96 
97  tmp32 = PLAT_htonl(port->getLinkUpCount());
98  memcpy(buf_ptr + AP_TEST_STATUS_COUNTER_LINKUP(AP_TEST_STATUS_OFFSET), &tmp32, sizeof(tmp32));
99 
100  tmp32 = PLAT_htonl(port->getLinkDownCount());
101  memcpy(buf_ptr + AP_TEST_STATUS_COUNTER_LINKDOWN(AP_TEST_STATUS_OFFSET), &tmp32, sizeof(tmp32));
102 
103  Timestamp system_time;
104  Timestamp device_time;
105  uint32_t local_clock, nominal_clock_rate;
106  port->getDeviceTime(system_time, device_time, local_clock, nominal_clock_rate);
107  tmp64 = PLAT_htonll(TIMESTAMP_TO_NS(system_time));
108  memcpy(buf_ptr + AP_TEST_STATUS_MESSAGE_TIMESTAMP(AP_TEST_STATUS_OFFSET), &tmp64, sizeof(tmp64));
109 
110  BIT_D2BHTONB(buf_ptr + AP_TEST_STATUS_STATION_STATE(AP_TEST_STATUS_OFFSET), (uint8_t)port->getStationState(), 0);
111 
112  port->sendGeneralPort(AVTP_ETHERTYPE, buf_t, messageLength, MCAST_TEST_STATUS, NULL);
113 
114  return;
115 }
