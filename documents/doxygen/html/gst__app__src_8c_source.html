46  /* Function to read data*/
47 static gboolean read_data(gst_app_t *app)
48 {
49  GstBuffer *buffer;
50  guint8 *ptr;
51  gint size;
52  GstFlowReturn ret;
53 
54  ptr = g_malloc(BUFF_SIZE);
55  g_assert(ptr);
56 
57  size = read_data_from_queue(ptr);
58  if(size == 0) {
59  ret = gst_app_src_end_of_stream(app->src);
60  g_debug("eos returned %d at %d\n", ret, __LINE__);
61  return FALSE;
62  }
63  buffer = gst_buffer_new();
64  buffer = gst_buffer_new_wrapped(ptr, size);
65  ret = gst_app_src_push_buffer(app->src, buffer);
66 
67  if(ret != GST_FLOW_OK) {
68  g_debug("push buffer returned %d for %d bytes \n", ret, size);
69  return FALSE;
70  }
71 
72  if(size != BUFF_SIZE) {
73  ret = gst_app_src_end_of_stream(app->src);
74  g_debug("eos returned %d at %d\n", ret, __LINE__);
75  return FALSE;
76  }
77 
78  return TRUE;
79 }

84 static void start_feed (GstElement * pipeline, guint size, gst_app_t *app)
85 {
86  if (app->sourceid == 0) {
87  GST_DEBUG("\nGSTREAMER CALLBACK : 1\n");
88  send_data_status(1);
89  GST_DEBUG ("start feeding");
90  app->sourceid = g_idle_add ((GSourceFunc) read_data, app);
91  }
92 }

94  /* This callback triggers when audio appsrc has enough data and we can stop sending.
95  * We remove the idle handler from the mainloop */
96 static void stop_feed (GstElement * pipeline, gst_app_t *app)
97 {
98  if (app->sourceid != 0) {
99  send_data_status(0);
100  GST_DEBUG("\nGSTREAMER CALLBACK : 0\n");
101  GST_DEBUG ("stop feeding");
102  g_source_remove (app->sourceid);
103  app->sourceid = 0;
104  }
105 }

107  /* Function to Parse Errors */
108 static gboolean bus_callback(GstBus *bus, GstMessage *message, gpointer *ptr)
109 {
110  gst_app_t *app = (gst_app_t*)ptr;
111 
112  switch(GST_MESSAGE_TYPE(message))
113  {
114  case GST_MESSAGE_ERROR:
115  {
116  gchar *debug;
117  GError *err;
118 
119  gst_message_parse_error(message, &err, &debug);
120  g_print("Error %s\n", err->message);
121  g_error_free(err);
122  g_free(debug);
123  g_main_loop_quit(app->loop);
124  }
125  break;
126 
127  case GST_MESSAGE_WARNING:
128  {
129  gchar *debug;
130  GError *err;
131  gchar *name;
132 
133  gst_message_parse_warning(message, &err, &debug);
134  g_print("Warning %s\nDebug %s\n", err->message, debug);
135 
136  name = GST_MESSAGE_SRC_NAME(message);
137 
138  g_print("Name of src %s\n", name ? name : "nil");
139  g_error_free(err);
140  g_free(debug);
141  }
142  break;
143 
144  case GST_MESSAGE_EOS:
145  g_print("End of stream\n");
146  g_main_loop_quit(app->loop);
147  flag_exit_app(1);
148  break;
149 
150  case GST_MESSAGE_STATE_CHANGED:
151  break;
152 
153  default:
154  GST_DEBUG("got message %s\n", \
155  gst_message_type_get_name (GST_MESSAGE_TYPE (message)));
156  break;
157  }
158 
159  return TRUE;
160 }

162 /* Function to be executed when source is found */
163 static void found_source (GObject * object, GObject * orig, GParamSpec * pspec, gst_app_t *app)
164 {
165  /* get a handle to the appsrc */
166  g_object_get (orig, pspec->name, &app->src, NULL);
167 
168  GST_DEBUG ("got appsrc %p", app->src);
169 
170  /* configure the appsrc, we will push a buffer to appsrc when it needs more data */
171  g_signal_connect (app->src, "need-data", G_CALLBACK (start_feed), app);
172  g_signal_connect (app->src, "enough-data", G_CALLBACK (stop_feed), app);
173 }

175 int gstreamer_main(void)
176 {
177  gst_app_t *app = &gst_app;
178  GstBus *bus;
179  GstStateChangeReturn state_ret;
180 
181  /* GStreamer Initialization */
182  gst_init(NULL, NULL);
183 
184  /* Create The Elements */
185  app->src = (GstAppSrc*)gst_element_factory_make("appsrc", "mysrc");
186  app->playbin = gst_element_factory_make("playbin", "myplaybin");
187 
188  /* Assert if Element Creation failed */
189  g_assert(app->src);
190  g_assert(app->playbin);
191 
192  /* Add the bus Handler */
193  bus = gst_pipeline_get_bus(GST_PIPELINE (app->playbin));
194  gst_bus_add_watch(bus, (GstBusFunc)bus_callback, app);
195 
196  /* Set uri property of playbin */
197  g_object_set (app->playbin, "uri", "appsrc://",NULL);
198 
199  /* Connect the source to playbin when notification is arrived */
200  g_signal_connect (app->playbin, "deep-notify::source",(GCallback) found_source, app);
201 
202  /* Set GStreamer Pipeline for Playing */
203  state_ret = gst_element_set_state((GstElement*)app->playbin, GST_STATE_PLAYING);
204  g_print("set state returned %d\n", state_ret);
205 
206  app->loop = g_main_loop_new(NULL, FALSE);
207  gst_app_src_set_max_bytes(app->src, 1024 * 1024 * 120);
208  g_main_loop_run(app->loop);
209 
210  /* Set GStreamer Pipeline to NULL */
211  state_ret = gst_element_set_state((GstElement*)app->playbin, GST_STATE_NULL);
212  gst_object_unref (bus);
213  g_main_loop_unref (app->loop);
214 
215  return 0;
216 }
