32 /*
33 * MODULE SUMMARY : The AVB endpoint is the "master process".
34 * It includes the SRP and QMgr libraries which handle SRP reservations
35 * and TX queuing.
36 *
37 * The actual AVTP talker/listener work is done in a separate process.
38 * The aim of using separate processes is to (1) reduce the
39 * complexity in the central process; and (2) to allow multiple types
40 * of children for different AVTP encapsulations and data sources.
41 *
42 * Streamer processes contact the endpoint process through an IPC to
43 * declare their streams. Currently, the IPC uses unix sockets. The
44 * IPC methods are implemented in openavb_endpoint_client.c and
45 * openavb_endpoint_server.c. The streamers (talkers and listeners)
46 * are referred to as our"clients", and the endpoint process is their
47 * "server".
48 *
49 * When SRP establishes a reservation (or when a reservation goes
50 * away), the endpoint communicates the event back to the streamer
51 * process through a callback (which also uses the IPC mechanism.)
52 */

124 clientStream_t* addStream(int h, AVBStreamID_t *streamID)
125 {
126  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
127  clientStream_t *newClientStream = NULL, **lpp;
128 
129  do {
130  newClientStream = (clientStream_t *)calloc(1, sizeof(clientStream_t));
131  if(newClientStream == NULL) {
132  AVB_LOG_ERROR("addStream: Failed to malloc stream");
133  break;
134  }
135 
136  memcpy(newClientStream->streamID.addr, streamID->addr, ETH_ALEN);
137  newClientStream->streamID.uniqueID = streamID->uniqueID;
138  newClientStream->clientHandle = h;
139  newClientStream->fwmark = INVALID_FWMARK;
140 
141  if(x_streamList == NULL) {
142  x_streamList = newClientStream;
143  }else {
144  // insert at end
145  for(lpp = &x_streamList; *lpp != NULL; lpp = &(*lpp)->next) {
146  if ((*lpp)->next == NULL) {
147  (*lpp)->next = newClientStream;
148  break;
149  }
150  }
151  }
152  } while (0);
153  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
154  return newClientStream;
155 }

169 /* Find a stream in the list of streams we're handling
170  */
171 clientStream_t* findStream(AVBStreamID_t *streamID)
172 {
173  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
174  clientStream_t* ps = NULL;
175 
176  // Check for default stream MAC address, and fill it in with
177  // interface MAC so that if the talker didn't fill in
178  // the stream MAC, we use the one that the endpoint is
179  // configured to use.
180  //
181  // Listener should never pass a default MAC in,
182  // because the config code forces the listener to specify MAC in
183  // its configuration. Talker may send a default (empty) MAC in
184  // the stream ID, in which case we fill it in.
185  //
186  // TODO: This is sketchy - it would probably be better to force every
187  // client to send fully populated stream IDs. I think the reason
188  // I didn't do that is that it would cause duplicate configuration
189  // (in the talker and in the endpoint.) Perhaps the filling in of
190  // the MAC could happen in the endpoint function which calls
191  // findstream for the talker.
192  //
193  static const U8 emptyMAC[ETH_ALEN] = { 0, 0, 0, 0, 0, 0 };
194  if (memcmp(streamID->addr, emptyMAC, ETH_ALEN) == 0) {
195  memcpy(streamID->addr, x_cfg.ifmac, ETH_ALEN);
196  AVB_LOGF_DEBUG("Replaced default streamID MAC with interface MAC "ETH_FORMAT, ETH_OCTETS(streamID->addr));
197  }
198 
199  clientStream_t **lpp;
200  for(lpp = &x_streamList; *lpp != NULL; lpp = &(*lpp)->next) {
201  if (memcmp(streamID->addr, (*lpp)->streamID.addr, ETH_ALEN) == 0
202  && streamID->uniqueID == (*lpp)->streamID.uniqueID)
203  {
204  ps = *lpp;
205  break;
206  }
207  }
208  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
209  return ps;
210 }

243  // Pass to SRP
244  rc = openavbSrpDeregisterStream(&ps->streamID);

246 
247  // Remove QMgr entry for stream
248  if (ps->fwmark != INVALID_FWMARK) {
249  openavbQmgrRemoveStream(ps->fwmark);
250  ps->fwmark = INVALID_FWMARK;
251  }
252 
253  // Release MAAP address allocation
254  if (ps->hndMaap) {
255  openavbMaapRelease(ps->hndMaap);
256  ps->hndMaap = NULL;
257  }
258 
259  // Finish Shaping
260  if (ps->hndShaper) {
261  openavbShaperRelease(ps->hndShaper);
262  ps->hndShaper = NULL;
263  }
264 
265  // remove record
266  delStream(ps);
267 
268  openavbEndPtLogAllStaticStreams();
269 
270  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
271  return IS_OPENAVB_SUCCESS(rc);

299 /* SRP tells us about a listener peer (Listener Ready or Failed)
300  */
301 openavbRC strmAttachCb(void* pv,
302  openavbSrpLsnrDeclSubtype_t lsnrDecl)
303 {
304  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
305 
306  openavbRC rc = OPENAVB_FAILURE;
307  clientStream_t *ps = (clientStream_t*)pv;
308 
309  AVB_LOGF_INFO("SRP talker callback uid=%d: lsnrDecl=%x", ps->streamID.uniqueID, lsnrDecl);
310 
311  if (lsnrDecl == openavbSrp_LDSt_Ready
312  || lsnrDecl == openavbSrp_LDSt_Ready_Failed)
313  {
314  // Somebody is listening - get ready to stream
315 
316  if (ps->fwmark != INVALID_FWMARK) {
317  AVB_LOG_DEBUG("attach callback: already setup queues");
318  rc = OPENAVB_SUCCESS;
319  }
320  else {
321  AVB_LOG_DEBUG("Attach callback: setting up queues for streaming");
322 
323  rc = openavbSrpGetClassParams(ps->srClass, &ps->priority, &ps->vlanID, &ps->classRate);
324  if (IS_OPENAVB_SUCCESS(rc)) {
325  ps->fwmark = openavbQmgrAddStream(ps->srClass, ps->classRate, ps->tSpec.maxIntervalFrames, ps->tSpec.maxFrameSize);
326  if (ps->fwmark == INVALID_FWMARK) {
327  AVB_LOG_ERROR("Error in attach callback: unable to setup stream queues");
328  rc = OPENAVB_FAILURE;
329  }
330  else {
331  rc = OPENAVB_SUCCESS;
332  }
333  }
334  else {
335  AVB_LOG_ERROR("Error in attach callback: unable to get class params");
336  rc = OPENAVB_FAILURE;
337  }
338  }
339  }
340  else {
341  // Nobody listening
342  if (ps->fwmark != INVALID_FWMARK) {
343  AVB_LOG_DEBUG("Attach callback: tearing down queues");
344  openavbQmgrRemoveStream(ps->fwmark);
345  ps->fwmark = INVALID_FWMARK;
346  }
347  rc = OPENAVB_SUCCESS;
348  }
349 
350  if (IS_OPENAVB_SUCCESS(rc)) {
351 
352  openavbEptSrvrNotifyTlkrOfSrpCb(ps->clientHandle,
353  &ps->streamID,
354  x_cfg.ifname,
355  ps->destAddr,
356  lsnrDecl,
357  ps->srClass,
358  ps->classRate,
359  ps->vlanID,
360  ps->priority,
361  ps->fwmark);
362  rc = OPENAVB_SUCCESS;
363  }
364 
365  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
366  return rc;
367 }

399 /*************************************************************
400  * MAAP Restart - our destination MAC has been changed.
401  *
402  * This is a clunky way to handle it - but it works, and this code
403  * won't be called often. (MAAP sends probes before settling on an
404  * address, so only a buggy or malicious peer should send us down this
405  * path.)
406  *
407  * A better way to handle this would require SRP and the
408  * talkers/listeners to look at destination addresses (in addition
409  * to StreamID and talker/listener declaration) and explicitly handle
410  * destination address changes.
411  */
412 static void maapRestartCallback(void* handle, struct ether_addr *addr)
413 {
414  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
415 
416  static clientStream_t* ps;
417  openavbRC rc;
418 
419  ps = findStreamMaap(handle);
420  AVB_LOGF_STATUS("MAAP restart callback: handle=%p, stream=%p", handle, ps);
421 
422  if (ps) {
423  memcpy(ps->destAddr, addr->ether_addr_octet, ETH_ALEN);
424 
425  // Pretend that our listeners went away
426  strmAttachCb(ps, (openavbSrpLsnrDeclSubtype_t)0);
427 
428  if(!x_cfg.noSrp) {
429  // Remove the old registration with SRP
430  openavbSrpDeregisterStream(&ps->streamID);
431 
432  // Re-register with the new address
433  rc = openavbSrpRegisterStream((void*)ps,
434  &ps->streamID,
435  ps->destAddr,
436  &ps->tSpec,
437  ps->srClass,
438  ps->srRank,
439  ps->latency);
440  } else {
441  rc = OPENAVB_SUCCESS;
442  }
443 
444  if (!IS_OPENAVB_SUCCESS(rc)) {
445  AVB_LOG_ERROR("MAAP restart: failed to re-register talker stream");
446  }
447  }
448  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
449 }

452 int avbEndpointLoop(void)
453 {
454  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
455 
456  int retVal = -1;
457  openavbRC rc = OPENAVB_SUCCESS;
458  do {
459 
460  if (!x_cfg.bypassAsCapableCheck && (startPTP() < 0)) {
461  // make sure ptp, a separate process, starts and is using the same interface as endpoint
462  AVB_LOG_ERROR("PTP failed to start - Exiting");
463  break;
464  } else if(x_cfg.bypassAsCapableCheck) {
465  AVB_LOG_WARNING(" ");
466  AVB_LOG_WARNING("Configuration 'gptp_asCapable_not_required = 1' is set.");
467  AVB_LOG_WARNING("This configuration bypasses the requirement for gPTP");
468  AVB_LOG_WARNING("and openavb_gptp is not started automatically.");
469  AVB_LOG_WARNING("An appropriate ptp MUST be started separately.");
470  AVB_LOG_WARNING("Any network which does not use ptp to synchronize time");
471  AVB_LOG_WARNING("on each and every network device is NOT an AVB network.");
472  AVB_LOG_WARNING("Such a network WILL NOT FUNCTION PROPERLY.");
473  AVB_LOG_WARNING(" ");
474  }
475 
476  x_streamList = NULL;
477 
478  if (!openavbQmgrInitialize(x_cfg.fqtss_mode, x_cfg.ifindex, x_cfg.ifname, x_cfg.mtu, x_cfg.link_kbit, x_cfg.nsr_kbit)) {
479  AVB_LOG_ERROR("Failed to initialize QMgr");
480  break;
481  }
482 
483  if (!openavbMaapInitialize(x_cfg.ifname, x_cfg.maapPort, &(x_cfg.maap_preferred), maapRestartCallback)) {
484  AVB_LOG_ERROR("Failed to initialize MAAP");
485  openavbQmgrFinalize();
486  break;
487  }
488 
489  if (!openavbShaperInitialize(x_cfg.ifname, x_cfg.shaperPort)) {
490  AVB_LOG_ERROR("Failed to initialize Shaper");
491  openavbMaapFinalize();
492  openavbQmgrFinalize();
493  break;

495

497  // Initialize SRP
498  rc = openavbSrpInitialize(strmAttachCb, strmRegCb, x_cfg.ifname, x_cfg.link_kbit, x_cfg.bypassAsCapableCheck);
499  } else {
500  rc = OPENAVB_SUCCESS;
501  AVB_LOG_WARNING(" ");
502  AVB_LOG_WARNING("Configuration 'preconfigured = 1' is set.");
503  AVB_LOG_WARNING("SRP is disabled. Streams MUST be configured manually");
504  AVB_LOG_WARNING("on each and every device in the network, without exception.");
505  AVB_LOG_WARNING("AN AVB NETWORK WILL NOT FUNCTION AS EXPECTED UNLESS ALL");
506  AVB_LOG_WARNING("STREAMS ARE PROPERLY CONFIGURED ON ALL NETWORK DEVICES.");
507  AVB_LOG_WARNING(" ");
508  }
509 
510  if (!IS_OPENAVB_SUCCESS(rc)) {
511  AVB_LOG_ERROR("Failed to initialize SRP");
512  openavbShaperFinalize();
513  openavbMaapFinalize();
514  openavbQmgrFinalize();
515  break;
516  }
517 
518  if (openavbEndpointServerOpen()) {
519 
520  retVal = 0;
521 
522  while (endpointRunning) {
523  openavbEptSrvrService();
524  }
525 
526  openavbEndpointServerClose();
527  }
528 
529  if(!x_cfg.noSrp) {
530  // Shutdown SRP
531  openavbSrpShutdown();
532  }
533 
534  openavbShaperFinalize();
535  openavbMaapFinalize();
536  openavbQmgrFinalize();
537 
538  } while (0);
539 
540  if (!x_cfg.bypassAsCapableCheck && (stopPTP() < 0)) {
541  AVB_LOG_WARNING("Failed to execute PTP stop command: killall -s SIGINT openavb_gptp");
542  }
543 
544  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
545  return retVal;
546 }
