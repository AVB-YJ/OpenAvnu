103 static bool setupHWQueue(int nClass, unsigned classBytesPerSec)
104 {
105  int err = 0;
106 #if (AVB_FEATURE_IGB)
107  U32 class_a_bytes_per_sec, class_b_bytes_per_sec;
108 #endif
109 
110  AVB_TRACE_ENTRY(AVB_TRACE_QUEUE_MANAGER);
111 
112 #if (AVB_FEATURE_IGB)
113  if (nClass == SR_CLASS_A) {
114  class_a_bytes_per_sec = classBytesPerSec;
115  class_b_bytes_per_sec = qmgr_classes[SR_CLASS_B].classBytesPerSec;
116  } else {
117  class_a_bytes_per_sec = qmgr_classes[SR_CLASS_A].classBytesPerSec;
118  class_b_bytes_per_sec = classBytesPerSec;
119  }
120  err = igb_set_class_bandwidth2(qdisc_data.igb_dev, class_a_bytes_per_sec, class_b_bytes_per_sec);
121  if (err)
122  AVB_LOGF_ERROR("Adding stream; igb_set_class_bandwidth failed: %s", strerror(err));
123 #endif
124 
125  AVB_TRACE_EXIT(AVB_TRACE_QUEUE_MANAGER);
126  return !err;
127 }

129 /* Add a stream.
130  *
131  * nClass = index of class (A, B, etc.)
132  * classRate = class observation interval (8000, 4000, etc.)
133  * maxIntervalFrames = frames per interval
134  * maxFrameSize = max size of frames
135  *
136  */
137 U16 openavbQmgrAddStream(SRClassIdx_t nClass, unsigned classRate, unsigned maxIntervalFrames, unsigned maxFrameSize)
138 {
139  unsigned fullFrameSize = maxFrameSize + OPENAVB_AVTP_ETHER_FRAME_OVERHEAD + 1;
140  unsigned long streamBytesPerSec = fullFrameSize * maxIntervalFrames * classRate;
141  int idx, nStream;
142  U16 fwmark = INVALID_FWMARK;
143 
144  AVB_TRACE_ENTRY(AVB_TRACE_QUEUE_MANAGER);
145  LOCK();
146 
147  AVB_LOGF_DEBUG("Adding stream; class=%d, rate=%u frames=%u, size=%u(%u), bytes/sec=%lu",
148  nClass, classRate, maxIntervalFrames, maxFrameSize,
149  fullFrameSize, streamBytesPerSec);
150 
151  if ((int)nClass < 0 || nClass >= MAX_AVB_SR_CLASSES || streamBytesPerSec == 0) {
152  AVB_LOG_ERROR("Adding stream; invalid argument");
153  }
154  else {
155  // Find an unused stream in the appropriate SR class
156  for (nStream = 0, idx = nClass * MAX_AVB_STREAMS_PER_CLASS; nStream < MAX_AVB_STREAMS_PER_CLASS; nStream++, idx++) {
157  if (0 == qmgr_streams[idx].streamBytesPerSec) {
158  fwmark = TC_AVB_MARK(nClass, nStream);
159  break;
160  }
161  }
162 
163  if (fwmark == INVALID_FWMARK) {
164  AVB_LOGF_ERROR("Adding stream; too many streams in class %d", nClass);
165  } else {
166 
167  if (qdisc_data.mode != AVB_SHAPER_DISABLED) {
168  if (!setupHWQueue(nClass, qmgr_classes[nClass].classBytesPerSec + streamBytesPerSec)) {
169  fwmark = INVALID_FWMARK;
170  }
171  }
172 
173  if (fwmark != INVALID_FWMARK) {
174  // good to go - update stream
175  qmgr_streams[idx].streamBytesPerSec = streamBytesPerSec;
176  qmgr_streams[idx].classRate = classRate;
177  qmgr_streams[idx].maxIntervalFrames = maxIntervalFrames;
178  qmgr_streams[idx].maxFrameSize = maxFrameSize;
179  // and class
180  qmgr_classes[nClass].classBytesPerSec += streamBytesPerSec;
181 
182  AVB_LOGF_DEBUG("Added stream; classBPS=%u, streamBPS=%u", qmgr_classes[nClass].classBytesPerSec, qmgr_streams[idx].streamBytesPerSec);
183  }
184  }
185  }
186 
187  UNLOCK();
188  AVB_TRACE_EXIT(AVB_TRACE_QUEUE_MANAGER);
189  return fwmark;
190 }

193 void openavbQmgrRemoveStream(U16 fwmark)
194 {
195  AVB_TRACE_ENTRY(AVB_TRACE_QUEUE_MANAGER);
196 
197  if (fwmark == INVALID_FWMARK) {
198  AVB_LOG_ERROR("Removing stream; invalid argument");
199  AVB_TRACE_EXIT(AVB_TRACE_QUEUE_MANAGER);
200  return;
201  }
202 
203  int nClass = TC_AVB_MARK_CLASS(fwmark);
204  int nStream = TC_AVB_MARK_STREAM(fwmark);
205  int idx = nClass * MAX_AVB_STREAMS_PER_CLASS + nStream;
206 
207  LOCK();
208 
209  if (nStream < 0
210  || nStream >= MAX_AVB_STREAMS
211  || nClass < 0
212  || nClass >= MAX_AVB_SR_CLASSES
213  || qmgr_streams[idx].streamBytesPerSec == 0)
214  {
215  // something is wrong
216  AVB_LOG_ERROR("Removing stream; invalid argument or data");
217  }
218  else {
219  if (qdisc_data.mode != AVB_SHAPER_DISABLED) {
220  setupHWQueue(nClass, qmgr_classes[nClass].classBytesPerSec - qmgr_streams[idx].streamBytesPerSec);
221  }
222 
223  // update class
224  qmgr_classes[nClass].classBytesPerSec -= qmgr_streams[idx].streamBytesPerSec;
225  AVB_LOGF_DEBUG("Removed stream; classBPS=%u, streamBPS=%u", qmgr_classes[nClass].classBytesPerSec, qmgr_streams[idx].streamBytesPerSec);
226  // and stream
227  memset(&qmgr_streams[idx], 0, sizeof(qmgrStream_t));
228  }
229 
230  UNLOCK();
231  AVB_TRACE_EXIT(AVB_TRACE_QUEUE_MANAGER);
232 }

234 bool openavbQmgrInitialize(int mode, int ifindex, const char* ifname, unsigned mtu, unsigned link_kbit, unsigned nsr_kbit)
235 {
236  AVB_TRACE_ENTRY(AVB_TRACE_QUEUE_MANAGER);
237  bool ret = FALSE;
238 
239 // if (!ifname || mtu == 0 || link_kbit == 0 || nsr_kbit == 0) {
240 // AVB_LOG_ERROR("Initializing QMgr; invalid argument");
241 // return FALSE;
242 // }
243 
244  LOCK();
245 
246  if (qdisc_data.ref++ > 0) {
247  AVB_LOG_DEBUG("Already initialized");
248 
249  UNLOCK();
250  AVB_TRACE_EXIT(AVB_TRACE_QUEUE_MANAGER);
251  return TRUE;
252  }
253 
254  if (mode >= 0)
255  qdisc_data.mode = mode;
256  else
257  qdisc_data.mode = AVB_DEFAULT_QDISC_MODE;
258 
259 
260  AVB_LOGF_DEBUG("Initializing QMgr; mode=%d, idx=%d, mtu=%u, link_kbit=%u, nsr_kbit=%u",
261  qdisc_data.mode, ifindex, mtu, link_kbit, nsr_kbit);
262 
263 #if (AVB_FEATURE_IGB)
264  if ( qdisc_data.mode != AVB_SHAPER_DISABLED
265  && (qdisc_data.igb_dev = igbAcquireDevice()) == 0)
266  {
267  AVB_LOG_ERROR("Initializing QMgr; unable to acquire igb device");
268  }
269  else
270 #endif
271  {
272  // Initialize data for classes and streams
273  memset(qmgr_classes, 0, sizeof(qmgr_classes));
274  memset(qmgr_streams, 0, sizeof(qmgr_streams));
275 
276  // Save the configuration
277  if (ifname)
278  strncpy(qdisc_data.ifname, ifname, sizeof(qdisc_data.ifname) - 1);
279  qdisc_data.ifindex = ifindex;
280  qdisc_data.linkKbit = link_kbit;
281  qdisc_data.linkMTU = mtu;
282  qdisc_data.nsrKbit = nsr_kbit;
283 
284  if (qdisc_data.mode == AVB_SHAPER_DISABLED) {
285  ret = TRUE;
286  } else {
287  ret = TRUE;
288  // igb device aquired, nothing more to do
289  }
290  }
291 
292  UNLOCK();
293  AVB_TRACE_EXIT(AVB_TRACE_QUEUE_MANAGER);
294  return ret;
295 }

297 void openavbQmgrFinalize(void)
298 {
299  AVB_TRACE_ENTRY(AVB_TRACE_QUEUE_MANAGER);
300  LOCK();
301 
302  if (--qdisc_data.ref == 0 && qdisc_data.mode != AVB_SHAPER_DISABLED) {
303  int nClass;
304  for (nClass = SR_CLASS_A; nClass < MAX_AVB_SR_CLASSES; nClass++) {
305  int nStream, idx;
306  for (nStream = 0, idx = nClass * MAX_AVB_STREAMS_PER_CLASS; nStream < MAX_AVB_STREAMS_PER_CLASS; nStream++, idx++) {
307  if (qmgr_streams[idx].streamBytesPerSec) {
308  U16 fwmark = TC_AVB_MARK(nClass, nStream);
309  openavbQmgrRemoveStream(fwmark);
310  }
311  }
312  }
313 
314 #if (AVB_FEATURE_IGB)
315  igbReleaseDevice(qdisc_data.igb_dev);
316  qdisc_data.igb_dev = NULL;
317 #endif
318  }
319 
320  UNLOCK();
321  AVB_TRACE_EXIT(AVB_TRACE_QUEUE_MANAGER);
322 }
