88 // Each configuration name value pair for this mapping will result in this callback being called.
89 void openavbIntfEchoCfgCB(media_q_t *pMediaQ, const char *name, const char *value)
90 {
91  AVB_TRACE_ENTRY(AVB_TRACE_INTF);
92  if (pMediaQ) {
93  char *pEnd;
94  long tmp;
95 
96  pvt_data_t *pPvtData = pMediaQ->pPvtIntfInfo;
97  if (!pPvtData) {
98  AVB_LOG_ERROR("Private interface module data not allocated.");
99  return;
100  }
101 
102  if (strcmp(name, "intf_nv_echo_string") == 0) {
103  if (pPvtData->pEchoString)
104  free(pPvtData->pEchoString);
105  pPvtData->pEchoString = strdup(value);
106  if (pPvtData->pEchoString) {
107  pPvtData->echoStringLen = strlen(pPvtData->pEchoString);
108  }
109  }
110  else if (strcmp(name, "intf_nv_echo_string_repeat") == 0) {
111  pPvtData->echoStringRepeat = strtol(value, &pEnd, 10);
112 
113  // Repeat the string if needed
114  if (pPvtData->echoStringRepeat > 1 && pPvtData->pEchoString) {
115  char *pEchoStringRepeat = calloc(1, (pPvtData->echoStringLen * pPvtData->echoStringRepeat) + 1);
116 
117  int i1;
118  for (i1 = 0; i1 < pPvtData->echoStringRepeat; i1++) {
119  strcat(pEchoStringRepeat, pPvtData->pEchoString);
120  }
121 
122  free(pPvtData->pEchoString);
123  pPvtData->pEchoString = pEchoStringRepeat;
124  pPvtData->echoStringLen = strlen(pPvtData->pEchoString);
125  }
126  }
127  else if (strcmp(name, "intf_nv_echo_increment") == 0) {
128  tmp = strtol(value, &pEnd, 10);
129  if (*pEnd == '\0' && tmp == 1) {
130  pPvtData->increment = (tmp == 1);
131  }
132  }
133  else if (strcmp(name, "intf_nv_tx_local_echo") == 0) {
134  tmp = strtol(value, &pEnd, 10);
135  if (*pEnd == '\0' && tmp == 1) {
136  pPvtData->txLocalEcho = (tmp == 1);
137  }
138  }
139  else if (strcmp(name, "intf_nv_echo_no_newline") == 0) {
140  tmp = strtol(value, &pEnd, 10);
141  if (*pEnd == '\0' && tmp == 1) {
142  pPvtData->noNewline = (tmp == 1);
143  }
144  }
145  else if (strcmp(name, "intf_nv_ignore_timestamp") == 0) {
146  tmp = strtol(value, &pEnd, 10);
147  if (*pEnd == '\0' && tmp == 1) {
148  pPvtData->ignoreTimestamp = (tmp == 1);
149  }
150  }
151  }
152 
153  AVB_TRACE_EXIT(AVB_TRACE_INTF);
154 }

184 // This callback will be called for each AVB transmit interval. Commonly this will be
185 // 4000 or 8000 times per second.
186 bool openavbIntfEchoTxCB(media_q_t *pMediaQ)
187 {
188  AVB_TRACE_ENTRY(AVB_TRACE_INTF_DETAIL);
189  if (pMediaQ) {
190  pvt_data_t *pPvtData = pMediaQ->pPvtIntfInfo;
191  if (!pPvtData) {
192  AVB_LOG_ERROR("Private interface module data not allocated.");
193  return FALSE;
194  }
195 
196  media_q_item_t *pMediaQItem = openavbMediaQHeadLock(pMediaQ);
197  if (pMediaQItem) {
198  if (pMediaQItem->itemSize >= (pPvtData->increment ? pPvtData->echoStringLen + 16 : pPvtData->echoStringLen)) {
199  if (pPvtData->increment) {
200  int len = sprintf(pMediaQItem->pPubData, "%s %u", pPvtData->pEchoString, pPvtData->Counter++);
201  pMediaQItem->dataLen = len;
202  }
203  else {
204  memcpy(pMediaQItem->pPubData, pPvtData->pEchoString, pPvtData->echoStringLen);
205  pMediaQItem->dataLen = pPvtData->echoStringLen;
206  }
207  }
208  else {
209  memcpy(pMediaQItem->pPubData, pPvtData->pEchoString, pMediaQItem->itemSize);
210  pMediaQItem->dataLen = pMediaQItem->itemSize;
211  }
212 
213  if (pPvtData->txLocalEcho) {
214  if (pPvtData->noNewline)
215  printf("%s ", (char *)pMediaQItem->pPubData);
216  else
217  printf("%s\n\r", (char *)pMediaQItem->pPubData);
218  }
219 
220  if (pPvtData->fixedTimestampEnabled) {
221  openavbAvtpTimeAddUSec(&pPvtData->walltime, pPvtData->tsIncrement);
222  *pMediaQItem->pAvtpTime = pPvtData->walltime;
223  } else {
224  openavbAvtpTimeSetToWallTime(pMediaQItem->pAvtpTime);
225  }
226  openavbMediaQHeadPush(pMediaQ);
227  AVB_TRACE_EXIT(AVB_TRACE_INTF_DETAIL);
228  return TRUE;
229  }
230  else {
231  AVB_TRACE_EXIT(AVB_TRACE_INTF_DETAIL);
232  return FALSE; // Media queue full
233  }
234  }
235  AVB_TRACE_EXIT(AVB_TRACE_INTF_DETAIL);
236  return FALSE;
237 }

258 // This callback is called when acting as a listener.
259 bool openavbIntfEchoRxCB(media_q_t *pMediaQ)
260 {
261  AVB_TRACE_ENTRY(AVB_TRACE_INTF_DETAIL);
262  if (pMediaQ) {
263  bool moreItems = TRUE;
264  pvt_data_t *pPvtData = pMediaQ->pPvtIntfInfo;
265  if (!pPvtData) {
266  AVB_LOG_ERROR("Private interface module data not allocated.");
267  return FALSE;
268  }
269 
270  media_q_item_t *pMediaQItem = NULL;
271  while (moreItems) {
272  pMediaQItem = openavbMediaQTailLock(pMediaQ, pPvtData->ignoreTimestamp);
273  if (pMediaQItem) {
274  if (pMediaQItem->dataLen) {
275  if (pMediaQItem->itemSize > pMediaQItem->dataLen) {
276  *((char*)pMediaQItem->pPubData + pMediaQItem->dataLen) = 0; //End string with 0
277  }
278  if (pPvtData->noNewline) {
279  printf("%s ", (char *)pMediaQItem->pPubData);
280  }
281  else {
282  if (pPvtData->increment) {
283  printf("%s : %u\n\r", (char *)pMediaQItem->pPubData, pPvtData->Counter++);
284  }
285  else {
286  printf("%s\n\r", (char *)pMediaQItem->pPubData);
287  }
288  }
289  }
290  openavbMediaQTailPull(pMediaQ);
291  }
292  else {
293  moreItems = FALSE;
294  }
295  }
296  }
297  AVB_TRACE_EXIT(AVB_TRACE_INTF_DETAIL);
298  return FALSE;
299 }

347 extern bool DLL_EXPORT openavbIntfEchoInitialize(media_q_t *pMediaQ, openavb_intf_cb_t *pIntfCB)
348 {
349  AVB_TRACE_ENTRY(AVB_TRACE_INTF);

358

360 
361  pIntfCB->intf_cfg_cb = openavbIntfEchoCfgCB;
362  pIntfCB->intf_gen_init_cb = openavbIntfEchoGenInitCB;
363  pIntfCB->intf_tx_init_cb = openavbIntfEchoTxInitCB;

368  pIntfCB->intf_gen_end_cb = openavbIntfEchoGenEndCB;
369  pIntfCB->intf_enable_fixed_timestamp = openavbIntfEchoEnableFixedTimestamp;
370 
371  pPvtData->pEchoString = NULL;
372  pPvtData->echoStringRepeat = 1;
373  pPvtData->echoStringLen = 0;
374  pPvtData->increment = FALSE;
375  pPvtData->txLocalEcho = FALSE;
376  pPvtData->noNewline = FALSE;
377  pPvtData->ignoreTimestamp = FALSE;
378  pPvtData->fixedTimestampEnabled = FALSE;
379  }
380 
381  AVB_TRACE_EXIT(AVB_TRACE_INTF);
382  return TRUE;
