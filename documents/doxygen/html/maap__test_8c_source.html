98 int unpack_maap(maap_packet_t *packet, uint8_t *stream) {
99  packet->DA = be64toh(*(uint64_t *)stream) >> 16;
100  stream += 6;
101  packet->SA = be64toh(*(uint64_t *)stream) >> 16;
102  stream += 6;
103  packet->Ethertype = be16toh(*(uint16_t *)stream);
104  stream += 2;
105  packet->subtype = *stream;
106  stream++;
107  packet->SV = (*stream & 0x80) >> 7;
108  packet->version = (*stream & 0x70) >> 4;
109  packet->message_type = *stream & 0x0f;
110  stream++;
111  packet->status = (*stream & 0xf8) >> 3;
112  packet->control_data_length = be16toh(*(uint16_t *)stream) & 0x07ff;
113  stream += 2;
114  packet->stream_id = be64toh(*(uint64_t *)stream);
115  stream += 8;
116  packet->requested_start_address = be64toh(*(uint64_t *)stream) >> 16;
117  stream += 6;
118  packet->requested_count = be16toh(*(uint16_t *)stream);
119  stream += 2;
120  packet->start_address = be64toh(*(uint64_t *)stream) >> 16;
121  stream += 6;
122  packet->count = be16toh(*(uint16_t *)stream);
123  return 0;
124 }

126 int pack_maap(maap_packet_t *packet, uint8_t *stream) {
127  *(uint64_t *)stream = htobe64(packet->DA << 16);
128  stream += 6;
129  *(uint64_t *)stream = htobe64(packet->SA << 16);
130  stream += 6;
131  *(uint16_t *)stream = htobe16(packet->Ethertype);
132  stream += 2;
133  *stream = packet->subtype;
134  stream++;
135  *stream = (packet->SV << 7) | ((packet->version & 0x07) << 4) |
136  (packet->message_type & 0x0f);
137  stream++;
138  *(uint16_t *)stream = htobe16(((packet->status & 0x001f) << 11) |
139  (packet->control_data_length & 0x07ff));
140  stream += 2;
141  *(uint64_t *)stream = htobe64(packet->stream_id);
142  stream += 8;
143  *(uint64_t *)stream = htobe64(packet->requested_start_address << 16);
144  stream += 6;
145  *(uint16_t *)stream = htobe16(packet->requested_count);
146  stream += 2;
147  *(uint64_t *)stream = htobe64(packet->start_address << 16);
148  stream += 6;
149  *(uint16_t *)stream = htobe16(packet->count);
150  return 0;
151 }

177 int bind_sock(char *device, int rawsock, int protocol) {
178  struct sockaddr_ll sll;
179  struct ifreq ifr;
180 
181  bzero(&sll, sizeof(sll));
182  bzero(&ifr, sizeof(ifr));
183 
184  /* First, get the interface index */
185  strncpy((char *)ifr.ifr_name, device, IFNAMSIZ);
186  if((ioctl(rawsock, SIOCGIFINDEX, &ifr)) == -1) {
187  printf("Error getting Interface index !\n");
188  printf("Problem interface: %s\n", device);
189  exit(-1);
190  }
191 
192  /* Bind our raw socket to this interface */
193  sll.sll_family = AF_PACKET;
194  sll.sll_ifindex = ifr.ifr_ifindex;
195  sll.sll_protocol = htons(protocol);
196 
197  /* Get the mac address */
198  if ((ioctl(rawsock, SIOCGIFHWADDR, &ifr)) == -1) {
199  printf("Error getting HW address\n");
200  exit(-1);
201  }
202  bcopy(ifr.ifr_hwaddr.sa_data, &test_state.hwaddr, 6);
203  test_state.hwaddr = be64toh(test_state.hwaddr) >> 16;
204  printf("Our MAC is %012llx\n", (unsigned long long int)test_state.hwaddr);
205 
206  if((bind(rawsock, (struct sockaddr*)&sll, sizeof(sll))) == -1) {
207  perror("Error binding raw socket to interface\n");
208  exit(-1);
209  }
210 
211  return 1;
212 }

349  break;
