56 openavbRC openavbAemDescriptorEntityToBuf(void *pVoidDescriptor, U16 bufLength, U8 *pBuf, U16 *descriptorSize)
57 {
58  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
59 
60  openavb_aem_descriptor_entity_t *pDescriptor = pVoidDescriptor;
61 
62  if (!pDescriptor || !pBuf || !descriptorSize) {
63  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_INVALID_ARGUMENT), AVB_TRACE_AEM);
64  }
65 
66  if (bufLength < OPENAVB_DESCRIPTOR_ENTITY_BASE_LENGTH) {
67  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVBAVDECC_RC_BUFFER_TOO_SMALL), AVB_TRACE_AEM);
68  }
69 
70  *descriptorSize = 0;
71 
72  openavb_aem_descriptor_entity_t *pSrc = pDescriptor;
73  U8 *pDst = pBuf;
74 
75  OCT_D2BHTONS(pDst, pSrc->descriptor_type);
76  OCT_D2BHTONS(pDst, pSrc->descriptor_index);
77  OCT_D2BMEMCP(pDst, pSrc->entity_id);
78  OCT_D2BMEMCP(pDst, pSrc->entity_model_id);
79  OCT_D2BHTONL(pDst, pSrc->entity_capabilities);
80  OCT_D2BHTONS(pDst, pSrc->talker_stream_sources);
81  OCT_D2BHTONS(pDst, pSrc->talker_capabilities);
82  OCT_D2BHTONS(pDst, pSrc->listener_stream_sinks);
83  OCT_D2BHTONS(pDst, pSrc->listener_capabilities);
84  OCT_D2BHTONL(pDst, pSrc->controller_capabilities);
85  OCT_D2BHTONL(pDst, pSrc->available_index);
86  OCT_D2BMEMCP(pDst, pSrc->association_id);
87  OCT_D2BMEMCP(pDst, pSrc->entity_name);
88  BIT_D2BHTONS(pDst, pSrc->vendor_name_string.offset, 3, 0);
89  BIT_D2BHTONS(pDst, pSrc->vendor_name_string.index, 0, 2);
90  BIT_D2BHTONS(pDst, pSrc->model_name_string.offset, 3, 0);
91  BIT_D2BHTONS(pDst, pSrc->model_name_string.index, 0, 2);
92  OCT_D2BMEMCP(pDst, pSrc->firmware_version);
93  OCT_D2BMEMCP(pDst, pSrc->group_name);
94  OCT_D2BMEMCP(pDst, pSrc->serial_number);
95  OCT_D2BHTONS(pDst, pSrc->configurations_count);
96  OCT_D2BHTONS(pDst, pSrc->current_configuration);
97 
98  *descriptorSize = pDst - pBuf;
99 
100  AVB_RC_TRACE_RET(OPENAVB_AVDECC_SUCCESS, AVB_TRACE_AEM);
101 }

164 extern DLL_EXPORT openavb_aem_descriptor_entity_t *openavbAemDescriptorEntityNew()
165 {
166  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
167 
168  openavb_aem_descriptor_entity_t *pDescriptor;
169 
170  pDescriptor = malloc(sizeof(*pDescriptor));
171 
172  if (!pDescriptor) {
173  AVB_RC_LOG(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_OUT_OF_MEMORY));
174  AVB_TRACE_EXIT(AVB_TRACE_AEM);
175  return NULL;
176  }
177  memset(pDescriptor, 0, sizeof(openavb_aem_descriptor_entity_t));
178 
179  pDescriptor->descriptorPvtPtr = malloc(sizeof(*pDescriptor->descriptorPvtPtr));
180  if (!pDescriptor->descriptorPvtPtr) {
181  free(pDescriptor);
182  pDescriptor = NULL;
183  AVB_RC_LOG(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_OUT_OF_MEMORY));
184  AVB_TRACE_EXIT(AVB_TRACE_AEM);
185  return NULL;
186  }
187  memset(pDescriptor->descriptorPvtPtr, 0, sizeof(*pDescriptor->descriptorPvtPtr));
188 
189  pDescriptor->descriptorPvtPtr->size = sizeof(openavb_aem_descriptor_entity_t);
190  pDescriptor->descriptorPvtPtr->bTopLevel = TRUE;
191  pDescriptor->descriptorPvtPtr->toBuf = openavbAemDescriptorEntityToBuf;
192  pDescriptor->descriptorPvtPtr->fromBuf = openavbAemDescriptorEntityFromBuf;
193  pDescriptor->descriptorPvtPtr->update = openavbAemDescriptorEntityUpdate;
194 
195  pDescriptor->descriptor_type = OPENAVB_AEM_DESCRIPTOR_ENTITY;
196  pDescriptor->descriptor_index = 0;
197 
198  // Default to no localized strings.
199  pDescriptor->vendor_name_string.offset = OPENAVB_AEM_NO_STRING_OFFSET;
200  pDescriptor->vendor_name_string.index = OPENAVB_AEM_NO_STRING_INDEX;
201  pDescriptor->model_name_string.offset = OPENAVB_AEM_NO_STRING_OFFSET;
202  pDescriptor->model_name_string.index = OPENAVB_AEM_NO_STRING_INDEX;
203 
204  AVB_TRACE_EXIT(AVB_TRACE_AEM);
205  return pDescriptor;
206 }

208 extern DLL_EXPORT bool openavbAemDescriptorEntitySet_entity_id(openavb_aem_descriptor_entity_t *pDescriptor, char *aMacAddr, U8 *nMacAddr, U16 id)
209 {
210  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
211 
212  if (!pDescriptor) {
213  AVB_RC_LOG(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_INVALID_ARGUMENT));
214  AVB_TRACE_EXIT(AVB_TRACE_AEM);
215  return FALSE;
216  }
217 
218  struct ether_addr macAddrBuffer;
219  if (aMacAddr) {
220  // If an ascii mac address is passed in parse and use it.
221  memset(macAddrBuffer.ether_addr_octet, 0, sizeof(macAddrBuffer));
222 
223  if (!ether_aton_r(aMacAddr, &macAddrBuffer)) {
224  AVB_RC_LOG(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVBAVDECC_RC_PARSING_MAC_ADDRESS));
225  AVB_TRACE_EXIT(AVB_TRACE_AEM);
226  return FALSE;
227  }
228  memcpy(pDescriptor->entity_id, macAddrBuffer.ether_addr_octet, 3);
229  memcpy(pDescriptor->entity_id + 3 + 2, macAddrBuffer.ether_addr_octet + 3, 3);
230  }
231  else if (nMacAddr) {
232  // If a network mac address is passed in use it.
233  memcpy(pDescriptor->entity_id, nMacAddr, 3);
234  memcpy(pDescriptor->entity_id + 3 + 2, nMacAddr + 3, 3);
235  }
236  else {
237  // If no mac address is passed in obtain it from the stack.
238  memcpy(pDescriptor->entity_id, gAvdeccCfg.ifmac, 3);
239  memcpy(pDescriptor->entity_id + 3 + 2, gAvdeccCfg.ifmac + 3, 3);
240  }
241 
242  U16 tmpU16 = htons(id);
243  memcpy(pDescriptor->entity_id + 3, &tmpU16, sizeof(tmpU16));
244 
245  AVB_TRACE_EXIT(AVB_TRACE_AEM);
246  return TRUE;
247 }
