114 U32 openavbTimestampEvalTimestamp(openavb_timestamp_eval_t tsEval, U32 ts)
115 {
116  U32 tsRet = ts;
117 
118  if (tsEval) {
119  // Determine the Jitter
120  if (tsEval->tsPrev > ts) {
121  tsEval->tsInterval = (((U32)-1) - tsEval->tsPrev) + ts;
122  }
123  else {
124  tsEval->tsInterval = ts - tsEval->tsPrev;
125  }
126 
127  // Save real ts for not interval
128  tsEval->tsPrev = ts;
129 
130  // Increment the main timestamp counter
131  tsEval->tsCnt++;
132 
133  // Accumulate totals
134  if (!tsEval->started) {
135  // First timestamp interval
136  tsEval->tsTtlCalc = 0;
137  tsEval->tsTtlReal = 0;
138  tsEval->started = TRUE;
139  }
140  else {
141  // All but first timestamp
142  tsEval->tsTtlCalc += tsEval->tsRateInterval;
143  tsEval->tsTtlReal += tsEval->tsInterval;
144 
145  tsEval->tsJitter = abs(tsEval->tsRateInterval - tsEval->tsInterval);
146  if (tsEval->tsJitter > tsEval->tsMaxJitter) {
147  tsEval->tsMaxJitter = tsEval->tsJitter;
148  }
149  tsEval->tsAccumJitter += tsEval->tsJitter;
150 
151  tsEval->tsDrift = abs(tsEval->tsTtlCalc - tsEval->tsTtlReal);
152 
153  // Reporting
154  if (tsEval->reportInterval) {
155  if ((tsEval->tsCnt % tsEval->reportInterval) == 0) {
156  //printf("Jitter:%9u AvgJitter:%9u MaxJitter:%9u Drift:%9u\n", tsEval->tsJitter, (U32)(tsEval->tsAccumJitter / tsEval->tsCnt), tsEval->tsMaxJitter, tsEval->tsDrift);
157  openavbPrintbufPrintf(tsEval->printbuf, "Jitter:%9u AvgJitter:%9u MaxJitter:%9u Drift:%9u\n", tsEval->tsJitter, (U32)(tsEval->tsAccumJitter / tsEval->tsCnt), tsEval->tsMaxJitter, tsEval->tsDrift);
158  }
159  }
160  }
161  }
162 
163  return tsRet;
164 }
