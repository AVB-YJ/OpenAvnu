35 // Macros to map stream/class into 16-bit fwmark
36 // Bottom 8 bits are used for stream index (allows 32 streams/class)
37 // Upper 8 bits are used for class index
38 //
39 // The fwmark is attached to the socket, and the kernel uses it to
40 // steer AVTP frames into the correct queues.
41 //
42 // If we ever want more than 256 classes (won't happen) or 256 streams
43 // per class (very unlikely) we'll need to shift the boundary between
44 // the bits used for the class idex and those used for stream idx.
45 //
46 // If the combination of the two needs more than 16 bits, we'll have
47 // to use something other than the FWMARK to communicate with the kernel.
48 //
49 #define TC_AVB_CLASS_SHIFT 8
50 #define TC_AVB_STREAM_MASK ((1 << TC_AVB_CLASS_SHIFT) - 1)
51 #define TC_AVB_MARK_CLASS(M) (((M) >> TC_AVB_CLASS_SHIFT) - 1)
52 #define TC_AVB_MARK_STREAM(M) ((M) & TC_AVB_STREAM_MASK)
53 #define TC_AVB_MARK(C,S) (((C + 1) << TC_AVB_CLASS_SHIFT) | (TC_AVB_STREAM_MASK & (S)))

61 // Modes for qdisc shaping
62 //
63 typedef enum {
64  // Disable FQTSS
65  AVB_SHAPER_DISABLED,
66  // Drop all frames
67  AVB_SHAPER_DROP_ALL,
68  // Treat everything as non-SR traffic
69  AVB_SHAPER_ALL_NONSR,
70  // Drop all AVB stream frames
71  AVB_SHAPER_DROP_SR,
72  // Shaping done in SW
73  // - Credit-based shaping per-class (in software)
74  // - Priority between classes
75  // - WRR for streams within each class
76  AVB_SHAPER_SW,
77 #ifdef AVB_ENABLE_HWQ_PER_CLASS
78  // Shaping in HW w/TXQ per AVB class
79  // - Each AVB class gets its own txq w/shaping in HW
80  // - SW does round-robin among classes to keep all TX queues fed
81  // - SW does WRR for streams within each class
82  AVB_SHAPER_HWQ_PER_CLASS,
83 #endif
84 #ifdef AVB_ENABLE_HWQ_PER_STREAM
85  // Shaping in HW w/TXQ per AVB stream
86  // - Each AVB stream gets its own txq w/shaping in HW
87  // - SW does round-robin among streams to keep all TX queues fed
88  AVB_SHAPER_HWQ_PER_STREAM
89 #endif
90 } avb_shaper_mode;
