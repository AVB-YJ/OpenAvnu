71 int log_imp::proc_logging_thread_callback()
72 {
73  DWORD dwEvent;
74 
75  while (true)
76  {
77  dwEvent = WaitForMultipleObjects(2, poll_events, FALSE, INFINITE);
78 
79  if (dwEvent == (WAIT_OBJECT_0 + LOG_EVENT))
80  {
81  if ((write_index - read_index) > 0)
82  {
83  callback_func(user_obj,
84  log_buf[read_index % LOG_BUF_COUNT].level,
85  log_buf[read_index % LOG_BUF_COUNT].msg,
86  log_buf[read_index % LOG_BUF_COUNT].time_stamp_ms); // Call callback function
87 
88  read_index++;
89  }
90  }
91  else
92  {
93  SetEvent(poll_events[KILL_EVENT]);
94  break;

96  }
