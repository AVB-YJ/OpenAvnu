102 int watchdog_setup(OSThreadFactory *thread_factory)
103 {
104 #ifdef SYSTEMD_WATCHDOG
105  SystemdWatchdogHandler *watchdog = new SystemdWatchdogHandler();
106  OSThread *watchdog_thread = thread_factory->createThread();
107  int watchdog_result;
108  long unsigned int watchdog_interval;
109  watchdog_interval = watchdog->getSystemdWatchdogInterval(&watchdog_result);
110  if (watchdog_result) {
111  GPTP_LOG_INFO("Watchtog interval read from service file: %lu us", watchdog_interval);
112  watchdog->update_interval = watchdog_interval / 2;
113  GPTP_LOG_STATUS("Starting watchdog handler (Update every: %lu us)", watchdog->update_interval);
114  watchdog_thread->start(watchdogUpdateThreadFunction, watchdog);
115  return 0;
116  } else if (watchdog_result < 0) {
117  GPTP_LOG_ERROR("Watchdog settings read error.");
118  return -1;
119  } else {
120  GPTP_LOG_STATUS("Watchdog disabled");
121  return 0;
122  }
123 #else
124  return 0;
125 #endif
126 }
