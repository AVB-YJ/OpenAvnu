120 bool openavbAdpOpenSocket(const char* ifname, U16 vlanID, U8 vlanPCP)
121 {
122  AVB_TRACE_ENTRY(AVB_TRACE_ADP);
123 
124  hdr_info_t hdr;
125 
126 #ifndef UBUNTU
127  // This is the normal case for most of our supported platforms
128  rxSock = openavbRawsockOpen(ifname, TRUE, FALSE, ETHERTYPE_8021Q, ADP_FRAME_LEN, ADP_NUM_BUFFERS);
129 #else
130  rxSock = openavbRawsockOpen(ifname, TRUE, FALSE, ETHERTYPE_AVTP, ADP_FRAME_LEN, ADP_NUM_BUFFERS);
131 #endif
132  txSock = openavbRawsockOpen(ifname, FALSE, TRUE, ETHERTYPE_AVTP, ADP_FRAME_LEN, ADP_NUM_BUFFERS);
133 
134  if (txSock && rxSock
135  && openavbRawsockGetAddr(txSock, ADDR_PTR(&intfAddr))
136  && ether_aton_r(ADP_PROTOCOL_ADDR, &adpAddr)
137  && openavbRawsockRxMulticast(rxSock, TRUE, ADDR_PTR(&adpAddr)))
138  {
139  if (!openavbRawsockRxAVTPSubtype(rxSock, OPENAVB_ADP_AVTP_SUBTYPE | 0x80)) {
140  AVB_LOG_DEBUG("RX AVTP Subtype not supported");
141  }
142 
143  memset(&hdr, 0, sizeof(hdr_info_t));
144  hdr.shost = ADDR_PTR(&intfAddr);
145  hdr.dhost = ADDR_PTR(&adpAddr);
146  hdr.ethertype = ETHERTYPE_AVTP;
147  if (vlanID != 0 || vlanPCP != 0) {
148  hdr.vlan = TRUE;
149  hdr.vlan_pcp = vlanPCP;
150  hdr.vlan_vid = vlanID;
151  AVB_LOGF_DEBUG("VLAN pcp=%d vid=%d", hdr.vlan_pcp, hdr.vlan_vid);
152  }
153  if (!openavbRawsockTxSetHdr(txSock, &hdr)) {
154  AVB_LOG_ERROR("TX socket Header Failure");
155  openavbAdpCloseSocket();
156  AVB_TRACE_EXIT(AVB_TRACE_ADP);
157  return false;
158  }
159 
160  AVB_TRACE_EXIT(AVB_TRACE_ADP);
161  return true;
162  }
163 
164  AVB_LOG_ERROR("Invalid socket");
165  openavbAdpCloseSocket();
166 
167  AVB_TRACE_EXIT(AVB_TRACE_ADP);
168  return false;
169 }

171 static void openavbAdpMessageRxFrameParse(U8* payload, int payload_len, hdr_info_t *hdr)
172 {
173  AVB_TRACE_ENTRY(AVB_TRACE_ADP);
174 
175  openavb_adp_control_header_t adpHeader;
176  openavb_adp_data_unit_t adpPdu;
177 
178 #if 0
179  AVB_LOGF_DEBUG("openavbAdpMessageRxFrameParse packet data (length %d):", payload_len);
180  AVB_LOG_BUFFER(AVB_LOG_LEVEL_DEBUG, payload, payload_len, 16);
181 #endif
182 
183  U8 *pSrc = payload;
184  {
185  // AVTP Control Header
186  openavb_adp_control_header_t *pDst = &adpHeader;
187 
188  BIT_B2DNTOHB(pDst->cd, pSrc, 0x80, 7, 0);
189  BIT_B2DNTOHB(pDst->subtype, pSrc, 0x7f, 0, 1);
190  BIT_B2DNTOHB(pDst->sv, pSrc, 0x80, 7, 0);
191  BIT_B2DNTOHB(pDst->version, pSrc, 0x70, 4, 0);
192  BIT_B2DNTOHB(pDst->message_type, pSrc, 0x0f, 0, 1);
193  BIT_B2DNTOHB(pDst->valid_time, pSrc, 0xf800, 11, 0);
194  BIT_B2DNTOHS(pDst->control_data_length, pSrc, 0x07ff, 0, 2);
195  OCT_B2DMEMCP(pDst->entity_id, pSrc);
196  }
197 
198  if (adpHeader.subtype == OPENAVB_ADP_AVTP_SUBTYPE &&
199  (adpHeader.message_type == OPENAVB_ADP_MESSAGE_TYPE_ENTITY_DISCOVER ||
200  (gAvdeccCfg.bFastConnectSupported && adpHeader.message_type == OPENAVB_ADP_MESSAGE_TYPE_ENTITY_AVAILABLE))) {
201  // ADP PDU
202  openavb_adp_data_unit_t *pDst = &adpPdu;
203 
204  OCT_B2DMEMCP(pDst->entity_model_id, pSrc);
205  OCT_B2DNTOHL(pDst->entity_capabilities, pSrc);
206  OCT_B2DNTOHS(pDst->talker_stream_sources, pSrc);
207  OCT_B2DNTOHS(pDst->talker_capabilities, pSrc);
208  OCT_B2DNTOHS(pDst->listener_stream_sinks, pSrc);
209  OCT_B2DNTOHS(pDst->listener_capabilities, pSrc);
210  OCT_B2DNTOHL(pDst->controller_capabilities, pSrc);
211  OCT_B2DNTOHL(pDst->available_index, pSrc);
212  OCT_B2DMEMCP(pDst->gptp_grandmaster_id, pSrc);
213  OCT_B2DNTOHB(pDst->gptp_domain_number, pSrc);
214  OCT_B2DMEMCP(pDst->reserved0, pSrc);
215  OCT_B2DNTOHS(pDst->identify_control_index, pSrc);
216  OCT_B2DNTOHS(pDst->interface_index, pSrc);
217  OCT_B2DMEMCP(pDst->association_id, pSrc);
218  OCT_B2DMEMCP(pDst->reserved1, pSrc);
219 
220  if (adpHeader.message_type == OPENAVB_ADP_MESSAGE_TYPE_ENTITY_DISCOVER) {
221  // Update the interface state machine
222  openavbAdpSMAdvertiseInterfaceSet_entityID(adpHeader.entity_id);
223  openavbAdpSMAdvertiseInterfaceSet_rcvdDiscover(TRUE);
224  }
225  else {
226  // See if Fast Connect is waiting for this device to be available.
227  if (adpPdu.talker_stream_sources > 0) {
228  openavbAcmpSMListenerSet_talkerTestFastConnect(adpHeader.entity_id);
229  }
230  }
231  }
232 
233  AVB_TRACE_EXIT(AVB_TRACE_ADP);
234 }

236 static void openavbAdpMessageRxFrameReceive(U32 timeoutUsec)
237 {
238  AVB_TRACE_ENTRY(AVB_TRACE_ADP);
239 
240  hdr_info_t hdrInfo;
241  unsigned int offset, len;
242  U8 *pBuf, *pFrame;
243 
244  memset(&hdrInfo, 0, sizeof(hdr_info_t));
245 
246  pBuf = (U8 *)openavbRawsockGetRxFrame(rxSock, timeoutUsec, &offset, &len);
247  if (pBuf) {
248  pFrame = pBuf + offset;
249 
250  offset = openavbRawsockRxParseHdr(rxSock, pBuf, &hdrInfo);
251  {
252 #ifndef UBUNTU
253  if (hdrInfo.ethertype == ETHERTYPE_8021Q) {
254  // Oh! Need to look past the VLAN tag
255  U16 vlan_bits = ntohs(*(U16 *)(pFrame + offset));
256  hdrInfo.vlan = TRUE;
257  hdrInfo.vlan_vid = vlan_bits & 0x0FFF;
258  hdrInfo.vlan_pcp = (vlan_bits >> 13) & 0x0007;
259  offset += 2;
260  hdrInfo.ethertype = ntohs(*(U16 *)(pFrame + offset));
261  offset += 2;
262  }
263 #endif
264 
265  // Make sure that this is an AVTP packet
266  // (Should always be AVTP if it's to our AVTP-specific multicast address)
267  if (hdrInfo.ethertype == ETHERTYPE_AVTP) {
268  // parse the PDU only for ADP messages
269  if (*(pFrame + offset) == (0x80 | OPENAVB_ADP_AVTP_SUBTYPE)) {
270  if (memcmp(hdrInfo.shost, ADDR_PTR(&intfAddr), 6) != 0) { // Not from us!
271  openavbAdpMessageRxFrameParse(pFrame + offset, len - offset, &hdrInfo);
272  }
273  }
274  }
275  else {
276  AVB_LOG_WARNING("Received non-AVTP frame!");
277  AVB_LOGF_DEBUG("Unexpected packet data (length %d):", len);
278  AVB_LOG_BUFFER(AVB_LOG_LEVEL_DEBUG, pFrame, len, 16);
279  }
280  }
281 
282  // Release the frame
283  openavbRawsockRelRxFrame(rxSock, pBuf);
284  }
285 
286  AVB_TRACE_EXIT(AVB_TRACE_ADP);
287 }

290 void openavbAdpMessageTxFrame(U8 msgType, U8 *destAddr)
291 {
292  AVB_TRACE_ENTRY(AVB_TRACE_ADP);
293 
294  U8 *pBuf;
295  U32 size;
296  unsigned int hdrlen = 0;
297 
298  pBuf = openavbRawsockGetTxFrame(txSock, TRUE, &size);
299 
300  if (!pBuf) {
301  AVB_LOG_ERROR("No TX buffer");
302  AVB_TRACE_EXIT(AVB_TRACE_ADP);
303  return;
304  }
305 
306  if (size < ADP_FRAME_LEN) {
307  AVB_LOG_ERROR("TX buffer too small");
308  openavbRawsockRelTxFrame(txSock, pBuf);
309  pBuf = NULL;
310  AVB_TRACE_EXIT(AVB_TRACE_ADP);
311  return;
312  }
313 
314  memset(pBuf, 0, ADP_FRAME_LEN);
315  openavbRawsockTxFillHdr(txSock, pBuf, &hdrlen);
316 
317  if (destAddr)
318  memcpy(pBuf, destAddr, ETH_ALEN);
319 
320  ADP_LOCK();
321  U8 *pDst = pBuf + hdrlen;
322  {
323  // AVTP Control Header
324  openavb_adp_control_header_t *pSrc = &openavbAdpSMGlobalVars.entityInfo.header;
325  BIT_D2BHTONB(pDst, pSrc->cd, 7, 0);
326  BIT_D2BHTONB(pDst, pSrc->subtype, 0, 1);
327  BIT_D2BHTONB(pDst, pSrc->sv, 7, 0);
328  BIT_D2BHTONB(pDst, pSrc->version, 4, 0);
329  BIT_D2BHTONB(pDst, msgType, 0, 1);
330  BIT_D2BHTONS(pDst, pSrc->valid_time, 11, 0);
331  BIT_D2BHTONS(pDst, pSrc->control_data_length, 0, 2);
332  OCT_D2BMEMCP(pDst, pSrc->entity_id);
333  }
334 
335  {
336  // ADP PDU
337  openavb_adp_data_unit_t *pSrc = &openavbAdpSMGlobalVars.entityInfo.pdu;
338  OCT_D2BMEMCP(pDst, pSrc->entity_model_id);
339  OCT_D2BHTONL(pDst, pSrc->entity_capabilities);
340  OCT_D2BHTONS(pDst, pSrc->talker_stream_sources);
341  OCT_D2BHTONS(pDst, pSrc->talker_capabilities);
342  OCT_D2BHTONS(pDst, pSrc->listener_stream_sinks);
343  OCT_D2BHTONS(pDst, pSrc->listener_capabilities);
344  OCT_D2BHTONL(pDst, pSrc->controller_capabilities);
345  OCT_D2BHTONL(pDst, pSrc->available_index);
346  OCT_D2BMEMCP(pDst, pSrc->gptp_grandmaster_id);
347  OCT_D2BHTONB(pDst, pSrc->gptp_domain_number);
348  OCT_D2BMEMCP(pDst, pSrc->reserved0);
349  OCT_D2BHTONS(pDst, pSrc->identify_control_index);
350  OCT_D2BHTONS(pDst, pSrc->interface_index);
351  OCT_D2BMEMCP(pDst, pSrc->association_id);
352  OCT_D2BMEMCP(pDst, pSrc->reserved1);
353  }
354  ADP_UNLOCK();
355 
356 #if 0
357  AVB_LOGF_DEBUG("openavbAdpMessageTxFrame packet data (length %d):", hdrlen + AVTP_HDR_LEN + ADP_DATA_LEN);
358  AVB_LOG_BUFFER(AVB_LOG_LEVEL_DEBUG, pBuf, hdrlen + AVTP_HDR_LEN + ADP_DATA_LEN, 16);
359 #endif
360 
361  openavbRawsockTxFrameReady(txSock, pBuf, hdrlen + AVTP_HDR_LEN + ADP_DATA_LEN, 0);
362  openavbRawsockSend(txSock);
363 
364  AVB_TRACE_EXIT(AVB_TRACE_ADP);
365 }

420 openavbRC openavbAdpMessageSend(U8 messageType)
421 {
422  AVB_TRACE_ENTRY(AVB_TRACE_ADP);
423 
424  // Note: this entire process of the GM ID is not as the 1722.1 spec expects.
425  // The openavbAdpSMAdvertiseInterfaceSet_advertisedGrandmasterID() should be called when the
426  // stack detects a GM change that will trigger an advertise. Instead we are detecting
427  // the GM change when we have are sending an advertise message. This means we will not have a timely
428  // new advertise in the event of a GM change. Additionally the handling the advertiseInterface var
429  // of GM ID is not normal since the GM ID is being placed directly into the PDU rather than getting
430  // pulled from the state machine var.
431  // AVDECC_TODO: This logic should change to detect GM change else where in the system and call the
432  // expected openavbAdpSMAdvertiseInterfaceSet_advertisedGrandmasterID() to start the advertise process.
433 #ifdef AVB_PTP_AVAILABLE
434  openavb_adp_data_unit_t *pPdu = &openavbAdpSMGlobalVars.entityInfo.pdu;
435  openavbRC retCode = OPENAVB_PTP_FAILURE;
436  U32 ptpSharedMemoryEntryIndex;
437  retCode = openavbPtpFindLatestSharedMemoryEntry(&ptpSharedMemoryEntryIndex);
438  if (IS_OPENAVB_FAILURE(retCode)) {
439  AVB_LOG_INFO("Failed to find PTP shared memory entry.");
440  }
441  else {
442  if (memcmp(pPdu->gptp_grandmaster_id, openavbPtpGMChageTable.entry[ptpSharedMemoryEntryIndex].gmId, sizeof(pPdu->gptp_grandmaster_id))) {
443  memcpy(pPdu->gptp_grandmaster_id, openavbPtpGMChageTable.entry[ptpSharedMemoryEntryIndex].gmId, sizeof(pPdu->gptp_grandmaster_id));
444  openavbAdpSMAdvertiseInterfaceSet_advertisedGrandmasterID(pPdu->gptp_grandmaster_id);
445  }
446  }
447 #else
448  openavb_adp_data_unit_t *pPdu = &openavbAdpSMGlobalVars.entityInfo.pdu;
449  uint8_t current_grandmaster_id[8];
450  if (!osalAVBGrandmasterGetCurrent(current_grandmaster_id, &(pPdu->gptp_domain_number)))
451  {
452  AVB_LOG_ERROR("osalAVBGrandmasterGetCurrent failure");
453  }
454  else if (memcmp(pPdu->gptp_grandmaster_id, current_grandmaster_id, sizeof(pPdu->gptp_grandmaster_id)))
455  {
456  memcpy(pPdu->gptp_grandmaster_id, current_grandmaster_id, sizeof(pPdu->gptp_grandmaster_id));
457  openavbAdpSMAdvertiseInterfaceSet_advertisedGrandmasterID(pPdu->gptp_grandmaster_id);
458  }
459 #endif
460 
461  openavbAdpMessageTxFrame(messageType, NULL);
462  AVB_RC_TRACE_RET(OPENAVB_AVDECC_SUCCESS, AVB_TRACE_ADP);
463 }
