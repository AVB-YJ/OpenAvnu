59 static int cfgCallback(void *user, const char *section, const char *name, const char *value)
60 {
61  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
62 
63  if (!user || !section || !name || !value) {
64  AVB_LOG_ERROR("Config: invalid arguments passed to callback");
65  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
66  return 0;
67  }
68 
69  openavb_avdecc_cfg_t *pCfg = (openavb_avdecc_cfg_t*)user;
70 
71  AVB_LOGF_DEBUG("name=[%s] value=[%s]", name, value);
72 
73  bool valOK = FALSE;
74  char *pEnd;
75 
76  if (MATCH(section, "network"))
77  {
78  if (MATCH(name, "ifname"))
79  {
80  if_info_t ifinfo;
81  if (openavbCheckInterface(value, &ifinfo)) {
82  strncpy(pCfg->ifname, value, sizeof(pCfg->ifname) - 1);
83  memcpy(pCfg->ifmac, &ifinfo.mac, ETH_ALEN);
84  valOK = TRUE;
85  }
86  }
87  }
88  else if (MATCH(section, "vlan"))
89  {
90  if (MATCH(name, "vlanID")) {
91  errno = 0;
92  pCfg->vlanID = strtoul(value, &pEnd, 10);
93  if (*pEnd == '\0' && errno == 0)
94  valOK = TRUE;
95  }
96  else if (MATCH(name, "vlanPCP")) {
97  errno = 0;
98  pCfg->vlanPCP = strtoul(value, &pEnd, 10);
99  if (*pEnd == '\0' && errno == 0)
100  valOK = TRUE;
101  }
102  else {
103  // unmatched item, fail
104  AVB_LOGF_ERROR("Unrecognized configuration item: section=%s, name=%s", section, name);
105  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
106  return 0;
107  }
108  }
109  else if (MATCH(section, "fast_connect"))
110  {
111  if (MATCH(name, "fast_connect")) {
112  errno = 0;
113  pCfg->bFastConnectSupported = (strtoul(value, &pEnd, 10) != 0);
114  if (*pEnd == '\0' && errno == 0)
115  valOK = TRUE;
116  }
117  else {
118  // unmatched item, fail
119  AVB_LOGF_ERROR("Unrecognized configuration item: section=%s, name=%s", section, name);
120  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
121  return 0;
122  }
123  }
124  else if (MATCH(section, "discovery"))
125  {
126  if (MATCH(name, "valid_time")) {
127  errno = 0;
128  pCfg->valid_time = strtoul(value, &pEnd, 10);
129  if (*pEnd == '\0' && errno == 0) {
130  if (pCfg->valid_time < 2 || pCfg->valid_time > 62) {
131  AVB_LOG_ERROR("valid_time must be between 2 and 62 (inclusive)");
132  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
133  return 0;
134  }
135  if ((pCfg->valid_time & 1) != 0) {
136  AVB_LOGF_WARNING("valid_time converted to an even number (%d to %d)", pCfg->valid_time, pCfg->valid_time - 1);
137  }
138  pCfg->valid_time /= 2; // Convert from seconds to 2-second units.
139  valOK = TRUE;
140  }
141  }
142  else {
143  // unmatched item, fail
144  AVB_LOGF_ERROR("Unrecognized configuration item: section=%s, name=%s", section, name);
145  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
146  return 0;
147  }
148  }
149  else if (MATCH(section, "descriptor_entity"))
150  {
151  if (MATCH(name, "avdeccId")) {
152  errno = 0;
153  pCfg->avdeccId = strtoul(value, &pEnd, 0);
154  if (*pEnd == '\0' && errno == 0)
155  valOK = TRUE;
156  }
157  else if (MATCH(name, "entity_model_id")) {
158  errno = 0;
159  U64 nTemp = strtoull(value, &pEnd, 0);
160  if (*pEnd == '\0' && errno == 0) {
161  valOK = TRUE;
162  int i;
163  for (i = 7; i >= 0; --i) {
164  pCfg->entity_model_id[i] = (U8) (nTemp & 0xFF);
165  nTemp = (nTemp >> 8);
166  }
167  AVB_LOGF_DEBUG("entity_model_id = " ENTITYID_FORMAT, ENTITYID_ARGS(pCfg->entity_model_id));
168  }
169  }
170  else if (MATCH(name, "entity_name"))
171  {
172  strncpy(pCfg->entity_name, value, sizeof(pCfg->entity_name));
173  // String does not need to be 0-terminated.
174  valOK = TRUE;
175  }
176  else if (MATCH(name, "firmware_version"))
177  {
178  strncpy(pCfg->firmware_version, value, sizeof(pCfg->firmware_version));
179  // String does not need to be 0-terminated.
180  valOK = TRUE;
181  }
182  else if (MATCH(name, "group_name"))
183  {
184  strncpy(pCfg->group_name, value, sizeof(pCfg->group_name));
185  // String does not need to be 0-terminated.
186  valOK = TRUE;
187  }
188  else if (MATCH(name, "serial_number"))
189  {
190  strncpy(pCfg->serial_number, value, sizeof(pCfg->serial_number));
191  // String does not need to be 0-terminated.
192  valOK = TRUE;
193  }
194  else {
195  // unmatched item, fail
196  AVB_LOGF_ERROR("Unrecognized configuration item: section=%s, name=%s", section, name);
197  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
198  return 0;
199  }
200  }
201  else if (MATCH(section, "localization"))
202  {
203  if (MATCH(name, "locale_identifier"))
204  {
205  strncpy(pCfg->locale_identifier, value, sizeof(pCfg->locale_identifier));
206  // String does not need to be 0-terminated.
207  valOK = TRUE;
208  }
209  else if (MATCH(name, "vendor_name"))
210  {
211  strncpy(pCfg->vendor_name, value, sizeof(pCfg->vendor_name));
212  // String does not need to be 0-terminated.
213  valOK = TRUE;
214  }
215  else if (MATCH(name, "model_name"))
216  {
217  strncpy(pCfg->model_name, value, sizeof(pCfg->model_name));
218  // String does not need to be 0-terminated.
219  valOK = TRUE;
220  }
221  else {
222  // unmatched item, fail
223  AVB_LOGF_ERROR("Unrecognized configuration item: section=%s, name=%s", section, name);
224  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
225  return 0;
226  }
227  }
228  else {
229  // unmatched item, fail
230  AVB_LOGF_ERROR("Unrecognized configuration item: section=%s, name=%s", section, name);
231  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
232  return 0;
233  }
234 
235  if (!valOK) {
236  cfgValErr(section, name, value);
237  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
238  return 0;
239  }
240 
241  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
242  return 1; // OK
243 }

245 // Parse ini file, and create config data
246 //
247 int openavbReadAvdeccConfig(const char *ini_file, openavb_avdecc_cfg_t *pCfg)
248 {
249  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
250 
251  // defaults - most are handled by setting everything to 0
252  memset(pCfg, 0, sizeof(openavb_avdecc_cfg_t));
253  pCfg->valid_time = 31; // See IEEE Std 1722.1-2013 clause 6.2.1.6
254  pCfg->avdeccId = 0xfffe;
255 
256  int result = ini_parse(ini_file, cfgCallback, pCfg);
257  if (result < 0) {
258  AVB_LOGF_ERROR("Couldn't parse INI file: %s", ini_file);
259  return -1;
260  }
261  if (result > 0) {
262  AVB_LOGF_ERROR("Error in INI file: %s, line %d", ini_file, result);
263  return -1;
264  }
265 
266  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
267 
268  // Yay, we did it.
269  return 0;
270 }
