28 /* AAF Listener example.
29  *
30  * This example implements a very simple AAF listener application which
31  * receives AFF packets from the network, retrieves the PCM samples, and
32  * writes them to stdout once the presentation time is reached.
33  *
34  * For simplicity, the example accepts only AAF packets with the following
35  * specification:
36  * - Sample format: 16-bit little endian
37  * - Sample rate: 48 kHz
38  * - Number of channels: 2 (stereo)
39  *
40  * TSN stream parameters such as destination mac address are passed via
41  * command-line arguments. Run 'aaf-listener --help' for more information.
42  *
43  * This example relies on the system clock to schedule PCM samples for
44  * playback. So make sure the system clock is synchronized with the PTP
45  * Hardware Clock (PHC) from your NIC and that the PHC is synchronized with
46  * the PTP time from the network. For further information on how to synchronize
47  * those clocks see ptp4l(8) and phc2sys(8) man pages.
48  *
49  * The easiest way to use this example is combining it with 'aplay' tool
50  * provided by alsa-utils. 'aplay' reads a PCM stream from stdin and sends it
51  * to a ALSA playback device (e.g. your speaker). So, to play Audio from a TSN
52  * stream, you should do something like this:
53  *
54  * $ aaf-listener <args> | aplay -f dat -t raw -D <playback-device>
55  */

144 /* Schedule 'pcm_sample' to be presented at time specified by 'tspec'. */
145 static int schedule_sample(int fd, struct timespec *tspec, uint8_t *pcm_sample)
146 {
147  struct sample_entry *entry;
148 
149  entry = malloc(sizeof(*entry));
150  if (!entry) {
151  fprintf(stderr, "Failed to allocate memory\n");
152  return -1;
153  }
154 
155  entry->tspec.tv_sec = tspec->tv_sec;
156  entry->tspec.tv_nsec = tspec->tv_nsec;
157  memcpy(entry->pcm_sample, pcm_sample, DATA_LEN);
158 
159  STAILQ_INSERT_TAIL(&samples, entry, entries);
160 
161  /* If this was the first entry inserted onto the queue, we need to arm
162  * the timer.
163  */
164  if (STAILQ_FIRST(&samples) == entry) {
165  int res;
166 
167  res = arm_timer(fd, tspec);
168  if (res < 0) {
169  STAILQ_REMOVE(&samples, entry, sample_entry, entries);
170  free(entry);
171  return -1;
172  }
173  }
174 
175  return 0;
176 }

203  if (fd < 0) {
204  perror("Failed to open socket");
205  return -1;
206  }
207 
208  snprintf(req.ifr_name, sizeof(req.ifr_name), "%s", ifname);
209  res = ioctl(fd, SIOCGIFINDEX, &req);
210  if (res < 0) {

212  goto err;
213  }
214 
215  sk_addr.sll_ifindex = req.ifr_ifindex;
216 
217  res = bind(fd, (struct sockaddr *) &sk_addr, sizeof(sk_addr));
218  if (res < 0) {
219  perror("Couldn't bind() to interface");
220  goto err;
221  }
222 
223  mreq.mr_ifindex = sk_addr.sll_ifindex;
224  mreq.mr_type = PACKET_MR_MULTICAST;
225  mreq.mr_alen = ETH_ALEN;
226  memcpy(&mreq.mr_address, macaddr, ETH_ALEN);
227 
228  res = setsockopt(fd, SOL_PACKET, PACKET_ADD_MEMBERSHIP,
229  &mreq, sizeof(struct packet_mreq));

232  goto err;
233  }

242 static bool is_valid_packet(struct avtp_stream_pdu *pdu)
243 {
244  struct avtp_common_pdu *common = (struct avtp_common_pdu *) pdu;
245  uint64_t val64;
246  uint32_t val32;
247  int res;
248 
249  res = avtp_pdu_get(common, AVTP_FIELD_SUBTYPE, &val32);
250  if (res < 0) {
251  fprintf(stderr, "Failed to get subtype field: %d\n", res);
252  return false;
253  }
254  if (val32 != AVTP_SUBTYPE_AAF) {
255  fprintf(stderr, "Subtype mismatch: expected %u, got %u\n",
256  AVTP_FIELD_SUBTYPE, val32);
257  return false;
258  }
259 
260  res = avtp_pdu_get(common, AVTP_FIELD_VERSION, &val32);
261  if (res < 0) {
262  fprintf(stderr, "Failed to get version field: %d\n", res);
263  return false;
264  }
265  if (val32 != 0) {
266  fprintf(stderr, "Version mismatch: expected %u, got %u\n",
267  0, val32);
268  return false;
269  }
270 
271  res = avtp_aaf_pdu_get(pdu, AVTP_AAF_FIELD_TV, &val64);
272  if (res < 0) {
273  fprintf(stderr, "Failed to get tv field: %d\n", res);
274  return false;
275  }
276  if (val64 != 1) {
277  fprintf(stderr, "tv mismatch: expected %u, got %lu\n",
278  1, val64);
279  return false;
280  }
281 
282  res = avtp_aaf_pdu_get(pdu, AVTP_AAF_FIELD_SP, &val64);
283  if (res < 0) {
284  fprintf(stderr, "Failed to get sp field: %d\n", res);
285  return false;
286  }
287  if (val64 != AVTP_AAF_PCM_SP_NORMAL) {
288  fprintf(stderr, "tv mismatch: expected %u, got %lu\n",
289  1, val64);
290  return false;
291  }
292 
293  res = avtp_aaf_pdu_get(pdu, AVTP_AAF_FIELD_STREAM_ID, &val64);
294  if (res < 0) {
295  fprintf(stderr, "Failed to get stream ID field: %d\n", res);
296  return false;
297  }
298  if (val64 != STREAM_ID) {
299  fprintf(stderr, "Stream ID mismatch: expected %lu, got %lu\n",
300  STREAM_ID, val64);
301  return false;
302  }
303 
304  res = avtp_aaf_pdu_get(pdu, AVTP_AAF_FIELD_SEQ_NUM, &val64);
305  if (res < 0) {
306  fprintf(stderr, "Failed to get sequence num field: %d\n", res);
307  return false;
308  }
309 
310  if (val64 != expected_seq) {
311  /* If we have a sequence number mismatch, we simply log the
312  * issue and continue to process the packet. We don't want to
313  * invalidate it since it is a valid packet after all.
314  */
315  fprintf(stderr, "Sequence number mismatch: expected %u, got %lu\n",
316  expected_seq, val64);
317  expected_seq = val64;
318  }
319 
320  expected_seq++;
321 
322  res = avtp_aaf_pdu_get(pdu, AVTP_AAF_FIELD_FORMAT, &val64);
323  if (res < 0) {
324  fprintf(stderr, "Failed to get format field: %d\n", res);
325  return false;
326  }
327  if (val64 != AVTP_AAF_FORMAT_INT_16BIT) {
328  fprintf(stderr, "Format mismatch: expected %u, got %lu\n",
329  AVTP_AAF_FORMAT_INT_16BIT, val64);
330  return false;
331  }
332 
333  res = avtp_aaf_pdu_get(pdu, AVTP_AAF_FIELD_NSR, &val64);
334  if (res < 0) {
335  fprintf(stderr, "Failed to get sample rate field: %d\n", res);
336  return false;
337  }
338  if (val64 != AVTP_AAF_PCM_NSR_48KHZ) {
339  fprintf(stderr, "Sample rate mismatch: expected %u, got %lu",
340  AVTP_AAF_PCM_NSR_48KHZ, val64);
341  return false;
342  }
343 
344  res = avtp_aaf_pdu_get(pdu, AVTP_AAF_FIELD_CHAN_PER_FRAME, &val64);
345  if (res < 0) {
346  fprintf(stderr, "Failed to get channels field: %d\n", res);
347  return false;
348  }
349  if (val64 != NUM_CHANNELS) {
350  fprintf(stderr, "Channels mismatch: expected %u, got %lu\n",
351  NUM_CHANNELS, val64);
352  return false;
353  }
354 
355  res = avtp_aaf_pdu_get(pdu, AVTP_AAF_FIELD_BIT_DEPTH, &val64);
356  if (res < 0) {
357  fprintf(stderr, "Failed to get depth field: %d\n", res);
358  return false;
359  }
360  if (val64 != 16) {
361  fprintf(stderr, "Depth mismatch: expected %u, got %lu\n",
362  16, val64);
363  return false;
364  }
365 
366  res = avtp_aaf_pdu_get(pdu, AVTP_AAF_FIELD_STREAM_DATA_LEN, &val64);
367  if (res < 0) {
368  fprintf(stderr, "Failed to get data_len field: %d\n", res);
369  return false;
370  }
371  if (val64 != DATA_LEN) {
372  fprintf(stderr, "Data len mismatch: expected %u, got %lu\n",
373  DATA_LEN, val64);
374  return false;
375  }
376 
377  return true;
378 }

380 static int get_presentation_time(struct avtp_stream_pdu *pdu,
381  struct timespec *tspec)
382 {
383  int res;
384  uint64_t avtp_time, ptime, now;
385 
386  res = avtp_aaf_pdu_get(pdu, AVTP_AAF_FIELD_TIMESTAMP, &avtp_time);
387  if (res < 0) {
388  fprintf(stderr, "Failed to get AVTP time from PDU\n");
389  return -1;
390  }
391 
392  res = clock_gettime(CLOCK_REALTIME, tspec);
393  if (res < 0) {
394  perror("Failed to get time from PHC");
395  return -1;
396  }
397 
398  now = (tspec->tv_sec * NSEC_PER_SEC) + tspec->tv_nsec;
399 
400  /* The avtp_timestamp within AAF packet is the lower part (32
401  * less-significant bits) from presentation time calculated by the
402  * talker.
403  */
404  ptime = (now & 0xFFFFFFFF00000000ULL) | avtp_time;
405 
406  /* If 'ptime' is less than the 'now', it means the higher part
407  * from 'ptime' needs to be incremented by 1 in order to recover the
408  * presentation time set by the talker.
409  */
410  if (ptime < now)
411  ptime += (1ULL << 32);
412 
413  tspec->tv_sec = ptime / NSEC_PER_SEC;
414  tspec->tv_nsec = ptime % NSEC_PER_SEC;
415 
416  return 0;
417 }

419 static int new_packet(int sk_fd, int timer_fd)
420 {
421  int res;
422  ssize_t n;
423  struct timespec tspec;
424  struct avtp_stream_pdu *pdu = alloca(PDU_SIZE);
425 
426  memset(pdu, 0, PDU_SIZE);
427 
428  n = recv(sk_fd, pdu, PDU_SIZE, 0);
429  if (n < 0 || n != PDU_SIZE) {
430  perror("Failed to receive data");
431  return -1;
432  }
433 
434  if (!is_valid_packet(pdu)) {
435  fprintf(stderr, "Dropping packet\n");
436  return 0;
437  }
438 
439  res = get_presentation_time(pdu, &tspec);
440  if (res < 0)
441  return -1;
442 
443  res = schedule_sample(timer_fd, &tspec, pdu->avtp_payload);
444  if (res < 0)
445  return -1;
446 
447  return 0;
448 }

450 static int timeout(int fd)
451 {
452  int res;
453  ssize_t n;
454  uint64_t expirations;
455  struct sample_entry *entry;
456 
457  n = read(fd, &expirations, sizeof(uint64_t));
458  if (n < 0) {
459  perror("Failed to read timerfd");
460  return -1;
461  }
462 
463  assert(expirations == 1);
464 
465  entry = STAILQ_FIRST(&samples);
466  assert(entry != NULL);
467 
468  res = present_sample(entry->pcm_sample);
469  if (res < 0)
470  return -1;
471 
472  STAILQ_REMOVE_HEAD(&samples, entries);
473  free(entry);
474 
475  if (!STAILQ_EMPTY(&samples)) {
476  entry = STAILQ_FIRST(&samples);
477 
478  res = arm_timer(fd, &entry->tspec);
479  if (res < 0)
480  return -1;
481  }
482 
483  return 0;
484 }

486 int main(int argc, char *argv[])
487 {
488  int sk_fd, timer_fd, res;
489  struct pollfd fds[2];
490 
491  argp_parse(&argp, argc, argv, 0, NULL, NULL);
492 
493  STAILQ_INIT(&samples);
494 
495  sk_fd = setup_socket();
496  if (sk_fd < 0)
497  return 1;
498 
499  timer_fd = timerfd_create(CLOCK_REALTIME, 0);
500  if (timer_fd < 0) {
501  close(sk_fd);
502  return 1;
503  }
504 
505  fds[0].fd = sk_fd;
506  fds[0].events = POLLIN;
507  fds[1].fd = timer_fd;
508  fds[1].events = POLLIN;
509 
510  while (1) {
511  res = poll(fds, 2, -1);
512  if (res < 0) {
513  perror("Failed to poll() fds");
514  goto err;
515  }
516 
517  if (fds[0].revents & POLLIN) {
518  res = new_packet(sk_fd, timer_fd);
519  if (res < 0)
520  goto err;
521  }
522 
523  if (fds[1].revents & POLLIN) {
524  res = timeout(timer_fd);
525  if (res < 0)
526  goto err;
527  }
528  }
529 
530  return 0;
531 
532 err:
533  close(sk_fd);
534  close(timer_fd);
535  return 1;
