100 - (id)pickDeviceImage:(AVB17221Entity *)entity {
101  NSImage *image;
102  // Macbook icon
103  if (entity.isLocalEntity) {
104  image = [[NSWorkspace sharedWorkspace] iconForFileType:NSFileTypeForHFSTypeCode(kComputerIcon)];
105  }
106  else {
107  unsigned vendorID = 0;
108  NSScanner *scanner = [NSScanner scannerWithString:[[self vendorFromEntity:entity] substringFromIndex:2]];
109  [scanner scanHexInt:&vendorID];
110  switch (vendorID) {
111  case kAvbioVendorId:
112  // Brick icon
113  if (entity.entityModelID == kAvbioBrickModelId) {
114  image = [NSApp applicationIconImage];
115  }
116  // Bullet icon placeholder
117  else {
118  image = [NSApp applicationIconImage];
119  }
120  break;
121  case kXmosVendorId:
122  image = [NSImage imageNamed:kXmosIcon];
123  break;
124  default:
125  image = [NSApp applicationIconImage];
126  break;
127  }
128  }
129  return image;
130 }

132 - (id)getTypeFromCapabilities:(AVB17221Entity *)entity ofTalker:(BOOL)isTalker {
133  NSString *type;
134  NSInteger typeValue;
135  if (isTalker) {
136  typeValue = [[[NSString stringWithFormat:@"%x", entity.talkerCapabilities] substringToIndex:1] integerValue];
137  }
138  else {
139  typeValue = [[[NSString stringWithFormat:@"%x", entity.listenerCapabilities] substringToIndex:1] integerValue];
140  }
141  switch (typeValue) {
142  case (NSInteger)4:
143  type = @"Audio";
144  break;
145  case (NSInteger)8:
146  type = @"Video";
147  break;
148  default:
149  type = @"Unknown";
150  break;
151  }
152  return type;
153 }
