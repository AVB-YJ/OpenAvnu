96 int pack_maap(const MAAP_Packet *packet, uint8_t *stream) {
97  uint64_t tmp64;
98  uint16_t tmp16;
99  uint8_t tmp8;
100 
101  assert(packet);
102  assert(stream);
103 
104  tmp64 = HTOBE64(packet->DA << 16);
105  memcpy(stream, &tmp64, 6);
106  stream += 6;
107 
108  tmp64 = HTOBE64(packet->SA << 16);
109  memcpy(stream, &tmp64, 6);
110  stream += 6;
111 
112  tmp16 = HTOBE16(packet->Ethertype);
113  memcpy(stream, &tmp16, 2);
114  stream += 2;
115 
116  tmp8 = (uint8_t) packet->subtype;
117  *stream = tmp8;
118  stream++;
119 
120  tmp8 = (packet->SV << 7) | ((packet->version & 0x07) << 4) |
121  (packet->message_type & 0x0f);
122  *stream = tmp8;
123  stream++;
124 
125  tmp16 = HTOBE16(((packet->maap_version & 0x001f) << 11) |
126  (packet->control_data_length & 0x07ff));
127  memcpy(stream, &tmp16, 2);
128  stream += 2;
129 
130  tmp64 = HTOBE64(packet->stream_id);
131  memcpy(stream, &tmp64, 8);
132  stream += 8;
133 
134  tmp64 = HTOBE64(packet->requested_start_address << 16);
135  memcpy(stream, &tmp64, 6);
136  stream += 6;
137 
138  tmp16 = HTOBE16(packet->requested_count);
139  memcpy(stream, &tmp16, 2);
140  stream += 2;
141 
142  tmp64 = HTOBE64(packet->conflict_start_address << 16);
143  memcpy(stream, &tmp64, 6);
144  stream += 6;
145 
146  tmp16 = HTOBE16(packet->conflict_count);
147  memcpy(stream, &tmp16, 2);

152 void init_packet(MAAP_Packet *packet, uint64_t dest_mac, uint64_t src_mac) {
153  assert(packet);
154  assert(dest_mac != 0);
155  assert(src_mac != 0);
156 
157  packet->DA = dest_mac;
158  packet->SA = src_mac;
159  packet->Ethertype = MAAP_TYPE;
160  packet->subtype = MAAP_SUBTYPE;
161  packet->SV = 0; /* SV is 0 - Defined in IEEE 1722-2016 Table F.23 */
162  packet->version = 0; /* AVTP version is 0 - Defined in IEEE 1722-2016 4.4.3.4 */
163  packet->message_type = 0;
164  packet->maap_version = 1; /* MAAP version is 1 - Defined in IEEE 1722-2016 B.2.3.1 */
165  packet->control_data_length = 16; /* Control data length is 16 - Defined in IEEE 1722-2016 B.2.1 */
166  packet->stream_id = 0; /* MAAP stream_id is 0 - Defined in IEEE 1722-2016 B.2.4 */
167  packet->requested_start_address = 0;
168  packet->requested_count = 0;
169  packet->conflict_start_address = 0;
170  packet->conflict_count = 0;
171 }

182 int compare_mac_addresses(uint64_t local_mac, uint64_t remote_mac)
183 {
184  int i;
185  uint8_t local_byte, remote_byte;
186 
187  for (i = 0; i < 6; ++i)
188  {
189  /* Test the next least-significant byte. */
190  local_byte = (uint8_t)(local_mac & 0xFF);
191  remote_byte = (uint8_t)(remote_mac & 0xFF);
192 
193  /* See if we have a difference. */
194  if (local_byte < remote_byte) return 1;
195  if (local_byte > remote_byte) return 0;
196 
197  /* Try the next lowest-byte in the next iteration. */
198  local_mac = local_mac >> 8;
199  remote_mac = remote_mac >> 8;
200  }
201 
202  /* Assume that our own packet was somehow reflected back at us.
203  * Return 1 to ignore this packet. */
204  return 1;
205 }
