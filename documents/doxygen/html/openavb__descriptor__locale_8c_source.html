52 openavbRC openavbAemDescriptorLocaleToBuf(void *pVoidDescriptor, U16 bufLength, U8 *pBuf, U16 *descriptorSize)
53 {
54  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
55 
56  openavb_aem_descriptor_locale_t *pDescriptor = pVoidDescriptor;
57 
58  if (!pDescriptor || !pBuf || !descriptorSize) {
59  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_INVALID_ARGUMENT), AVB_TRACE_AEM);
60  }
61 
62  if (bufLength < OPENAVB_DESCRIPTOR_LOCALE_BASE_LENGTH) {
63  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVBAVDECC_RC_BUFFER_TOO_SMALL), AVB_TRACE_AEM);
64  }
65 
66  *descriptorSize = 0;
67 
68  openavb_aem_descriptor_locale_t *pSrc = pDescriptor;
69  U8 *pDst = pBuf;
70 
71  OCT_D2BHTONS(pDst, pSrc->descriptor_type);
72  OCT_D2BHTONS(pDst, pSrc->descriptor_index);
73  OCT_D2BMEMCP(pDst, pSrc->locale_identifier);
74  OCT_D2BHTONS(pDst, pSrc->number_of_strings);
75  OCT_D2BHTONS(pDst, pSrc->base_strings);
76 
77  *descriptorSize = pDst - pBuf;
78 
79  AVB_RC_TRACE_RET(OPENAVB_AVDECC_SUCCESS, AVB_TRACE_AEM);
80 }

126 extern DLL_EXPORT openavb_aem_descriptor_locale_t *openavbAemDescriptorLocaleNew(void)
127 {
128  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
129 
130  openavb_aem_descriptor_locale_t *pDescriptor;
131 
132  pDescriptor = malloc(sizeof(*pDescriptor));
133 
134  if (!pDescriptor) {
135  AVB_RC_LOG(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_OUT_OF_MEMORY));
136  AVB_TRACE_EXIT(AVB_TRACE_AEM);
137  return NULL;
138  }
139  memset(pDescriptor, 0, sizeof(*pDescriptor));
140 
141  pDescriptor->descriptorPvtPtr = malloc(sizeof(*pDescriptor->descriptorPvtPtr));
142  if (!pDescriptor->descriptorPvtPtr) {
143  free(pDescriptor);
144  pDescriptor = NULL;
145  AVB_RC_LOG(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_OUT_OF_MEMORY));
146  AVB_TRACE_EXIT(AVB_TRACE_AEM);
147  return NULL;
148  }
149 
150  pDescriptor->descriptorPvtPtr->size = sizeof(openavb_aem_descriptor_locale_t);
151  pDescriptor->descriptorPvtPtr->bTopLevel = TRUE;
152  pDescriptor->descriptorPvtPtr->toBuf = openavbAemDescriptorLocaleToBuf;
153  pDescriptor->descriptorPvtPtr->fromBuf = openavbAemDescriptorLocaleFromBuf;
154  pDescriptor->descriptorPvtPtr->update = openavbAemDescriptorLocaleUpdate;
155 
156  pDescriptor->descriptor_type = OPENAVB_AEM_DESCRIPTOR_LOCALE;
157 
158  AVB_TRACE_EXIT(AVB_TRACE_AEM);
159  return pDescriptor;
160 }
