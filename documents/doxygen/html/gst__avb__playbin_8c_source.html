354 static void freequeue()
355 {
356  struct tailq_entry *q, *next_item;
357 
358  for (q = TAILQ_FIRST(&buffer_queue); q != NULL; q = next_item)
359  {
360  next_item = TAILQ_NEXT(q, entries);
361  /* Remove the item from the tail queue. */
362  TAILQ_REMOVE(&buffer_queue, q, entries);
363  free(q->payload_data);
364  free(q);
365  }
366  for (q = TAILQ_FIRST(&free_queue); q != NULL; q = next_item)
367  {
368  next_item = TAILQ_NEXT(q, entries);
369  /* Remove the item from the tail queue. */
370  TAILQ_REMOVE(&free_queue, q, entries);
371  free(q->payload_data);
372  free(q);
373  }
374 
375  return;
376 }

402  TAILQ_INSERT_TAIL(&free_queue, q, entries);
403  pthread_mutex_unlock(&(free_queue_lock));
404  }
405  return i;

433 static GstFlowReturn
434 gst_avbsrc_create (GstPushSrc * psrc, GstBuffer ** buf)
435 {
436 
437  if(!avb_init)
438  {
439  struct packet_mreq mreq;
440  struct ifreq device;
441  uint32_t buff_size;
442  pthread_t tid;
443  int ifindex;
444 
445  /* add a signal handler for interruption Ctl+c */
446  signal(SIGINT, sigint_handler);
447 
448  buff_size = PAYLOAD_SIZE;
449 
450  start_feed_socket_init();
451 
452  err = pci_connect(&igb_dev);
453  if (err) {
454  printf("connect failed (%s) - are you running as root?\n", strerror(errno));
455  return errno;
456  }
457 
458  err = igb_init(&igb_dev);
459  if (err) {
460  printf("init failed (%s) - is the driver really loaded?\n", strerror(errno));
461  return errno;
462  }
463 
464  socket_d = socket(AF_PACKET, SOCK_RAW, htons(ETHER_TYPE_AVTP));
465  if (socket_d == -1) {
466  printf("failed to open event socket: %s \n", strerror(errno));
467  return -1;
468  }
469 
470  memset(&device, 0, sizeof(device));
471  memcpy(device.ifr_name, interface1, IFNAMSIZ);
472  err = ioctl(socket_d, SIOCGIFINDEX, &device);
473  if (err == -1) {
474  printf("Failed to get interface index: %s\n", strerror(errno));
475  return -1;
476  }
477 
478  ifindex = device.ifr_ifindex;
479  memset(&ifsock_addr, 0, sizeof(ifsock_addr));
480  ifsock_addr.sll_family = AF_PACKET;
481  ifsock_addr.sll_ifindex = ifindex;
482  ifsock_addr.sll_protocol = htons(ETHER_TYPE_AVTP);
483  err = bind(socket_d, (struct sockaddr *) & ifsock_addr, sizeof(ifsock_addr));
484  if (err == -1) {
485  printf("Call to bind() failed: %s\n", strerror(errno));
486  return -1;
487  }
488 
489  memset(&mreq, 0, sizeof(mreq));
490  mreq.mr_ifindex = ifindex;
491  mreq.mr_type = PACKET_MR_MULTICAST;
492  mreq.mr_alen = 6;
493  memcpy(mreq.mr_address, DEST_ADDR, mreq.mr_alen);
494  err = setsockopt(socket_d, SOL_PACKET, PACKET_ADD_MEMBERSHIP, &mreq, sizeof(mreq));
495  if (err == -1) {
496  printf ("Unable to add PTP multicast addresses to port id: %u\n", ifindex);
497  return -1;
498  }
499 
500  TAILQ_INIT(&buffer_queue);
501  TAILQ_INIT(&free_queue);
502  pthread_mutex_init(&(buffer_queue_lock), NULL);
503  pthread_mutex_init(&(free_queue_lock), NULL);
504 
505  if (initiliaze_queue(buff_size) < 0)
506  return -EINVAL;
507 
508  err = pthread_create(&tid, NULL, gstreamer_main_loop, NULL);
509  if (err != 0)
510  printf("can't create thread :[%s]\n", strerror(err));
511 
512  memset(frame, 0, sizeof(frame));
513 
514  size = sizeof(ifsock_addr);
515  frame_sequence = 0;
516  start_of_input_data = 1;
517 
518  memset(&sched, 0 , sizeof (sched));
519  sched.sched_priority = 25;
520  sched_setscheduler(0, SCHED_RR, &sched);
521  }
522 
523  while (1) {
524  if (g_exit_app)
525  break;
526 
527  err = recvfrom(socket_d, frame, FRAME_SIZE, 0, (struct sockaddr *) &ifsock_addr, (socklen_t *)&size);
528  if (err > 0) {
529  while (!free_queue.tqh_first) {
530  usleep(1);
531  }
532  fprintf(stderr,"frame sequence = %lld\n", frame_sequence++);
533 
534  pthread_mutex_lock(&(free_queue_lock));
535  qptr = free_queue.tqh_first;
536  TAILQ_REMOVE(&free_queue, qptr, entries);
537  h1722 = (seventeen22_header *)((uint8_t*)frame + sizeof(eth_header));
538  qptr->payload_length = ntohs(h1722->length) - sizeof(six1883_header);
539  memcpy(qptr->payload_data, (uint8_t *)((uint8_t*)frame + sizeof(eth_header) + sizeof(seventeen22_header)
540  + sizeof(six1883_header)), qptr->payload_length);
541 
542  pthread_mutex_lock(&(buffer_queue_lock));
543  TAILQ_INSERT_TAIL(&buffer_queue, qptr, entries);
544  pthread_mutex_unlock(&(buffer_queue_lock));
545  pthread_mutex_unlock(&(free_queue_lock));
546  start_of_input_data = 1;
547  } else {
548  printf("Failed to receive frame !!!\n");
549  }
550  }
551 
552  usleep(100);
553  pthread_exit(NULL);
554  close(socket_d);
555  freequeue();
556 
557  return GST_FLOW_OK;
558 }
