66 static struct igb_packet* alloc_page(device_t* dev, struct igb_dma_alloc *a_page)
67 {
68  int err = igb_dma_malloc_page(dev, a_page);
69  if (err) {
70  AVB_LOGF_ERROR("igb_dma_malloc_page failed: %s", strerror(err));
71  return NULL;
72  }
73 
74  struct igb_packet *free_packets;
75  struct igb_packet a_packet;
76 
77  a_packet.dmatime = a_packet.attime = a_packet.flags = 0;
78  a_packet.map.paddr = a_page->dma_paddr;
79  a_packet.map.mmap_size = a_page->mmap_size;
80  a_packet.offset = 0;
81  a_packet.vaddr = a_page->dma_vaddr + a_packet.offset;
82  a_packet.len = IGB_MTU;
83  a_packet.next = NULL;
84 
85  free_packets = NULL;
86 
87  /* divide the dma page into buffers for packets */
88  int i;
89  for (i = 0; i < a_page->mmap_size / IGB_MTU; i++) {
90  struct igb_packet *tmp_packet = malloc(sizeof(struct igb_packet));
91  if (!tmp_packet) {
92  AVB_LOG_ERROR("failed to allocate igb_packet memory!");
93  return false;
94  }
95  *tmp_packet = a_packet;
96  tmp_packet->offset = (i * IGB_MTU);
97  tmp_packet->vaddr += tmp_packet->offset;
98  tmp_packet->next = free_packets;
99  memset(tmp_packet->vaddr, 0, IGB_MTU);
100  free_packets = tmp_packet;
101  }
102  return free_packets;
103 }

105 device_t *igbAcquireDevice()
106 {
107  AVB_TRACE_ENTRY(AVB_TRACE_HAL_ETHER);
108 
109  LOCK();
110  if (!igb_dev) {
111  device_t *tmp_dev = calloc(1, sizeof(device_t));
112  if (!tmp_dev) {
113  AVB_LOGF_ERROR("Cannot allocate memory for device: %s", strerror(errno));
114  goto unlock;
115  }
116 
117  int err = pci_connect(tmp_dev);
118  if (err) {
119  AVB_LOGF_ERROR("connect failed (%s) - are you running as root?", strerror(err));
120  goto unlock;
121  }
122 
123  err = igb_init(tmp_dev);
124  if (err) {
125  AVB_LOGF_ERROR("init failed (%s) - is the driver really loaded?", strerror(err));
126  igb_detach(tmp_dev);
127  goto unlock;
128  }
129 
130 
131  int i;
132  for (i = 0; i < IGB_PAGES; i++) {
133  struct igb_packet* free_packets = alloc_page(tmp_dev, &g_pages[i]);
134  if (!g_free_packets) {
135  g_free_packets = free_packets;
136  } else {
137  struct igb_packet* last_packet = g_free_packets;
138  while (last_packet->next) {
139  last_packet = last_packet->next;
140  }
141  last_packet->next = free_packets;
142  }
143  }
144 
145  g_totalBuffers = count_packets(g_free_packets);
146 
147  AVB_LOGF_INFO("TX buffers: %d", g_totalBuffers);
148 
149  igbControlLaunchTime(tmp_dev, IGB_LAUNCHTIME_ENABLED);
150 
151  AVB_LOGF_INFO("IGB launch time feature is %s", IGB_LAUNCHTIME_ENABLED ? "ENABLED" : "DISABLED");
152 
153  igb_dev = tmp_dev;
154 unlock:
155  if (!igb_dev)
156  free(tmp_dev);
157  }
158 
159  if (igb_dev) {
160  igb_dev_users += 1;
161  AVB_LOGF_DEBUG("igb_dev_users %d", igb_dev_users);
162  }
163  UNLOCK();
164 
165  AVB_TRACE_EXIT(AVB_TRACE_HAL_ETHER);
166  return igb_dev;
167 }

169 void igbReleaseDevice(device_t* dev)
170 {
171  AVB_TRACE_ENTRY(AVB_TRACE_HAL_ETHER);
172 
173  LOCK();
174 
175  igb_dev_users -= 1;
176  AVB_LOGF_DEBUG("igb_dev_users %d", igb_dev_users);
177 
178  if (igb_dev && igb_dev_users <= 0) {
179  int i;
180  for (i = 0; i < IGB_PAGES; i++)
181  igb_dma_free_page(igb_dev, &g_pages[i]);
182 
183  igb_detach(igb_dev);
184  free(igb_dev);
185  igb_dev = NULL;
186  }
187 
188  UNLOCK();
189 
190  AVB_TRACE_EXIT(AVB_TRACE_HAL_ETHER);
191 }

193 struct igb_packet *igbGetTxPacket(device_t* dev)
194 {
195  AVB_TRACE_ENTRY(AVB_TRACE_HAL_ETHER);
196 
197  LOCK();
198 
199  struct igb_packet* tx_packet = g_free_packets;
200 
201  if (!tx_packet) {
202  struct igb_packet *cleaned_packets;
203  igb_clean(dev, &cleaned_packets);
204  while (cleaned_packets) {
205  struct igb_packet *tmp_packet = cleaned_packets;
206  cleaned_packets = cleaned_packets->next;
207  tmp_packet->next = g_free_packets;
208  g_free_packets = tmp_packet;
209  }
210  tx_packet = g_free_packets;
211 
212  g_usedBuffers = g_totalBuffers - count_packets(g_free_packets);
213  }
214 
215  if (tx_packet) {
216  g_free_packets = tx_packet->next;
217  }
218 
219  UNLOCK();
220 
221  AVB_TRACE_EXIT(AVB_TRACE_HAL_ETHER);
222  return tx_packet;
223 }

255 bool igbControlLaunchTime(device_t *dev, int enable)
256 {
257  AVB_TRACE_ENTRY(AVB_TRACE_HAL_ETHER);
258 
259  const uint32_t E1000_TQAVCTRL = 0x03570;
260  const uint32_t E1000_TQAVCTRL_LAUNCH_VALID = 0x00000200;
261 
262  int err;
263  uint32_t regVal;
264 
265  err = igb_lock(dev);
266  if (err)
267  goto error;
268 
269  igb_readreg(dev, E1000_TQAVCTRL, &regVal);
270 
271  if (enable)
272  regVal |= E1000_TQAVCTRL_LAUNCH_VALID;
273  else
274  regVal &= ~E1000_TQAVCTRL_LAUNCH_VALID;
275 
276  igb_writereg(dev, E1000_TQAVCTRL, regVal);
277 
278  err = igb_unlock(dev);
279 
280 error:
281  AVB_TRACE_EXIT(AVB_TRACE_HAL_ETHER);
282  return !err;
283 }
