81 void MockCheckedActualCall::copyOutputParameters(MockCheckedExpectedCall* expectedCall)
82 {
83  for (MockOutputParametersListNode* p = outputParameterExpectations_; p; p = p->next_)
84  {
85  MockNamedValue outputParameter = expectedCall->getOutputParameter(p->name_);
86  MockNamedValueCopier* copier = outputParameter.getCopier();
87  if (copier)
88  {
89  copier->copy(p->ptr_, outputParameter.getObjectPointer());
90  }
91  else if ((outputParameter.getType() == "const void*") && (p->type_ == "void*"))
92  {
93  const void* data = outputParameter.getConstPointerValue();
94  size_t size = outputParameter.getSize();
95  PlatformSpecificMemCpy(p->ptr_, data, size);
96  }
97  else if (outputParameter.getName() != "")
98  {
99  SimpleString type = expectedCall->getOutputParameter(p->name_).getType();
100  MockNoWayToCopyCustomTypeFailure failure(getTest(), type);
101  failTest(failure);
102  }
103  }
104 }

106 void MockCheckedActualCall::completeCallWhenMatchIsFound()
107 {
108  // Expectations that don't ignore parameters have higher fulfillment preference than those that ignore parameters
109 
110  matchingExpectation_ = potentiallyMatchingExpectations_.removeFirstFinalizedMatchingExpectation();
111  if (matchingExpectation_) {
112  copyOutputParameters(matchingExpectation_);
113  callHasSucceeded();
114  } else {
115  MockCheckedExpectedCall* matchingExpectationWithIgnoredParameters = potentiallyMatchingExpectations_.getFirstMatchingExpectation();
116  if (matchingExpectationWithIgnoredParameters) {
117  copyOutputParameters(matchingExpectationWithIgnoredParameters);
118  }
119  }
120 }

337 void MockCheckedActualCall::checkExpectations()
338 {
339  if (state_ != CALL_IN_PROGRESS)
340  {
341  potentiallyMatchingExpectations_.resetActualCallMatchingState();
342  return;
343  }
344 
345  if (potentiallyMatchingExpectations_.hasFinalizedMatchingExpectations())
346  FAIL("Actual call is in progress, but there are finalized matching expectations when checking expectations. This cannot happen.") // LCOV_EXCL_LINE
347 
348  matchingExpectation_ = potentiallyMatchingExpectations_.removeFirstMatchingExpectation();
349  if (matchingExpectation_) {
350  matchingExpectation_->finalizeActualCallMatch();
351  callHasSucceeded();
352  potentiallyMatchingExpectations_.resetActualCallMatchingState();
353  return;
354  }
355 
356  if (potentiallyMatchingExpectations_.hasUnmatchingExpectationsBecauseOfMissingParameters()) {
357  MockExpectedParameterDidntHappenFailure failure(getTest(), getName(), allExpectations_);
358  failTest(failure);
359  }
360  else {
361  MockExpectedObjectDidntHappenFailure failure(getTest(), getName(), allExpectations_);
362  failTest(failure);
363  }
364 }

514 MockActualCall& MockCheckedActualCall::onObject(const void* objectPtr)
515 {
516  if(hasFailed()) {
517  return *this;
518  }
519 
520  callIsInProgress();
521 
522  potentiallyMatchingExpectations_.onlyKeepExpectationsOnObject(objectPtr);
523 
524  if (potentiallyMatchingExpectations_.isEmpty()) {
525  MockUnexpectedObjectFailure failure(getTest(), getName(), objectPtr, allExpectations_);
526  failTest(failure);
527  return *this;
528  }
529 
530  potentiallyMatchingExpectations_.wasPassedToObject();
531  completeCallWhenMatchIsFound();
532 
533  return *this;
534 }

536 void MockCheckedActualCall::addOutputParameter(const SimpleString& name, const SimpleString& type, void* ptr)
537 {
538  MockOutputParametersListNode* newNode = new MockOutputParametersListNode(name, type, ptr);
539 
540  if (outputParameterExpectations_ == NULL)
541  outputParameterExpectations_ = newNode;
542  else {
543  MockOutputParametersListNode* lastNode = outputParameterExpectations_;
544  while (lastNode->next_) lastNode = lastNode->next_;
545  lastNode->next_ = newNode;
546  }
547 }

549 void MockCheckedActualCall::cleanUpOutputParameterList()
550 {
551  MockOutputParametersListNode* current = outputParameterExpectations_;
552  MockOutputParametersListNode* toBeDeleted = NULL;
553 
554  while (current) {
555  toBeDeleted = current;
556  outputParameterExpectations_ = current = current->next_;
557  delete toBeDeleted;
558  }
559 }
