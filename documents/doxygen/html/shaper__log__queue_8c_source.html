64 shaper_log_queue_t shaperLogQueueNewQueue(uint32_t elemSize, uint32_t queueSize)
65 {
66  if (elemSize < 1 || queueSize < 1)
67  return NULL;
68 
69  shaper_log_queue_t retQueue = calloc(1, sizeof(struct shaper_log_queue));
70  if (retQueue) {
71  retQueue->elemArray = calloc(queueSize, sizeof(struct shaper_log_queue_elem));
72  if (retQueue->elemArray) {
73  uint32_t i1;
74  for (i1 = 0; i1 < queueSize; i1++) {
75  retQueue->elemArray[i1].data = calloc(1, elemSize);
76  if (!retQueue->elemArray[i1].data) {
77  shaperLogQueueDeleteQueue(retQueue);
78  return NULL;
79  }
80  }
81  }
82  else {
83  shaperLogQueueDeleteQueue(retQueue);
84  return NULL;
85  }
86 
87  retQueue->elemSize = elemSize;
88  retQueue->queueSize = queueSize;
89  retQueue->head = 0;
90  retQueue->tail = 0;
91  }
92 
93  return retQueue;
94 }
