49 void *igbRawsockOpen(igb_rawsock_t* rawsock, const char *ifname, bool rx_mode, bool tx_mode, U16 ethertype, U32 frame_size, U32 num_frames)
50 {
51  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK);
52 
53  if (!pcapRawsockOpen((pcap_rawsock_t*)rawsock, ifname, rx_mode,
54  tx_mode, ethertype, frame_size, num_frames))
55  {
56  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
57  return NULL;
58  }
59 
60  if (tx_mode) {
61  // Deal with frame size.
62  if (frame_size == 0) {
63  // use interface MTU as max frames size, if none specified
64  rawsock->base.frameSize = IGB_MTU;
65  }
66  else if (frame_size > IGB_MTU) {
67  AVB_LOGF_ERROR("Creating rawsock; requested frame size exceeds %d", IGB_MTU);
68  igbRawsockClose(rawsock);
69  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
70  return NULL;
71  } else {
72  rawsock->base.frameSize = frame_size;
73  }
74 
75  // IGB setup
76  rawsock->igb_dev = igbAcquireDevice();
77 
78  // select class B queue by default
79  rawsock->queue = 1;
80  }
81 
82  // fill virtual functions table
83  rawsock_cb_t *cb = &rawsock->base.cb;
84  cb->close = igbRawsockClose;
85  cb->getTxFrame = igbRawsockGetTxFrame;
86  cb->relTxFrame = igbRawsockRelTxFrame;
87  cb->txSetMark = igbRawsockTxSetMark;
88  cb->txFrameReady = igbRawsockTxFrameReady;
89  cb->send = igbRawsockSend;
90  cb->txBufLevel = igbRawsockTxBufLevel;
91  cb->getTXOutOfBuffers = igbRawsockGetTXOutOfBuffers;
92  cb->getTXOutOfBuffersCyclic = igbRawsockGetTXOutOfBuffersCyclic;
93 
94  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
95  return rawsock;
96 }

136 // Get a buffer from the ring to use for TX
137 U8 *igbRawsockGetTxFrame(void *pvRawsock, bool blocking, unsigned int *len)
138 {
139  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK_DETAIL);
140 
141  igb_rawsock_t *rawsock = (igb_rawsock_t*)pvRawsock;
142 
143  if (!VALID_TX_RAWSOCK(rawsock) || len == NULL) {
144  AVB_LOG_ERROR("Getting TX frame; bad arguments");
145  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
146  return NULL;
147  }
148 
149  U8 *ret = NULL;
150  int bBufferBusyReported = 0;
151 
152  do {
153  rawsock->tx_packet = igbGetTxPacket(rawsock->igb_dev);
154  if (!rawsock->tx_packet && blocking) {
155  if (0 == bBufferBusyReported) {
156  if (!rawsock->txOutOfBuffer) {
157  AVB_LOGF_DEBUG("Getting TX frame (%p): TX buffer busy", rawsock);
158  }
159  ++rawsock->txOutOfBuffer;
160  ++rawsock->txOutOfBufferCyclic;
161  } else if (1 == bBufferBusyReported) {
162  AVB_LOGF_DEBUG("Getting TX frame (%p): TX buffer busy after usleep(10) verify if there are any late frames", rawsock);
163  }
164  ++bBufferBusyReported;
165  usleep(10);
166  }
167  } while (!rawsock->tx_packet && blocking);
168 
169  if (rawsock->tx_packet) {
170  *len = rawsock->base.frameSize;
171  ret = rawsock->tx_packet->vaddr;
172  }
173 
174  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
175  return ret;
176 }

195 // Release a TX frame, and mark it as ready to send
196 bool igbRawsockTxFrameReady(void *pvRawsock, U8 *pBuffer, unsigned int len, U64 timeNsec)
197 {
198  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK_DETAIL);
199  igb_rawsock_t *rawsock = (igb_rawsock_t*)pvRawsock;
200  if (!VALID_TX_RAWSOCK(rawsock)) {
201  AVB_LOG_ERROR("Send; invalid argument");
202  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
203  return -1;
204  }
205 
206  int err;
207 
208  rawsock->tx_packet->len = len;
209 
210 #if IGB_LAUNCHTIME_ENABLED
211  gptpmaster2local(&gPtpTD, timeNsec, &rawsock->tx_packet->attime);
212 #endif
213 
214  err = igb_xmit(rawsock->igb_dev, rawsock->queue, rawsock->tx_packet);
215  if (err) {
216  AVB_LOGF_ERROR("igb_xmit failed: %s", strerror(err));
217  }
218 
219  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
220  return !err;
221 }
