50 // Base MPEG2 Transport Stream packet size
51 #define MPEG2_TS_PKT_SIZE 188
52 // MPEG2TS sync byte
53 #define MPEG2_TS_SYNC_BYTE 0x47
54 
55 // GStreamer likes to pass 4096-byte buffers, so we want a buffer
56 // bigger than that. And, we're more efficient if the interface layer
57 // passes us whole TS packets, so let's use a default value that's a
58 // multiple of both 188 and 192
59 #define MPEG2TS_MQITEM_SIZE 9024
60 #define MPEG2TS_MQITEM_COUNT 20
61 
62 // MPEG2TS packets with 4 byte timestamp (192 bytes) are called "source packets" in AVTP
63 #define MPEGTS_SRC_PKT_HDR_SIZE 4
64 #define MPEGTS_SRC_PKT_SIZE (MPEG2_TS_PKT_SIZE + MPEGTS_SRC_PKT_HDR_SIZE)
65 
66 // AVTP payload: we allow for 5 source packets (192 * 5) plus the CIP header (8) = 968
67 // TODO: make the number of packet configurable
68 #define AVTP_DFLT_SRC_PKTS 7
69 
70 // AVTP Header sizes
71 #define AVTP_V0_HEADER_SIZE 12
72 #define MAP_HEADER_SIZE 12
73 #define CIP_HEADER_SIZE 8
74 
75 #define TOTAL_HEADER_SIZE (AVTP_V0_HEADER_SIZE + MAP_HEADER_SIZE + CIP_HEADER_SIZE)
76 
77 #define BLOCKS_PER_SRC_PKT 8
78 #define BLOCKS_PER_AVTP_PKT (BLOCKS_PER_SRC_PKT * AVTP_NUM_SRC_PKTS)
79 
81 // AVTP Version 0 Header
83 
84 // This mapping does not directly set or read these.
85 #define HIDX_AVTP_VERZERO96 0
86 
87 // - 1 Byte - TV bit (timestamp valid)
88 #define HIDX_AVTP_HIDE7_TV1 1
89 
90 // - 1 Byte - TU bit (timestamp uncertain)
91 #define HIDX_AVTP_HIDE7_TU1 3
92 
93 
95 // Mapping specific header
97 
98 // - 4 bytes avtp_timestamp
99 #define HIDX_AVTP_TIMESTAMP32 12
100 
101 // - 4 bytes gateway_info
102 #define HIDX_GATEWAY32 16
103 
104 // - 2 bytes stream_data_len (save a pointer for later use)
105 #define HIDX_DATALEN16 20
106 
107 // 2 bit tag = binary 01 (CIP header - included)
108 // 6 bit channel = 31 / 0x1F (Native AVB)
109 #define HIDX_TAG2_CHANNEL6 22
110 
111 // 4 bits tcode = 0xA
112 // 4 bits sy (application-specific) = 0
113 #define HIDX_TCODE4_SY4 23
114 
116 // CIP Header - 2 quadlets (8 bytes)
118 
119 // 2 bits cip flag = binary 00
120 // 6 bits sid (source identifier) = 63 / 0x3F (On the AVB network)
121 #define HIDX_CIP2_SID6 24
122 
123 // 8 bits dbs (data block size) = 6 (6 quadlets per block)
124 #define HIDX_DBS8 25
125 
126 // 2 bits FN (fraction number) = Bx11 (8 data blocks)
127 // 3 bits QPC (quadlet padding count) = Bx000
128 // 1 bit SPH (source packet header) = Bx1 (using source packet headers)
129 // 2 bits RSV (reserved) = Bx00
130 #define HIDX_FN2_QPC3_SPH1_RSV2 26
131 
132 // 8 bits DBC (data block counter) = counter
133 #define HIDX_DBC8 27
134 
135 // 2 bits cip flag (2nd quadlet) = binary 10
136 // 6 bits fmt (stream format) = binary 100000 (MPEG2-TS)
137 #define HIDX_CIP2_FMT6 28
138 
139 // 3 bytes fdf (format dependant)
140 // i.e. for MPEG2-TS
141 // 1 bit tsf (time shift flag) = 0
142 // 23 bits reserved = 0
143 #define HIDX_TSF1_RESA7 29
144 #define HIDX_RESB8 30
145 #define HIDX_RESC8 31
146 
147 #define DEFAULT_SRC_PKTS_PER_AVTP_FRAME 5

193 void openavbMapMpeg2tsCfgCB(media_q_t *pMediaQ, const char *name, const char *value)
194 {
195  AVB_TRACE_ENTRY(AVB_TRACE_MAP);
196 
197  if (pMediaQ) {
198  pvt_data_t *pPvtData = pMediaQ->pPvtMapInfo;
199  if (!pPvtData) {
200  AVB_LOG_ERROR("Private mapping module data not allocated.");
201  return;
202  }
203 
204  char *pEnd;
205  unsigned long tmp;
206  bool nameOK = TRUE, valueOK = FALSE;
207 
208  if (strcmp(name, "map_nv_item_count") == 0) {
209  tmp = strtoul(value, &pEnd, 10);
210  if (pEnd != value && *pEnd == '\0') {
211  pPvtData->itemCount = tmp;
212  valueOK = TRUE;
213  }
214  }
215  else if (strcmp(name, "map_nv_item_size") == 0) {
216  tmp = strtoul(value, &pEnd, 10);
217  if (pEnd != value && *pEnd == '\0') {
218  pPvtData->itemSize = tmp;
219  valueOK = TRUE;
220  }
221  }
222  else if (strcmp(name, "map_nv_ts_packet_size") == 0) {
223  tmp = strtoul(value, &pEnd, 10);
224  if (pEnd != value && *pEnd == '\0' && (tmp == 188 || tmp == 192)) {
225  pPvtData->tsPacketSize = tmp;
226  valueOK = TRUE;
227  }
228  }
229  else if (strcmp(name, "map_nv_num_source_packets") == 0) {
230  tmp = strtoul(value, &pEnd, 10);
231  if (pEnd != value && *pEnd == '\0') {
232  pPvtData->numSourcePackets = tmp;
233  valueOK = TRUE;
234  }
235  }
236  else if (strcmp(name, "map_nv_tx_rate") == 0
237  || strcmp(name, "map_nv_tx_interval") == 0) {
238  tmp = strtoul(value, &pEnd, 10);
239  if (pEnd != value && *pEnd == '\0') {
240  pPvtData->txRate = tmp;
241  valueOK = TRUE;
242  }
243  }
244  else {
245  AVB_LOGF_WARNING("Unknown configuration item: %s", name);
246  nameOK = FALSE;
247  }
248 
249  if (nameOK && !valueOK) {
250  AVB_LOGF_WARNING("Bad value for configuration item: %s = %s", name, value);
251  }
252  }
253 
254  AVB_TRACE_EXIT(AVB_TRACE_MAP);
255 }

329 static int syncScan(media_q_item_t *pMediaQItem, int startIdx)
330 {
331  char *data = pMediaQItem->pPubData;
332  int offset;
333  for (offset = startIdx; offset < pMediaQItem->dataLen; offset++) {
334  if (data[offset] == MPEG2_TS_SYNC_BYTE)
335  break;
336  }
337 
338  AVB_LOGF_WARNING("Dropped %d bytes", offset - startIdx);
339  if (offset >= pMediaQItem->dataLen)
340  offset = -1;
341 
342  return offset;
343 }

345 // This talker callback will be called for each AVB observation interval.
346 tx_cb_ret_t openavbMapMpeg2tsTxCB(media_q_t *pMediaQ, U8 *pData, U32 *dataLen)
347 {
348  AVB_TRACE_ENTRY(AVB_TRACE_MAP_DETAIL);
349  if (pMediaQ && pData && dataLen) {
350  U8 *pHdr = pData;
351  U8 *pPayload = pData + TOTAL_HEADER_SIZE;
352  pvt_data_t *pPvtData = pMediaQ->pPvtMapInfo;
353  if (!pPvtData) {
354  AVB_LOG_ERROR("Private mapping module data not allocated.");
355  return TX_CB_RET_PACKET_NOT_READY;
356  }
357 
358  //pHdr[HIDX_AVTP_TIMESTAMP32] = 0x00; // Set later
359  *(U32 *)(&pHdr[HIDX_GATEWAY32]) = 0x00000000;
360  //pHdr[HIDX_DATALEN16] = 0x00; // Set Later
361  pHdr[HIDX_TAG2_CHANNEL6] = (1 << 6) | 0x1f;
362  pHdr[HIDX_TCODE4_SY4] = (0x0a << 4) | 0;
363 
364  // Set the majority of the CIP header now.
365  pHdr[HIDX_CIP2_SID6] = (0x00 << 6) | 0x3f;
366  pHdr[HIDX_DBS8] = 0x06;
367  pHdr[HIDX_FN2_QPC3_SPH1_RSV2] = (0x03 << 6) | (0x00 << 3) | (0x01 << 2) | 0x00;
368  // pHdr[HIDX_DBC8] = 0; // This will be set later.
369  pHdr[HIDX_CIP2_FMT6] = (0x02 << 6) | 0xa0;
370  pHdr[HIDX_TSF1_RESA7] = (0x01 << 7) | 0x00;
371  pHdr[HIDX_RESB8] = 0x00;
372  pHdr[HIDX_RESC8] = 0x00;
373 
374  media_q_item_t *pMediaQItem = openavbMediaQTailLock(pMediaQ, TRUE);
375  int sourcePacketsAdded = 0;
376  int nItemBytes, nAvailBytes;
377  int offset, bytesNeeded;
378  U32 timestamp;
379  bool moreSourcePackets = TRUE;
380 
381  while (pMediaQItem && moreSourcePackets) {
382 
383  if (pPvtData->unsynched) {
384  // Scan forward, looking for next sync byte.
385  offset = syncScan(pMediaQItem, 0);
386  if (offset >= 0)
387  pMediaQItem->readIdx = offset;
388  else {
389  pPvtData->unsynched = TRUE;
390  pMediaQItem->dataLen = 0;
391  pMediaQItem->readIdx = 0;
392  }
393  }
394 
395  nItemBytes = pMediaQItem->dataLen - pMediaQItem->readIdx;
396  nAvailBytes = nItemBytes + pPvtData->nSavedBytes;
397 
398  if (nItemBytes == 0) {
399  // Empty MQ item, ignore
400  }
401  else if (nAvailBytes >= pPvtData->tsPacketSize) {
402  // PTP walltime already set in the interface module. Just add the max transit time.
403  // If this is a new mq item, add the AVTP transit time to the timestamp
404  if (pMediaQItem->readIdx == 0) {
405  openavbAvtpTimeAddUSec(pMediaQItem->pAvtpTime, pPvtData->maxTransitUsec);
406  }
407 
408  timestamp = openavbAvtpTimeGetAvtpTimestamp(pMediaQItem->pAvtpTime);
409 
410  // Set timestamp valid flag
411  if (openavbAvtpTimeTimestampIsValid(pMediaQItem->pAvtpTime))
412  pHdr[HIDX_AVTP_HIDE7_TV1] |= 0x01; // Set
413  else {
414  pHdr[HIDX_AVTP_HIDE7_TV1] &= ~0x01; // Clear
415  }
416 
417  // Set timestamp uncertain flag
418  if (openavbAvtpTimeTimestampIsUncertain(pMediaQItem->pAvtpTime))
419  pHdr[HIDX_AVTP_HIDE7_TU1] |= 0x01; // Set
420  else pHdr[HIDX_AVTP_HIDE7_TU1] &= ~0x01; // Clear
421 
422  // Set the timestamp.
423  if (sourcePacketsAdded == 0) {
424  // TODO: I think this is wrong with source packets
425  *(U32 *)(&pHdr[HIDX_AVTP_TIMESTAMP32]) = htonl(timestamp);
426  }
427 
428  /* Copy the TS packets into the outgoing AVTP frame
429  */
430  offset = 0, bytesNeeded = pPvtData->tsPacketSize;
431 
432  // If getting 188-byte packets from interface, need to add source packet header
433  if (pPvtData->tsPacketSize == MPEG2_TS_PKT_SIZE) {
434  // Set the timestamp on this source packet
435  *((U32 *)pPayload) = htonl(timestamp);
436  offset = MPEGTS_SRC_PKT_HDR_SIZE;
437  }
438 
439  // Use any leftover data from last MQ item
440  if (pPvtData->nSavedBytes) {
441  memcpy(pPayload + offset, pPvtData->savedBytes, pPvtData->nSavedBytes);
442  offset += pPvtData->nSavedBytes;
443  bytesNeeded -= pPvtData->nSavedBytes;
444  pPvtData->nSavedBytes = 0;
445  }
446 
447  // Now, copy data from current MQ item
448  memcpy(pPayload + offset, pMediaQItem->pPubData + pMediaQItem->readIdx, bytesNeeded);
449 
450  // Check that the data we've copied is synchronized
452  // where we think it should
453  if (pPayload[4] == MPEG2_TS_SYNC_BYTE) {
454  // OK, now we can update the read index
455  pMediaQItem->readIdx += bytesNeeded;
456  // and move the payload ptr for the next source packet
457  pPayload += MPEGTS_SRC_PKT_SIZE;
458 
459  // Keep track of how many source packets have been added to the outgoing packet
460  sourcePacketsAdded++;
461  }
462  else {
463  AVB_LOG_WARNING("Alignment problem");
464 
465  // Scan forward, looking for next sync byte.
466  // Start scan after 4-byte header if getting 192 byte packets.
467  // Ignore saved data if there was any, start from what's in current item.
468  offset = syncScan(pMediaQItem, pMediaQItem->readIdx + (pPvtData->tsPacketSize - MPEG2_TS_PKT_SIZE) + 1);
469  if (offset >= 0)
470  pMediaQItem->readIdx = offset;
471  else {
472  pPvtData->unsynched = TRUE;
473  pMediaQItem->dataLen = 0;
474  pMediaQItem->readIdx = 0;
475  }
476  }
477  }
478  else {
479  // Arghhh - a partial packet.
480  assert(pPvtData->nSavedBytes + nItemBytes <= MPEG2_TS_PKT_SIZE);
481 
482  memcpy(pPvtData->savedBytes + pPvtData->nSavedBytes,
483  pMediaQItem->pPubData + pMediaQItem->readIdx,
484  nItemBytes);
485  pPvtData->nSavedBytes += nItemBytes;
486 
487  pMediaQItem->dataLen = 0;
488  pMediaQItem->readIdx = 0;
489  }
490 
491  // Have we reached our target?
492  if (sourcePacketsAdded >= pPvtData->numSourcePackets)
493  moreSourcePackets = FALSE;
494 
495  // Have we used up the data in the MQ item?
496  if (pMediaQItem->dataLen - pMediaQItem->readIdx == 0) {
497  // release used-up item
498  openavbMediaQTailPull(pMediaQ);
499 
500  // and get a new one, if needed
501  if (moreSourcePackets)
502  pMediaQItem = openavbMediaQTailLock(pMediaQ, TRUE);
503  else pMediaQItem = NULL;
504  }
505 
506  } // while (pMediaQItem && moreSourcePackets)
507 
508  if (pMediaQItem) {
509  // holds more data, leave it in the queue
510  openavbMediaQTailUnlock(pMediaQ);
511  }
512 
513  if (sourcePacketsAdded > 0) {
514 
515  // Set the block continuity and data length
516  pHdr[HIDX_DBC8] = pPvtData->DBC;
517  pPvtData->DBC = (pPvtData->DBC + (sourcePacketsAdded * BLOCKS_PER_SRC_PKT)) & 0x000000ff;
518 
519  U16 datalen = (sourcePacketsAdded * MPEGTS_SRC_PKT_SIZE) + CIP_HEADER_SIZE;
520  *(U16 *)(pHdr + HIDX_DATALEN16) = htons(datalen);
521 
522  // Set out bound data length (entire packet length)
523  *dataLen = (sourcePacketsAdded * MPEGTS_SRC_PKT_SIZE) + TOTAL_HEADER_SIZE;
524 
525  AVB_TRACE_LINE(AVB_TRACE_MAP_LINE);
526  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
527  return TX_CB_RET_PACKET_READY;
528  }
529  }
530 
531  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
532  return TX_CB_RET_PACKET_NOT_READY;
533 }

543 // This callback occurs when running as a listener and data is available.
544 bool openavbMapMpeg2tsRxCB(media_q_t *pMediaQ, U8 *pData, U32 dataLen)
545 {
546  AVB_TRACE_ENTRY(AVB_TRACE_MAP_DETAIL);
547  if (pMediaQ && pData) {
548  U8 *pHdr = pData;
549  U8 *pPayload = pData + TOTAL_HEADER_SIZE;
550  U8 sourcePacketCount = 0;
551  pvt_data_t *pPvtData = pMediaQ->pPvtMapInfo;
552  if (!pPvtData) {
553  AVB_LOG_ERROR("Private mapping module data not allocated.");
554  return FALSE;
555  }
556 
557  //pHdr[HIDX_AVTP_TIMESTAMP32];
558  //pHdr[HIDX_GATEWAY32];
559  //pHdr[HIDX_DATALEN16];
560 
561  //pHdr[HIDX_TAG2_CHANNEL6];
562  //pHdr[HIDX_TCODE4_SY4];
563  //pHdr[HIDX_CIP2_SID6];
564  //pHdr[HIDX_DBS8];
565  //pHdr[HIDX_FN2_QPC3_SPH1_RSV2];
566 
567  // Only support full source packets.
568  // TODO: This mapper could be enhanced to support partial source packets and assemble the datablocks
569  // recieved across multiple avtp packets.
570  if ((pHdr[HIDX_DBC8] % 8) > 0) {
571  AVB_LOG_ERROR("Unsupported MPEG2TS block count received, payload discarded.");
572  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
573  return FALSE;
574  }
575 
576  // TODO: validate DBC
577 
578  // Determine the number of source packets bundled into the AVTP payload.
579  U16 datalen = ntohs(*(U16 *)(pHdr + HIDX_DATALEN16));
580  sourcePacketCount = (datalen - CIP_HEADER_SIZE) / MPEGTS_SRC_PKT_SIZE;
581 
582  //pHdr[HIDX_CIP2_FMT6];
583  //pHdr[HIDX_TSF1_RESA7];
584  //pHdr[HIDX_RESB8];
585  //pHdr[HIDX_RESC8];
586 
587  if (sourcePacketCount > 0) {
588  // Get item in media queue
589  media_q_item_t *pMediaQItem = openavbMediaQHeadLock(pMediaQ);
590  if (pMediaQItem) {
591  if (pMediaQItem->dataLen == 0) {
592  // Get the timestamp and place it in the media queue item.
593  U32 timestamp = ntohl(*(U32 *)(&pHdr[HIDX_AVTP_TIMESTAMP32]));
594  openavbAvtpTimeSetToTimestamp(pMediaQItem->pAvtpTime, timestamp);
595 
596  // Set timestamp valid and timestamp uncertain flags
597  openavbAvtpTimeSetTimestampValid(pMediaQItem->pAvtpTime, (pHdr[HIDX_AVTP_HIDE7_TV1] & 0x01) ? TRUE : FALSE);
598  openavbAvtpTimeSetTimestampUncertain(pMediaQItem->pAvtpTime, (pHdr[HIDX_AVTP_HIDE7_TU1] & 0x01) ? TRUE : FALSE);
599  }
600 
601  int i;
602  for (i = 0; i < sourcePacketCount; i++) {
603  if (pMediaQItem->itemSize - pMediaQItem->dataLen >= pPvtData->tsPacketSize) {
604  memcpy(pMediaQItem->pPubData + pMediaQItem->dataLen,
605  pPayload + MPEGTS_SRC_PKT_SIZE - pPvtData->tsPacketSize,
606  pPvtData->tsPacketSize);
607  pMediaQItem->dataLen += pPvtData->tsPacketSize;
608  pPayload += MPEGTS_SRC_PKT_SIZE;
609  }
610  else {
611  AVB_LOG_ERROR("Data too large for media queue");
612  pMediaQItem->dataLen = 0;
613  break;
614  }
615  }
616 
617  // TODO: try to pack more data into MQ item?
618  openavbMediaQHeadPush(pMediaQ);
619  }
620  else {
621  // Media queue full, data dropped
622  IF_LOG_INTERVAL(1000) AVB_LOG_ERROR("Media queue full");
623  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
624  return FALSE;
625  }
626  }
627  }
628  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
629  return FALSE;
630 }
