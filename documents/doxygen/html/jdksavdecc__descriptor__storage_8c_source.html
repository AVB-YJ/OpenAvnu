109 uint16_t jdksavdecc_descriptor_storage_buffer_get_configuration_count( struct jdksavdecc_entity_model *self_ )
110 {
111  uint16_t r = 0;
112  struct jdksavdecc_descriptor_storage *self = (struct jdksavdecc_descriptor_storage *)self_;
113  uint32_t num_items = self->header.toc_count;
114 
115  if ( num_items > 0 )
116  {
117  // Find the offset for the last item
118  uint32_t item = num_items - 1;
119  uint32_t offset = self->header.toc_offset + sizeof( struct jdksavdecc_descriptor_storage_item ) * item;
120  if ( offset + JDKSAVDECC_DESCRIPTOR_STORAGE_ITEM_CONFIGURATION_INDEX_OFFSET <= self->storage_length )
121  {
122  // Read the configuration number from this last item
123  r = jdksavdecc_uint16_get( self->user_ptr, offset + JDKSAVDECC_DESCRIPTOR_STORAGE_ITEM_CONFIGURATION_INDEX_OFFSET );
124  // the count of configurations is one larger than the last configuration number
125  r = r + 1;
126  }
127  }
128 
129  return r;
130 }

132 static int jdksavdecc_descriptor_storage_buffer_compare_item( const void *lhs_, const void *rhs_ )
133 {
134  int r = 0;
135  struct jdksavdecc_descriptor_storage_item lhs;
136  struct jdksavdecc_descriptor_storage_item rhs;
137  uint64_t lhsv;
138  uint64_t rhsv;
139 
140  jdksavdecc_descriptor_storage_item_read( &lhs, lhs_, 0, JDKSAVDECC_DESCRIPTOR_STORAGE_ITEM_LENGTH );
141  jdksavdecc_descriptor_storage_item_read( &rhs, rhs_, 0, JDKSAVDECC_DESCRIPTOR_STORAGE_ITEM_LENGTH );
142 
143  // items are expected to be ordered by configuration, then descriptor type, then descriptor index
144  lhsv = ( ( (uint64_t)lhs.configuration_index ) << 32 ) + ( ( (uint64_t)lhs.descriptor_type ) << 16 )
145  + ( lhs.descriptor_index );
146 
147  rhsv = ( ( (uint64_t)rhs.configuration_index ) << 32 ) + ( ( (uint64_t)rhs.descriptor_type ) << 16 )
148  + ( rhs.descriptor_index );
149 
150  if ( lhsv < rhsv )
151  {
152  r = -1;
153  }
154  else if ( lhsv > rhsv )
155  {
156  r = 1;
157  }
158  return r;
159 }

161 uint16_t jdksavdecc_descriptor_storage_buffer_read_descriptor( struct jdksavdecc_entity_model *self_,
162  uint16_t configuration_number,
163  uint16_t descriptor_type,
164  uint16_t descriptor_index,
165  uint8_t *result_buffer,
166  uint16_t result_buffer_len )
167 {
168  uint16_t r = 0;
169  struct jdksavdecc_descriptor_storage *self = (struct jdksavdecc_descriptor_storage *)self_;
170  void *p;
171  void *descriptor_items;
172  struct jdksavdecc_descriptor_storage_item key;
173 
174  descriptor_items = ( (uint8_t *)self->user_ptr ) + self->header.toc_offset;
175  key.configuration_index = configuration_number;
176  key.descriptor_type = descriptor_type;
177  key.descriptor_index = descriptor_index;
178  key.length = 0;
179  key.offset = 0;
180 
181  p = bsearch( &key,
182  descriptor_items,
183  self->header.toc_count,
184  JDKSAVDECC_DESCRIPTOR_STORAGE_ITEM_LENGTH,
185  jdksavdecc_descriptor_storage_buffer_compare_item );
186 
187  if ( p )
188  {
189  // found the match, now get the item offset and length and give it to the caller
190  jdksavdecc_descriptor_storage_item_read( &key, p, 0, JDKSAVDECC_DESCRIPTOR_STORAGE_ITEM_LENGTH );
191  if ( key.length <= result_buffer_len )
192  {
193  r = key.length;
194  memcpy( result_buffer, ( (uint8_t *)self->user_ptr ) + key.offset, key.length );
195  }
196  }
197  return r;
198 }
