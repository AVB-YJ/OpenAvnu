184 static void countdown()
185 {
186  if (malloc_out_of_memory_counter <= NO_COUNTDOWN)
187  return;
188 
189  if (malloc_out_of_memory_counter == OUT_OF_MEMORRY)
190  return;
191 
192  malloc_out_of_memory_counter--;
193 
194  if (malloc_out_of_memory_counter == OUT_OF_MEMORRY)
195  cpputest_malloc_set_out_of_memory();
196 }
