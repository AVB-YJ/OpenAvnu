36 ssize_t jdksavdecc_frame_read( struct jdksavdecc_frame *p, void const *base, ssize_t pos, size_t len )
37 {
38  ssize_t r = jdksavdecc_validate_range( pos, len, 14 );
39  if ( r >= 0 )
40  {
41  size_t payload_offset;
42  uint16_t tag;
43 
44  p->dest_address = jdksavdecc_eui48_get( base, pos + 0 );
45  p->src_address = jdksavdecc_eui48_get( base, pos + 6 );
46  tag = jdksavdecc_uint16_get( base, pos + 12 );
47 
48  if ( tag == 0x8100 )
49  {
50  payload_offset = 18;
51  p->tpid = tag;
52  r = jdksavdecc_validate_range( pos, len, pos + payload_offset );
53  if ( r >= 0 )
54  {
55  uint16_t tci = jdksavdecc_uint16_get( base, pos + 14 );
56 
57  p->pcp = ( tci >> 13 ) & 0x7;
58  p->dei = ( tci >> 12 ) & 1;
59  p->vid = tci & 0xfff;
60 
61  p->ethertype = jdksavdecc_uint16_get( base, pos + 16 );
62  }
63  }
64  else
65  {
66  payload_offset = 14;
67  p->tpid = 0;
68  p->dei = 0;
69  p->pcp = 0;
70  p->vid = 0;
71 
72  p->ethertype = tag;
73  }
74 
75  if ( r >= 0 )
76  {
77  p->length = ( uint16_t )( len - payload_offset );
78  memcpy( p->payload, ( (uint8_t *)base ) + pos + payload_offset, p->length );
79  r = len;
80  }
81  }
82  return r;
83 }

85 ssize_t jdksavdecc_frame_write( struct jdksavdecc_frame const *p, void *base, ssize_t pos, size_t len )
86 {
87  ssize_t r = jdksavdecc_validate_range( pos, len, 14 );
88  if ( r >= 0 )
89  {
90  size_t payload_offset;
91 
92  jdksavdecc_eui48_set( p->dest_address, base, pos + 0 );
93  jdksavdecc_eui48_set( p->src_address, base, pos + 6 );
94  if ( p->tpid == 0x8100 )
95  {
96  payload_offset = 18;
97  r = jdksavdecc_validate_range( pos, len, payload_offset );
98  if ( r >= 0 )
99  {
100  uint16_t tci = ( ( p->pcp & 0x7 ) << 13 ) | ( ( p->dei & 1 ) << 12 ) | ( ( p->vid ) & 0xfff );
101  jdksavdecc_uint16_set( p->tpid, base, pos + 12 );
102  jdksavdecc_uint16_set( tci, base, pos + 14 );
103  jdksavdecc_uint16_set( p->ethertype, base, pos + 16 );
104  }
105  }
106  else
107  {
108  payload_offset = 14;
109  jdksavdecc_uint16_set( p->ethertype, base, pos + 12 );
110  }
111 
112  r = jdksavdecc_validate_range( pos, len, pos + payload_offset + p->length );
113  if ( r >= 0 )
114  {
115  memcpy( ( (uint8_t *)base ) + pos + payload_offset, p->payload, p->length );
116  }
117  }
118 
119  return r;
120 }
