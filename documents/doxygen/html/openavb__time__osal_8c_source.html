56 static bool x_timeInit(void) {
57  AVB_TRACE_ENTRY(AVB_TRACE_TIME);
58 
59  if (gptpinit(&gPtpShmFd, &gPtpMmap) < 0) {
60  AVB_LOG_ERROR("GPTP init failed");
61  AVB_TRACE_EXIT(AVB_TRACE_TIME);
62  return FALSE;
63  }
64 
65  if (gptpgetdata(gPtpMmap, &gPtpTD) < 0) {
66  AVB_LOG_ERROR("GPTP data fetch failed");
67  AVB_TRACE_EXIT(AVB_TRACE_TIME);
68  return FALSE;
69  }
70 
71  AVB_LOGF_INFO("local_time = %" PRIu64, gPtpTD.local_time);
72  AVB_LOGF_INFO("ml_phoffset = %" PRId64 ", ls_phoffset = %" PRId64, gPtpTD.ml_phoffset, gPtpTD.ls_phoffset);
73  AVB_LOGF_INFO("ml_freqffset = %Lf, ls_freqoffset = %Lf", gPtpTD.ml_freqoffset, gPtpTD.ls_freqoffset);

75  AVB_TRACE_EXIT(AVB_TRACE_TIME);

79 static bool x_getPTPTime(U64 *timeNsec) {
80  AVB_TRACE_ENTRY(AVB_TRACE_TIME);
81 
82  if (gptpgetdata(gPtpMmap, &gPtpTD) < 0) {
83  AVB_LOG_ERROR("GPTP data fetch failed");
84  AVB_TRACE_EXIT(AVB_TRACE_TIME);
85  return FALSE;
86  }
87 
88  uint64_t now_local;
89  uint64_t update_8021as;
90  int64_t delta_8021as;
91  int64_t delta_local;
92 
93  if (gptplocaltime(&gPtpTD, &now_local)) {
94  update_8021as = gPtpTD.local_time - gPtpTD.ml_phoffset;
95  delta_local = now_local - gPtpTD.local_time;
96  delta_8021as = gPtpTD.ml_freqoffset * delta_local;
97  *timeNsec = update_8021as + delta_8021as;
98 
99  AVB_TRACE_EXIT(AVB_TRACE_TIME);
100  return TRUE;
101  }
102 
103  AVB_TRACE_EXIT(AVB_TRACE_TIME);
104  return FALSE;
105 }

130 bool osalClockGettime(openavb_clockId_t openavbClockId, struct timespec *getTime) {
131  AVB_TRACE_ENTRY(AVB_TRACE_TIME);
132 
133  if (openavbClockId < OPENAVB_CLOCK_WALLTIME)
134  {
135  clockid_t clockId = CLOCK_MONOTONIC;
136  switch (openavbClockId) {
137  case OPENAVB_CLOCK_REALTIME:
138  clockId = CLOCK_REALTIME;
139  break;
140  case OPENAVB_CLOCK_MONOTONIC:
141  clockId = CLOCK_MONOTONIC;
142  break;
143  case OPENAVB_TIMER_CLOCK:
144  clockId = CLOCK_MONOTONIC;
145  break;
146  case OPENAVB_CLOCK_WALLTIME:
147  break;
148  }
149  if (!clock_gettime(clockId, getTime)) return TRUE;
150  }
151  else if (openavbClockId == OPENAVB_CLOCK_WALLTIME) {
152  U64 timeNsec;
153  if (!x_getPTPTime(&timeNsec)) {
154  AVB_TRACE_EXIT(AVB_TRACE_TIME);
155  return FALSE;
156  }
157  getTime->tv_sec = timeNsec / NANOSECONDS_PER_SECOND;
158  getTime->tv_nsec = timeNsec % NANOSECONDS_PER_SECOND;
159  AVB_TRACE_EXIT(AVB_TRACE_TIME);
160  return TRUE;
161  }
162  AVB_TRACE_EXIT(AVB_TRACE_TIME);
163  return FALSE;
164 }

166 bool osalClockGettime64(openavb_clockId_t openavbClockId, U64 *timeNsec) {

173  break;
174  case OPENAVB_CLOCK_MONOTONIC:
175  clockId = CLOCK_MONOTONIC;
176  break;
177  case OPENAVB_TIMER_CLOCK:
178  clockId = CLOCK_MONOTONIC;
179  break;
180  case OPENAVB_CLOCK_WALLTIME:
181  break;
182  }
183  struct timespec getTime;
184  if (!clock_gettime(clockId, &getTime)) {
185  *timeNsec = ((U64)getTime.tv_sec * (U64)NANOSECONDS_PER_SECOND) + (U64)getTime.tv_nsec;
186  AVB_TRACE_EXIT(AVB_TRACE_TIME);
187  return TRUE;
188  }
189  }
190  else if (openavbClockId == OPENAVB_CLOCK_WALLTIME) {
191  AVB_TRACE_EXIT(AVB_TRACE_TIME);
192  return x_getPTPTime(timeNsec);
193  }
194  AVB_TRACE_EXIT(AVB_TRACE_TIME);
195  return FALSE;
196 }
