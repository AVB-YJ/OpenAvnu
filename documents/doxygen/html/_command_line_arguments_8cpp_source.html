51 bool CommandLineArguments::parse(TestPlugin* plugin)
52 {
53  bool correctParameters = true;
54  for (int i = 1; i < ac_; i++) {
55  SimpleString argument = av_[i];
56 
57  if (argument == "-v") verbose_ = true;
58  else if (argument == "-c") color_ = true;
59  else if (argument == "-p") runTestsAsSeperateProcess_ = true;
60  else if (argument == "-lg") listTestGroupNames_ = true;
61  else if (argument == "-ln") listTestGroupAndCaseNames_ = true;
62  else if (argument == "-ri") runIgnored_ = true;
63  else if (argument.startsWith("-r")) SetRepeatCount(ac_, av_, i);
64  else if (argument.startsWith("-g")) AddGroupFilter(ac_, av_, i);
65  else if (argument.startsWith("-sg")) AddStrictGroupFilter(ac_, av_, i);
66  else if (argument.startsWith("-xg")) AddExcludeGroupFilter(ac_, av_, i);
67  else if (argument.startsWith("-xsg")) AddExcludeStrictGroupFilter(ac_, av_, i);
68  else if (argument.startsWith("-n")) AddNameFilter(ac_, av_, i);
69  else if (argument.startsWith("-sn")) AddStrictNameFilter(ac_, av_, i);
70  else if (argument.startsWith("-xn")) AddExcludeNameFilter(ac_, av_, i);
71  else if (argument.startsWith("-xsn")) AddExcludeStrictNameFilter(ac_, av_, i);
72  else if (argument.startsWith("TEST(")) AddTestToRunBasedOnVerboseOutput(ac_, av_, i, "TEST(");
73  else if (argument.startsWith("IGNORE_TEST(")) AddTestToRunBasedOnVerboseOutput(ac_, av_, i, "IGNORE_TEST(");
74  else if (argument.startsWith("-o")) correctParameters = SetOutputType(ac_, av_, i);
75  else if (argument.startsWith("-p")) correctParameters = plugin->parseAllArguments(ac_, av_, i);
76  else if (argument.startsWith("-k")) SetPackageName(ac_, av_, i);
77  else correctParameters = false;
78 
79  if (correctParameters == false) {
80  return false;
81  }
82  }
83  return true;
84 }

137 void CommandLineArguments::SetRepeatCount(int ac, const char** av, int& i)
138 {
139  repeat_ = 0;
140 
141  SimpleString repeatParameter(av[i]);
142  if (repeatParameter.size() > 2) repeat_ = SimpleString::AtoI(av[i] + 2);
143  else if (i + 1 < ac) {
144  repeat_ = SimpleString::AtoI(av[i + 1]);
145  if (repeat_ != 0) i++;
146  }
147 
148  if (0 == repeat_) repeat_ = 2;
149 
150 }

217 void CommandLineArguments::AddTestToRunBasedOnVerboseOutput(int ac, const char** av, int& index, const char* parameterName)
218 {
219  SimpleString wholename = getParameterField(ac, av, index, parameterName);
220  SimpleString testname = wholename.subStringFromTill(',', ')');
221  testname = testname.subString(2);
222  TestFilter* namefilter = new TestFilter(testname);
223  TestFilter* groupfilter = new TestFilter(wholename.subStringFromTill(wholename.at(0), ','));
224  namefilter->strictMatching();
225  groupfilter->strictMatching();
226  groupFilters_ = groupfilter->add(groupFilters_);
227  nameFilters_ = namefilter->add(nameFilters_);
228 }

238 bool CommandLineArguments::SetOutputType(int ac, const char** av, int& i)
239 {
240  SimpleString outputType = getParameterField(ac, av, i, "-o");
241  if (outputType.size() == 0) return false;
242 
243  if (outputType == "normal" || outputType == "eclipse") {
244  outputType_ = OUTPUT_ECLIPSE;
245  return true;
246  }
247  if (outputType == "junit") {
248  outputType_ = OUTPUT_JUNIT;
249  return true;
250  }
251  if (outputType == "teamcity") {
252  outputType_ = OUTPUT_TEAMCITY;
253  return true;
254  }
