144 // Each configuration name value pair for this mapping will result in this callback being called.
145 void openavbMapH264CfgCB(media_q_t *pMediaQ, const char *name, const char *value)
146 {
147  AVB_TRACE_ENTRY(AVB_TRACE_MAP);
148 
149  if (pMediaQ) {
150  pvt_data_t *pPvtData = pMediaQ->pPvtMapInfo;
151  if (!pPvtData) {
152  AVB_LOG_ERROR("Private mapping module data not allocated.");
153  return;
154  }
155 
156  char *pEnd;
157 
158  if (strcmp(name, "map_nv_item_count") == 0) {
159  pPvtData->itemCount = strtol(value, &pEnd, 10);
160  }
161  else if (strcmp(name, "map_nv_tx_rate") == 0
162  || strcmp(name, "map_nv_tx_interval") == 0) {
163  pPvtData->txInterval = strtol(value, &pEnd, 10);
164  }
165  else if (strcmp(name, "map_nv_max_payload_size") == 0) {
166  U32 size = strtol(value, &pEnd, 10);
167  if (size > MAX_PAYLOAD_SIZE) {
168  AVB_LOGF_WARNING("map_nv_max_payload_size too large. Parameter set to default: %d", MAX_PAYLOAD_SIZE);
169  size = MAX_PAYLOAD_SIZE;
170  }
171  pPvtData->maxPayloadSize = size;
172  pPvtData->maxDataSize = (pPvtData->maxPayloadSize + TOTAL_HEADER_SIZE);
173  pPvtData->itemSize = pPvtData->maxPayloadSize;
174  }
175  }
176 
177  AVB_TRACE_EXIT(AVB_TRACE_MAP);
178 }

254 // This talker callback will be called for each AVB observation interval.
255 tx_cb_ret_t openavbMapH264TxCB(media_q_t *pMediaQ, U8 *pData, U32 *dataLen)
256 {
257  AVB_TRACE_ENTRY(AVB_TRACE_MAP_DETAIL);
258  if (pMediaQ && pData && dataLen) {
259  U8 *pHdr = pData;
260  U8 *pPayload = pData + TOTAL_HEADER_SIZE;
261  pvt_data_t *pPvtData = pMediaQ->pPvtMapInfo;
262  if (!pPvtData) {
263  AVB_LOG_ERROR("Private mapping module data not allocated.");
264  return TX_CB_RET_PACKET_NOT_READY;
265  }
266 
267  //pHdr[HIDX_AVTP_TIMESTAMP32] = 0x00; // Set later
268  pHdr[HIDX_FORMAT8] = 0x02; // RTP Payload type
269  pHdr[HIDX_FORMAT_SUBTYPE8] = 0x01; // H.264 subtype
270  pHdr[HIDX_RESV16] = 0x0000; // Reserved
271  //pHdr[HIDX_STREAM_DATA_LEN16] = 0x0000; // Set later
272  //pHdr[HIDX_M31_M21_M11_M01_EVT2_RESV2] = 0x00; // M0 set later
273  pHdr[HIDX_RESV8] = 0x00;
274 
275  media_q_item_t *pMediaQItem = openavbMediaQTailLock(pMediaQ, TRUE);
276  if (pMediaQItem) {
277  if (pMediaQItem->dataLen > 0) {
278  if (pMediaQItem->dataLen > pPvtData->itemSize) {
279  AVB_LOGF_ERROR("Media queue data item size too large. Reported size: %d Max Size: %d", pMediaQItem->dataLen, pPvtData->itemSize);
280  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
281  openavbMediaQTailPull(pMediaQ);
282  return TX_CB_RET_PACKET_NOT_READY;
283  }
284 
285  // PTP walltime already set in the interface module. Just add the max transit time.
286  openavbAvtpTimeAddUSec(pMediaQItem->pAvtpTime, pPvtData->maxTransitUsec);
287 
288  // Set timestamp valid flag
289  if (openavbAvtpTimeTimestampIsValid(pMediaQItem->pAvtpTime))
290  pHdr[HIDX_AVTP_HIDE7_TV1] |= 0x01; // Set
291  else {
292  pHdr[HIDX_AVTP_HIDE7_TV1] &= ~0x01; // Clear
293  }
294 
295  // Set timestamp uncertain flag
296  if (openavbAvtpTimeTimestampIsUncertain(pMediaQItem->pAvtpTime))
297  pHdr[HIDX_AVTP_HIDE7_TU1] |= 0x01; // Set
298  else pHdr[HIDX_AVTP_HIDE7_TU1] &= ~0x01; // Clear
299 
300  // Set the timestamp.
301  *(U32 *)(&pHdr[HIDX_AVTP_TIMESTAMP32]) = htonl(openavbAvtpTimeGetAvtpTimestamp(pMediaQItem->pAvtpTime));
302 
303  if (((media_q_item_map_h264_pub_data_t *)pMediaQItem->pPubMapData)->lastPacket) {
304  pHdr[HIDX_M31_M21_M11_M01_EVT2_RESV2] = 0x10;;
305  }
306  else {
307  pHdr[HIDX_M31_M21_M11_M01_EVT2_RESV2] = 0x00;
308  }
309 
310  // Set h264_timestamp
311  *(U32 *)(&pHdr[HIDX_H264_TIMESTAMP32]) =
312  htonl(((media_q_item_map_h264_pub_data_t *)pMediaQItem->pPubMapData)->timestamp);
313 
314  // Copy the h264 rtp payload into the outgoing avtp packet.
315  memcpy(pPayload, pMediaQItem->pPubData, pMediaQItem->dataLen);
316 
317  *(U16 *)(&pHdr[HIDX_STREAM_DATA_LEN16]) = htons(pMediaQItem->dataLen);
318 
319  // Set out bound data length (entire packet length)
320  *dataLen = pMediaQItem->dataLen + TOTAL_HEADER_SIZE;
321 
322  AVB_TRACE_LINE(AVB_TRACE_MAP_LINE);
323  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
324  openavbMediaQTailPull(pMediaQ);
325  return TX_CB_RET_PACKET_READY;
326  }
327  openavbMediaQTailPull(pMediaQ);
328  }
329  }
330 
331  if (dataLen) {
332  // Set out bound data length (entire packet length)
333  *dataLen = 0;
334  }
335 
336  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
337  return TX_CB_RET_PACKET_NOT_READY;
338 }

348 // This callback occurs when running as a listener and data is available.
349 bool openavbMapH264RxCB(media_q_t *pMediaQ, U8 *pData, U32 dataLen)
350 {
351  AVB_TRACE_ENTRY(AVB_TRACE_MAP_DETAIL);
352  if (pMediaQ && pData) {
353  U8 *pHdr = pData;
354  U8 *pPayload = pData + TOTAL_HEADER_SIZE;
355 
356 
357  //pHdr[HIDX_AVTP_TIMESTAMP32]
358  //pHdr[HIDX_FORMAT8]
359  //pHdr[HIDX_FORMAT_SUBTYPE8]
360  //pHdr[HIDX_RESV16]
361  //pHdr[HIDX_STREAM_DATA_LEN16]
362  U16 payloadLen = ntohs(*(U16 *)(&pHdr[HIDX_STREAM_DATA_LEN16]));
363  //pHdr[HIDX_M31_M21_M11_M01_EVT2_RESV2]
364  //pHdr[HIDX_RESV8]
365 
366  // validate header
367  if (payloadLen > dataLen - TOTAL_HEADER_SIZE) {
368  IF_LOG_INTERVAL(1000) AVB_LOG_ERROR("header data len > actual data len");
369  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
370  return FALSE;
371  }
372 
373  // Get item pointer in media queue
374  media_q_item_t *pMediaQItem = openavbMediaQHeadLock(pMediaQ);
375  if (pMediaQItem) {
376  // Get the timestamp and place it in the media queue item.
377  U32 timestamp = ntohl(*(U32 *)(&pHdr[HIDX_AVTP_TIMESTAMP32]));
378  openavbAvtpTimeSetToTimestamp(pMediaQItem->pAvtpTime, timestamp);
379 
380  // Set timestamp valid and timestamp uncertain flags
381  openavbAvtpTimeSetTimestampValid(pMediaQItem->pAvtpTime, (pHdr[HIDX_AVTP_HIDE7_TV1] & 0x01) ? TRUE : FALSE);
382  openavbAvtpTimeSetTimestampUncertain(pMediaQItem->pAvtpTime, (pHdr[HIDX_AVTP_HIDE7_TU1] & 0x01) ? TRUE : FALSE);
383 
384  if (pHdr[HIDX_M31_M21_M11_M01_EVT2_RESV2] & 0x10)
385  ((media_q_item_map_h264_pub_data_t *)pMediaQItem->pPubMapData)->lastPacket = TRUE;
386  else
387  ((media_q_item_map_h264_pub_data_t *)pMediaQItem->pPubMapData)->lastPacket = FALSE;
388 
389  ((media_q_item_map_h264_pub_data_t *)pMediaQItem->pPubMapData)->timestamp =
390  ntohl(*(U32 *)(&pHdr[HIDX_H264_TIMESTAMP32]));
391 
392  if (pMediaQItem->itemSize >= payloadLen) {
393  memcpy(pMediaQItem->pPubData, pPayload, payloadLen);
394  pMediaQItem->dataLen = payloadLen;
395  }
396  else {
397  AVB_LOGF_ERROR("Data to large for media queue (itemSize %d, data size %d).", pMediaQItem->itemSize, payloadLen);
398  pMediaQItem->dataLen = 0;
399  }
400 
401  openavbMediaQHeadPush(pMediaQ);
402  AVB_TRACE_LINE(AVB_TRACE_MAP_LINE);
403  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
404  return TRUE;
405  }
406  else {
407  IF_LOG_INTERVAL(1000) AVB_LOG_ERROR("Media queue full");
408  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
409  return FALSE; // Media queue full
410  }
411  }
412  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
413  return FALSE;
414 }
