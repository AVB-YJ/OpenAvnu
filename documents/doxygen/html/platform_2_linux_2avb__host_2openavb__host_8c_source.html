80 /***********************************************
81  * Signal handler - used to respond to signals.
82  * Allows graceful cleanup.
83  */
84 static void openavbTLSigHandler(int signal)
85 {
86  AVB_TRACE_ENTRY(AVB_TRACE_HOST);
87 
88  if (signal == SIGINT || signal == SIGTERM) {
89  if (bRunning) {
90  AVB_LOG_INFO("Host shutting down");
91  bRunning = FALSE;
92  }
93  else {
94  // Force shutdown
95  exit(2);
96  }
97  }
98  else if (signal == SIGUSR1) {
99  AVB_LOG_DEBUG("Waking up streaming thread");
100  }
101  else {
102  AVB_LOG_ERROR("Unexpected signal");
103  }
104 
105  AVB_TRACE_EXIT(AVB_TRACE_HOST);
106 }

129 /**********************************************
130  * main
131  */
132 int main(int argc, char *argv[])
133 {
134  AVB_TRACE_ENTRY(AVB_TRACE_HOST);
135 
136  int iniIdx = 0;
137  char *programName;
138  char *optIfnameGlobal = NULL;
139  char *optLogFileName = NULL;
140 
141  programName = strrchr(argv[0], '/');
142  programName = programName ? programName + 1 : argv[0];
143 
144  if (argc < 2) {
145  openavbTlHostUsage(programName);
146  exit(-1);
147  }
148 
149  tl_handle_t *tlHandleList = NULL;
150  int i1;
151 
152  // Process command line
153  bool optDone = FALSE;
154  while (!optDone) {
155  int opt = getopt(argc, argv, "hI:l:");
156  if (opt != EOF) {
157  switch (opt) {
158  case 'I':
159  optIfnameGlobal = strdup(optarg);
160  break;
161  case 'l':
162  optLogFileName = strdup(optarg);
163  break;
164  case 'h':
165  default:
166  openavbTlHostUsage(programName);
167  exit(-1);
168  }
169  }
170  else {
171  optDone = TRUE;
172  }
173  }
174 
175  osalAVBInitialize(optLogFileName, optIfnameGlobal);
176 
177  iniIdx = optind;
178  U32 tlCount = argc - iniIdx;
179 
180  if (!openavbTLInitialize(tlCount)) {
181  AVB_LOG_ERROR("Unable to initialize talker listener library");
182  osalAVBFinalize();
183  exit(-1);
184  }
185 
186  // Setup signal handler
187  // We catch SIGINT and shutdown cleanly
188  bool err;
189  struct sigaction sa;
190  sa.sa_handler = openavbTLSigHandler;
191  sigemptyset(&sa.sa_mask);
192  sa.sa_flags = 0;
193  err = sigaction(SIGINT, &sa, NULL);
194  if (err)
195  {
196  AVB_LOG_ERROR("Failed to setup SIGINT handler");
197  osalAVBFinalize();
198  exit(-1);
199  }
200  err = sigaction(SIGTERM, &sa, NULL);
201  if (err)
202  {
203  AVB_LOG_ERROR("Failed to setup SIGTERM handler");
204  osalAVBFinalize();
205  exit(-1);
206  }
207  err = sigaction(SIGUSR1, &sa, NULL);
208  if (err)
209  {
210  AVB_LOG_ERROR("Failed to setup SIGUSR1 handler");
211  osalAVBFinalize();
212  exit(-1);
213  }
214 
215  // Ignore SIGPIPE signals.
216  signal(SIGPIPE, SIG_IGN);
217 
218  registerStaticMapModule(openavbMapPipeInitialize);
219  registerStaticMapModule(openavbMapAVTPAudioInitialize);
220  registerStaticMapModule(openavbMapCtrlInitialize);
221  registerStaticMapModule(openavbMapH264Initialize);
222  registerStaticMapModule(openavbMapMjpegInitialize);
223  registerStaticMapModule(openavbMapMpeg2tsInitialize);
224  registerStaticMapModule(openavbMapNullInitialize);
225  registerStaticMapModule(openavbMapUncmpAudioInitialize);
226 
227  registerStaticIntfModule(openavbIntfEchoInitialize);
228  registerStaticIntfModule(openavbIntfCtrlInitialize);
229  registerStaticIntfModule(openavbIntfLoggerInitialize);
230  registerStaticIntfModule(openavbIntfNullInitialize);
231  registerStaticIntfModule(openavbIntfToneGenInitialize);
232  registerStaticIntfModule(openavbIntfViewerInitialize);
233  registerStaticIntfModule(openavbIntfAlsaInitialize);
234  registerStaticIntfModule(openavbIntfMpeg2tsFileInitialize);
235  registerStaticIntfModule(openavbIntfWavFileInitialize);
236 #ifdef AVB_FEATURE_GSTREAMER
237  registerStaticIntfModule(openavbIntfMjpegGstInitialize);
238  registerStaticIntfModule(openavbIntfMpeg2tsGstInitialize);
239  registerStaticIntfModule(openavbIntfH264RtpGstInitialize);
240 #endif
241  tlHandleList = calloc(1, sizeof(tl_handle_t) * tlCount);
242 
243  // Open all streams
244  for (i1 = 0; i1 < tlCount; i1++) {
245  tlHandleList[i1] = openavbTLOpen();
246  }
247 
248  // Parse ini and configure all streams
249  for (i1 = 0; i1 < tlCount; i1++) {
250  openavb_tl_cfg_t cfg;
251  openavb_tl_cfg_name_value_t NVCfg;
252  char iniFile[1024];
253 
254  snprintf(iniFile, sizeof(iniFile), "%s", argv[i1 + iniIdx]);
255 
256  if (optIfnameGlobal && !strcasestr(iniFile, ",ifname=")) {
257  snprintf(iniFile + strlen(iniFile), sizeof(iniFile), ",ifname=%s", optIfnameGlobal);
258  }
259 
260  openavbTLInitCfg(&cfg);
261  memset(&NVCfg, 0, sizeof(NVCfg));
262 
263  if (!openavbTLReadIniFileOsal(tlHandleList[i1], iniFile, &cfg, &NVCfg)) {
264  AVB_LOGF_ERROR("Error reading ini file: %s\n", argv[i1 + 1]);
265  osalAVBFinalize();
266  exit(-1);
267  }
268  if (!openavbTLConfigure(tlHandleList[i1], &cfg, &NVCfg)) {
269  AVB_LOGF_ERROR("Error configuring: %s\n", argv[i1 + 1]);
270  osalAVBFinalize();
271  exit(-1);
272  }
273 
274  int i2;
275  for (i2 = 0; i2 < NVCfg.nLibCfgItems; i2++) {
276  free(NVCfg.libCfgNames[i2]);
277  free(NVCfg.libCfgValues[i2]);
278  }
279  }
280 
281 #ifdef AVB_FEATURE_GSTREAMER
282  // If we're supporting the interface modules which use GStreamer,
283  // initialize GStreamer here to avoid errors.
284  gst_init(0, NULL);
285 #endif
286 
287  // Run any streams where the stop initial state was not requested.
288  for (i1 = 0; i1 < tlCount; i1++) {
289  if (openavbTLGetInitialState(tlHandleList[i1]) != TL_INIT_STATE_STOPPED) {
290  openavbTLRun(tlHandleList[i1]);
291  }
292  }
293 
294  while (bRunning) {
295  SLEEP_MSEC(1);
296  }
297 
298  for (i1 = 0; i1 < tlCount; i1++) {
299  openavbTLStop(tlHandleList[i1]);
300  }
301 
302  for (i1 = 0; i1 < tlCount; i1++) {
303  openavbTLClose(tlHandleList[i1]);
304  }
305 
306  openavbTLCleanup();
307 
308  if (optLogFileName) {
309  free(optLogFileName);
310  optLogFileName = NULL;
311  }
312 
313 #ifdef AVB_FEATURE_GSTREAMER
314  // If we're supporting the interface modules which use GStreamer,
315  // De-initialize GStreamer to clean up resources.
316  gst_deinit();
317 #endif
318 
319  osalAVBFinalize();
320 
321  AVB_TRACE_EXIT(AVB_TRACE_HOST);
322  exit(0);
323 }
