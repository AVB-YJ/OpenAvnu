42 // Open a rawsock for TX or RX
43 void* ringRawsockOpen(ring_rawsock_t *rawsock, const char *ifname, bool rx_mode, bool tx_mode, U16 ethertype, U32 frame_size, U32 num_frames)
44 {
45  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK);
46 
47  if (!simpleRawsockOpen((simple_rawsock_t*)rawsock, ifname, rx_mode,
48  tx_mode, ethertype, frame_size, num_frames))
49  {
50  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
51  return NULL;
52  }
53 
54  rawsock->pMem = (void*)(-1);
55 
56  // Use version 2 headers for the MMAP packet stuff - avoids 32/64
57  // bit problems, gives nanosecond timestamps, and allows rx of vlan id
58  int val = TPACKET_V2;
59  if (setsockopt(rawsock->sock, SOL_PACKET, PACKET_VERSION, &val, sizeof(val)) < 0) {
60  AVB_LOGF_ERROR("Creating rawsock; get PACKET_VERSION: %s", strerror(errno));
61  ringRawsockClose(rawsock);
62  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
63  return NULL;
64  }
65 
66  // Get the size of the headers in the ring
67  unsigned len = sizeof(val);
68  if (getsockopt(rawsock->sock, SOL_PACKET, PACKET_HDRLEN, &val, &len) < 0) {
69  AVB_LOGF_ERROR("Creating rawsock; get PACKET_HDRLEN: %s", strerror(errno));
70  ringRawsockClose(rawsock);
71  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
72  return NULL;
73  }
74  rawsock->bufHdrSize = TPACKET_ALIGN(val);
75 
76  if (rawsock->base.rxMode) {
77  rawsock->bufHdrSize = rawsock->bufHdrSize + TPACKET_ALIGN(sizeof(struct sockaddr_ll));
78  }
79  rawsock->bufferSize = rawsock->base.frameSize + rawsock->bufHdrSize;
80  rawsock->frameCount = num_frames;
81  AVB_LOGF_DEBUG("frameSize=%d, bufHdrSize=%d(%d+%zu) bufferSize=%d, frameCount=%d",
82  rawsock->base.frameSize, rawsock->bufHdrSize, val, sizeof(struct sockaddr_ll),
83  rawsock->bufferSize, rawsock->frameCount);
84 
85  // Get number of bytes in a memory page. The blocks we ask for
86  // must be a multiple of pagesize. (Actually, it should be
87  // (pagesize * 2^N) to avoid wasting memory.)
88  int pagesize = getpagesize();
89  rawsock->blockSize = pagesize * 4;
90  AVB_LOGF_DEBUG("pagesize=%d blockSize=%d", pagesize, rawsock->blockSize);
91 
92  // Calculate number of buffers and frames based on blocks
93  int buffersPerBlock = rawsock->blockSize / rawsock->bufferSize;
94  rawsock->blockCount = rawsock->frameCount / buffersPerBlock + 1;
95  rawsock->frameCount = buffersPerBlock * rawsock->blockCount;
96 
97  AVB_LOGF_DEBUG("frameCount=%d, buffersPerBlock=%d, blockCount=%d",
98  rawsock->frameCount, buffersPerBlock, rawsock->blockCount);
99 
100  // Fill in the kernel structure with our calculated values
101  struct tpacket_req s_packet_req;
102  memset(&s_packet_req, 0, sizeof(s_packet_req));
103  s_packet_req.tp_block_size = rawsock->blockSize;
104  s_packet_req.tp_frame_size = rawsock->bufferSize;
105  s_packet_req.tp_block_nr = rawsock->blockCount;
106  s_packet_req.tp_frame_nr = rawsock->frameCount;
107 
108  // Ask the kernel to create the TX_RING or RX_RING
109  if (rawsock->base.txMode) {
110  if (setsockopt(rawsock->sock, SOL_PACKET, PACKET_TX_RING,
111  (char*)&s_packet_req, sizeof(s_packet_req)) < 0) {
112  AVB_LOGF_ERROR("Creating rawsock; TX_RING: %s", strerror(errno));
113  ringRawsockClose(rawsock);
114  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
115  return NULL;
116  }
117  AVB_LOGF_DEBUG("PACKET_%s_RING OK", "TX");
118  }
119  else {
120  if (setsockopt(rawsock->sock, SOL_PACKET, PACKET_RX_RING,
121  (char*)&s_packet_req, sizeof(s_packet_req)) < 0) {
122  AVB_LOGF_ERROR("Creating rawsock, RX_RING: %s", strerror(errno));
123  ringRawsockClose(rawsock);
124  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
125  return NULL;
126  }
127  AVB_LOGF_DEBUG("PACKET_%s_RING OK", "TX");
128  }
129 
130  // Call MMAP to get access to the memory used for the ring
131  rawsock->memSize = rawsock->blockCount * rawsock->blockSize;
132  AVB_LOGF_DEBUG("memSize=%zu (%d, %d), sock=%d",
133  rawsock->memSize,
134  rawsock->blockCount,
135  rawsock->blockSize,
136  rawsock->sock);
137  rawsock->pMem = mmap((void*)0, rawsock->memSize, PROT_READ|PROT_WRITE, MAP_SHARED, rawsock->sock, (off_t)0);
138  if (rawsock->pMem == (void*)(-1)) {
139  AVB_LOGF_ERROR("Creating rawsock; MMAP: %s", strerror(errno));
140  ringRawsockClose(rawsock);
141  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
142  return NULL;
143  }
144  AVB_LOGF_DEBUG("mmap: %p", rawsock->pMem);
145 
146  // Initialize the memory
147  memset(rawsock->pMem, 0, rawsock->memSize);
148 
149  // Initialize the state of the ring
150  rawsock->blockIndex = 0;
151  rawsock->bufferIndex = 0;
152  rawsock->buffersOut = 0;
153  rawsock->buffersReady = 0;
154 
155  // fill virtual functions table
156  rawsock_cb_t *cb = &rawsock->base.cb;
157  cb->close = ringRawsockClose;
158  cb->getTxFrame = ringRawsockGetTxFrame;
159  cb->relTxFrame = ringRawsockRelTxFrame;
160  cb->txFrameReady = ringRawsockTxFrameReady;
161  cb->send = ringRawsockSend;
162  cb->txBufLevel = ringRawsockTxBufLevel;
163  cb->rxBufLevel = ringRawsockRxBufLevel;
164  cb->getRxFrame = ringRawsockGetRxFrame;
165  cb->rxParseHdr = ringRawsockRxParseHdr;
166  cb->relRxFrame = ringRawsockRelRxFrame;
167  cb->getTXOutOfBuffers = ringRawsockGetTXOutOfBuffers;
168  cb->getTXOutOfBuffersCyclic = ringRawsockGetTXOutOfBuffersCyclic;
169 
170  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
171  return rawsock;
172 }

193 // Get a buffer from the ring to use for TX
194 U8* ringRawsockGetTxFrame(void *pvRawsock, bool blocking, unsigned int *len)
195 {
196  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK_DETAIL);
197  ring_rawsock_t *rawsock = (ring_rawsock_t*)pvRawsock;
198 
199  // Displays only warning when buffer busy after second try
200  int bBufferBusyReported = 0;
201 
202 
203  if (!VALID_TX_RAWSOCK(rawsock) || len == NULL) {
204  AVB_LOG_ERROR("Getting TX frame; bad arguments");
205  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
206  return NULL;
207  }
208  if (rawsock->buffersOut >= rawsock->frameCount) {
209  AVB_LOG_ERROR("Getting TX frame; too many TX buffers in use");
210  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
211  return NULL;
212  }
213 
214  // Get pointer to next framebuf.
215  volatile struct tpacket2_hdr *pHdr =
216  (struct tpacket2_hdr*)(rawsock->pMem
217  + (rawsock->blockIndex * rawsock->blockSize)
218  + (rawsock->bufferIndex * rawsock->bufferSize));
219  // And pointer to portion of buffer to be filled with frame
220  volatile U8 *pBuffer = (U8*)pHdr + rawsock->bufHdrSize;
221 
222  AVB_LOGF_VERBOSE("block=%d, buffer=%d, out=%d, pBuffer=%p, pHdr=%p",
223  rawsock->blockIndex, rawsock->bufferIndex, rawsock->buffersOut,
224  pBuffer, pHdr);
225 
226  // Check if buffer ready for user
227  // In send mode, we want to see TP_STATUS_AVAILABLE
228  while (pHdr->tp_status != TP_STATUS_AVAILABLE)
229  {
230  switch (pHdr->tp_status) {
231  case TP_STATUS_SEND_REQUEST:
232  case TP_STATUS_SENDING:
233  if (blocking) {
234 #if 0
235 // We should be able to poll on the socket to wait for the buffer to
236 // be ready, but it doesn't work (at least on 2.6.37).
237 // Keep this code, because it may work on newer kernels
238  // poll until tx buffer is ready
239  struct pollfd pfd;
240  pfd.fd = rawsock->sock;
241  pfd.events = POLLWRNORM;
242  pfd.revents = 0;
243  int ret = poll(&pfd, 1, -1);
244  if (ret < 0 && errno != EINTR) {
245  AVB_LOGF_DEBUG("getting TX frame; poll failed: %s", strerror(errno));
246  }
247 #else
248  // Can't poll, so sleep instead to avoid tight loop
249  if(0 == bBufferBusyReported) {
250  if(!rawsock->txOutOfBuffer) {
251  // Display this info only once just to let know that something like this happened
252  AVB_LOGF_INFO("Getting TX frame (sock=%d): TX buffer busy", rawsock->sock);
253  }
254 
255  ++rawsock->txOutOfBuffer;
256  ++rawsock->txOutOfBufferCyclic;
257  } else if(1 == bBufferBusyReported) {
258  //Display this warning if buffer was busy more than once because it might influence late/lost
259  AVB_LOGF_WARNING("Getting TX frame (sock=%d): TX buffer busy after usleep(50) verify if there are any lost/late frames", rawsock->sock);
260  }
261 
262  ++bBufferBusyReported;
263 
264  usleep(50);
265 #endif
266  }
267  else {
268  AVB_LOG_DEBUG("Non-blocking, return NULL");
269  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
270  return NULL;
271  }
272  break;
273  case TP_STATUS_WRONG_FORMAT:
274  default:
275  pHdr->tp_status = TP_STATUS_AVAILABLE;
276  break;
277  }
278  }
279 
280  // Remind client how big the frame buffer is
281  if (len)
282  *len = rawsock->base.frameSize;
283 
284  // increment indexes to point to next buffer
285  if (++(rawsock->bufferIndex) >= (rawsock->frameCount/rawsock->blockCount)) {
286  rawsock->bufferIndex = 0;
287  if (++(rawsock->blockIndex) >= rawsock->blockCount) {
288  rawsock->blockIndex = 0;
289  }
290  }
291 
292  // increment the count of buffers held by client
293  rawsock->buffersOut += 1;
294 
295  // warn if too many are out
296  if (rawsock->buffersOut >= (rawsock->frameCount * 4)/5) {
297  AVB_LOGF_WARNING("Getting TX frame; consider increasing buffers: count=%d, out=%d",
298  rawsock->frameCount, rawsock->buffersOut);
299  }
300 
301  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
302  return (U8*)pBuffer;
303 }

305 // Release a TX frame, without marking it as ready to send
306 bool ringRawsockRelTxFrame(void *pvRawsock, U8 *pBuffer)
307 {
308  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK_DETAIL);
309  ring_rawsock_t *rawsock = (ring_rawsock_t*)pvRawsock;
310  if (!VALID_TX_RAWSOCK(rawsock) || pBuffer == NULL) {
311  AVB_LOG_ERROR("Releasing TX frame; invalid argument");
312  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
313  return FALSE;
314  }
315 
316  volatile struct tpacket2_hdr *pHdr = (struct tpacket2_hdr*)(pBuffer - rawsock->bufHdrSize);
317  AVB_LOGF_VERBOSE("pBuffer=%p, pHdr=%p", pBuffer, pHdr);
318 
319  pHdr->tp_len = 0;
320  pHdr->tp_status = TP_STATUS_KERNEL;
321  rawsock->buffersOut -= 1;
322 
323  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
324  return TRUE;
325 }

327 // Release a TX frame, and mark it as ready to send
328 bool ringRawsockTxFrameReady(void *pvRawsock, U8 *pBuffer, unsigned int len, U64 timeNsec)
329 {
330  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK_DETAIL);
331  ring_rawsock_t *rawsock = (ring_rawsock_t*)pvRawsock;
332 
333  if (!VALID_TX_RAWSOCK(rawsock)) {
334  AVB_LOG_ERROR("Marking TX frame ready; invalid argument");
335  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
336  return FALSE;
337  }
338 
339  if (timeNsec) {
340  IF_LOG_INTERVAL(1000) AVB_LOG_WARNING("launch time is unsupported in ring_rawsock");
341  }
342 
343 
344  volatile struct tpacket2_hdr *pHdr = (struct tpacket2_hdr*)(pBuffer - rawsock->bufHdrSize);
345  AVB_LOGF_VERBOSE("pBuffer=%p, pHdr=%p szFrame=%d, len=%d", pBuffer, pHdr, rawsock->base.frameSize, len);
346 
347  assert(len <= rawsock->bufferSize);
348  pHdr->tp_len = len;
349  pHdr->tp_status = TP_STATUS_SEND_REQUEST;
350  rawsock->buffersReady += 1;
351 
352  if (rawsock->buffersReady >= rawsock->frameCount) {
353  AVB_LOG_WARNING("All buffers in ready/unsent state, calling send");
354  ringRawsockSend(pvRawsock);
355  }
356 
357  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
358  return TRUE;
359 }

361 // Send all packets that are ready (i.e. tell kernel to send them)
362 int ringRawsockSend(void *pvRawsock)
363 {
364  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK_DETAIL);
365  ring_rawsock_t *rawsock = (ring_rawsock_t*)pvRawsock;
366  if (!VALID_TX_RAWSOCK(rawsock)) {
367  AVB_LOG_ERROR("Send; invalid argument");
368  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
369  return -1;
370  }
371 
372  // Linux does something dumb to wait for frames to be sent.
373  // Without MSG_DONTWAIT, CPU usage is bad.
374  int flags = MSG_DONTWAIT;
375  int sent = send(rawsock->sock, NULL, 0, flags);
376  if (errno == EINTR) {
377  // ignore
378  }
379  else if (sent < 0) {
380  AVB_LOGF_ERROR("Send failed: %s", strerror(errno));
381  assert(0);
382  }
383  else {
384  AVB_LOGF_VERBOSE("Sent %d bytes, %d frames", sent, rawsock->buffersReady);
385  rawsock->buffersOut -= rawsock->buffersReady;
386  rawsock->buffersReady = 0;
387  }
388 
389  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
390  return sent;
391 }

393 // Count used TX buffers in ring
394 int ringRawsockTxBufLevel(void *pvRawsock)
395 {
396  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK_DETAIL);
397  ring_rawsock_t *rawsock = (ring_rawsock_t*)pvRawsock;
398 
399  int buffersPerBlock = rawsock->blockSize / rawsock->bufferSize;
400 
401  int iBlock, iBuffer, nInUse = 0;
402  volatile struct tpacket2_hdr *pHdr;
403 
404  if (!VALID_RAWSOCK(rawsock)) {
405  AVB_LOG_ERROR("getting buffer level; invalid arguments");
406  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
407  return FALSE;
408  }
409 
410  for (iBlock = 0; iBlock < rawsock->blockCount; iBlock++) {
411  for (iBuffer = 0; iBuffer < buffersPerBlock; iBuffer++) {
412 
413  pHdr = (struct tpacket2_hdr*)(rawsock->pMem
414  + (iBlock * rawsock->blockSize)
415  + (iBuffer * rawsock->bufferSize));
416 
417  if (rawsock->base.txMode) {
418  if (pHdr->tp_status == TP_STATUS_SEND_REQUEST
419  || pHdr->tp_status == TP_STATUS_SENDING)
420  nInUse++;
421  }
422  }
423  }
424 
425  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
426  return nInUse;
427 }

465 // Get a RX frame
466 U8* ringRawsockGetRxFrame(void *pvRawsock, U32 timeout, unsigned int *offset, unsigned int *len)
467 {
468  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK_DETAIL);
469  ring_rawsock_t *rawsock = (ring_rawsock_t*)pvRawsock;
470  if (!VALID_RX_RAWSOCK(rawsock)) {
471  AVB_LOG_ERROR("Getting RX frame; invalid arguments");
472  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
473  return NULL;
474  }
475  if (rawsock->buffersOut >= rawsock->frameCount) {
476  AVB_LOG_ERROR("Too many RX buffers in use");
477  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
478  return NULL;
479  }
480 
481  // Get pointer to active buffer in ring
482  volatile struct tpacket2_hdr *pHdr =
483  (struct tpacket2_hdr*)(rawsock->pMem
484  + (rawsock->blockIndex * rawsock->blockSize)
485  + (rawsock->bufferIndex * rawsock->bufferSize));
486  volatile U8 *pBuffer = (U8*)pHdr + rawsock->bufHdrSize;
487 
488  AVB_LOGF_VERBOSE("block=%d, buffer=%d, out=%d, pBuffer=%p, pHdr=%p",
489  rawsock->blockIndex, rawsock->bufferIndex, rawsock->buffersOut,
490  pBuffer, pHdr);
491 
492  // Check if buffer ready for user
493  // In receive mode, we want TP_STATUS_USER flag set
494  if ((pHdr->tp_status & TP_STATUS_USER) == 0)
495  {
496  struct timespec ts, *pts = NULL;
497  struct pollfd pfd;
498 
499  // Use poll to wait for "ready to read" condition
500 
501  // Poll even if our timeout is 0 - to catch the case where
502  // kernel is writing to the wrong slot (see below.)
503  if (timeout != OPENAVB_RAWSOCK_BLOCK) {
504  ts.tv_sec = timeout / MICROSECONDS_PER_SECOND;
505  ts.tv_nsec = (timeout % MICROSECONDS_PER_SECOND) * NANOSECONDS_PER_USEC;
506  pts = &ts;
507  }
508 
509  pfd.fd = rawsock->sock;
510  pfd.events = POLLIN;
511  pfd.revents = 0;
512 
513  int ret = ppoll(&pfd, 1, pts, NULL);
514  if (ret < 0) {
515  if (errno != EINTR) {
516  AVB_LOGF_ERROR("Getting RX frame; poll failed: %s", strerror(errno));
517  }
518  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
519  return NULL;
520  }
521  if ((pfd.revents & POLLIN) == 0) {
522  // timeout
523  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
524  return NULL;
525  }
526 
527  if ((pHdr->tp_status & TP_STATUS_USER) == 0) {
528  // Hmmm, this is unexpected. poll indicated that the
529  // socket was ready to read, but the slot in the TX ring
530  // that we're looking for the kernel to fill isn't filled.
531 
532  // If there aren't any RX buffers held by the application,
533  // we can try to fix this sticky situation...
534  if (rawsock->buffersOut == 0) {
535  // Scan forward through the RX ring, and look for a
536  // buffer that's ready for us to read. The kernel has
537  // a bad habit of not starting at the beginning of the
538  // ring when the listener process is restarted.
539  int nSkipped = 0;
540  while((pHdr->tp_status & TP_STATUS_USER) == 0) {
541  // Move to next slot in ring.
542  // (Increment buffer/block indexes)
543  if (++(rawsock->bufferIndex) >= (rawsock->frameCount/rawsock->blockCount)) {
544  rawsock->bufferIndex = 0;
545  if (++(rawsock->blockIndex) >= rawsock->blockCount) {
546  rawsock->blockIndex = 0;
547  }
548  }
549 
550  // Adjust pHdr, pBuffer to point to the new slot
551  pHdr = (struct tpacket2_hdr*)(rawsock->pMem
552  + (rawsock->blockIndex * rawsock->blockSize)
553  + (rawsock->bufferIndex * rawsock->bufferSize));
554  pBuffer = (U8*)pHdr + rawsock->bufHdrSize;
555 
556  // If we've scanned all the way around the ring, bail out.
557  if (++nSkipped > rawsock->frameCount) {
558  AVB_LOG_WARNING("Getting RX frame; no frame after poll");
559  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
560  return NULL;
561  }
562  }
563 
564  // We found a slot that's ready. Hopefully, we're good now.
565  AVB_LOGF_WARNING("Getting RX frame; skipped %d empty slots (rawsock=%p)", nSkipped, rawsock);
566  }
567  else {
568  AVB_LOG_WARNING("Getting RX frame; no frame after poll");
569  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
570  return NULL;
571  }
572  }
573  }
574 
575  AVB_LOGF_VERBOSE("Buffer status=0x%4.4lx", (unsigned long)pHdr->tp_status);
576  if (pHdr->tp_status & TP_STATUS_COPY) {
577  AVB_LOG_WARNING("Frame too big for receive buffer");
578  }
579 
580  // Check the "losing" flag. That indicates that the ring is full,
581  // and the kernel had to toss some frames. There is no "winning" flag.
582  if ((pHdr->tp_status & TP_STATUS_LOSING)) {
583  if (!rawsock->bLosing) {
584  AVB_LOG_WARNING("Getting RX frame; mmap buffers full");
585  rawsock->bLosing = TRUE;
586  }
587  }
588  else {
589  rawsock->bLosing = FALSE;
590  }
591 
592  // increment indexes for next time
593  if (++(rawsock->bufferIndex) >= (rawsock->frameCount/rawsock->blockCount)) {
594  rawsock->bufferIndex = 0;
595  if (++(rawsock->blockIndex) >= rawsock->blockCount) {
596  rawsock->blockIndex = 0;
597  }
598  }
599 
600  // Remember that the client has another buffer
601  rawsock->buffersOut += 1;
602 
603  if (pHdr->tp_snaplen < pHdr->tp_len) {
604 #if (AVB_LOG_LEVEL >= AVB_LOG_LEVEL_VERBOSE)
605  AVB_LOGF_WARNING("Getting RX frame; partial frame ignored (len %d, snaplen %d)", pHdr->tp_len, pHdr->tp_snaplen);
606  AVB_LOG_BUFFER(AVB_LOG_LEVEL_VERBOSE, (const U8 *) pBuffer + (pHdr->tp_mac - rawsock->bufHdrSize), pHdr->tp_len, 16);
607 #else
608  IF_LOG_INTERVAL(1000) AVB_LOGF_WARNING("Getting RX frame; partial frame ignored (len %d, snaplen %d)", pHdr->tp_len, pHdr->tp_snaplen);
609 #endif
610  ringRawsockRelRxFrame(rawsock, (U8*)pBuffer);
611  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
612  return NULL;
613  }
614 
615  // Return pointer to the buffer and length
616  *offset = pHdr->tp_mac - rawsock->bufHdrSize;
617  *len = pHdr->tp_snaplen;
618  AVB_LOGF_VERBOSE("Good RX frame (len %d, snaplen %d)", pHdr->tp_len, pHdr->tp_snaplen);
619 
620  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
621  return (U8*)pBuffer;
622 }

624 // Parse the ethernet frame header. Returns length of header, or -1 for failure
625 int ringRawsockRxParseHdr(void *pvRawsock, U8 *pBuffer, hdr_info_t *pInfo)
626 {
627  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK_DETAIL);
628  ring_rawsock_t *rawsock = (ring_rawsock_t*)pvRawsock;
629  int hdrLen;
630  if (!VALID_RX_RAWSOCK(rawsock)) {
631  AVB_LOG_ERROR("Parsing Ethernet headers; invalid arguments");
632  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
633  return -1;
634  }
635 
636  volatile struct tpacket2_hdr *pHdr = (struct tpacket2_hdr*)(pBuffer - rawsock->bufHdrSize);
637  AVB_LOGF_VERBOSE("ringRawsockRxParseHdr: pBuffer=%p, pHdr=%p", pBuffer, pHdr);
638 
639  memset(pInfo, 0, sizeof(hdr_info_t));
640 
641  eth_hdr_t *pNoTag = (eth_hdr_t*)((U8*)pHdr + pHdr->tp_mac);
642  hdrLen = pHdr->tp_net - pHdr->tp_mac;
643  pInfo->shost = pNoTag->shost;
644  pInfo->dhost = pNoTag->dhost;
645  pInfo->ethertype = ntohs(pNoTag->ethertype);
646  pInfo->ts.tv_sec = pHdr->tp_sec;
647  pInfo->ts.tv_nsec = pHdr->tp_nsec;
648 
649  if (pInfo->ethertype == ETHERTYPE_8021Q) {
650  pInfo->vlan = TRUE;
651  pInfo->vlan_vid = pHdr->tp_vlan_tci & 0x0FFF;
652  pInfo->vlan_pcp = (pHdr->tp_vlan_tci >> 13) & 0x0007;
653  pInfo->ethertype = ntohs(*(U16*)( ((U8*)(&pNoTag->ethertype)) + 4));
654  hdrLen += 4;
655  }
656 
657  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
658  return hdrLen;
