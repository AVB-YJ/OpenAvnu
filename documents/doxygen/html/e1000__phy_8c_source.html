259 s32 e1000_read_phy_reg_mdic(struct e1000_hw *hw, u32 offset, u16 *data)
260 {
261  struct e1000_phy_info *phy = &hw->phy;
262  u32 i, mdic = 0;
263 
264  DEBUGFUNC("e1000_read_phy_reg_mdic");
265 
266  if (offset > MAX_PHY_REG_ADDRESS) {
267  DEBUGOUT1("PHY Address %d is out of range\n", offset);
268  return -E1000_ERR_PARAM;
269  }
270 
271  /* Set up Op-code, Phy Address, and register offset in the MDI
272  * Control register. The MAC will take care of interfacing with the
273  * PHY to retrieve the desired data.
274  */
275  mdic = ((offset << E1000_MDIC_REG_SHIFT) |
276  (phy->addr << E1000_MDIC_PHY_SHIFT) |
277  (E1000_MDIC_OP_READ));
278 
279  E1000_WRITE_REG(hw, E1000_MDIC, mdic);
280 
281  /* Poll the ready bit to see if the MDI read completed
282  * Increasing the time out as testing showed failures with
283  * the lower time out
284  */
285  for (i = 0; i < (E1000_GEN_POLL_TIMEOUT * 3); i++) {
286  usec_delay_irq(50);
287  mdic = E1000_READ_REG(hw, E1000_MDIC);
288  if (mdic & E1000_MDIC_READY)
289  break;
290  }
291  if (!(mdic & E1000_MDIC_READY)) {
292  DEBUGOUT("MDI Read did not complete\n");
293  return -E1000_ERR_PHY;
294  }
295  if (mdic & E1000_MDIC_ERROR) {
296  DEBUGOUT("MDI Error\n");
297  return -E1000_ERR_PHY;
298  }
299  if (((mdic & E1000_MDIC_REG_MASK) >> E1000_MDIC_REG_SHIFT) != offset) {
300  DEBUGOUT2("MDI Read offset error - requested %d, returned %d\n",
301  offset,
302  (mdic & E1000_MDIC_REG_MASK) >> E1000_MDIC_REG_SHIFT);
303  return -E1000_ERR_PHY;
304  }
305  *data = (u16) mdic;
306 
307  return E1000_SUCCESS;
308 }

394

541 s32 e1000_write_sfp_data_byte(struct e1000_hw *hw, u16 offset, u8 data)
542 {
543  u32 i = 0;
544  u32 i2ccmd = 0;
545  u32 data_local = 0;
546 
547  DEBUGFUNC("e1000_write_sfp_data_byte");
548 
549  if (offset > E1000_I2CCMD_SFP_DIAG_ADDR(255)) {
550  DEBUGOUT("I2CCMD command address exceeds upper limit\n");
551  return -E1000_ERR_PHY;
552  }
553  /* The programming interface is 16 bits wide
554  * so we need to read the whole word first
555  * then update appropriate byte lane and write
556  * the updated word back.
557  */
558  /* Set up Op-code, EEPROM Address,in the I2CCMD
559  * register. The MAC will take care of interfacing
560  * with an EEPROM to write the data given.
561  */
562  i2ccmd = ((offset << E1000_I2CCMD_REG_ADDR_SHIFT) |
563  E1000_I2CCMD_OPCODE_READ);
564  /* Set a command to read single word */
565  E1000_WRITE_REG(hw, E1000_I2CCMD, i2ccmd);
566  for (i = 0; i < E1000_I2CCMD_PHY_TIMEOUT; i++) {
567  usec_delay(50);
568  /* Poll the ready bit to see if lastly
569  * launched I2C operation completed
570  */
571  i2ccmd = E1000_READ_REG(hw, E1000_I2CCMD);
572  if (i2ccmd & E1000_I2CCMD_READY) {
573  /* Check if this is READ or WRITE phase */
574  if ((i2ccmd & E1000_I2CCMD_OPCODE_READ) ==
575  E1000_I2CCMD_OPCODE_READ) {
576  /* Write the selected byte
577  * lane and update whole word
578  */
579  data_local = i2ccmd & 0xFF00;
580  data_local |= data;
581  i2ccmd = ((offset <<
582  E1000_I2CCMD_REG_ADDR_SHIFT) |
583  E1000_I2CCMD_OPCODE_WRITE | data_local);
584  E1000_WRITE_REG(hw, E1000_I2CCMD, i2ccmd);
585  } else {
586  break;
587  }
588  }
589  }
590  if (!(i2ccmd & E1000_I2CCMD_READY)) {
591  DEBUGOUT("I2CCMD Write did not complete\n");
592  return -E1000_ERR_PHY;
593  }
594  if (i2ccmd & E1000_I2CCMD_ERROR) {
595  DEBUGOUT("I2CCMD Error bit set\n");
596  return -E1000_ERR_PHY;
597  }
598  return E1000_SUCCESS;
599 }
