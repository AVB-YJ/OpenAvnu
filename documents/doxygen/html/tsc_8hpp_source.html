73 inline uint32_t FindFrequencyByModel(uint8_t model_query) {
74  ModelNumberSystemClockRateMapping *map = ModelNumberSystemClockRateMap;
75  uint32_t ret = 0;
76 
77  while (map->clock_rate != 0)
78  {
79  short *model = map->model_list;
80 
81  while (*model != -1)
82  {
83  if (*model == model_query) {
84  ret = map->clock_rate;
85  break;
86  }
87  ++model;
88  }
89  ++map;
90  }
91 
92  return ret;
93 }

101 inline uint64_t getTSCFrequency( bool builtin ) {
102  int max_cpuid_level;
103  int tmp[4];
104  BOOL is_windows_10;
105  LARGE_INTEGER freq;
106 
107  // Find the max cpuid level, and if possible find clock info
108  __cpuid(tmp, 0);
109  max_cpuid_level = tmp[0];
110  if (max_cpuid_level >= CLOCK_INFO_CPUID_LEAF)
111  __cpuid(tmp, CLOCK_INFO_CPUID_LEAF);
112 
113  // Determine if the OS is Windows 10 or later.
114  {
115  OSVERSIONINFOEXW osvi = { sizeof(osvi), 0, 0, 0, 0, {0}, 0, 0 };
116  DWORDLONG const dwlConditionMask = VerSetConditionMask(0, VER_MAJORVERSION, VER_GREATER_EQUAL);
117  osvi.dwMajorVersion = 10;
118 
119  // NOTE: VerifyVersionInfo returns false when called by applications that do not have a compatibility manifest for Windows 10,
120  // even when the current operating system version is Windows 10.
121  is_windows_10 = VerifyVersionInfoW(&osvi, VER_MAJORVERSION, dwlConditionMask) != FALSE;
122  }
123 
124  // For pre-Win10 on 6th Generation or greater Intel CPUs the raw hardware
125  // clock will be returned, *else* use QPC for everything else
126  //
127  // EAX (tmp[0]) must be >= 1, See Intel SDM 17.15.4 "Invariant Time-keeping"
128  if (!is_windows_10 &&
129  max_cpuid_level >= CLOCK_INFO_CPUID_LEAF &&
130  tmp[0] >= 1 &&
131  builtin )
132  {
133  SYSTEM_INFO info;
134 
135  GetSystemInfo(&info);
136 
137  // Look at processor model (upper byte of revision) number to determine clock rate
138  return FindFrequencyByModel(info.wProcessorRevision >> 8);
139  }
140 
141  if (QueryPerformanceFrequency(&freq))
142  return freq.QuadPart;
143 
144  return 0;
145 }
