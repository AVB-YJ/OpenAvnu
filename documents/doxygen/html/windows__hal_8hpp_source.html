182  OSLockResult lock_l( DWORD timeout ) {
183  DWORD wait_result = WaitForSingleObject( lock_c, timeout );
184  if( wait_result == WAIT_TIMEOUT ) return oslock_held;
185  else if( wait_result == WAIT_OBJECT_0 ) return oslock_ok;
186  else return oslock_fail;
187 
188  }
189  OSLockResult nonreentrant_lock_l( DWORD timeout ) {
190  OSLockResult result;
191  DWORD wait_result;
192  wait_result = WaitForSingleObject( lock_c, timeout );
193  if( wait_result == WAIT_OBJECT_0 ) {
194  if( thread_id == GetCurrentThreadId() ) {
195  result = oslock_self;
196  ReleaseMutex( lock_c );
197  } else {
198  result = oslock_ok;
199  thread_id = GetCurrentThreadId();
200  }
201  } else if( wait_result == WAIT_TIMEOUT ) result = oslock_held;
202  else result = oslock_fail;
203 
204  return result;
205  }
206 protected:
210  WindowsLock() {
211  lock_c = NULL;
212  }
217  bool initialize( OSLockType type ) {
218  lock_c = CreateMutex( NULL, false, NULL );
219  if( lock_c == NULL ) return false;
220  this->type = type;
221  return true;
222  }
227  OSLockResult lock() {
228  if( type == oslock_recursive ) {
229  return lock_l( INFINITE );
230  }
231  return nonreentrant_lock_l( INFINITE );
232  }
237  OSLockResult trylock() {
238  if( type == oslock_recursive ) {
239  return lock_l( 0 );
240  }

424  cleanupRetiredTimers();
