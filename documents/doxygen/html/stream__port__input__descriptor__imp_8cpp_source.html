102 
103  /******************************************** AECP Common Data *********************************************/

105  // Fill aem_cmd_get_sampling_rate.sequence_id in AEM Controller State Machine

107 
108  /****************** AECP Message Specific Data *****************/
109  aem_cmd_get_audio_map.descriptor_type = descriptor_type();
110  aem_cmd_get_audio_map.descriptor_index = descriptor_index();

112 
113  /******************************* Fill frame payload with AECP data and send the frame **************************/
114  aecp_controller_state_machine_ref->ether_frame_init(base_end_station_imp_ref->mac(), &cmd_frame,
115  ETHER_HDR_SIZE + JDKSAVDECC_AEM_COMMAND_GET_AUDIO_MAP_COMMAND_LEN);
116  aem_cmd_get_audio_map_returned = jdksavdecc_aem_command_get_audio_map_write(&aem_cmd_get_audio_map,

119  sizeof(cmd_frame.payload));
120  if (aem_cmd_get_audio_map_returned < 0)
121  {

124  return -1;
125  }
126 
127  aecp_controller_state_machine_ref->common_hdr_init(JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_COMMAND,
128  &cmd_frame,
129  base_end_station_imp_ref->entity_id(),
130  JDKSAVDECC_AEM_COMMAND_GET_AUDIO_MAP_COMMAND_LEN -
131  JDKSAVDECC_COMMON_CONTROL_HEADER_LEN);

171 int STDCALL stream_port_input_descriptor_imp::send_add_audio_mappings_cmd(void * notification_id)
172 {
173  int i = 0;
174  struct jdksavdecc_frame cmd_frame;
175  struct jdksavdecc_aem_command_add_audio_mappings aem_cmd_add_audio_mappings;
176  ssize_t aem_cmd_add_audio_mappings_returned;
177  memset(&aem_cmd_add_audio_mappings, 0, sizeof(aem_cmd_add_audio_mappings));
178 
179  /******************************************** AECP Common Data *********************************************/
180  aem_cmd_add_audio_mappings.aem_header.aecpdu_header.controller_entity_id = base_end_station_imp_ref->get_adp()->get_controller_entity_id();
181  // Fill aem_cmd_get_sampling_rate.sequence_id in AEM Controller State Machine
182  aem_cmd_add_audio_mappings.aem_header.command_type = JDKSAVDECC_AEM_COMMAND_ADD_AUDIO_MAPPINGS;
183 
184  /****************** AECP Message Specific Data *****************/
185  aem_cmd_add_audio_mappings.descriptor_type = descriptor_type();
186  aem_cmd_add_audio_mappings.descriptor_index = descriptor_index();
187  if (pending_maps.size() > AEM_MAX_MAPS)
188  {
189  aem_cmd_add_audio_mappings.number_of_mappings = AEM_MAX_MAPS;
190  }
191  else
192  {
193  aem_cmd_add_audio_mappings.number_of_mappings = (uint16_t)pending_maps.size();
194  }
195 
196  /******************************* Fill frame payload with AECP data and send the frame **************************/
197  aecp_controller_state_machine_ref->ether_frame_init(base_end_station_imp_ref->mac(), &cmd_frame,
198  ETHER_HDR_SIZE + JDKSAVDECC_AEM_COMMAND_ADD_AUDIO_MAPPINGS_COMMAND_LEN +
199  (uint16_t)pending_maps.size() * JDKSAVDECC_AUDIO_MAPPING_LEN);
200  aem_cmd_add_audio_mappings_returned = jdksavdecc_aem_command_add_audio_mappings_write(&aem_cmd_add_audio_mappings,
201  cmd_frame.payload,
202  ETHER_HDR_SIZE,
203  sizeof(cmd_frame.payload));
204  if (aem_cmd_add_audio_mappings_returned < 0)
205  {
206  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "aem_cmd_add_audio_mappings_write error\n");
207  assert(aem_cmd_add_audio_mappings_returned >= 0);
208  return -1;
209  }
210 
211  size_t num_pending_maps = pending_maps.size();
212  std::vector<struct audio_map_mapping>::iterator it = pending_maps.begin();
213  while (it != pending_maps.end())
214  {
215  if (i >= AEM_MAX_MAPS)
216  break;
217 
218  struct avdecc_lib::audio_map_mapping map = *it;
219 
220  jdksavdecc_uint16_set(map.stream_index, cmd_frame.payload, ETHER_HDR_SIZE +
221  JDKSAVDECC_AEM_COMMAND_ADD_AUDIO_MAPPINGS_COMMAND_OFFSET_MAPPINGS + JDKSAVDECC_AUDIO_MAPPING_LEN * i +
222  JDKSAVDECC_AUDIO_MAPPING_OFFSET_MAPPING_STREAM_INDEX);
223  jdksavdecc_uint16_set(map.stream_channel, cmd_frame.payload, ETHER_HDR_SIZE +
224  JDKSAVDECC_AEM_COMMAND_ADD_AUDIO_MAPPINGS_COMMAND_OFFSET_MAPPINGS +
225  (JDKSAVDECC_AUDIO_MAPPING_LEN * i + JDKSAVDECC_AUDIO_MAPPING_OFFSET_MAPPING_STREAM_CHANNEL));
226  jdksavdecc_uint16_set(map.cluster_offset, cmd_frame.payload, ETHER_HDR_SIZE +
227  JDKSAVDECC_AEM_COMMAND_ADD_AUDIO_MAPPINGS_COMMAND_OFFSET_MAPPINGS +
228  (JDKSAVDECC_AUDIO_MAPPING_LEN * i + JDKSAVDECC_AUDIO_MAPPING_OFFSET_MAPPING_CLUSTER_OFFSET));
229  jdksavdecc_uint16_set(map.cluster_channel, cmd_frame.payload, ETHER_HDR_SIZE +
230  JDKSAVDECC_AEM_COMMAND_ADD_AUDIO_MAPPINGS_COMMAND_OFFSET_MAPPINGS +
231  (JDKSAVDECC_AUDIO_MAPPING_LEN * i + JDKSAVDECC_AUDIO_MAPPING_OFFSET_MAPPING_CLUSTER_CHANNEL));
232 
233  pending_maps.erase(it);
234  it = pending_maps.begin();
235  i++;
236  }
237 
238  aecp_controller_state_machine_ref->common_hdr_init(JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_COMMAND,
239  &cmd_frame,
240  base_end_station_imp_ref->entity_id(),
241  JDKSAVDECC_AEM_COMMAND_ADD_AUDIO_MAPPINGS_COMMAND_LEN -
242  JDKSAVDECC_COMMON_CONTROL_HEADER_LEN);
243  system_queue_tx(notification_id, CMD_WITH_NOTIFICATION, cmd_frame.payload, cmd_frame.length);
244 
245  if (num_pending_maps > AEM_MAX_MAPS)
246  {
247  return 1;
248  }
249  else
250  {
251  return 0;
252  }
253 }
