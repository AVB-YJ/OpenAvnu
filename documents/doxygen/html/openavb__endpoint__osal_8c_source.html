53 bool startEndpoint(int mode, int ifindex, const char* ifname, unsigned mtu, unsigned link_kbit, unsigned nsr_kbit)
54 {
55  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
56  LOG_EAVB_CORE_VERSION();
57 
58  // Ensure that we're running as root
59  // (need to be root to use raw sockets)
60  uid_t euid = geteuid();
61  if (euid != (uid_t)0) {
62  fprintf(stderr, "Error: needs to run as root\n\n");
63  goto error;
64  }
65 
66  // Set endpoint configuration
67  memset(&x_cfg, 0, sizeof(openavb_endpoint_cfg_t));
68  x_cfg.fqtss_mode = mode;
69  x_cfg.ifindex = ifindex;
70  x_cfg.mtu = mtu;
71 
72  x_cfg.link_kbit = link_kbit;
73  x_cfg.nsr_kbit = nsr_kbit;
74 
75  openavbReadConfig(DEFAULT_INI_FILE, DEFAULT_SAVE_INI_FILE, &x_cfg);
76 
77  if_info_t ifinfo;
78  if (ifname && openavbCheckInterface(ifname, &ifinfo)) {
79  strncpy(x_cfg.ifname, ifname, sizeof(x_cfg.ifname));
80  memcpy(x_cfg.ifmac, &ifinfo.mac, ETH_ALEN);
81  x_cfg.ifindex = ifinfo.index;
82  x_cfg.mtu = ifinfo.mtu;
83  }
84 
85  endpointRunning = TRUE;
86  int err = pthread_create(&endpointServerHandle, NULL, endpointServerThread, NULL);
87  if (err) {
88  AVB_LOGF_ERROR("Failed to start endpoint thread: %s", strerror(err));
89  goto error;
90  }
91 
92  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
93  return true;
94 
95 error:
96  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
97  return false;
98 }
