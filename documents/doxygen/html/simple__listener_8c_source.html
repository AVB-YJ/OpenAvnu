95 void pcap_callback(u_char* args, const struct pcap_pkthdr* packet_header, const u_char* packet)
96 {
97  unsigned char* test_stream_id;
98  struct ethernet_header* eth_header;
99  uint32_t* buf;
100  uint32_t frame[2] = { 0 , 0 };
101 
102  int i;
103  (void) args; /* unused */
104  (void) packet_header; /* unused */
105 
106 #if DEBUG
107  fprintf(stdout,"Got packet.\n");
108 #endif /* DEBUG*/
109 
110  eth_header = (struct ethernet_header*)(packet);
111 
112 #if DEBUG
113  fprintf(stdout,"Ether Type: 0x%02x%02x\n", eth_header->type[0], eth_header->type[1]);
114 #endif /* DEBUG*/
115 
116  if (0 == memcmp(glob_ether_type,eth_header->type,sizeof(eth_header->type)))
117  {
118  test_stream_id = (unsigned char*)(packet + ETHERNET_HEADER_SIZE + SEVENTEEN22_HEADER_PART1_SIZE);
119 
120 #if DEBUG
121  fprintf(stderr, "Received stream id: %02x%02x%02x%02x%02x%02x%02x%02x\n ",
122  test_stream_id[0], test_stream_id[1],
123  test_stream_id[2], test_stream_id[3],
124  test_stream_id[4], test_stream_id[5],
125  test_stream_id[6], test_stream_id[7]);
126 #endif /* DEBUG*/
127 
128  if (0 == memcmp(test_stream_id, glob_target_stream_id, sizeof(STREAM_ID_SIZE)))
129  {
130 
131 #if DEBUG
132  fprintf(stdout,"Stream ids matched.\n");
133 #endif /* DEBUG*/
134  buf = (uint32_t*)(packet + HEADER_SIZE);
135  for(i = 0; i < SAMPLES_PER_FRAME * CHANNELS; i += 2)
136  {
137  memcpy(&frame[0], &buf[i], sizeof(frame));
138 
139  frame[0] = ntohl(frame[0]); /* convert to host-byte order */
140  frame[1] = ntohl(frame[1]);
141  frame[0] &= 0x00ffffff; /* ignore leading label */
142  frame[1] &= 0x00ffffff;
143  frame[0] <<= 8; /* left-align remaining PCM-24 sample */
144  frame[1] <<= 8;
145 
146  sf_writef_int(glob_snd_file, (const int *)frame, 1);
147  }
148  }
149  }
150 }

204 void sigint_handler(int signum)
205 {
206  int ret;
207 
208  fprintf(stdout,"Received signal %d:leaving...\n", signum);
209 
210  if (0 != ctx_sig->talker) {
211  ret = send_leave(ctx_sig);
212  if (ret)
213  printf("send_leave failed\n");
214  }
215 
216  if (2 > ctx_sig->control_socket)
217  {
218  close(ctx_sig->control_socket);
219  ret = mrp_disconnect(ctx_sig);
220  if (ret)
221  printf("mrp_disconnect failed\n");
222  }
223 
224 #if PCAP
225  if (NULL != glob_pcap_handle)
226  {
227  pcap_breakloop(glob_pcap_handle);
228  pcap_close(glob_pcap_handle);
229  }
230 #endif /* PCAP */
231 
232 #if LIBSND
233  sf_write_sync(glob_snd_file);
234  sf_close(glob_snd_file);
235 #endif /* LIBSND */
236 }
