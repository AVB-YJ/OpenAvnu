28 /* AAF Talker example.
29  *
30  * This example implements a very simple AAF talker application which reads
31  * a PCM stream from stdin, creates AAF packets and transmit them via the
32  * network.
33  *
34  * For simplicity, the example supports only one set of PCM parameters:
35  * - Sample format: 16-bit little endian
36  * - Sample rate: 48 kHz
37  * - Number of channels: 2 (stereo)
38  *
39  * TSN stream parameters (e.g. destination mac address, traffic priority) are
40  * passed via command-line arguments. Run 'aaf-talker --help' for more
41  * information.
42  *
43  * In order to have this example working properly, make sure you have
44  * configured FQTSS feature from your NIC according (for further information
45  * see tc-cbs(8)). Also, this example relies on system clock to set the AVTP
46  * timestamp so make sure it is synchronized with the PTP Hardware Clock (PHC)
47  * from your NIC and that the PHC is synchronized with the network clock. For
48  * further information see ptp4l(8) and phc2sys(8).
49  *
50  * The easiest way to use this example is combining it with 'arecord' tool
51  * provided by alsa-utils. 'arecord' reads the PCM stream from a capture ALSA
52  * device (e.g. your microphone) and writes it to stdout. So to stream Audio
53  * captured from your mic to a TSN network you should do something like this:
54  *
55  * $ arecord -f dat -t raw -D <capture-device> | aaf-talker <args>
56  */

95 static error_t parser(int key, char *arg, struct argp_state *state)
96 {
97  int res;
98 
99  switch (key) {
100  case 'd':
101  res = sscanf(arg, "%hhx:%hhx:%hhx:%hhx:%hhx:%hhx",
102  &macaddr[0], &macaddr[1], &macaddr[2],
103  &macaddr[3], &macaddr[4], &macaddr[5]);
104  if (res != 6) {
105  fprintf(stderr, "Invalid address\n");
106  exit(EXIT_FAILURE);
107  }
108 
109  break;
110  case 'i':
111  strncpy(ifname, arg, sizeof(ifname) - 1);
112  break;
113  case 'm':
114  max_transit_time = atoi(arg);
115  break;
116  case 'p':
117  priority = atoi(arg);
118  break;
119  }
120 
121  return 0;
122 }

153 static int calculate_avtp_time(uint32_t *avtp_time)
154 {
155  int res;
156  struct timespec tspec;
157  uint64_t ptime;
158 
159  res = clock_gettime(CLOCK_REALTIME, &tspec);
160  if (res < 0) {
161  perror("Failed to get time");
162  return -1;
163  }
164 
165  ptime = (tspec.tv_sec * NSEC_PER_SEC) +
166  (max_transit_time * NSEC_PER_MSEC) + tspec.tv_nsec;
167 
168  *avtp_time = ptime % (1ULL << 32);
169 
170  return 0;
171 }

188 
189  res = avtp_aaf_pdu_set(pdu, AVTP_AAF_FIELD_FORMAT,

196  if (res < 0)

200  NUM_CHANNELS);
201  if (res < 0)
202  return -1;
203

205  if (res < 0)
206  return -1;
207 
208  res = avtp_aaf_pdu_set(pdu, AVTP_AAF_FIELD_STREAM_DATA_LEN, DATA_LEN);
209  if (res < 0)
210  return -1;
211 
212  res = avtp_aaf_pdu_set(pdu, AVTP_AAF_FIELD_SP, AVTP_AAF_PCM_SP_NORMAL);
213  if (res < 0)

219 int main(int argc, char *argv[])
220 {
221  int fd, res;
222  struct ifreq req;
223  struct sockaddr_ll sk_addr;
224  struct avtp_stream_pdu *pdu = alloca(PDU_SIZE);
225  uint8_t seq_num = 0;
226 
227  argp_parse(&argp, argc, argv, 0, NULL, NULL);
228 
229  fd = setup_socket();
230  if (fd < 0)
231  return 1;
232 
233  snprintf(req.ifr_name, sizeof(req.ifr_name), "%s", ifname);
234  res = ioctl(fd, SIOCGIFINDEX, &req);
235  if (res < 0) {
236  perror("Failed to get interface index");
237  goto err;
238  }
239 
240  sk_addr.sll_family = AF_PACKET;
241  sk_addr.sll_protocol = htons(ETH_P_TSN);
242  sk_addr.sll_halen = ETH_ALEN;
243  sk_addr.sll_ifindex = req.ifr_ifindex;
244  memcpy(&sk_addr.sll_addr, macaddr, ETH_ALEN);
245 
246  res = init_pdu(pdu);
247  if (res < 0)
248  goto err;
249 
250  while (1) {
251  ssize_t n;
252  uint32_t avtp_time;
253 
254  memset(pdu->avtp_payload, 0, DATA_LEN);
255 
256  n = read(STDIN_FILENO, pdu->avtp_payload, DATA_LEN);
257  if (n == 0)
258  break;
259 
260  if (n != DATA_LEN) {
261  fprintf(stderr, "read %zd bytes, expected %d\n",
262  n, DATA_LEN);
263  }
264 
265  res = calculate_avtp_time(&avtp_time);
266  if (res < 0) {
267  fprintf(stderr, "Failed to calculate avtp time\n");
268  goto err;
269  }
270 
271  res = avtp_aaf_pdu_set(pdu, AVTP_AAF_FIELD_TIMESTAMP,
272  avtp_time);
273  if (res < 0)
274  goto err;
275 
276  res = avtp_aaf_pdu_set(pdu, AVTP_AAF_FIELD_SEQ_NUM, seq_num++);
277  if (res < 0)
278  goto err;
279 
280  n = sendto(fd, pdu, PDU_SIZE, 0,
281  (struct sockaddr *) &sk_addr, sizeof(sk_addr));
282  if (n < 0) {
283  perror("Failed to send data");
284  goto err;
285  }
286 
287  if (n != PDU_SIZE) {
288  fprintf(stderr, "wrote %zd bytes, expected %zd\n",
289  n, PDU_SIZE);
290  }
291  }
292 
293  close(fd);
294  return 0;
295 
296 err:
297  close(fd);
298  return 1;
299 }
