93 void jdksavdecc_aecp_common_control_header_print( struct jdksavdecc_printer *self,
94  struct jdksavdecc_aecpdu_common_control_header const *p )
95 {
96  struct jdksavdecc_uint16_name *status_name_table = jdksavdecc_aecp_print_status;
97  jdksavdecc_printer_print_label( self, "message_type" );
98  jdksavdecc_printer_print_uint16_name( self, jdksavdecc_aecp_print_message_type, p->message_type );
99  jdksavdecc_printer_print_eol( self );
100 
101  jdksavdecc_printer_print_label( self, "status" );
102  switch ( p->message_type )
103  {
104  case JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_COMMAND:
105  case JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_RESPONSE:
106  status_name_table = jdksavdecc_aecp_aem_print_status;
107  break;
108  case JDKSAVDECC_AECP_MESSAGE_TYPE_ADDRESS_ACCESS_COMMAND:
109  case JDKSAVDECC_AECP_MESSAGE_TYPE_ADDRESS_ACCESS_RESPONSE:
110  status_name_table = jdksavdecc_aecp_aa_print_status;
111  break;
112  case JDKSAVDECC_AECP_MESSAGE_TYPE_AVC_COMMAND:
113  case JDKSAVDECC_AECP_MESSAGE_TYPE_AVC_RESPONSE:
114  status_name_table = jdksavdecc_aecp_avc_print_status;
115  break;
116  case JDKSAVDECC_AECP_MESSAGE_TYPE_HDCP_APM_COMMAND:
117  case JDKSAVDECC_AECP_MESSAGE_TYPE_HDCP_APM_RESPONSE:
118  status_name_table = jdksavdecc_aecp_hdcp_apm_print_status;
119  break;
120  case JDKSAVDECC_AECP_MESSAGE_TYPE_VENDOR_UNIQUE_COMMAND:
121  case JDKSAVDECC_AECP_MESSAGE_TYPE_VENDOR_UNIQUE_RESPONSE:
122  status_name_table = jdksavdecc_aecp_vendor_print_status;
123  break;
124  case JDKSAVDECC_AECP_MESSAGE_TYPE_EXTENDED_COMMAND:
125  case JDKSAVDECC_AECP_MESSAGE_TYPE_EXTENDED_RESPONSE:
126  status_name_table = jdksavdecc_aecp_print_status;
127  break;
128  default:
129  status_name_table = jdksavdecc_aecp_print_status;
130  break;
131  }
132  jdksavdecc_printer_print_uint16_name( self, status_name_table, p->status );
133  jdksavdecc_printer_print_eol( self );
134 
135  jdksavdecc_printer_print_label( self, "control_data_length" );
136  jdksavdecc_printer_print_uint16( self, p->control_data_length );
137  jdksavdecc_printer_print_eol( self );
138 
139  jdksavdecc_printer_print_label( self, "target_entity_id" );
140  jdksavdecc_printer_print_eui64( self, p->target_entity_id );
141  jdksavdecc_printer_print_eol( self );
142 }

169  }
170  break;
171  }
172  case JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_RESPONSE:
