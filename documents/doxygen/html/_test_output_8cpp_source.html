140 void TestOutput::printTestsEnded(const TestResult& result)
141 {
142  print("\n");
143  if (result.getFailureCount() > 0) {
144  if (color_) {
145  print("\033[31;1m");
146  }
147  print("Errors (");
148  print(result.getFailureCount());
149  print(" failures, ");
150  }
151  else {
152  if (color_) {
153  print("\033[32;1m");
154  }
155  print("OK (");
156  }
157  print(result.getTestCount());
158  print(" tests, ");
159  print(result.getRunCount());
160  print(" ran, ");
161  print(result.getCheckCount());
162  print(" checks, ");
163  print(result.getIgnoredCount());
164  print(" ignored, ");
165  print(result.getFilteredOutCount());
166  print(" filtered out, ");
167  print(result.getTotalExecutionTime());
168  print(" ms)");
169  if (color_) {
170  print("\033[m");
171  }
172  print("\n\n");
173 
174  dotCount_ = 0;
175 }
