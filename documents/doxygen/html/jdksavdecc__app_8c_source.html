36 ssize_t jdksavdecc_appdu_read( struct jdksavdecc_appdu *p, void const *base, ssize_t pos, size_t len )
37 {
38  ssize_t r = jdksavdecc_validate_range( pos, len, JDKSAVDECC_APPDU_HEADER_LEN );
39  // clear the structure
40  memset( p, 0, sizeof( *p ) );
41  if ( r >= 0 )
42  {
43  p->version = jdksavdecc_uint8_get( base, pos + JDKSAVDECC_APPDU_OFFSET_VERSION );
44  if ( p->version == JDKSAVDECC_APPDU_VERSION )
45  {
46  p->message_type = jdksavdecc_uint8_get( base, pos + JDKSAVDECC_APPDU_OFFSET_MESSAGE_TYPE );
47 
48  if ( p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_NOP
49  || p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_ENTITY_ID_REQUEST
50  || p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_ENTITY_ID_RESPONSE
51  || p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_LINK_UP
52  || p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_LINK_DOWN
53  || p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_AVDECC_FROM_APS
54  || p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_AVDECC_FROM_APC
55  || p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_VENDOR )
56  {
57 
58  p->payload_length = jdksavdecc_uint16_get( base, pos + JDKSAVDECC_APPDU_OFFSET_PAYLOAD_LENGTH );
59  p->address = jdksavdecc_eui48_get( base, pos + JDKSAVDECC_APPDU_OFFSET_ADDRESS );
60  p->reserved = jdksavdecc_uint16_get( base, pos + JDKSAVDECC_APPDU_OFFSET_RESERVED );
61 
62  // make sure that the payload_length is a reasonable number
63  if ( p->payload_length <= JDKSAVDECC_APPDU_MAX_PAYLOAD_LENGTH )
64  {
65  // make sure we have enough data in the incoming buffer for the entire payload
66  r = jdksavdecc_validate_range( pos + JDKSAVDECC_APPDU_OFFSET_PAYLOAD, len, p->payload_length );
67 
68  if ( r >= 0 )
69  {
70  // The payload is complete.
71  // Make payload pointer point directly within incoming raw buffer
72  p->payload = ( (uint8_t *)base ) + JDKSAVDECC_APPDU_OFFSET_PAYLOAD;
73  }
74  else
75  {
76  // the raw buffer is incomplete.
77  p->payload = 0;
78  r = JDKSAVDECC_APPDU_ERROR_PAYLOAD_INCOMPLETE;
79  }
80  }
81  else
82  {
83  // The payload_length field is out of range.
84  r = JDKSAVDECC_APPDU_ERROR_INVALID_HEADER;
85  }
86  }
87  else
88  {
89  // The message type is unrecognized.
90  r = JDKSAVDECC_APPDU_ERROR_INVALID_HEADER;
91  }
92  }
93  else
94  {
95  // The protocol version is unsupported
96  r = JDKSAVDECC_APPDU_ERROR_INVALID_HEADER;
97  }
98  }
99  return r;
100 }

102 ssize_t jdksavdecc_appdu_write( struct jdksavdecc_appdu const *p, void *base, size_t pos, size_t len )
103 {
104  ssize_t r = jdksavdecc_validate_range( pos, len, JDKSAVDECC_APPDU_HEADER_LEN );
105  if ( r >= 0 )
106  {
107  jdksavdecc_uint8_set( p->version, base, pos + JDKSAVDECC_APPDU_OFFSET_VERSION );
108  if ( p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_NOP
109  || p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_ENTITY_ID_REQUEST
110  || p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_ENTITY_ID_RESPONSE
111  || p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_LINK_UP
112  || p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_LINK_DOWN
113  || p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_AVDECC_FROM_APS
114  || p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_AVDECC_FROM_APC
115  || p->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_VENDOR )
116  {
117 
118  jdksavdecc_uint8_set( p->message_type, base, pos + JDKSAVDECC_APPDU_OFFSET_MESSAGE_TYPE );
119  jdksavdecc_uint16_set( p->payload_length, base, pos + JDKSAVDECC_APPDU_OFFSET_PAYLOAD_LENGTH );
120  jdksavdecc_eui48_set( p->address, base, pos + JDKSAVDECC_APPDU_OFFSET_ADDRESS );
121  jdksavdecc_uint16_set( p->reserved, base, pos + JDKSAVDECC_APPDU_OFFSET_RESERVED );
122  if ( p->payload_length > 0 )
123  {
124  if ( p->payload && p->payload_length <= JDKSAVDECC_APPDU_MAX_PAYLOAD_LENGTH )
125  {
126  // We have a payload and the payload length is non zero, make sure we have space in the output buffer
127  r = jdksavdecc_validate_range( pos, len, JDKSAVDECC_APPDU_OFFSET_PAYLOAD + p->payload_length );
128  if ( r > 0 )
129  {
130  // we have space so copy it over
131  memcpy( ( (uint8_t *)base ) + pos + JDKSAVDECC_APPDU_OFFSET_PAYLOAD, p->payload, p->payload_length );
132  }
133  else
134  {
135  // we don't have space, so return error
136  r = JDKSAVDECC_APPDU_ERROR_BUFFER_LENGTH_INSUFFICIENT;
137  }
138  }
139  else
140  {
141  // we have payload length but no payload? that is invalid.
142  r = JDKSAVDECC_APPDU_ERROR_INVALID_HEADER;
143  }
144  }
145  }
146  else
147  {
148  // The message type is unrecognized.
149  r = JDKSAVDECC_APPDU_ERROR_INVALID_HEADER;
150  }
151  }
152  return r;
153 }

157 ssize_t jdksavdecc_appdu_parse_buffer( struct jdksavdecc_appdu *self, us_buffer_t *buffer )
158 {
159  ssize_t r = JDKSAVDECC_APPDU_ERROR_BUFFER_LENGTH_INSUFFICIENT;
160  // store the current buffer position in case parsing fails
161  size_t pos = buffer->m_next_out;
162 
163  // is there enough bytes in the buffer for a full appdu heade?
164  if ( us_buffer_readable_count( buffer ) >= 12 )
165  {
166 
167  // now validate the header
168  r = JDKSAVDECC_APPDU_ERROR_INVALID_HEADER;
169 
170  // extract version
171  self->version = us_buffer_read_byte( buffer );
172 
173  // validate version
174  if ( self->version == JDKSAVDECC_APPDU_VERSION )
175  {
176 
177  // extract message_type
178  self->message_type = us_buffer_read_byte( buffer );
179 
180  // validate message_type
181  if ( self->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_NOP
182  || self->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_ENTITY_ID_REQUEST
183  || self->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_ENTITY_ID_RESPONSE
184  || self->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_LINK_UP
185  || self->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_LINK_DOWN
186  || self->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_AVDECC_FROM_APS
187  || self->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_AVDECC_FROM_APC
188  || self->message_type == JDKSAVDECC_APPDU_MESSAGE_TYPE_VENDOR )
189  {
190 
191  size_t cnt = 0;
192 
193  // read the payload length field
194  if ( us_buffer_read_uint16( buffer, &self->payload_length ) )
195  {
196 
197  // validate the payload length is sane
198  if ( self->payload_length <= JDKSAVDECC_APPDU_MAX_PAYLOAD_LENGTH )
199  {
200 
201  // read the 6 byte address field
202  us_buffer_read_data( buffer, self->address.value, 6, &cnt );
203  if ( cnt == 6 )
204  {
205 
206  // read the reserved field
207  if ( us_buffer_read_uint16( buffer, &self->reserved ) )
208  {
209 
210  // check to see if the buffer contains all the required payload data
211  r = JDKSAVDECC_APPDU_ERROR_PAYLOAD_INCOMPLETE;
212 
213  // apparently it has the data
214  if ( us_buffer_readable_count( buffer ) >= self->payload_length )
215  {
216 
217  // read the data into the payload buffer
218  us_buffer_read_data( buffer, self->payload, self->payload_length, &cnt );
219 
220  if ( cnt == self->payload_length )
221  {
222 
223  // success reading! return how many bytes in total were parsed
224  r = JDKSAVDECC_APPDU_HEADER_LEN + self->payload_length;
225  }
226  }
227  }
228  }
229  }
230  }
231  }
232  }
233  }
234 
235  if ( r < 0 )
236  {
237  // parsing failed, rewind the buffer
238  buffer->m_next_out = pos;
239  }
240  return r;
241 }
