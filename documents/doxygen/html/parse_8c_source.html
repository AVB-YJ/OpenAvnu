70  /* look for a null starting at the end of the buffer until
71  we find one or hit the beginning of the buffer */
72  for (i = len - 1; s[i] != 0 && i >= 0; --i) {}
73 
74  /* If a null wasn't found before the element at offset 0, counter will be negative */
75  if (i < 0)
76  return -1;

84  if (NULL == param) {

88  return -1;

95  delimiter++;

98

101  break;

182  data);
183  }
184  break;

196  count++;
197  }
