101 // callback function - called for each name/value pair by ini parsing library
102 static int openavbIniCfgCallback(void *user, const char *tlSection, const char *name, const char *value)
103 {
104  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC);
105 
106  openavb_tl_data_cfg_t *pCfg = (openavb_tl_data_cfg_t *)user;
107 
108  AVB_LOGF_DEBUG("name=[%s] value=[%s]", name, value);
109 
110  bool valOK = FALSE;
111  char *pEnd;
112 
113  if (MATCH(name, "role")) {
114  if (MATCH(value, "talker")) {
115  pCfg->role = AVB_ROLE_TALKER;
116  valOK = TRUE;
117  }
118  else if (MATCH(value, "listener")) {
119  pCfg->role = AVB_ROLE_LISTENER;
120  valOK = TRUE;
121  }
122  }
123  else if (MATCH(name, "initial_state")) {
124  if (MATCH(value, "running")) {
125  pCfg->initial_state = TL_INIT_STATE_RUNNING;
126  valOK = TRUE;
127  }
128  else if (MATCH(value, "stopped")) {
129  pCfg->initial_state = TL_INIT_STATE_STOPPED;
130  valOK = TRUE;
131  }
132  }
133  else if (MATCH(name, "dest_addr")) {
134  valOK = parse_mac(value, &pCfg->dest_addr);
135  }
136  else if (MATCH(name, "stream_addr")) {
137  valOK = parse_mac(value, &pCfg->stream_addr);
138  }
139  else if (MATCH(name, "stream_uid")) {
140  errno = 0;
141  pCfg->stream_uid = strtol(value, &pEnd, 10);
142  if (*pEnd == '\0' && errno == 0
143  && pCfg->stream_uid <= UINT16_MAX)
144  valOK = TRUE;
145  }
146  else if (MATCH(name, "max_interval_frames")) {
147  errno = 0;
148  pCfg->max_interval_frames = strtol(value, &pEnd, 10);
149  if (*pEnd == '\0' && errno == 0
150  && pCfg->max_interval_frames <= UINT16_MAX)
151  valOK = TRUE;
152  }
153  else if (MATCH(name, "max_frame_size")) {
154  errno = 0;
155  pCfg->max_frame_size = strtol(value, &pEnd, 10);
156  if (*pEnd == '\0' && errno == 0
157  && pCfg->max_interval_frames <= UINT16_MAX)
158  valOK = TRUE;
159  }
160  else if (MATCH(name, "sr_class")) {
161  if (strlen(value) == 1) {
162  if (tolower(value[0]) == 'a') {
163  pCfg->sr_class = SR_CLASS_A;
164  valOK = TRUE;
165  }
166  else if (tolower(value[0]) == 'b') {
167  pCfg->sr_class = SR_CLASS_B;
168  valOK = TRUE;
169  }
170  }
171  }
172  else if (MATCH(name, "sr_rank")) {
173  if (strlen(value) == 1) {
174  if (value[0] == '1') {
175  pCfg->sr_rank = SR_RANK_REGULAR;
176  valOK = TRUE;
177  }
178  else if (value[0] == '0') {
179  pCfg->sr_rank = SR_RANK_EMERGENCY;
180  valOK = TRUE;
181  }
182  }
183  }
184  else if (MATCH(name, "max_transit_usec")) {
185  errno = 0;
186  pCfg->max_transit_usec = strtol(value, &pEnd, 10);
187  if (*pEnd == '\0' && errno == 0
188  && pCfg->max_transit_usec <= UINT32_MAX)
189  valOK = TRUE;
190  }
191  else if (MATCH(name, "max_transmit_deficit_usec")) {
192  errno = 0;
193  pCfg->max_transmit_deficit_usec = strtol(value, &pEnd, 10);
194  if (*pEnd == '\0' && errno == 0
195  && pCfg->max_transmit_deficit_usec <= UINT32_MAX)
196  valOK = TRUE;
197  }
198  else if (MATCH(name, "internal_latency")) {
199  errno = 0;
200  pCfg->internal_latency = strtol(value, &pEnd, 10);
201  if (*pEnd == '\0' && errno == 0
202  && pCfg->internal_latency <= UINT32_MAX)
203  valOK = TRUE;
204  }
205  else if (MATCH(name, "batch_factor")) {
206  errno = 0;
207  pCfg->batch_factor = strtol(value, &pEnd, 10);
208  if (*pEnd == '\0' && errno == 0
209  && pCfg->batch_factor > 0
210  && pCfg->batch_factor <= INT32_MAX)
211  valOK = TRUE;
212  }
213  else if (MATCH(name, "max_stale")) {
214  errno = 0;
215  pCfg->max_stale = strtol(value, &pEnd, 10);
216  if (*pEnd == '\0' && errno == 0
217  && pCfg->max_stale >= 0
218  && pCfg->max_stale <= INT32_MAX)
219  valOK = TRUE;
220  }
221  else if (MATCH(name, "raw_tx_buffers")) {
222  errno = 0;
223  pCfg->raw_tx_buffers = strtol(value, &pEnd, 10);
224  if (*pEnd == '\0' && errno == 0
225  && pCfg->raw_tx_buffers <= UINT32_MAX)
226  valOK = TRUE;
227  }
228  else if (MATCH(name, "raw_rx_buffers")) {
229  errno = 0;
230  pCfg->raw_rx_buffers = strtol(value, &pEnd, 10);
231  if (*pEnd == '\0' && errno == 0
232  && pCfg->raw_rx_buffers <= UINT32_MAX)
233  valOK = TRUE;
234  }
235  else if (MATCH(name, "report_seconds")) {
236  errno = 0;
237  pCfg->report_seconds = strtol(value, &pEnd, 10);
238  if (*pEnd == '\0' && errno == 0
239  && (int)pCfg->report_seconds >= 0
240  && pCfg->report_seconds <= INT32_MAX)
241  valOK = TRUE;
242  }
243  else if (MATCH(name, "start_paused")) {
244  // ignore this item - tl_host doesn't use it because
245  // it pauses before reading any of its streams.
246  errno = 0;
247  long tmp;
248  tmp = strtol(value, &pEnd, 10);
249  if (*pEnd == '\0' && errno == 0
250  && tmp >= 0
251  && tmp <= 1) {
252  pCfg->start_paused = (tmp == 1);
253  valOK = TRUE;
254  }
255  }
256  else if (MATCH(name, "vlan_id")) {
257  errno = 0;
258  long tmp;
259  tmp = strtol(value, &pEnd, 0);
260  // vlanID is 12 bit field
261  if (*pEnd == '\0' && errno == 0
262  && tmp >= 0x0
263  && tmp <= 0xFFF) {
264  pCfg->vlan_id = tmp;
265  valOK = TRUE;
266  }
267  }
268  else if (MATCH(name, "fixed_timestamp")) {
269  errno = 0;
270  long tmp;
271  tmp = strtol(value, &pEnd, 0);
272  if (*pEnd == '\0' && errno == 0) {
273  pCfg->fixed_timestamp = tmp;
274  valOK = TRUE;
275  }
276  }
277  else if (MATCH(name, "spin_wait")) {
278  errno = 0;
279  long tmp;
280  tmp = strtol(value, &pEnd, 0);
281  if (*pEnd == '\0' && errno == 0) {
282  pCfg->spin_wait = (tmp == 1);
283  valOK = TRUE;
284  }
285  }
286  else if (MATCH(name, "tx_blocking_in_intf")) {
287  errno = 0;
288  long tmp;
289  tmp = strtol(value, &pEnd, 0);
290  if (*pEnd == '\0' && errno == 0) {
291  pCfg->tx_blocking_in_intf = tmp;
292  valOK = TRUE;
293  }
294  }
295  else if (MATCH(name, "thread_rt_priority")) {
296  errno = 0;
297  long tmp;
298  tmp = strtol(value, &pEnd, 0);
299  if (*pEnd == '\0' && errno == 0) {
300  pCfg->thread_rt_priority = tmp;
301  valOK = TRUE;
302  }
303  }
304  else if (MATCH(name, "thread_affinity")) {
305  errno = 0;
306  unsigned long tmp;
307  tmp = strtoul(value, &pEnd, 0);
308  if (*pEnd == '\0' && errno == 0) {
309  pCfg->thread_affinity = tmp;
310  valOK = TRUE;
311  }
312  }
313 
314  else if (MATCH(name, "friendly_name")) {
315  strncpy(pCfg->friendly_name, value, FRIENDLY_NAME_SIZE - 1);
316  valOK = TRUE;
317  }
318 
319  else if (MATCH(name, "current_sampling_rate")) {
320  errno = 0;
321  pCfg->current_sampling_rate = strtol(value, &pEnd, 10);
322  if (*pEnd == '\0' && errno == 0
323  && pCfg->current_sampling_rate <= UINT32_MAX)
324  valOK = TRUE;
325  }
326  else if (MATCH(name, "sampling_rates")) {
327  errno = 0;
328  memset(pCfg->sampling_rates,0,sizeof(pCfg->sampling_rates));
329  char *rate, *copy;
330  copy = strdup(value);
331  rate = strtok(copy,",");
332  int i = 0,break_flag = 0;
333  while (rate != NULL )
334  {
335  pCfg->sampling_rates[i] = strtol(rate,&pEnd, 10);
336  if (*pEnd != '\0' || errno != 0
337  || pCfg->sampling_rates[i] > UINT32_MAX)
338  {
339  break_flag = 1;
340  break;
341  }
342  rate = strtok(NULL,",");
343  i++;
344  }
345  if (break_flag != 1)
346  {
347  valOK = TRUE;
348  pCfg->sampling_rates_count = i;
349  }
350  }
351  else if (strcmp(name, "intf_nv_audio_rate") == 0) {
352  long int val = strtol(value, &pEnd, 10);
353  if (val >= AVB_AUDIO_RATE_8KHZ && val <= AVB_AUDIO_RATE_192KHZ) {
354  pCfg->audioRate = val;
355  valOK = TRUE;
356  }
357  else {
358  AVB_LOG_ERROR("Invalid audio rate configured for intf_nv_audio_rate.");
359  pCfg->audioRate = AVB_AUDIO_RATE_44_1KHZ;
360  }
361  }
362  else if (strcmp(name, "intf_nv_audio_bit_depth") == 0) {
363  long int val = strtol(value, &pEnd, 10);
364  if (val >= AVB_AUDIO_BIT_DEPTH_1BIT && val <= AVB_AUDIO_BIT_DEPTH_64BIT) {
365  pCfg->audioBitDepth = val;
366  valOK = TRUE;
367  }
368  else {
369  AVB_LOG_ERROR("Invalid audio type configured for intf_nv_audio_bits.");
370  pCfg->audioBitDepth = AVB_AUDIO_BIT_DEPTH_24BIT;
371  }
372  }
373  else if (strcmp(name, "intf_nv_audio_channels") == 0) {
374  long int val = strtol(value, &pEnd, 10);
375  if (val >= AVB_AUDIO_CHANNELS_1 && val <= AVB_AUDIO_CHANNELS_8) {
376  pCfg->audioChannels = val;
377  valOK = TRUE;
378  }
379  else {
380  AVB_LOG_ERROR("Invalid audio channels configured for intf_nv_audio_channels.");
381  pCfg->audioChannels = AVB_AUDIO_CHANNELS_2;
382  }
383  }
384  else if (MATCH(name, "map_fn")) {
385  errno = 0;
386  memset(pCfg->map_fn,0,sizeof(pCfg->map_fn));
387  strncpy(pCfg->map_fn,value,sizeof(pCfg->map_fn)-1);
388  valOK = TRUE;
389  }
390 
391  else {
392  // Ignored item.
393  AVB_LOGF_DEBUG("Unhandled configuration item: name=%s, value=%s", name, value);
394 
395  // Don't abort for this item.
396  valOK = TRUE;
397  }
398 
399  if (!valOK) {
400  // bad value
401  AVB_LOGF_ERROR("Invalid value: name=%s, value=%s", name, value);
402  return 0;
403  }
404 
405  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
406 
407  return 1; // OK
408 }

410 bool openavbReadTlDataIniFile(const char *fileName, openavb_tl_data_cfg_t *pCfg)
411 {
412  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC);
413 
414  openavbIniCfgInit(pCfg);
415 
416  // Use the .INI file name as the default friendly name.
417  strncpy(pCfg->friendly_name, fileName, FRIENDLY_NAME_SIZE - 1);
418  char * pszComma = strchr(pCfg->friendly_name, ',');
419  if (pszComma) {
420  // Get rid of anything following the file name.
421  *pszComma = '\0';
422  }
423  char * pszExtension = strrchr(pCfg->friendly_name, '.');
424  if (pszExtension && strcasecmp(pszExtension, ".ini") == 0) {
425  // Get rid of the .INI file extension.
426  *pszExtension = '\0';
427  }
428 
429  int result = ini_parse(fileName, openavbIniCfgCallback, pCfg);
430  if (result < 0) {
431  AVB_LOGF_ERROR("Couldn't parse INI file: %s", fileName);
432  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
433  return FALSE;
434  }
435  if (result > 0) {
436  AVB_LOGF_ERROR("Error in INI file: %s, line %d", fileName, result);
437  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
438  return FALSE;
439  }
440 
441  if (pCfg->current_sampling_rate != 0 && pCfg->audioRate != 0 &&
442  pCfg->current_sampling_rate != pCfg->audioRate)
443  {
444  AVB_LOGF_ERROR("current_sampling_rate(%u) and intf_nv_audio_rate(%u) do not match.", pCfg->current_sampling_rate, pCfg->audioRate);
445  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
446  return FALSE;
447  }
448 
449  if (pCfg->current_sampling_rate == 0)
450  {
451  /* Make sure we have a default sampling rate. */
452  if (pCfg->audioRate != 0) {
453  pCfg->current_sampling_rate = pCfg->audioRate;
454  } else if (pCfg->sampling_rates[0] != 0) {
455  pCfg->current_sampling_rate = pCfg->sampling_rates[0];
456  } else {
457  pCfg->current_sampling_rate = AVB_AUDIO_RATE_48KHZ;
458  }
459  if (pCfg->audioRate == 0) {
460  AVB_LOGF_WARNING("current_sampling_rate not specified in %s. Defaulting to %u", fileName, pCfg->current_sampling_rate);
461  }
462  }
463  if (pCfg->sampling_rates_count == 0)
464  {
465  /* Set the list of sampling rates to the current sampling rate. */
466  pCfg->sampling_rates[0] = pCfg->current_sampling_rate;
467  pCfg->sampling_rates_count = 1;
468  if (pCfg->audioRate == 0) {
469  AVB_LOGF_WARNING("sampling_rates not specified in %s. Defaulting to %u", fileName, pCfg->current_sampling_rate);
470  }
471  } else {
472  /* Make sure the current sampling rate is in the list of sampling rates. */
473  U16 i;
474  for (i = 0; i < pCfg->sampling_rates_count; ++i) {
475  if (pCfg->sampling_rates[i] == pCfg->current_sampling_rate) break;
476  }
477  if (i >= pCfg->sampling_rates_count) {
478  AVB_LOGF_ERROR("current_sampling_rate(%u) not in list of sampling_rates.", pCfg->current_sampling_rate);
479  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
480  return FALSE;
481  }
482  }
483 
484  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
485  return TRUE;
486 }

489 // Save the connection to the saved state
490 //
491 bool openavbAvdeccSaveState(const openavb_tl_data_cfg_t *pListener, U16 flags, U16 talker_unique_id, const U8 talker_entity_id[8], const U8 controller_entity_id[8])
492 {
493  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC);
494 
495  int i;
496 
497  // Don't add to the saved state list if fast connect support is not enabled.
498  if (!gAvdeccCfg.bFastConnectSupported) {
499  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
500  return false;
501  }
502 
503  // If the supplied saved state matches one of the ones already saved, do nothing and return.
504  // If the Talker or Controller has changed, delete the old information.
505  for (i = 0; i < 1000; ++i) {
506  const openavb_saved_state_t * pTest = openavbAvdeccGetSavedState(i);
507  if (!pTest) {
508  break;
509  }
510 
511  if (strcmp(pTest->listener_friendly_name, pListener->friendly_name) == 0) {
512  if (pTest->flags == flags &&
513  pTest->talker_unique_id == talker_unique_id &&
514  memcmp(pTest->talker_entity_id, talker_entity_id, 8) == 0 &&
515  memcmp(pTest->controller_entity_id, controller_entity_id, 8) == 0) {
516  // The supplied data is a match for the existing item. Do nothing.
517  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
518  return true;
519  }
520 
521  // Delete this item. We will create a new item with the updated information.
522  openavbAvdeccDeleteSavedState(i);
523  break;
524  }
525  }
526 
527  // Add the supplied state to the list of states.
528  openavbAvdeccAddSavedState(pListener->friendly_name, flags, talker_unique_id, talker_entity_id, controller_entity_id);
529 
530  AVB_LOGF_DEBUG("New saved state: listener_id=%s, flags=0x%04x, talker_unique_id=0x%04x, talker_entity_id=" ENTITYID_FORMAT ", controller_entity_id=" ENTITYID_FORMAT,
531  pListener->friendly_name,
532  flags,
533  talker_unique_id,
534  ENTITYID_ARGS(talker_entity_id),
535  ENTITYID_ARGS(controller_entity_id));
536 
537  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
538  return true;
539 }

541 // Delete a connection with saved state
542 //
543 bool openavbAvdeccClearSavedState(const openavb_tl_data_cfg_t *pListener)
544 {
545  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC);
546 
547  int i;
548 
549  // Delete the saved state matching the supplied one.
550  // If the supplied saved state does not match any of the ones already saved, do nothing and return.
551  for (i = 0; i < 1000; ++i) {
552  const openavb_saved_state_t * pTest = openavbAvdeccGetSavedState(i);
553  if (!pTest) {
554  break;
555  }
556 
557  if (strcmp(pTest->listener_friendly_name, pListener->friendly_name) == 0) {
558  // We found the index for the item to delete.
559  // Delete the item and save the updated file.
560  openavbAvdeccDeleteSavedState(i);
561  AVB_LOGF_DEBUG("Cleared saved state: listener_id=\"%s\"", pListener->friendly_name);
562 
563  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
564  return true;
565  }
566  }
567 
568  AVB_LOGF_WARNING("Unable to find saved state to clear: listener_id=\"%s\"", pListener->friendly_name);
569  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
570  return false;
571 }

573 // Determine if the connection has a saved state
574 //
575 bool openavbAvdeccGetSaveStateInfo(const openavb_tl_data_cfg_t *pListener, U16 *p_flags, U16 *p_talker_unique_id, U8 (*p_talker_entity_id)[8], U8 (*p_controller_entity_id)[8])
576 {
577  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC);
578 
579  int i;
580 
581  // Don't return anything from the saved state list if fast connect support is not enabled.
582  if (!gAvdeccCfg.bFastConnectSupported) {
583  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
584  return false;
585  }
