31 int32_t avb_get_iface_mac_address(int8_t *iface, uint8_t *addr)
32 {
33  struct ifreq ifreq;
34  int fd, ret;
35 
36  /* Create a socket */
37  fd = socket(PF_PACKET, SOCK_RAW, htons(0x800));
38  if (fd < 0)
39  return -1;
40 
41  memset(&ifreq, 0, sizeof(ifreq));
42 
43  strncpy(ifreq.ifr_name, (const char*)iface, sizeof(ifreq.ifr_name));
44  ret = ioctl(fd, SIOCGIFHWADDR, &ifreq);
45  if (ret < 0) {
46  close(fd);
47  return -1;
48  }
49 
50  memcpy(addr, ifreq.ifr_hwaddr.sa_data, ETH_ALEN);
51  close(fd);
52 
53  return 0;
54 }
