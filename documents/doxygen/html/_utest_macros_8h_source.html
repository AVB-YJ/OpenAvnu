333 #define CHECK_THROWS(expected, expression) \
334  { \
335  SimpleString failure_msg("expected to throw "#expected "\nbut threw nothing"); \
336  bool caught_expected = false; \
337  try { \
338  (expression); \
339  } catch(const expected &) { \
340  caught_expected = true; \
341  } catch(...) { \
342  failure_msg = "expected to throw " #expected "\nbut threw a different type"; \
343  } \
344  if (!caught_expected) { \
345  UtestShell::getCurrent()->fail(failure_msg.asCharString(), __FILE__, __LINE__); \
346  } \
347  else { \
348  UtestShell::getCurrent()->countCheck(); \
349  } \
350  }
