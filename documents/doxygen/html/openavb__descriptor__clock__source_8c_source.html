52 openavbRC openavbAemDescriptorClockSourceToBuf(void *pVoidDescriptor, U16 bufLength, U8 *pBuf, U16 *descriptorSize)
53 {
54  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
55 
56  openavb_aem_descriptor_clock_source_t *pDescriptor = pVoidDescriptor;
57 
58  if (!pDescriptor || !pBuf || !descriptorSize) {
59  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_INVALID_ARGUMENT), AVB_TRACE_AEM);
60  }
61 
62  if (bufLength < OPENAVB_DESCRIPTOR_CLOCK_SOURCE_BASE_LENGTH) {
63  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVBAVDECC_RC_BUFFER_TOO_SMALL), AVB_TRACE_AEM);
64  }
65 
66  *descriptorSize = 0;
67 
68  openavb_aem_descriptor_clock_source_t *pSrc = pDescriptor;
69  U8 *pDst = pBuf;
70 
71  OCT_D2BHTONS(pDst, pSrc->descriptor_type);
72  OCT_D2BHTONS(pDst, pSrc->descriptor_index);
73  OCT_D2BMEMCP(pDst, pSrc->object_name);
74  BIT_D2BHTONS(pDst, pSrc->localized_description.offset, 3, 0);
75  BIT_D2BHTONS(pDst, pSrc->localized_description.index, 0, 2);
76  OCT_D2BHTONS(pDst, pSrc->clock_source_flags);
77  OCT_D2BHTONS(pDst, pSrc->clock_source_type);
78  OCT_D2BMEMCP(pDst, pSrc->clock_source_identifier);
79  OCT_D2BHTONS(pDst, pSrc->clock_source_location_type);
80  OCT_D2BHTONS(pDst, pSrc->clock_source_location_index);
81 
82  *descriptorSize = pDst - pBuf;
83 
84  AVB_RC_TRACE_RET(OPENAVB_AVDECC_SUCCESS, AVB_TRACE_AEM);
85 }

89  AVB_TRACE_ENTRY(AVB_TRACE_AEM);

99  }
100 
101  U8 *pSrc = pBuf;
102  openavb_aem_descriptor_clock_source_t *pDst = pDescriptor;
103

108  BIT_B2DNTOHS(pDst->localized_description.index, pSrc, 0x0007, 0, 2);
109  OCT_B2DNTOHS(pDst->clock_source_flags, pSrc);
110  OCT_B2DNTOHS(pDst->clock_source_type, pSrc);
111  OCT_B2DMEMCP(pDst->clock_source_identifier, pSrc);

136 extern DLL_EXPORT openavb_aem_descriptor_clock_source_t *openavbAemDescriptorClockSourceNew()
137 {
138  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
139 
140  openavb_aem_descriptor_clock_source_t *pDescriptor;
141 
142  pDescriptor = malloc(sizeof(*pDescriptor));
143 
144  if (!pDescriptor) {
145  AVB_RC_LOG(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_OUT_OF_MEMORY));
146  AVB_TRACE_EXIT(AVB_TRACE_AEM);
147  return NULL;
148  }
149  memset(pDescriptor, 0, sizeof(*pDescriptor));
150 
151  pDescriptor->descriptorPvtPtr = malloc(sizeof(*pDescriptor->descriptorPvtPtr));
152  if (!pDescriptor->descriptorPvtPtr) {
153  free(pDescriptor);
154  pDescriptor = NULL;
155  AVB_RC_LOG(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_OUT_OF_MEMORY));
156  AVB_TRACE_EXIT(AVB_TRACE_AEM);
157  return NULL;
158  }
159 
160  pDescriptor->descriptorPvtPtr->size = sizeof(openavb_aem_descriptor_clock_source_t);
161  pDescriptor->descriptorPvtPtr->bTopLevel = TRUE;
162  pDescriptor->descriptorPvtPtr->toBuf = openavbAemDescriptorClockSourceToBuf;
163  pDescriptor->descriptorPvtPtr->fromBuf = openavbAemDescriptorClockSourceFromBuf;
164  pDescriptor->descriptorPvtPtr->update = openavbAemDescriptorClockSourceUpdate;
165 
166  pDescriptor->descriptor_type = OPENAVB_AEM_DESCRIPTOR_CLOCK_SOURCE;
167 
168  // Default to no localized strings.
169  pDescriptor->localized_description.offset = OPENAVB_AEM_NO_STRING_OFFSET;
170  pDescriptor->localized_description.index = OPENAVB_AEM_NO_STRING_INDEX;
171 
172  AVB_TRACE_EXIT(AVB_TRACE_AEM);
173  return pDescriptor;
174 }

176 extern DLL_EXPORT bool openavbAemDescriptorClockSourceInitialize(openavb_aem_descriptor_clock_source_t *pDescriptor, U16 nConfigIdx, const openavb_avdecc_configuration_cfg_t *pConfig)
177 {
178  (void) nConfigIdx;
179  if (!pDescriptor || !pConfig) {
180  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_INVALID_ARGUMENT), AVB_TRACE_AEM);
181  }
182 
183  // AVDECC_TODO: These values need to be verified.
184  if (pConfig->stream->role == AVB_ROLE_TALKER)
185  {
186  // Make this an internal clock.
187  strcpy((char *) pDescriptor->object_name, "Internal Clock");
188  pDescriptor->clock_source_flags = 0;
189  pDescriptor->clock_source_type = OPENAVB_AEM_CLOCK_SOURCE_TYPE_INTERNAL;
190  pDescriptor->clock_source_location_type = OPENAVB_AEM_DESCRIPTOR_STREAM_OUTPUT;
191  pDescriptor->clock_source_location_index = 0;
192  }
193  else if (pConfig->stream->role == AVB_ROLE_LISTENER)
194  {
195  // Use the Talker's clock.
196  strcpy((char *) pDescriptor->object_name, "Input Stream");
197  pDescriptor->clock_source_flags = OPENAVB_AEM_CLOCK_SOURCE_FLAG_STREAM_ID;
198  pDescriptor->clock_source_type = OPENAVB_AEM_CLOCK_SOURCE_TYPE_INPUT_STREAM;
199  memcpy(pDescriptor->clock_source_identifier, pConfig->stream->stream_addr.buffer.ether_addr_octet, ETH_ALEN);
200  pDescriptor->clock_source_location_type = OPENAVB_AEM_DESCRIPTOR_STREAM_INPUT;
201  pDescriptor->clock_source_location_index = 0;
202  }
203 
204  return TRUE;
205 }
