220 // This talker callback will be called for each AVB observation interval.
221 tx_cb_ret_t openavbMapMjpegTxCB(media_q_t *pMediaQ, U8 *pData, U32 *dataLen)
222 {
223  AVB_TRACE_ENTRY(AVB_TRACE_MAP_DETAIL);
224  if (pMediaQ && pData && dataLen) {
225  U8 *pHdr = pData;
226  U8 *pPayload = pData + TOTAL_HEADER_SIZE;
227  pvt_data_t *pPvtData = pMediaQ->pPvtMapInfo;
228  if (!pPvtData) {
229  AVB_LOG_ERROR("Private mapping module data not allocated.");
230  return TX_CB_RET_PACKET_NOT_READY;
231  }
232 
233  //pHdr[HIDX_AVTP_TIMESTAMP32] = 0x00; // Set later
234  pHdr[HIDX_FORMAT8] = 0x02; // RTP Payload type
235  *(U16 *)(&pHdr[HIDX_FORMAT_SUBTYPE8]) = 0x00; // MJPEG format (RFC 2435)
236  //pHdr[HIDX_STREAM_DATA_LEN16] = 0x00; // Set later
237  pHdr[HIDX_RESV16] = 0x00;
238  pHdr[HIDX_RESV16+1] = 0x00;
239  pHdr[HIDX_M11_M01_EVT2_RESV2] = 0x00;
240  pHdr[HIDX_RESV8] = 0x00;
241 
242  media_q_item_t *pMediaQItem = openavbMediaQTailLock(pMediaQ, TRUE);
243  if (pMediaQItem) {
244  if (pMediaQItem->dataLen > 0) {
245  if (pMediaQItem->dataLen > ITEM_SIZE) {
246  AVB_LOGF_ERROR("Media queue data item size too large. Reported size: %d Max Size: %d", pMediaQItem->dataLen, ITEM_SIZE);
247  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
248  openavbMediaQTailPull(pMediaQ);
249  return TX_CB_RET_PACKET_NOT_READY;
250  }
251 
252  // PTP walltime already set in the interface module. Just add the max transit time.
253  openavbAvtpTimeAddUSec(pMediaQItem->pAvtpTime, pPvtData->maxTransitUsec);
254 
255  // Set timestamp valid flag
256  if (openavbAvtpTimeTimestampIsValid(pMediaQItem->pAvtpTime)) {
257  pHdr[HIDX_AVTP_HIDE7_TV1] |= 0x01; // Set
258  }
259  else {
260  pHdr[HIDX_AVTP_HIDE7_TV1] &= ~0x01; // Clear
261  }
262 
263  // Set timestamp uncertain flag
264  if (openavbAvtpTimeTimestampIsUncertain(pMediaQItem->pAvtpTime)) {
265  pHdr[HIDX_AVTP_HIDE7_TU1] |= 0x01; // Set
266  }
267  else {
268  pHdr[HIDX_AVTP_HIDE7_TU1] &= ~0x01; // Clear
269  }
270 
271  // Set the timestamp.
272  // 1722a-D6: The avtp_timestamp represents the presentation time associated with the given frame. The same avtp_timestamp
273  // shall appear in each fragment of a given frame. The M0 marker bit shall be set in the last packet of a frame.
274  if (!pPvtData->tsvalid)
275  {
276  pPvtData->timestamp = openavbAvtpTimeGetAvtpTimestamp(pMediaQItem->pAvtpTime);
277  pPvtData->tsvalid = TRUE;
278  }
279  *(U32 *)(&pHdr[HIDX_AVTP_TIMESTAMP32]) = htonl(pPvtData->timestamp);
280 
281  if (((media_q_item_map_mjpeg_pub_data_t *)pMediaQItem->pPubMapData)->lastFragment) {
282  pHdr[HIDX_M11_M01_EVT2_RESV2] = 0x00 | (1 << 4);
283  pPvtData->tsvalid = FALSE;
284  }
285  else {
286  pHdr[HIDX_M11_M01_EVT2_RESV2] = 0x00;
287  }
288 
289  // Copy the JPEG fragment into the outgoing avtp packet.
290  memcpy(pPayload, pMediaQItem->pPubData, pMediaQItem->dataLen);
291 
292  *(U16 *)(&pHdr[HIDX_STREAM_DATA_LEN16]) = pMediaQItem->dataLen;
293 
294  // Set out bound data length (entire packet length)
295  *dataLen = pMediaQItem->dataLen + TOTAL_HEADER_SIZE;
296 
297  AVB_TRACE_LINE(AVB_TRACE_MAP_LINE);
298  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
299  openavbMediaQTailPull(pMediaQ);
300  return TX_CB_RET_PACKET_READY;
301  }
302  openavbMediaQTailPull(pMediaQ);
303  }
304  }
305 
306  if (dataLen) {
307  // Set out bound data length (entire packet length)
308  *dataLen = 0;
309  }
310 
311  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
312  return TX_CB_RET_PACKET_NOT_READY;
313 }

323 // This callback occurs when running as a listener and data is available.
324 bool openavbMapMjpegRxCB(media_q_t *pMediaQ, U8 *pData, U32 dataLen)
325 {
326  AVB_TRACE_ENTRY(AVB_TRACE_MAP_DETAIL);
327  if (pMediaQ && pData) {
328  U8 *pHdr = pData;
329  U8 *pPayload = pData + TOTAL_HEADER_SIZE;
330 
331 // U16 payloadLen = ntohs(*(U16 *)(&pHdr[HIDX_STREAM_DATA_LEN16]));
332  U16 payloadLen = *(U16 *)(&pHdr[HIDX_STREAM_DATA_LEN16]);
333 
334  // Get item pointer in media queue
335  media_q_item_t *pMediaQItem = openavbMediaQHeadLock(pMediaQ);
336  if (pMediaQItem) {
337  // Get the timestamp and place it in the media queue item.
338  U32 timestamp = ntohl(*(U32 *)(&pHdr[HIDX_AVTP_TIMESTAMP32]));
339  openavbAvtpTimeSetToTimestamp(pMediaQItem->pAvtpTime, timestamp);
340 
341  // Set timestamp valid and timestamp uncertain flags
342  openavbAvtpTimeSetTimestampValid(pMediaQItem->pAvtpTime, (pHdr[HIDX_AVTP_HIDE7_TV1] & 0x01) ? TRUE : FALSE);
343  openavbAvtpTimeSetTimestampUncertain(pMediaQItem->pAvtpTime, (pHdr[HIDX_AVTP_HIDE7_TU1] & 0x01) ? TRUE : FALSE);
344 
345  if (pHdr[HIDX_M11_M01_EVT2_RESV2] & 0x10) {
346  ((media_q_item_map_mjpeg_pub_data_t *)pMediaQItem->pPubMapData)->lastFragment = TRUE;
347  }
348  else {
349  ((media_q_item_map_mjpeg_pub_data_t *)pMediaQItem->pPubMapData)->lastFragment = FALSE;
350  }
351 
352  if (pMediaQItem->itemSize >= dataLen - TOTAL_HEADER_SIZE) {
353  memcpy(pMediaQItem->pPubData, pPayload, payloadLen);
354  pMediaQItem->dataLen = dataLen - TOTAL_HEADER_SIZE;
355  }
356  else {
357  AVB_LOG_ERROR("Data to large for media queue.");
358  pMediaQItem->dataLen = 0;
359  }
360 
361  openavbMediaQHeadPush(pMediaQ);
362  AVB_TRACE_LINE(AVB_TRACE_MAP_LINE);
363  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
364  return TRUE;
365  }
366  else {
367  IF_LOG_INTERVAL(1000) AVB_LOG_ERROR("Media queue full.");
368  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
369  return FALSE; // Media queue full
370  }
371  }
372  AVB_TRACE_EXIT(AVB_TRACE_MAP_DETAIL);
373  return FALSE;
374 }
