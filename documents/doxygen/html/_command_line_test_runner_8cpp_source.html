40 int CommandLineTestRunner::RunAllTests(int ac, const char** av)
41 {
42  int result = 0;
43  ConsoleTestOutput backupOutput;
44 
45  MemoryLeakWarningPlugin memLeakWarn(DEF_PLUGIN_MEM_LEAK);
46  memLeakWarn.destroyGlobalDetectorAndTurnOffMemoryLeakDetectionInDestructor(true);
47  TestRegistry::getCurrentRegistry()->installPlugin(&memLeakWarn);
48 
49  {
50  CommandLineTestRunner runner(ac, av, TestRegistry::getCurrentRegistry());
51  result = runner.runAllTestsMain();
52  }
53 
54  if (result == 0) {
55  backupOutput << memLeakWarn.FinalReport(0);
56  }
57  TestRegistry::getCurrentRegistry()->removePluginByName(DEF_PLUGIN_MEM_LEAK);
58  return result;
59 }

98 int CommandLineTestRunner::runAllTests()
99 {
100  initializeTestRun();
101  int loopCount = 0;
102  int failureCount = 0;
103  int repeat_ = arguments_->getRepeatCount();
104 
105  if (arguments_->isListingTestGroupNames())
106  {
107  TestResult tr(*output_);
108  registry_->listTestGroupNames(tr);
109  return 0;
110  }
111 
112  if (arguments_->isListingTestGroupAndCaseNames())
113  {
114  TestResult tr(*output_);
115  registry_->listTestGroupAndCaseNames(tr);
116  return 0;
117  }
118 
119  while (loopCount++ < repeat_) {
120  output_->printTestRun(loopCount, repeat_);
121  TestResult tr(*output_);
122  registry_->runAllTests(tr);
123  failureCount += tr.getFailureCount();
124  }
125 
126  return failureCount;
127 }

156 bool CommandLineTestRunner::parseArguments(TestPlugin* plugin)
157 {
158  if (!arguments_->parse(plugin)) {
159  output_ = createConsoleOutput();
160  output_->print(arguments_->usage());
161  return false;
162  }
163 
164  if (arguments_->isJUnitOutput()) {
165  output_= createJUnitOutput(arguments_->getPackageName());
166  if (arguments_->isVerbose())
167  output_ = createCompositeOutput(output_, createConsoleOutput());
168  } else if (arguments_->isTeamCityOutput()) {
169  output_ = createTeamCityOutput();
170  } else
171  output_ = createConsoleOutput();
172  return true;
173 }
