114 static void process_maap_notify(Maap_Notify *mn, int socketfd)
115 {
116  assert(mn);
117 
118  switch (mn->result)
119  {
120  case MAAP_NOTIFY_ERROR_NONE:
121  /* No error. Don't display anything. */
122  break;
123  case MAAP_NOTIFY_ERROR_REQUIRES_INITIALIZATION:
124  AVB_LOG_ERROR("MAAP is not initialized, so the command cannot be performed.");
125  break;
126  case MAAP_NOTIFY_ERROR_ALREADY_INITIALIZED:
127  AVB_LOG_ERROR("MAAP is already initialized, so the values cannot be changed.");
128  break;
129  case MAAP_NOTIFY_ERROR_RESERVE_NOT_AVAILABLE:
130  AVB_LOG_ERROR("The MAAP reservation is not available, or yield cannot allocate a replacement block. "
131  "Try again with a smaller address block size.");
132  break;
133  case MAAP_NOTIFY_ERROR_RELEASE_INVALID_ID:
134  AVB_LOG_ERROR("The MAAP reservation ID is not valid, so cannot be released or report its status.");
135  break;
136  case MAAP_NOTIFY_ERROR_OUT_OF_MEMORY:
137  AVB_LOG_ERROR("The MAAP application is out of memory.");
138  break;
139  case MAAP_NOTIFY_ERROR_INTERNAL:
140  AVB_LOG_ERROR("The MAAP application experienced an internal error.");
141  break;
142  default:
143  AVB_LOGF_ERROR("The MAAP application returned an unknown error %d.", mn->result);
144  break;
145  }
146 
147  switch (mn->kind)
148  {
149  case MAAP_NOTIFY_INITIALIZED:
150  if (mn->result == MAAP_NOTIFY_ERROR_NONE) {
151  AVB_LOGF_DEBUG("MAAP initialized: 0x%012llx-0x%012llx (Size: %d)",
152  (unsigned long long) mn->start,
153  (unsigned long long) mn->start + mn->count - 1,
154  (unsigned int) mn->count);
155 
156  // We successfully initialized. Reserve a block of addresses.
157  Maap_Cmd maapcmd;
158  memset(&maapcmd, 0, sizeof(Maap_Cmd));
159  maapcmd.kind = MAAP_CMD_RESERVE;
160  maapcmd.start = 0; // No preferred address
161  maapcmd.count = MAX_AVB_STREAMS;
162  if (maapPreferredAddress != NULL) {
163  // Suggest the addresses from the previous time this application was run.
164  maapcmd.start = MaapMacAddrToLongLong(maapPreferredAddress);
165  AVB_LOGF_DEBUG("MAAP Preferred Address: 0x%012llx", maapcmd.start);
166  }
167  if (send(socketfd, (char *) &maapcmd, sizeof(Maap_Cmd), 0) < 0)
168  {
169  /* Something went wrong. Abort! */
170  AVB_LOGF_ERROR("MAAP: Error %d writing to network socket (%s)", errno, strerror(errno));
171  maapState = MAAP_STATE_ERROR;
172  } else {
173  maapState = MAAP_STATE_REQUESTING;
174  }
175  } else {
176  AVB_LOGF_ERROR("MAAP previously initialized to 0x%012llx-0x%012llx (Size: %d)",
177  (unsigned long long) mn->start,
178  (unsigned long long) mn->start + mn->count - 1,
179  (unsigned int) mn->count);
180  maapState = MAAP_STATE_ERROR;
181  }
182  break;
183  case MAAP_NOTIFY_ACQUIRING:
184  if (mn->result == MAAP_NOTIFY_ERROR_NONE) {
185  AVB_LOGF_DEBUG("MAAP address range %d querying: 0x%012llx-0x%012llx (Size %d)",
186  mn->id,
187  (unsigned long long) mn->start,
188  (unsigned long long) mn->start + mn->count - 1,
189  mn->count);
190  } else {
191  AVB_LOGF_ERROR("MAAP unknown address range %d acquisition error", mn->id);
192  maapState = MAAP_STATE_ERROR;
193  }
194  break;
195  case MAAP_NOTIFY_ACQUIRED:
196  if (mn->result == MAAP_NOTIFY_ERROR_NONE) {
197  AVB_LOGF_INFO("MAAP address range %d acquired: 0x%012llx-0x%012llx (Size %d)",
198  mn->id,
199  (unsigned long long) mn->start,
200  (unsigned long long) mn->start + mn->count - 1,
201  mn->count);
202 
203  // Update the stored addresses.
204  MAAP_LOCK();
205  int i = 0;
206  for (i = 0; i < mn->count && i < MAX_AVB_STREAMS; i++) {
207  MaapResultToMacAddr(mn->start + i, &(maapAllocList[i].destAddr));
208  if (maapAllocList[i].taken && maapRestartCallback) {
209  // Use the callback to notify that a change has occurred.
210  MAAP_UNLOCK();
211  maapRestartCallback(&(maapAllocList[i]), &(maapAllocList[i].destAddr));
212  MAAP_LOCK();
213  }
214  }
215  MAAP_UNLOCK();
216 
217  // Save the address so we can request it in the future.
218  if (maapPreferredAddress != NULL) {
219  struct ether_addr assignedAddress;
220  MaapResultToMacAddr((unsigned long long) mn->start, &assignedAddress);
221  if (memcmp(maapPreferredAddress, &assignedAddress, sizeof(struct ether_addr)) != 0) {
222  // Save the updated settings. Done immediately, so they won't get lost on device reboot.
223  memcpy(maapPreferredAddress, &assignedAddress, sizeof(struct ether_addr));
224  openavbSaveConfig(DEFAULT_SAVE_INI_FILE, &x_cfg);
225  }
226  }
227 
228  // We now have addresses we can use.
229  maapReservationId = mn->id;
230  maapState = MAAP_STATE_CONNECTED;
231  } else if (mn->id != -1) {
232  AVB_LOGF_ERROR("MAAP address range %d of size %d not acquired",
233  mn->id, mn->count);
234  maapState = MAAP_STATE_ERROR;
235  } else {
236  AVB_LOGF_ERROR("MAAP address range of size %d not acquired",
237  mn->count);
238  maapState = MAAP_STATE_ERROR;
239  }
240  break;
241  case MAAP_NOTIFY_RELEASED:
242  if (mn->result == MAAP_NOTIFY_ERROR_NONE) {
243  AVB_LOGF_DEBUG("MAAP address range %d released: 0x%012llx-0x%012llx (Size %d)",
244  mn->id,
245  (unsigned long long) mn->start,
246  (unsigned long long) mn->start + mn->count - 1,
247  mn->count);
248  maapState = MAAP_STATE_RELEASED;
249  } else {
250  AVB_LOGF_WARNING("MAAP address range %d not released",
251  mn->id);
252  }
253  break;
254  case MAAP_NOTIFY_STATUS:
255  if (mn->result == MAAP_NOTIFY_ERROR_NONE) {
256  AVB_LOGF_DEBUG("MAAP ID %d is address range 0x%012llx-0x%012llx (Size %d)",
257  mn->id,
258  (unsigned long long) mn->start,
259  (unsigned long long) mn->start + mn->count - 1,
260  mn->count);
261  } else {
262  AVB_LOGF_DEBUG("MAAP ID %d is not valid",
263  mn->id);
264  }
265  break;
266  case MAAP_NOTIFY_YIELDED:
267  if (mn->result != MAAP_NOTIFY_ERROR_REQUIRES_INITIALIZATION && mn->result != MAAP_NOTIFY_ERROR_RELEASE_INVALID_ID) {
268  if (mn->result != MAAP_NOTIFY_ERROR_NONE) {
269  AVB_LOGF_ERROR("MAAP Address range %d yielded: 0x%012llx-0x%012llx (Size %d). A new address range will not be allocated.",
270  mn->id,
271  (unsigned long long) mn->start,
272  (unsigned long long) mn->start + mn->count - 1,
273  mn->count);
274  maapState = MAAP_STATE_ERROR;
275  }
276  else {
277  AVB_LOGF_WARNING("MAAP Address range %d yielded: 0x%012llx-0x%012llx (Size %d)",
278  mn->id,
279  (unsigned long long) mn->start,
280  (unsigned long long) mn->start + mn->count - 1,
281  mn->count);
282  }
283  } else {
284  AVB_LOGF_ERROR("ID %d is not valid", mn->id);
285  }
286  break;
287  default:
288  AVB_LOGF_ERROR("MAAP notification type %d not recognized", mn->kind);
289  break;
290  }
291 }

295 static void* maapThread(void *arg)
296 {
297  int socketfd;
298  struct addrinfo hints, *ai, *p;
299  int ret;
300 
301  fd_set master, read_fds;
302  int fdmax;
303 
304  char recvbuffer[200];
305  int recvbytes;
306  Maap_Cmd maapcmd;
307 
308  AVB_LOG_DEBUG("MAAP Thread Starting");
309 
310  /* Create a localhost socket. */
311  memset(&hints, 0, sizeof hints);
312  hints.ai_family = AF_UNSPEC;
313  hints.ai_socktype = SOCK_STREAM;
314  hints.ai_flags = 0;
315  if ((ret = getaddrinfo("localhost", maapDaemonPort, &hints, &ai)) != 0) {
316  AVB_LOGF_ERROR("getaddrinfo failure %s", gai_strerror(ret));
317  maapState = MAAP_STATE_ERROR;
318  return NULL;
319  }
320 
321  for(p = ai; p != NULL; p = p->ai_next) {
322  socketfd = socket(p->ai_family, p->ai_socktype, p->ai_protocol);
323  if (socketfd == -1) {
324  continue;
325  }
326  ret = connect(socketfd, p->ai_addr, p->ai_addrlen);
327  if (ret == -1) {
328  close(socketfd);
329  continue;
330  } else {
331  break;
332  }
333  }
334 
335  freeaddrinfo(ai);
336 
337  if (p == NULL) {
338  AVB_LOGF_ERROR("MAAP: Unable to connect to the daemon, error %d (%s)", errno, strerror(errno));
339  maapState = MAAP_STATE_ERROR;
340  return NULL;
341  }
342 
343  if (fcntl(socketfd, F_SETFL, O_NONBLOCK) < 0)
344  {
345  AVB_LOG_ERROR("MAAP: Could not set the socket to non-blocking");
346  close(socketfd);
347  maapState = MAAP_STATE_ERROR;
348  return NULL;
349  }
350 
351  FD_ZERO(&read_fds);
352  FD_ZERO(&master);
353  FD_SET(STDIN_FILENO, &master);
354  FD_SET(socketfd, &master);
355  fdmax = socketfd;
356 
357  // Initialize the MAAP daemon.
358  // We will request a block of addresses when we get a response to the initialization.
359  maapState = MAAP_STATE_INITIALIZING;
360  memset(&maapcmd, 0, sizeof(Maap_Cmd));
361  maapcmd.kind = MAAP_CMD_INIT;
362  maapcmd.start = MAAP_DYNAMIC_POOL_BASE;
363  maapcmd.count = MAAP_DYNAMIC_POOL_SIZE;
364  if (send(socketfd, (char *) &maapcmd, sizeof(Maap_Cmd), 0) < 0)
365  {
366  /* Something went wrong. Abort! */
367  AVB_LOGF_ERROR("MAAP: Error %d writing to network socket (%s)", errno, strerror(errno));
368  maapState = MAAP_STATE_ERROR;
369  return NULL;
370  }
371 
372 
373  /*
374  * Main event loop
375  */
376 
377  while (maapRunning || maapState <= MAAP_STATE_CONNECTED)
378  {
379  if (!maapRunning && maapState == MAAP_STATE_CONNECTED)
380  {
381  // Tell the MAAP daemon to release the addresses.
382  memset(&maapcmd, 0, sizeof(Maap_Cmd));
383  maapcmd.kind = MAAP_CMD_RELEASE;
384  maapcmd.id = maapReservationId;
385  if (send(socketfd, (char *) &maapcmd, sizeof(Maap_Cmd), 0) > 0)
386  {
387  maapState = MAAP_STATE_RELEASING;
388  // Use the wait below to give the daemon time to respond.
389  }
390  else
391  {
392  maapState = MAAP_STATE_NOT_AVAILABLE;
393  break;
394  }
395  }
396 
397  /* Wait for something to happen. */
398  struct timeval tv_timeout = { 1, 0 };
399  read_fds = master;
400  ret = select(fdmax+1, &read_fds, NULL, NULL, &tv_timeout);
401  if (ret < 0)
402  {
403  AVB_LOGF_ERROR("MAAP: select() error %d (%s)", errno, strerror(errno));
404  maapState = MAAP_STATE_ERROR;
405  break;
406  }
407 
408  /* Handle any responses received. */
409  if (FD_ISSET(socketfd, &read_fds))
410  {
411  while ((recvbytes = recv(socketfd, recvbuffer, sizeof(Maap_Notify), 0)) > 0)
412  {
413  recvbuffer[recvbytes] = '\0';
414 
415  /* Process the response data (will be binary). */
416  if (recvbytes == sizeof(Maap_Notify))
417  {
418  process_maap_notify((Maap_Notify *) recvbuffer, socketfd);
419  }
420  else
421  {
422  AVB_LOGF_WARNING("MAAP: Received unexpected response of size %d", recvbytes);
423  }
424  }
425  if (recvbytes == 0)
426  {
427  /* The MAAP daemon closed the connection. Assume it shut down, and we should too. */
428  // AVDECC_TODO: Should we try to reconnect?
429  AVB_LOG_ERROR("MAAP daemon exited.");
430  maapState = MAAP_STATE_ERROR;
431  break;
432  }
433  if (recvbytes < 0 && errno != EWOULDBLOCK)
434  {
435  /* Something went wrong. Abort! */
436  AVB_LOGF_ERROR("MAAP: Error %d reading from network socket (%s)", errno, strerror(errno));
437  maapState = MAAP_STATE_ERROR;
438  break;
439  }
440  }
441  }
442 
443  if (maapState < MAAP_STATE_NOT_AVAILABLE) {
444  maapState = MAAP_STATE_NOT_AVAILABLE;
445  }
446 
447  close(socketfd);
448 
449  AVB_LOG_DEBUG("MAAP Thread Done");
450  return NULL;
451 }

546 void* openavbMaapAllocate(int count, /* out */ struct ether_addr *addr)
547 {
548  AVB_TRACE_ENTRY(AVB_TRACE_MAAP);
549  assert(count == 1);
550 
551  // Wait up to 10 seconds for the daemon to give us a block of addresses.
552  int j;
553  for (j = 0; j < 10 * 1000 / 50 && maapState < MAAP_STATE_CONNECTED; ++j) {
554  AVB_LOG_DEBUG("Waiting for MAAP Daemon status");
555  SLEEP_MSEC(50);
556  }
557 
558  MAAP_LOCK();
559 
560  // Find the next non-allocated address.
561  int i = 0;
562  while (i < MAX_AVB_STREAMS && maapAllocList[i].taken) {
563  i++;
564  }
565 
566  // Allocate an address from the pool.
567  if (i < MAX_AVB_STREAMS) {
568  maapAllocList[i].taken = true;
569  memcpy(addr, maapAllocList[i].destAddr.ether_addr_octet, sizeof(struct ether_addr));
570  AVB_LOGF_INFO("Allocated MAAP address " ETH_FORMAT, ETH_OCTETS(addr->ether_addr_octet));
571 
572  MAAP_UNLOCK();
573 
574  AVB_TRACE_EXIT(AVB_TRACE_MAAP);
575  return &maapAllocList[i];
576  }
577 
578  MAAP_UNLOCK();
579 
580  AVB_LOG_ERROR("All MAAP addresses already allocated");
581  AVB_TRACE_EXIT(AVB_TRACE_MAAP);
582  return NULL;
583 }
