213 void EtherPort::processMessage
214 ( char *buf, int length, LinkLayerAddress *remote, uint32_t link_speed )
215 {
216  GPTP_LOG_VERBOSE("Processing network buffer");
217 
218  PTPMessageCommon *msg =
219  buildPTPMessage( buf, (int)length, remote, this );
220 
221  if (msg == NULL)
222  {
223  GPTP_LOG_ERROR("Discarding invalid message");
224  return;
225  }
226  GPTP_LOG_VERBOSE("Processing message");
227 
228  if( msg->isEvent() )
229  {
230  Timestamp rx_timestamp = msg->getTimestamp();
231  Timestamp phy_compensation = getRxPhyDelay( link_speed );
232  GPTP_LOG_DEBUG( "RX PHY compensation: %s sec",
233  phy_compensation.toString().c_str() );
234  phy_compensation._version = rx_timestamp._version;
235  rx_timestamp = rx_timestamp - phy_compensation;
236  msg->setTimestamp( rx_timestamp );
237  }
238 
239  msg->processMessage(this);
240  if (msg->garbage())
241  delete msg;
242 }

244 void *EtherPort::openPort( EtherPort *port )
245 {
246  port_ready_condition->signal();
247 
248  while (1) {
249  uint8_t buf[128];
250  LinkLayerAddress remote;
251  net_result rrecv;
252  size_t length = sizeof(buf);
253  uint32_t link_speed;
254 
255  if ( ( rrecv = recv( &remote, buf, length, link_speed ))
256  == net_succeed )
257  {
258  processMessage
259  ((char *)buf, (int)length, &remote, link_speed );
260  } else if (rrecv == net_fatal) {
261  GPTP_LOG_ERROR("read from network interface failed");
262  this->processEvent(FAULT_DETECTED);
263  break;
264  }
265  }
266 
267  return NULL;
268 }

322 bool EtherPort::_processEvent( Event e )
323 {
324  bool ret;
325 
326  switch (e) {
327  case POWERUP:
328  case INITIALIZE:
329  if( !getAutomotiveProfile( ))
330  {
331  if ( getPortState() != PTP_SLAVE &&
332  getPortState() != PTP_MASTER )
333  {
334  GPTP_LOG_STATUS("Starting PDelay");
335  startPDelay();
336  }
337  }
338  else {
339  startPDelay();
340  }
341 
342  port_ready_condition->wait_prelock();
343 
344  if( !linkWatch(watchNetLinkWrapper, (void *)this) )
345  {
346  GPTP_LOG_ERROR("Error creating port link thread");
347  ret = false;
348  break;
349  }
350 
351  if( !linkOpen(openPortWrapper, (void *)this) )
352  {
353  GPTP_LOG_ERROR("Error creating port thread");
354  ret = false;
355  break;
356  }
357 
358  port_ready_condition->wait();
359 
360  if( getAutomotiveProfile( ))
361  {
362  setStationState(STATION_STATE_ETHERNET_READY);
363  if (getTestMode())
364  {
365  APMessageTestStatus *testStatusMsg = new APMessageTestStatus(this);
366  if (testStatusMsg) {
367  testStatusMsg->sendPort(this);
368  delete testStatusMsg;
369  }
370  }
371  if (!isGM) {
372  // Send an initial signalling message
373  PTPMessageSignalling *sigMsg = new PTPMessageSignalling(this);
374  if (sigMsg) {
375  sigMsg->setintervals(PTPMessageSignalling::sigMsgInterval_NoSend, getSyncInterval(), PTPMessageSignalling::sigMsgInterval_NoSend);
376  sigMsg->sendPort(this, NULL);
377  delete sigMsg;
378  }
379 
380  startSyncReceiptTimer((unsigned long long)
381  (SYNC_RECEIPT_TIMEOUT_MULTIPLIER *
382  ((double) pow((double)2, getSyncInterval()) *
383  1000000000.0)));
384  }
385  }
386 
387  ret = true;
388  break;
389  case STATE_CHANGE_EVENT:
390  // If the automotive profile is enabled, handle the event by
391  // doing nothing and returning true, preventing the default
392  // action from executing
393  if( getAutomotiveProfile( ))
394  ret = true;
395  else
396  ret = false;
397 
398  break;
399  case LINKUP:
400  haltPdelay(false);
401  startPDelay();
402  if( getAutomotiveProfile( ))
403  {
404  GPTP_LOG_EXCEPTION("LINKUP");
405  }
406  else {
407  GPTP_LOG_STATUS("LINKUP");
408  }
409 
410  if( clock->getPriority1() == 255 || getPortState() == PTP_SLAVE ) {
411  becomeSlave( true );
412  } else if( getPortState() == PTP_MASTER ) {
413  becomeMaster( true );
414  } else {
415  clock->addEventTimerLocked
416  ( this, ANNOUNCE_RECEIPT_TIMEOUT_EXPIRES,
417  (uint64_t)
418  ( ANNOUNCE_RECEIPT_TIMEOUT_MULTIPLIER *
419  pow( 2.0, getAnnounceInterval( )) *
420  1000000000.0 ));
421  }
422 
423  if( getAutomotiveProfile( ))
424  {
425  setAsCapable( true );
426 
427  setStationState(STATION_STATE_ETHERNET_READY);
428  if (getTestMode())
429  {
430  APMessageTestStatus *testStatusMsg = new APMessageTestStatus(this);
431  if (testStatusMsg) {
432  testStatusMsg->sendPort(this);
433  delete testStatusMsg;
434  }
435  }
436 
437  resetInitSyncInterval();
438  setAnnounceInterval( 0 );
439  resetInitPDelayInterval();
440 
441  if (!isGM) {
442  // Send an initial signaling message
443  PTPMessageSignalling *sigMsg = new PTPMessageSignalling(this);
444  if (sigMsg) {
445  sigMsg->setintervals(PTPMessageSignalling::sigMsgInterval_NoSend, getSyncInterval(), PTPMessageSignalling::sigMsgInterval_NoSend);
446  sigMsg->sendPort(this, NULL);
447  delete sigMsg;
448  }
449 
450  startSyncReceiptTimer((unsigned long long)
451  (SYNC_RECEIPT_TIMEOUT_MULTIPLIER *
452  ((double) pow((double)2, getSyncInterval()) *
453  1000000000.0)));
454  }
455 
456  // Reset Sync count and pdelay count
457  setPdelayCount(0);
458  setSyncCount(0);
459 
460  // Start AVB SYNC at 2. It will decrement after each sync. When it reaches 0 the Test Status message
461  // can be sent
462  if (isGM) {
463  avbSyncState = 1;
464  }
465  else {
466  avbSyncState = 2;
467  }
468 
469  if (getTestMode())
470  {
471  linkUpCount++;
472  }
473  }
474  this->timestamper_reset();
475 
476  ret = true;
477  break;
478  case LINKDOWN:
479  stopPDelay();
480  if( getAutomotiveProfile( ))
481  {
482  GPTP_LOG_EXCEPTION("LINK DOWN");
483  }
484  else {
485  setAsCapable(false);
486  GPTP_LOG_STATUS("LINK DOWN");
487  }
488  if (getTestMode())
489  {
490  linkDownCount++;
491  }
492 
493  ret = true;
494  break;
495  case ANNOUNCE_RECEIPT_TIMEOUT_EXPIRES:
496  case SYNC_RECEIPT_TIMEOUT_EXPIRES:
497  if( !getAutomotiveProfile( ))
498  {
499  ret = false;
500  break;
501  }
502 
503  // Automotive Profile specific action
504  if (e == SYNC_RECEIPT_TIMEOUT_EXPIRES) {
505  GPTP_LOG_EXCEPTION("SYNC receipt timeout");
506 
507  startSyncReceiptTimer((unsigned long long)
508  (SYNC_RECEIPT_TIMEOUT_MULTIPLIER *
509  ((double) pow((double)2, getSyncInterval()) *
510  1000000000.0)));
511  }
512  ret = true;
513  break;
514  case PDELAY_INTERVAL_TIMEOUT_EXPIRES:
515  GPTP_LOG_DEBUG("PDELAY_INTERVAL_TIMEOUT_EXPIRES occured");
516  {
517  Timestamp req_timestamp;
518 
519  PTPMessagePathDelayReq *pdelay_req =
520  new PTPMessagePathDelayReq(this);
521  PortIdentity dest_id;
522  getPortIdentity(dest_id);
523  pdelay_req->setPortIdentity(&dest_id);
524 
525  {
526  Timestamp pending =
527  PDELAY_PENDING_TIMESTAMP;
528  pdelay_req->setTimestamp(pending);
529  }
530 
531  if (last_pdelay_req != NULL) {
532  delete last_pdelay_req;
533  }
534  setLastPDelayReq(pdelay_req);
535 
536  getTxLock();
537  pdelay_req->sendPort(this, NULL);
538  GPTP_LOG_DEBUG("*** Sent PDelay Request message");
539  putTxLock();
540 
541  {
542  long long timeout;
543  long long interval;
544 
545  timeout = PDELAY_RESP_RECEIPT_TIMEOUT_MULTIPLIER *
546  ((long long)
547  (pow((double)2,getPDelayInterval())*1000000000.0));
548 
549  timeout = timeout > EVENT_TIMER_GRANULARITY ?
550  timeout : EVENT_TIMER_GRANULARITY;
551  clock->addEventTimerLocked
552  (this, PDELAY_RESP_RECEIPT_TIMEOUT_EXPIRES, timeout );
553  GPTP_LOG_DEBUG("Schedule PDELAY_RESP_RECEIPT_TIMEOUT_EXPIRES, "
554  "PDelay interval %d, timeout %lld",
555  getPDelayInterval(), timeout);
556 
557  interval =
558  ((long long)
559  (pow((double)2,getPDelayInterval())*1000000000.0));
560  interval = interval > EVENT_TIMER_GRANULARITY ?
561  interval : EVENT_TIMER_GRANULARITY;
562  startPDelayIntervalTimer(interval);
563  }
564  }
565  break;
566  case SYNC_INTERVAL_TIMEOUT_EXPIRES:
567  {
568  /* Set offset from master to zero, update device vs
569  system time offset */
570 
571  // Send a sync message and then a followup to broadcast
572  PTPMessageSync *sync = new PTPMessageSync(this);
573  PortIdentity dest_id;
574  bool tx_succeed;
575  getPortIdentity(dest_id);
576  sync->setPortIdentity(&dest_id);
577  getTxLock();
578  tx_succeed = sync->sendPort(this, NULL);
579  GPTP_LOG_DEBUG("Sent SYNC message");
580 
581  if( getAutomotiveProfile() &&
582  getPortState() == PTP_MASTER )
583  {
584  if (avbSyncState > 0) {
585  avbSyncState--;
586  if (avbSyncState == 0) {
587  // Send Avnu Automotive Profile status message
588  setStationState(STATION_STATE_AVB_SYNC);
589  if (getTestMode()) {
590  APMessageTestStatus *testStatusMsg = new APMessageTestStatus(this);
591  if (testStatusMsg) {
592  testStatusMsg->sendPort(this);
593  delete testStatusMsg;
594  }
595  }
596  }
597  }
598  }
599  putTxLock();
600 
601  if ( tx_succeed )
602  {
603  Timestamp sync_timestamp = sync->getTimestamp();
604 
605  GPTP_LOG_VERBOSE("Successful Sync timestamp");
606  GPTP_LOG_VERBOSE("Seconds: %u",
607  sync_timestamp.seconds_ls);
608  GPTP_LOG_VERBOSE("Nanoseconds: %u",
609  sync_timestamp.nanoseconds);
610 
611  PTPMessageFollowUp *follow_up = new PTPMessageFollowUp(this);
612  PortIdentity dest_id;
613  getPortIdentity(dest_id);
614 
615  follow_up->setClockSourceTime(getClock()->getFUPInfo());
616  follow_up->setPortIdentity(&dest_id);
617  follow_up->setSequenceId(sync->getSequenceId());
618  follow_up->setPreciseOriginTimestamp
619  (sync_timestamp);
620  follow_up->sendPort(this, NULL);
621  delete follow_up;
622  } else {
623  GPTP_LOG_ERROR
624  ("*** Unsuccessful Sync timestamp");
625  }
626  delete sync;
627  }
628  break;
629  case FAULT_DETECTED:
630  GPTP_LOG_ERROR("Received FAULT_DETECTED event");
631  if( !getAutomotiveProfile( ))
632  {
633  setAsCapable(false);
634  }
635  break;
636  case PDELAY_DEFERRED_PROCESSING:
637  GPTP_LOG_DEBUG("PDELAY_DEFERRED_PROCESSING occured");
638  pdelay_rx_lock->lock();
639  if (last_pdelay_resp_fwup == NULL) {
640  GPTP_LOG_ERROR("PDelay Response Followup is NULL!");
641  abort();
642  }
643  last_pdelay_resp_fwup->processMessage(this);
644  if (last_pdelay_resp_fwup->garbage()) {
645  delete last_pdelay_resp_fwup;
646  this->setLastPDelayRespFollowUp(NULL);
647  }
648  pdelay_rx_lock->unlock();
649  break;
650  case PDELAY_RESP_RECEIPT_TIMEOUT_EXPIRES:
651  if( !getAutomotiveProfile( ))
652  {
653  GPTP_LOG_EXCEPTION("PDelay Response Receipt Timeout");
654  setAsCapable(false);
655  }
656  setPdelayCount( 0 );
657  break;
658 
659  case PDELAY_RESP_PEER_MISBEHAVING_TIMEOUT_EXPIRES:
660  GPTP_LOG_EXCEPTION("PDelay Resp Peer Misbehaving timeout expired! Restarting PDelay");
661 
662  haltPdelay(false);
663  if( getPortState() != PTP_SLAVE &&
664  getPortState() != PTP_MASTER )
665  {
666  GPTP_LOG_STATUS("Starting PDelay" );
667  startPDelay();
668  }
669  break;
670  case SYNC_RATE_INTERVAL_TIMEOUT_EXPIRED:
671  {
672  GPTP_LOG_INFO("SYNC_RATE_INTERVAL_TIMEOUT_EXPIRED occured");
673 
674  sync_rate_interval_timer_started = false;
675 
676  bool sendSignalMessage = false;
677  if ( getSyncInterval() != operLogSyncInterval )
678  {
679  setSyncInterval( operLogSyncInterval );
680  sendSignalMessage = true;
681  }
682 
683  if( getPDelayInterval() != operLogPdelayReqInterval)
684  {
685  setPDelayInterval( operLogPdelayReqInterval );
686  sendSignalMessage = true;
687  }
688 
689  if (sendSignalMessage) {
690  if (!isGM) {
691  // Send operational signalling message
692  PTPMessageSignalling *sigMsg = new PTPMessageSignalling(this);
693  if (sigMsg) {
694  if( getAutomotiveProfile( ))
695  sigMsg->setintervals(PTPMessageSignalling::sigMsgInterval_NoChange, getSyncInterval(), PTPMessageSignalling::sigMsgInterval_NoChange);
696  else
697  sigMsg->setintervals(getPDelayInterval(), getSyncInterval(), PTPMessageSignalling::sigMsgInterval_NoChange);
698  sigMsg->sendPort(this, NULL);
699  delete sigMsg;
700  }
701 
702  startSyncReceiptTimer((unsigned long long)
703  (SYNC_RECEIPT_TIMEOUT_MULTIPLIER *
704  ((double) pow((double)2, getSyncInterval()) *
705  1000000000.0)));
706  }
707  }
708  }
709 
710  break;
711  default:
712  GPTP_LOG_ERROR
713  ( "Unhandled event type in "
714  "EtherPort::processEvent(), %d", e );
715  ret = false;
716  break;
717  }
718 
719  return ret;
720 }
