259 int process_ctl_msg(char *buf, int buflen, struct sockaddr_in *client)
260 {
261 
262  char respbuf[8];
263  /*
264  * Inbound/output commands from/to a client:
265  *
266  * When sent from a client, indicates an operation on the
267  * internal attribute databases. When sent by the daemon to
268  * a client, indicates an event such as a new attribute arrival,
269  * or a leaveall timer to force clients to re-advertise continued
270  * interest in an attribute.
271  *
272  * BYE Client detaches from daemon
273  *
274  * M+? - JOIN_MT a MAC address or service declaration
275  * M++ JOIN_IN a MAC Address (XXX: MMRP doesn't use 'New' though?)
276  * M-- - LV a MAC address or service declaration
277  * V+? - JOIN_MT a VID (VLAN ID)
278  * V++ - JOIN_IN a VID (VLAN ID)
279  * V-- - LV a VID (VLAN ID)
280  * S+? - JOIN_MT a Stream
281  * S++ - JOIN_IN a Stream
282  * S-- - LV a Stream
283  * I+S Add a stream id to the interesting talker stream id list
284  * I-S Remove a stream id from the interesting talker stream id list
285  * I-A Remove all stream ids from the interesting talker and listener stream id lists
286  *
287  * Outbound messages
288  * ERC - error, unrecognized command
289  * ERP - error, unrecognized parameter
290  * ERI - error, internal
291  * OK+ - success
292  *
293  * Registrar Commands
294  *
295  * M?? - query MMRP Registrar MAC Address database
296  * V?? - query MVRP Registrar VID database
297  * S?? - query MSRP Registrar database
298  *
299  * Registrar Responses (to ?? commands)
300  *
301  * MIN - Registered
302  * MMT - Registered, Empty
303  * MLV - Registered, Leaving
304  * MNE - New attribute notification
305  * MJO - JOIN attribute notification
306  * MLV - LEAVE attribute notification
307  * VIN - Registered
308  * VMT - Registered, Empty
309  * VLV - Registered, Leaving
310  * SIN - Registered
311  * SMT - Registered, Empty
312  * SLV - Registered, Leaving
313  *
314  */
315 
316  memset(respbuf, 0, sizeof(respbuf));
317 
318 #if LOG_CLIENT_RECV
319  if (logging_enable)
320  mrpd_log_printf("CMD:%s from CLNT %d\n", buf, client->sin_port);
321 #endif
322 
323  if (buflen < 3) {
324  printf("buflen = %d!\b", buflen);
325 
326  return -1;
327  }
328 
329  switch (buf[0]) {
330  case 'M':
331  return mmrp_recv_cmd(buf, buflen, client);
332  break;
333  case 'V':
334  return mvrp_recv_cmd(buf, buflen, client);
335  break;
336  case 'S':
337  case 'I':
338  return msrp_recv_cmd(buf, buflen, client);
339  break;
340  case 'B':
341  mmrp_bye(client);
342  mvrp_bye(client);
343  msrp_bye(client);
344  break;
345  default:
346  printf("unrecognized command %s\n", buf);
347  snprintf(respbuf, sizeof(respbuf) - 1, "ERC MRP parse %s", buf);
348  mrpd_send_ctl_msg(client, respbuf, sizeof(respbuf));
349  return -1;
350  break;
351  }
352 
353  return 0;
354 }

356 int recv_ctl_msg()
357 {
358  char *msgbuf;
359  struct sockaddr_in client_addr;
360  struct msghdr msg;
361  struct iovec iov;
362  int bytes = 0;
363 
364  msgbuf = (char *)malloc(MAX_MRPD_CMDSZ);
365  if (NULL == msgbuf)
366  return -1;
367 
368  memset(&msg, 0, sizeof(msg));
369  memset(&client_addr, 0, sizeof(client_addr));
370  memset(msgbuf, 0, MAX_MRPD_CMDSZ);
371 
372  iov.iov_len = MAX_MRPD_CMDSZ;
373  iov.iov_base = msgbuf;
374  msg.msg_name = &client_addr;
375  msg.msg_namelen = sizeof(client_addr);
376  msg.msg_iov = &iov;
377  msg.msg_iovlen = 1;
378 
379  bytes = recvmsg(control_socket, &msg, 0);
380  if (bytes <= 0)
381  goto out;
382 
383  process_ctl_msg(msgbuf, bytes, &client_addr);
384  out:
385  free(msgbuf);
386 
387  return -1;
388 }

431 mrpd_init_protocol_socket(u_int16_t etype, int *sock,
432  unsigned char *multicast_addr)
433 {
434  struct sockaddr_ll addr;
435  struct ifreq if_request;
436  int lsock;
437  int rc;
438  struct packet_mreq multicast_req;
439 
440  if (NULL == sock)
441  return -1;
442  if (NULL == multicast_addr)
443  return -1;
444 
445  memset(&multicast_req, 0, sizeof(multicast_req));
446  *sock = -1;
447 
448  lsock = socket(PF_PACKET, SOCK_RAW, htons(etype));
449  if (lsock < 0)
450  return -1;
451 
452  memset(&if_request, 0, sizeof(if_request));
453 
454  strncpy(if_request.ifr_name, interface, sizeof(if_request.ifr_name) - 1);
455 
456  rc = ioctl(lsock, SIOCGIFHWADDR, &if_request);
457  if (rc < 0) {
458  close(lsock);
459  return -1;
460  }
461 
462  memcpy(STATION_ADDR, if_request.ifr_hwaddr.sa_data,
463  sizeof(STATION_ADDR));
464 
465  memset(&if_request, 0, sizeof(if_request));
466 
467  strncpy(if_request.ifr_name, interface, sizeof(if_request.ifr_name)-1);
468 
469  rc = ioctl(lsock, SIOCGIFINDEX, &if_request);
470  if (rc < 0) {
471  close(lsock);
472  return -1;
473  }
474 
475  memset(&addr, 0, sizeof(addr));
476  addr.sll_ifindex = if_request.ifr_ifindex;
477  addr.sll_family = AF_PACKET;
478  addr.sll_protocol = htons(etype);
479 
480  rc = bind(lsock, (struct sockaddr *)&addr, sizeof(addr));
481  if (0 != rc) {
482 #if LOG_ERRORS
483  fprintf(stderr, "%s - Error on bind %s", __FUNCTION__, strerror(errno));
484 #endif
485  close(lsock);
486  return -1;
487  }
488 
489  rc = setsockopt(lsock, SOL_SOCKET, SO_BINDTODEVICE, interface,
490  strlen(interface));
491  if (0 != rc) {
492  close(lsock);
493  return -1;
494  }
495 
496  multicast_req.mr_ifindex = if_request.ifr_ifindex;
497  multicast_req.mr_type = PACKET_MR_MULTICAST;
498  multicast_req.mr_alen = 6;
499  memcpy(multicast_req.mr_address, multicast_addr, 6);
500 
501  rc = setsockopt(lsock, SOL_PACKET, PACKET_ADD_MEMBERSHIP,
502  &multicast_req, sizeof(multicast_req));
503  if (0 != rc) {
504  close(lsock);
505  return -1;
506  }
507 
508  *sock = lsock;
509 
510  return 0;
511 }

601 void process_events(void)
602 {
603 
604  fd_set fds, sel_fds;
605  int rc;
606  int max_fd;
607 
608  /* wait for events, demux the received packets, process packets */
609 
610  FD_ZERO(&fds);
611  FD_SET(control_socket, &fds);
612 
613  max_fd = control_socket;
614 
615  if (mmrp_enable) {
616  FD_SET(mmrp_socket, &fds);
617  if (mmrp_socket > max_fd)
618  max_fd = mmrp_socket;
619 
620  if (NULL == MMRP_db)
621  return;
622 
623  rc = mrp_register_timers(&(MMRP_db->mrp_db), &fds);
624  if (rc > max_fd)
625  max_fd = rc;
626  }
627  if (mvrp_enable) {
628  FD_SET(mvrp_socket, &fds);
629  if (mvrp_socket > max_fd)
630  max_fd = mvrp_socket;
631 
632  if (NULL == MVRP_db)
633  return;
634  rc = mrp_register_timers(&(MVRP_db->mrp_db), &fds);
635  if (rc > max_fd)
636  max_fd = rc;
637 
638  }
639  if (msrp_enable) {
640  FD_SET(msrp_socket, &fds);
641  if (msrp_socket > max_fd)
642  max_fd = msrp_socket;
643 
644  if (NULL == MSRP_db)
645  return;
646  rc = mrp_register_timers(&(MSRP_db->mrp_db), &fds);
647  if (rc > max_fd)
648  max_fd = rc;
649 
650  }
651 
652  FD_SET(periodic_timer, &fds);
653  if (periodic_timer > max_fd)
654  max_fd = periodic_timer;
655 
656  rc = mrp_periodictimer_fsm(&mrp_periodic_state, MRP_EVENT_BEGIN);
657  if (rc)
658  return;
659 
660  FD_SET(gc_timer, &fds);
661  if (gc_timer > max_fd)
662  max_fd = gc_timer;
663 
664  do {
665 
666  sel_fds = fds;
667  rc = select(max_fd + 1, &sel_fds, NULL, NULL, NULL);
668 
669  if (-1 == rc) {
670 #if LOG_ERRORS
671  fprintf(stderr, "Error on select %s\r\n", strerror(errno));
672 #endif
673  return; /* exit on error */
674  }
675  else {
676  if (FD_ISSET(control_socket, &sel_fds)) {
677 #if LOG_POLL_EVENTS
678  mrpd_log_printf("== EVENT recv_ctl_msg ==\n");
679 #endif
680  recv_ctl_msg();
681  }
682  if (mmrp_enable) {
683  if FD_ISSET
684  (mmrp_socket, &sel_fds) {
685 #if LOG_POLL_EVENTS
686  mrpd_log_printf("== EVENT mmrp_recv_msg ==\n");
687 #endif
688  mmrp_recv_msg();
689  }
690  if FD_ISSET
691  (MMRP_db->mrp_db.lva_timer, &sel_fds) {
692  mrpd_log_timer_event("MMRP",
693  MRP_EVENT_LVATIMER);
694  mmrp_event(MRP_EVENT_LVATIMER, NULL);
695  }
696  if FD_ISSET
697  (MMRP_db->mrp_db.lv_timer, &sel_fds) {
698  mrpd_log_timer_event("MMRP",
699  MRP_EVENT_LVTIMER);
700  mmrp_event(MRP_EVENT_LVTIMER, NULL);
701  }
702  if FD_ISSET
703  (MMRP_db->mrp_db.join_timer, &sel_fds) {
704  mrpd_log_timer_event("MMRP",
