73  printf( "Failed to open gptp handle, %d\n", GetLastError() );
74  }
75 
76  printf( "TSC Frequency: %llu\n", tsc_frequency );
77  while( !exit_flag ) {
78  uint64_t now_tscns, now_8021as;

91  break;

93

95  offset = omsg.read( pipe, 0 );
96 
97  printf( "Master-Local Offset = %lld\n", omsg.getMasterLocalOffset() );
98  printf( "Master-Local Frequency Offset = %Lf\n", omsg.getMasterLocalFreqOffset() );
99  printf( "Local-System Offset = %lld\n", omsg.getLocalSystemOffset() );
100  printf( "Local-System Frequency Offset = %Lf\n", omsg.getLocalSystemFreqOffset() );
101  printf( "Local Time = %llu\n\n", omsg.getLocalTime() );
102 
103  now_tscns = scaleTSCClockToNanoseconds( PLAT_rdtsc(), tsc_frequency );
104  update_tscns = omsg.getLocalTime() + omsg.getLocalSystemOffset();
105  delta_tscns = (unsigned)(now_tscns - update_tscns);
106  printf( "Time now in terms of TSC scaled to nanoseconds time: %llu\n", now_tscns );
107  printf( "TSC delta scaled to ns: %u\n", delta_tscns );
108  ml_ratio = omsg.getMasterLocalFreqOffset();
109  ls_ratio = omsg.getLocalSystemFreqOffset();
110  delta_8021as = (unsigned)(ml_ratio*ls_ratio*delta_tscns);
111  printf( "8021as delta scaled: %u\n", delta_8021as );
112  update_8021as = omsg.getLocalTime() - omsg.getMasterLocalOffset();
113  now_8021as = update_8021as + delta_8021as;
114  printf( "Last update time in terms of 802.1AS time: %llu\n", update_8021as );
115  printf( "Time now in terms of 802.1AS time: %llu\n", now_8021as );
116 
117  exit_flag = true;
118  }
119 
120  printf( "Closing pipe\n" );
121  CloseHandle( pipe );
122 
123  return 0;
