74 openavb_list_node_t openavbListAdd(openavb_list_t list, void *data)
75 {
76  openavb_list_node_t retNode = NULL;
77  if (list) {
78  retNode = calloc(1, sizeof(struct openavb_list_node));
79  if (retNode) {
80  retNode->data = data;
81  retNode->prev = NULL;
82  if (!list->head) {
83  list->head = retNode;
84  list->tail = retNode;
85  }
86  else {
87  retNode->prev = list->tail;
88  list->tail->next = retNode;
89  list->tail = retNode;
90  }
91  }
92  }
93  return retNode;
94 }

114 openavb_list_node_t openavbListDelete(openavb_list_t list, openavb_list_node_t node)
115 {
116  openavb_list_node_t retNode = NULL;
117  if (list && node) {
118  // Update head and tail if needed
119  if (node == list->head) {
120  list->head = node->next;
121  }
122  if (node == list->tail) {
123  list->tail = node->prev;
124  }
125 
126  // Unchain ourselves
127  if (node->prev) {
128  node->prev->next = node->next;
129  }
130  if (node->next) {
131  node->next->prev = node->prev;
132  retNode = node->next;
133  }
134 
135  // Free managed memory
136  if (node->managed && node->data) {
137  free(node->data);
138  node->data = NULL;
139  }
140  free(node);
141  }
142  return retNode;
143 }
