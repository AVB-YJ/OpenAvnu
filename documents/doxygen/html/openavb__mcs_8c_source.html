52 void openavbMcsAdvance(mcs_t *mediaClockSynth)
53 {
54  if (mediaClockSynth->firstTimeSet == FALSE) {
55  CLOCK_GETTIME64(OPENAVB_CLOCK_WALLTIME, &mediaClockSynth->edgeTime);
56  if (mediaClockSynth->edgeTime) {
57  mediaClockSynth->firstTimeSet = TRUE;
58  mediaClockSynth->startTime = mediaClockSynth->edgeTime;
59  }
60  }
61  else
62  {
63  mediaClockSynth->edgeTime += mediaClockSynth->nsPerAdvance;
64  mediaClockSynth->tickCount++;
65  if (mediaClockSynth->tickCount % mediaClockSynth->correctionInterval == 0) {
66  mediaClockSynth->edgeTime += mediaClockSynth->correctionAmount;
67  }
68 #if !IGB_LAUNCHTIME_ENABLED
69  IF_LOG_INTERVAL(8000) {
70  U64 nowNS;
71  CLOCK_GETTIME64(OPENAVB_CLOCK_WALLTIME, &nowNS);
72  S64 fixedRealDelta = mediaClockSynth->edgeTime - nowNS;
73  AVB_LOGF_INFO("Fixed/Real TS Delta: %lld", fixedRealDelta);
74  }
75 #endif
76  }
77 }
