56 bool startAvdecc(const char* ifname, const char **inifiles, int numfiles)
57 {
58  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC);
59  LOG_EAVB_CORE_VERSION();
60 
61  // Ensure that we're running as root
62  // (need to be root to use raw sockets)
63  uid_t euid = geteuid();
64  if (euid != (uid_t)0) {
65  fprintf(stderr, "Error: needs to run as root\n\n");
66  goto error;
67  }
68 
69  // Get the AVDECC configuration
70  memset(&gAvdeccCfg, 0, sizeof(openavb_avdecc_cfg_t));
71  openavbReadAvdeccConfig(DEFAULT_AVDECC_INI_FILE, &gAvdeccCfg);
72 
73  // Determine which interface to use.
74  if (ifname) {
75  strncpy(gAvdeccCfg.ifname, ifname, sizeof(gAvdeccCfg.ifname));
76  } else if (gAvdeccCfg.ifname[0] == '\0') {
77  AVB_LOG_ERROR("No interface specified. Use the -I flag, or add one to " DEFAULT_AVDECC_INI_FILE ".");
78  goto error;
79  }
80 
81  // Read the information from the supplied INI files.
82  openavb_tl_data_cfg_t * prevStream = NULL, * newStream;
83  U32 i1;
84  for (i1 = 0; i1 < numfiles; i1++) {
85 
86  char iniFile[1024];
87  snprintf(iniFile, sizeof(iniFile), "%s", inifiles[i1]);
88  // Create a new item with this INI information.
89  newStream = malloc(sizeof(openavb_tl_data_cfg_t));
90  if (!newStream) {
91  AVB_LOG_ERROR("Out of memory");
92  goto error;
93  }
94  memset(newStream, 0, sizeof(openavb_tl_data_cfg_t));
95  if (!openavbReadTlDataIniFile(iniFile, newStream)) {
96  AVB_LOGF_ERROR("Error reading ini file: %s", inifiles[i1]);
97  goto error;
98  }
99  // Append this item to the list of items.
100  if (!prevStream) {
101  // First item.
102  streamList = newStream;
103  } else {
104  // Subsequent item.
105  prevStream->next = newStream;
106  }
107  prevStream = newStream;
108  }
109 
110  /* Run AVDECC in its own thread. */
111  avdeccRunning = TRUE;
112  avdeccInitSucceeded = FALSE;
113  int err = pthread_create(&avdeccServerHandle, NULL, avdeccServerThread, NULL);
114  if (err) {
115  AVB_LOGF_ERROR("Failed to start AVDECC thread: %s", strerror(err));
116  goto error;
117  }
118 
119  /* Wait a while to see if the thread was able to start AVDECC. */
120  int i;
121  for (i = 0; avdeccRunning && !avdeccInitSucceeded && i < 5000; ++i) {
122  SLEEP_MSEC(1);
123  }
124  if (!avdeccInitSucceeded) {
125  goto error;
126  }
127 
128  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
129  return true;
130 
131 error:
132  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
133  return false;
134 }

155 static void* avdeccServerThread(void *arg)
156 {
157  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC);
158 
159  while (avdeccRunning) {
160  if (!openavbAvdeccInitialize()) {
161  AVB_LOG_ERROR("Failed to initialize AVDECC");
162  break;
163  }
164 
165  AVB_LOG_DEBUG("AVDECC Initialized");
166 
167  if (!openavbAvdeccStart()) {
168  AVB_LOG_ERROR("Failed to start AVDECC");
169  openavbAvdeccStop();
170  break;
171  }
172 
173  if (!openavbAvdeccMsgServerOpen()) {
174  AVB_LOG_ERROR("Failed to start AVDECC Server");
175  openavbAvdeccStop();
176  break;
177  }
178  else {
179  AVB_LOG_INFO("AVDECC Started");
180  avdeccInitSucceeded = TRUE;
181 
182  // Wait until AVDECC is finished.
183  while (avdeccRunning) {
184  openavbAvdeccMsgSrvrService();
185  }
186 
187  openavbAvdeccMsgServerClose();
188  }
189 
190  // Stop AVDECC
191  AVB_LOG_DEBUG("AVDECC Stopping");
192  openavbAvdeccStop();
193  AVB_LOG_INFO("AVDECC Stopped");
194  }
195 
196  // Shutdown AVDECC
197  openavbAvdeccCleanup();
198 
199  avdeccRunning = FALSE;
200 
201  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
202  return NULL;
203 }
