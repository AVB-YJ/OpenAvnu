133 void CommonPort::recommendState
134 ( PortState state, bool changed_external_master )
135 {
136  bool reset_sync = false;
137  switch (state) {
138  case PTP_MASTER:
139  if ( getPortState() != PTP_MASTER )
140  {
141  setPortState( PTP_MASTER );
142  // Start announce receipt timeout timer
143  // Start sync receipt timeout timer
144  becomeMaster( true );
145  reset_sync = true;
146  }
147  break;
148  case PTP_SLAVE:
149  if ( getPortState() != PTP_SLAVE )
150  {
151  becomeSlave( true );
152  reset_sync = true;
153  } else {
154  if( changed_external_master ) {
155  GPTP_LOG_STATUS("Changed master!" );
156  clock->newSyntonizationSetPoint();
157  clock->updateFUPInfo();
158  reset_sync = true;
159  }
160  }
161  break;
162  default:
163  GPTP_LOG_ERROR
164  ("Invalid state change requested by call to "
165  "1588Port::recommendState()");
166  break;
167  }
168  if( reset_sync ) sync_count = 0;
169  return;
170 }

172 bool CommonPort::serializeState( void *buf, off_t *count )
173 {
174  bool ret = true;
175 
176  if( buf == NULL ) {
177  *count = sizeof(port_state)+sizeof(_peer_rate_offset)+
178  sizeof(asCapable)+sizeof(one_way_delay);
179  return true;
180  }
181 
182  if( port_state != PTP_MASTER && port_state != PTP_SLAVE ) {
183  *count = 0;
184  ret = false;
185  goto bail;
186  }
187 
188  /* asCapable */
189  if( ret && *count >= (off_t) sizeof( asCapable )) {
190  memcpy( buf, &asCapable, sizeof( asCapable ));
191  *count -= sizeof( asCapable );
192  buf = ((char *)buf) + sizeof( asCapable );
193  } else if( ret == false ) {
194  *count += sizeof( asCapable );
195  } else {
196  *count = sizeof( asCapable )-*count;
197  ret = false;
198  }
199 
200  /* Port State */
201  if( ret && *count >= (off_t) sizeof( port_state )) {
202  memcpy( buf, &port_state, sizeof( port_state ));
203  *count -= sizeof( port_state );
204  buf = ((char *)buf) + sizeof( port_state );
205  } else if( ret == false ) {
206  *count += sizeof( port_state );
207  } else {
208  *count = sizeof( port_state )-*count;
209  ret = false;
210  }
211 
212  /* Link Delay */
213  if( ret && *count >= (off_t) sizeof( one_way_delay )) {
214  memcpy( buf, &one_way_delay, sizeof( one_way_delay ));
215  *count -= sizeof( one_way_delay );
216  buf = ((char *)buf) + sizeof( one_way_delay );
217  } else if( ret == false ) {
218  *count += sizeof( one_way_delay );
219  } else {
220  *count = sizeof( one_way_delay )-*count;
221  ret = false;
222  }
223 
224  /* Neighbor Rate Ratio */
225  if( ret && *count >= (off_t) sizeof( _peer_rate_offset )) {
226  memcpy( buf, &_peer_rate_offset, sizeof( _peer_rate_offset ));
227  *count -= sizeof( _peer_rate_offset );
228  buf = ((char *)buf) + sizeof( _peer_rate_offset );
229  } else if( ret == false ) {
230  *count += sizeof( _peer_rate_offset );
231  } else {
232  *count = sizeof( _peer_rate_offset )-*count;
233  ret = false;
234  }
235 
236  bail:
237  return ret;
238 }

338 bool CommonPort::processStateChange( Event e )
339 {
340  bool changed_external_master;
341  uint8_t LastEBestClockIdentity[PTP_CLOCK_IDENTITY_LENGTH];
342  int number_ports, j;
343  PTPMessageAnnounce *EBest = NULL;
344  char EBestClockIdentity[PTP_CLOCK_IDENTITY_LENGTH];
345  CommonPort **ports;
346 
347  // Nothing to do if we are slave only
348  if ( clock->getPriority1() == 255 )
349  return true;
350 
351  clock->getPortList(number_ports, ports);
352 
353  /* Find EBest for all ports */
354  j = 0;
355  for (int i = 0; i < number_ports; ++i) {
356  while (ports[j] == NULL)
357  ++j;
358  if ( ports[j]->getPortState() == PTP_DISABLED ||
359  ports[j]->getPortState() == PTP_FAULTY )
360  {
361  continue;
362  }
363  if( EBest == NULL )
364  {
365  EBest = ports[j]->calculateERBest();
366  }
367  else if( ports[j]->calculateERBest() )
368  {
369  if( ports[j]->
370  calculateERBest()->isBetterThan(EBest))
371  {
372  EBest = ports[j]->calculateERBest();
373  }
374  }
375  }
376 
377  if (EBest == NULL)
378  {
379  return true;
380  }
381 
382  /* Check if we've changed */
383  clock->getLastEBestIdentity().
384  getIdentityString( LastEBestClockIdentity );
385  EBest->getGrandmasterIdentity( EBestClockIdentity );
386  if( memcmp( EBestClockIdentity, LastEBestClockIdentity,
387  PTP_CLOCK_IDENTITY_LENGTH ) != 0 )
388  {
389  ClockIdentity newGM;
390  changed_external_master = true;
391  newGM.set((uint8_t *) EBestClockIdentity );
392  clock->setLastEBestIdentity( newGM );
393  }
394  else
395  {
396  changed_external_master = false;
397  }
398 
399  if( clock->isBetterThan( EBest ))
400  {
401  // We're Grandmaster, set grandmaster info to me
402  ClockIdentity clock_identity;
403  unsigned char priority1;
404  unsigned char priority2;
405  ClockQuality clock_quality;
406 
407  clock_identity = getClock()->getClockIdentity();
408  getClock()->setGrandmasterClockIdentity( clock_identity );
409  priority1 = getClock()->getPriority1();
410  getClock()->setGrandmasterPriority1( priority1 );
411  priority2 = getClock()->getPriority2();
412  getClock()->setGrandmasterPriority2( priority2 );
413  clock_quality = getClock()->getClockQuality();
414  getClock()->setGrandmasterClockQuality( clock_quality );
415  }
416 
417  j = 0;
418  for( int i = 0; i < number_ports; ++i )
419  {
420  while (ports[j] == NULL)
421  ++j;
422  if ( ports[j]->getPortState() ==
423  PTP_DISABLED ||
424  ports[j]->getPortState() ==
425  PTP_FAULTY)
426  {
427  continue;
428  }
429  if (clock->isBetterThan(EBest))
430  {
431  // We are the GrandMaster, all ports are master
432  EBest = NULL; // EBest == NULL : we were grandmaster
433  ports[j]->recommendState( PTP_MASTER,
434  changed_external_master );
435  } else {
436  if( EBest == ports[j]->calculateERBest() ) {
437  // The "best" Announce was received on this
438  // port
439  ClockIdentity clock_identity;
440  unsigned char priority1;
441  unsigned char priority2;
442  ClockQuality *clock_quality;
443 
444  ports[j]->recommendState
445  ( PTP_SLAVE, changed_external_master );
446  clock_identity =
447  EBest->getGrandmasterClockIdentity();
448  getClock()->setGrandmasterClockIdentity
449  ( clock_identity );
450  priority1 = EBest->getGrandmasterPriority1();
451  getClock()->setGrandmasterPriority1
452  ( priority1 );
453  priority2 =
454  EBest->getGrandmasterPriority2();
455  getClock()->setGrandmasterPriority2
456  ( priority2 );
457  clock_quality =
458  EBest->getGrandmasterClockQuality();
459  getClock()->setGrandmasterClockQuality
460  (*clock_quality);
461  } else {
462  /* Otherwise we are the master because we have
463  sync'd to a better clock */
464  ports[j]->recommendState
465  (PTP_MASTER, changed_external_master);
466  }
467  }
468  }
469 
470  return true;
471 }

474 bool CommonPort::processSyncAnnounceTimeout( Event e )
475 {
476  // We're Grandmaster, set grandmaster info to me
477  ClockIdentity clock_identity;
478  unsigned char priority1;
479  unsigned char priority2;
480  ClockQuality clock_quality;
481 
482  Timestamp system_time;
483  Timestamp device_time;
484  uint32_t local_clock, nominal_clock_rate;
485 
486  // Nothing to do
487  if( clock->getPriority1() == 255 )
488  return true;
489 
490  // Restart timer
491  if( e == ANNOUNCE_RECEIPT_TIMEOUT_EXPIRES ) {
492  clock->addEventTimerLocked
493  (this, ANNOUNCE_RECEIPT_TIMEOUT_EXPIRES,
494  (ANNOUNCE_RECEIPT_TIMEOUT_MULTIPLIER*
495  (unsigned long long)
496  (pow((double)2,getAnnounceInterval())*
497  1000000000.0)));
498  } else {
499  startSyncReceiptTimer
500  ((unsigned long long)
501  (SYNC_RECEIPT_TIMEOUT_MULTIPLIER *
502  ((double) pow((double)2, getSyncInterval()) *
503  1000000000.0)));
504  }
505 
506  if( getPortState() == PTP_MASTER )
507  return true;
508 
509  GPTP_LOG_STATUS(
510  "*** %s Timeout Expired - Becoming Master",
511  e == ANNOUNCE_RECEIPT_TIMEOUT_EXPIRES ? "Announce" :
512  "Sync" );
513 
514  clock_identity = getClock()->getClockIdentity();
515  getClock()->setGrandmasterClockIdentity( clock_identity );
516  priority1 = getClock()->getPriority1();
517  getClock()->setGrandmasterPriority1( priority1 );
518  priority2 = getClock()->getPriority2();
519  getClock()->setGrandmasterPriority2( priority2 );
520  clock_quality = getClock()->getClockQuality();
521  getClock()->setGrandmasterClockQuality( clock_quality );
522 
523  setPortState( PTP_MASTER );
524 
525  getDeviceTime( system_time, device_time,
526  local_clock, nominal_clock_rate );
527 
528  (void) clock->calcLocalSystemClockRateDifference
529  ( device_time, system_time );
530 
531  setQualifiedAnnounce( NULL );
532 
533  clock->addEventTimerLocked
534  ( this, SYNC_INTERVAL_TIMEOUT_EXPIRES,
535  16000000 );
536 
537  startAnnounce();
538 
539  return true;
540 }

542 bool CommonPort::processEvent( Event e )
543 {
544  bool ret;
545 
546  switch( e )
547  {
548  default:
549  // Unhandled event
550  ret = _processEvent( e );
551  break;
552 
553  case POWERUP:
554  case INITIALIZE:
555  GPTP_LOG_DEBUG("Received POWERUP/INITIALIZE event");
556 
557  // If port has been configured as master or slave, run media
558  // specific configuration. If it hasn't been configured
559  // start listening for announce messages
560  if( clock->getPriority1() == 255 ||
561  port_state == PTP_SLAVE )
562  {
563  becomeSlave( true );
564  }
565  else if( port_state == PTP_MASTER )
566  {
567  becomeMaster( true );
568  }
569  else
570  {
571  clock->addEventTimerLocked(this, ANNOUNCE_RECEIPT_TIMEOUT_EXPIRES,
572  (uint64_t) ( ANNOUNCE_RECEIPT_TIMEOUT_MULTIPLIER * pow(2.0, getAnnounceInterval()) * 1000000000.0 ));
573  }
574 
575  // Do any media specific initialization
576  ret = _processEvent( e );
577  break;
578 
579  case STATE_CHANGE_EVENT:
580  ret = _processEvent( e );
581 
582  // If this event wasn't handled in a media specific way, call
583  // the default action
584  if( !ret )
585  ret = processStateChange( e );
586  break;
587 
588  case ANNOUNCE_RECEIPT_TIMEOUT_EXPIRES:
589  case SYNC_RECEIPT_TIMEOUT_EXPIRES:
590  if (e == ANNOUNCE_RECEIPT_TIMEOUT_EXPIRES) {
591  incCounter_ieee8021AsPortStatAnnounceReceiptTimeouts();
592  }
593  else if (e == SYNC_RECEIPT_TIMEOUT_EXPIRES) {
594  incCounter_ieee8021AsPortStatRxSyncReceiptTimeouts();
595  }
596 
597  ret = _processEvent( e );
598 
599  // If this event wasn't handled in a media specific way, call
600  // the default action
601  if( !ret )
602  ret = processSyncAnnounceTimeout( e );
603  break;
604 
605  case ANNOUNCE_INTERVAL_TIMEOUT_EXPIRES:
606  GPTP_LOG_DEBUG("ANNOUNCE_INTERVAL_TIMEOUT_EXPIRES occured");
607 
608  // Send an announce message
609  if ( asCapable)
610  {
611  PTPMessageAnnounce *annc =
612  new PTPMessageAnnounce(this);
613  PortIdentity dest_id;
614  PortIdentity gmId;
615  ClockIdentity clock_id = clock->getClockIdentity();
616  gmId.setClockIdentity(clock_id);
617  getPortIdentity( dest_id );
618  annc->setPortIdentity( &dest_id );
619  annc->sendPort( this, NULL );
620  delete annc;
621  }
622 
623  startAnnounceIntervalTimer
624  ((uint64_t)( pow((double)2, getAnnounceInterval()) *
625  1000000000.0 ));
626  ret = true;
627  break;
628 
629  case SYNC_INTERVAL_TIMEOUT_EXPIRES:
630  GPTP_LOG_DEBUG("SYNC_INTERVAL_TIMEOUT_EXPIRES occured");
631  // If asCapable is true attempt some media specific action
632  ret = true;
633  if( asCapable )
634  ret = _processEvent( e );
635 
636  /* Do getDeviceTime() after transmitting sync frame
637  causing an update to local/system timestamp */
638  {
639  Timestamp system_time;
640  Timestamp device_time;
641  uint32_t local_clock, nominal_clock_rate;
642  FrequencyRatio local_system_freq_offset;
643  int64_t local_system_offset;
644 
645  getDeviceTime
646  ( system_time, device_time,
647  local_clock, nominal_clock_rate );
648 
649  GPTP_LOG_VERBOSE
650  ( "port::processEvent(): System time: %u,%u "
651  "Device Time: %u,%u",
652  system_time.seconds_ls,
653  system_time.nanoseconds,
654  device_time.seconds_ls,
655  device_time.nanoseconds );
656 
657  local_system_offset =
658  TIMESTAMP_TO_NS(system_time) -
659  TIMESTAMP_TO_NS(device_time);
660  local_system_freq_offset =
661  clock->calcLocalSystemClockRateDifference
662  ( device_time, system_time );
663  clock->setMasterOffset
664  ( this, 0, device_time, 1.0,
665  local_system_offset, system_time,
666  local_system_freq_offset, getSyncCount(),
667  pdelay_count, port_state, asCapable );
668  }
669 
670  // Call media specific action for completed sync
671  syncDone();
672 
673  // Restart the timer
674  startSyncIntervalTimer
675  ((uint64_t)( pow((double)2, getSyncInterval()) *
676  1000000000.0 ));
677 
678  break;
679  }
680 
681  return ret;
682 }
