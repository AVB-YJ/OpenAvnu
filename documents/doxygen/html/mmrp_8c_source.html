85 int mmrp_add(struct mmrp_attribute *rattrib)
86 {
87  struct mmrp_attribute *attrib;
88  struct mmrp_attribute *attrib_tail;
89  int mac_eq;
90 
91  /* XXX do a lookup first to guarantee uniqueness? */
92 
93  attrib_tail = attrib = MMRP_db->attrib_list;
94 
95  while (NULL != attrib) {
96  /* sort list into types, then sorted in order within types */
97  if (rattrib->type == attrib->type) {
98  if (MMRP_SVCREQ_TYPE == attrib->type) {
99  if (attrib->attribute.svcreq <
100  rattrib->attribute.svcreq) {
101  /* possible tail insertion ... */
102 
103  if (NULL != attrib->next) {
104  if (MMRP_SVCREQ_TYPE ==
105  attrib->next->type) {
106  attrib = attrib->next;
107  continue;
108  }
109  }
110 
111  rattrib->next = attrib->next;
112  rattrib->prev = attrib;
113  attrib->next = rattrib;
114  if (rattrib->next)
115  rattrib->next->prev = rattrib;
116  return 0;
117 
118  } else {
119  /* head insertion ... */
120  rattrib->next = attrib;
121  rattrib->prev = attrib->prev;
122  attrib->prev = rattrib;
123  if (NULL != rattrib->prev)
124  rattrib->prev->next = rattrib;
125  else
126  MMRP_db->attrib_list = rattrib;
127 
128  return 0;
129  }
130  } else {
131  mac_eq = memcmp(attrib->attribute.macaddr,
132  rattrib->attribute.macaddr, 6);
133 
134  if (mac_eq < 0) {
135  /* possible tail insertion ... */
136 
137  if (NULL != attrib->next) {
138  if (MMRP_MACVEC_TYPE ==
139  attrib->next->type) {
140  attrib = attrib->next;
141  continue;
142  }
143  }
144 
145  rattrib->next = attrib->next;
146  rattrib->prev = attrib;
147  attrib->next = rattrib;
148  if (rattrib->next)
149  rattrib->next->prev = rattrib;
150  return 0;
151 
152  } else {
153  /* head insertion ... */
154  rattrib->next = attrib;
155  rattrib->prev = attrib->prev;
156  attrib->prev = rattrib;
157  if (NULL != rattrib->prev)
158  rattrib->prev->next = rattrib;
159  else
160  MMRP_db->attrib_list = rattrib;
161 
162  return 0;
163  }
164  }
165  }
166  attrib_tail = attrib;
167  attrib = attrib->next;
168  }
169 
170  /* if we are here we didn't need to stitch in a a sorted entry
171  * so append it onto the tail (if it exists)
172  */
173 
174  if (NULL == attrib_tail) {
175  rattrib->next = NULL;
176  rattrib->prev = NULL;
177  MMRP_db->attrib_list = rattrib;
178  } else {
179  rattrib->next = NULL;
180  rattrib->prev = attrib_tail;
181  attrib_tail->next = rattrib;
182  }
183 
184  return 0;
185 }

201 int mmrp_event(int event, struct mmrp_attribute *rattrib)
202 {
203  struct mmrp_attribute *attrib;
204  int count = 0;
205 
206  switch (event) {
207  case MRP_EVENT_LVATIMER:
208  mrp_lvatimer_stop(&(MMRP_db->mrp_db));
209  mrp_jointimer_stop(&(MMRP_db->mrp_db));
210  /* update state */
211  attrib = MMRP_db->attrib_list;
212 
213  while (NULL != attrib) {
214 #if LOG_MMRP
215  mrpd_log_printf("MMRP -> mrp_applicant_fsm\n");
216 #endif
217  mrp_applicant_fsm(&(MMRP_db->mrp_db),
218  &(attrib->applicant), MRP_EVENT_TXLA,
219  mrp_registrar_in(&(attrib->registrar)));
220  mrp_registrar_fsm(&(attrib->registrar),
221  &(MMRP_db->mrp_db), MRP_EVENT_TXLA);
222  attrib = attrib->next;
223  }
224 
225  mrp_lvatimer_fsm(&(MMRP_db->mrp_db), MRP_EVENT_LVATIMER);
226 
227  MMRP_db->send_empty_LeaveAll_flag = 1;
228  mrp_lvatimer_fsm(&(MMRP_db->mrp_db), MRP_EVENT_TX);
229  mmrp_txpdu();
230  MMRP_db->send_empty_LeaveAll_flag = 0;
231  break;
232  case MRP_EVENT_RLA:
233  mrp_jointimer_start(&(MMRP_db->mrp_db));
234  if (NULL == rattrib)
235  return -1; /* XXX internal fault */
236 
237  /* update state */
238  attrib = MMRP_db->attrib_list;
239 
240  while (NULL != attrib) {
241  if (attrib->type == rattrib->type) {
242 #if LOG_MMRP
243  mrpd_log_printf("MMRP -> mrp_applicant_fsm\n");
244 #endif
245  mrp_applicant_fsm(&(MMRP_db->mrp_db),
246  &(attrib->applicant), MRP_EVENT_RLA,
247  mrp_registrar_in(&(attrib->registrar)));
248  mrp_registrar_fsm(&(attrib->registrar),
249  &(MMRP_db->mrp_db), MRP_EVENT_RLA);
250  }
251  attrib = attrib->next;
252  }
253 
254  mrp_lvatimer_fsm(&(MMRP_db->mrp_db), MRP_EVENT_RLA);
255 
256  break;
257  case MRP_EVENT_TX:
258  mrp_jointimer_stop(&(MMRP_db->mrp_db));
259  attrib = MMRP_db->attrib_list;
260 
261  while (NULL != attrib) {
262 #if LOG_MMRP
263  mrpd_log_printf("MMRP -> mrp_applicant_fsm\n");
264 #endif
265  mrp_applicant_fsm(&(MMRP_db->mrp_db),
266  &(attrib->applicant), MRP_EVENT_TX,
267  mrp_registrar_in(&(attrib->registrar)));
268  count += mrp_applicant_state_transition_implies_tx(&(attrib->applicant));
269  attrib = attrib->next;
270  }
271 
272  mmrp_txpdu();
273 
274  /*
275  * Certain state transitions imply we need to request another tx
276  * opportunity.
277  */
278  if (count) {
279  mrp_jointimer_start(&(MMRP_db->mrp_db));
280  }
281 
282  break;
283  case MRP_EVENT_LVTIMER:
284  mrp_lvtimer_stop(&(MMRP_db->mrp_db));
285  attrib = MMRP_db->attrib_list;
286 
287  while (NULL != attrib) {
288  mrp_registrar_fsm(&(attrib->registrar),
289  &(MMRP_db->mrp_db),
290  MRP_EVENT_LVTIMER);
291 
292  attrib = attrib->next;
293  }
294  break;
295  case MRP_EVENT_PERIODIC:
296  attrib = MMRP_db->attrib_list;
297 
298  if (NULL != attrib) {
299  mrp_jointimer_start(&(MMRP_db->mrp_db));
300  }
301 
302  while (NULL != attrib) {
303  mrp_applicant_fsm(&(MMRP_db->mrp_db),
304  &(attrib->applicant),
305  MRP_EVENT_PERIODIC,
306  mrp_registrar_in(&(attrib->registrar)));
307 #if LOG_MMRP
308  mmrp_print_debug_info(event, attrib);
309 #endif
310  attrib = attrib->next;
311  }
312  break;
313  case MRP_EVENT_NEW:
314  case MRP_EVENT_JOIN:
315  case MRP_EVENT_RNEW:
316  case MRP_EVENT_RJOININ:
317  case MRP_EVENT_RJOINMT:
318  case MRP_EVENT_LV:
319  case MRP_EVENT_RIN:
320  case MRP_EVENT_RMT:
321  case MRP_EVENT_RLV:
322  mrp_jointimer_start(&(MMRP_db->mrp_db));
323  if (NULL == rattrib)
324  return -1; /* XXX internal fault */
325 
326  /* update state */
327  attrib = mmrp_lookup(rattrib);
328 
329  if (NULL == attrib) {
330  /* ignore rMT! if attribute does not already exist */
331  if (MRP_EVENT_RMT == event) {
332  free(rattrib);
333  return 0;
334  }
335  mmrp_add(rattrib);
336  attrib = rattrib;
337  } else {
338  mmrp_merge(rattrib);
339  free(rattrib);
340  }
341 
342 #if LOG_MMRP
343  mrpd_log_printf("MMRP -> mrp_applicant_fsm\n");
344 #endif
345  mrp_applicant_fsm(&(MMRP_db->mrp_db), &(attrib->applicant),
346  event,
347  mrp_registrar_in(&(attrib->registrar)));
348  /* remap local requests into registrar events */
349  switch (event) {
350  case MRP_EVENT_NEW:
351  mrp_registrar_fsm(&(attrib->registrar),
352  &(MMRP_db->mrp_db), MRP_EVENT_BEGIN);
353  attrib->registrar.notify = MRP_NOTIFY_NEW;
354  break;
355  case MRP_EVENT_JOIN:
356  if (MRP_IN_STATE == attrib->registrar.mrp_state)
357  mrp_registrar_fsm(&(attrib->registrar),
358  &(MMRP_db->mrp_db),
359  MRP_EVENT_RJOININ);
360  else
361  mrp_registrar_fsm(&(attrib->registrar),
362  &(MMRP_db->mrp_db),
363  MRP_EVENT_RJOINMT);
364  break;
365  case MRP_EVENT_LV:
366  mrp_registrar_fsm(&(attrib->registrar),
367  &(MMRP_db->mrp_db), MRP_EVENT_RLV);
368  break;
369  default:
370  mrp_registrar_fsm(&(attrib->registrar),
371  &(MMRP_db->mrp_db), event);
372  break;
373  }
374  break;
375  default:
376  break;
377  }
378 
379  /*
380  * XXX should honor the MMRP_db->mrp_db.registration and
381  * MMRP_db->mrp_db.participant controls
382  */
383 
384  /* generate local notifications */
385  attrib = MMRP_db->attrib_list;
386 
387  while (NULL != attrib) {
388  if (MRP_NOTIFY_NONE != attrib->registrar.notify) {
389  mmrp_send_notifications(attrib,
390  attrib->registrar.notify);
391  attrib->registrar.notify = MRP_NOTIFY_NONE;
392  }
393  attrib = attrib->next;
394  }
395 
396  return 0;
397 }

442 int mmrp_recv_msg()
443 {
444  char *msgbuf;
445  int bytes = 0;
446  eth_hdr_t *eth;
447  mrpdu_t *mrpdu;
448  mrpdu_message_t *mrpdu_msg;
449  unsigned char *mrpdu_msg_ptr;
450  unsigned char *mrpdu_msg_eof;
451  mrpdu_vectorattrib_t *mrpdu_vectorptr;
452  uint16_t numvalues;
453  uint16_t numvalues_processed;
454  int numvectorbytes;
455  uint8_t vect_3pack;
456  int vectidx;
457  int vectevt[3];
458  int vectevt_idx;
459  uint8_t svcreq_firstval;
460  uint8_t macvec_firstval[6];
461  struct mmrp_attribute *attrib;
462  int endmarks;
463  int saw_lva = 0;
464 
465  bytes = mrpd_recvmsgbuf(mmrp_socket, &msgbuf);
466  if (bytes <= 0)
467  goto out;
468 
469  if ((unsigned int)bytes < (sizeof(eth_hdr_t) + sizeof(mrpdu_t) +
470  sizeof(mrpdu_message_t)))
471  goto out;
472 
473  eth = (eth_hdr_t *) msgbuf;
474 
475  /* note that MMRP frames should always arrive untagged (no vlan) */
476  if (MMRP_ETYPE != ntohs(eth->typelen))
477  goto out;
478 
479  /* check dest mac address too */
480  if (memcmp(eth->destaddr, MMRP_ADDR, sizeof(eth->destaddr)))
481  goto out;
482 
483  mrpdu = (mrpdu_t *) (msgbuf + sizeof(struct eth_hdr));
484 
485  /*
486  if (MMRP_PROT_VER != mrpdu->ProtocolVersion)
487  goto out;
488  */
489 
490  mrpdu_msg_ptr = MRPD_GET_MRPDU_MESSAGE_LIST(mrpdu);
491 
492  mrpdu_msg_eof = (unsigned char *)mrpdu_msg_ptr;
493  mrpdu_msg_eof += bytes;
494  mrpdu_msg_eof -= sizeof(eth_hdr_t);
495  mrpdu_msg_eof -= MRPD_OFFSETOF_MRPD_GET_MRPDU_MESSAGE_LIST;
496 
497  endmarks = 0;
498 
499  while (mrpdu_msg_ptr < (mrpdu_msg_eof - MRPDU_ENDMARK_SZ)) {
500  mrpdu_msg = (mrpdu_message_t *) mrpdu_msg_ptr;
501  if ((mrpdu_msg->AttributeType == 0) &&
502  (mrpdu_msg->AttributeLength == 0)) {
503  mrpdu_msg_ptr += 2;
504  endmarks++;
505  if (endmarks < 2)
506  continue; /* end-mark */
507  else
508  break; /* two endmarks - end of messages */
509  }
510 
511  endmarks = 0; /* reset the endmark counter */
512 
513  switch (mrpdu_msg->AttributeType) {
514  case MMRP_SVCREQ_TYPE:
515  if (mrpdu_msg->AttributeLength != 1) {
516  /* we could seek for an endmark to recover
517  */
518  goto out;
519  }
520  /* AttributeListLength not used for MMRP, hence
521  * Data points to the beginning of the VectorAttributes
522  */
523  mrpdu_vectorptr =
524  (mrpdu_vectorattrib_t *) mrpdu_msg->Data;
525  mrpdu_msg_ptr = (uint8_t *) mrpdu_vectorptr;
526 
527  while (!((mrpdu_msg_ptr[0] == 0)
528  && (mrpdu_msg_ptr[1] == 0))) {
529  numvalues =
530  MRPDU_VECT_NUMVALUES(ntohs
531  (mrpdu_vectorptr->
532  VectorHeader));
533 
534  if (MRPDU_VECT_LVA(ntohs(mrpdu_vectorptr->VectorHeader)) && !saw_lva) {
535  saw_lva = 1;
536  attrib = mmrp_alloc();
537  if (NULL == attrib)
538  goto out; /* oops - internal error */
539 
540  attrib->type = MMRP_SVCREQ_TYPE;
541  mmrp_event(MRP_EVENT_RLA, attrib);
542  free(attrib);
543  }
544 
545  if (0 == numvalues)
546  /* Malformed - cant tell how long the trailing vectors are */
547  goto out;
548 
549  if ((mrpdu_vectorptr->FirstValue_VectorEvents +
550  numvalues / 3) >= mrpdu_msg_eof)
551  /* Malformed - runs off the end of the pdu */
552  goto out;
553 
554  svcreq_firstval =
555  mrpdu_vectorptr->FirstValue_VectorEvents[0];
556 
557  /* if not an even multiple ... */
558  if (numvalues != ((numvalues / 3) * 3))
559  numvectorbytes = (numvalues / 3) + 1;
560  else
561  numvectorbytes = (numvalues / 3);
562 
563  for (vectidx = 1;
564  vectidx <= (numvectorbytes + 1);
565  vectidx++) {
566  vect_3pack =
567  mrpdu_vectorptr->
568  FirstValue_VectorEvents[vectidx];
569  vectevt[0] = vect_3pack / 36;
570  vectevt[1] =
571  (vect_3pack - vectevt[0] * 36) / 6;
572  vectevt[2] =
573  vect_3pack - (36 * vectevt[0]) -
574  (6 * vectevt[1]);
575 
576  numvalues_processed =
577  (numvalues > 3 ? 3 : numvalues);
578 
579  for (vectevt_idx = 0;
580  vectevt_idx < numvalues_processed;
581  vectevt_idx++) {
582 
583  if (1 < svcreq_firstval) /*discard junk */
584  continue;
585 
586  attrib = mmrp_alloc();
587  if (NULL == attrib)
588  goto out; /* oops - internal error */
589 
590  attrib->type = MMRP_SVCREQ_TYPE;
591  attrib->attribute.svcreq =
592  svcreq_firstval;
593  svcreq_firstval++;
594  memcpy(attrib->registrar.
595  macaddr, eth->srcaddr,
596  6);
597 
598  switch (vectevt[vectevt_idx]) {
599  case MRPDU_NEW:
600  mmrp_event
601  (MRP_EVENT_RNEW,
602  attrib);
603  break;
604  case MRPDU_JOININ:
605  mmrp_event
606  (MRP_EVENT_RJOININ,
607  attrib);
608  break;
609  case MRPDU_IN:
610  mmrp_event
611  (MRP_EVENT_RIN,
612  attrib);
613  break;
614  case MRPDU_JOINMT:
615  mmrp_event
616  (MRP_EVENT_RJOINMT,
617  attrib);
618  break;
619  case MRPDU_MT:
620  mmrp_event
621  (MRP_EVENT_RMT,
622  attrib);
623  break;
624  case MRPDU_LV:
625  mmrp_event
626  (MRP_EVENT_RLV,
627  attrib);
628  break;
629  default:
630  free(attrib);
631  break;
632  }
633  }
634  numvalues -= numvalues_processed;
635  }
636 
637  /* as either firstval is either 0 or 1 ... the common
638  * case will be nulls for evt_2 and evt_3 ...
639  */
640 
641  /* 2 byte numvalues + 34 byte FirstValue + (n) vector bytes */
642  mrpdu_msg_ptr = (uint8_t *) mrpdu_vectorptr;
643  mrpdu_msg_ptr += 3 + numvectorbytes;
644  mrpdu_vectorptr =
645  (mrpdu_vectorattrib_t *) mrpdu_msg_ptr;
646  }
647 
648  break;
649  case MMRP_MACVEC_TYPE:
650  if (mrpdu_msg->AttributeLength != 6) {
651  /* we can seek for an endmark to recover .. but this version
652  * dumps the entire packet as malformed
653  */
654  goto out;
655  }
656  /* AttributeListLength not used for MMRP, hence
657  * Data points to the beginning of the VectorAttributes
658  */
659  mrpdu_vectorptr =
660  (mrpdu_vectorattrib_t *) mrpdu_msg->Data;
661  mrpdu_msg_ptr = (uint8_t *) mrpdu_vectorptr;
662 
663  while (!((mrpdu_msg_ptr[0] == 0)
664  && (mrpdu_msg_ptr[1] == 0))) {
665  numvalues =
666  MRPDU_VECT_NUMVALUES(ntohs
667  (mrpdu_vectorptr->
668  VectorHeader));
669 
670  if (MRPDU_VECT_LVA(ntohs(mrpdu_vectorptr->VectorHeader))) {
671  attrib = mmrp_alloc();
672  if (NULL == attrib)
673  goto out; /* oops - internal error */
674 
675  attrib->type = MMRP_MACVEC_TYPE;
676  mmrp_event(MRP_EVENT_RLA, attrib);
677  free(attrib);
678  }
679 
680  if (0 == numvalues)
681  /* Malformed - cant tell how long the trailing vectors are */
682  goto out;
683
