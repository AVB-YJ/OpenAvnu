56 int STDCALL memory_object_descriptor_imp::start_operation_cmd(void * notification_id, uint16_t operation_type)
57 {
58  struct jdksavdecc_frame cmd_frame;
59  struct jdksavdecc_aem_command_start_operation aem_cmd_start_operation;
60  memset(&aem_cmd_start_operation, 0, sizeof(aem_cmd_start_operation));
61 
62  if (operation_type > JDKSAVDECC_MEMORY_OBJECT_OPERATION_UPLOAD)
63  {
64  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, " Invalid operation type %x on memory object\n", operation_type);
65  return -1;
66  }
67 
68  aem_cmd_start_operation.aem_header.aecpdu_header.controller_entity_id = base_end_station_imp_ref->get_adp()->get_controller_entity_id();
69  aem_cmd_start_operation.aem_header.command_type = JDKSAVDECC_AEM_COMMAND_START_OPERATION;
70 
71  aem_cmd_start_operation.descriptor_type = descriptor_type();
72  aem_cmd_start_operation.descriptor_index = descriptor_index();
73  aem_cmd_start_operation.operation_id = 0;
74  aem_cmd_start_operation.operation_type = operation_type;
75 
76  aecp_controller_state_machine_ref->ether_frame_init(base_end_station_imp_ref->mac(), &cmd_frame,
77  ETHER_HDR_SIZE + JDKSAVDECC_AEM_COMMAND_START_OPERATION_COMMAND_LEN);
78  ssize_t aem_cmd_start_operation_returned = jdksavdecc_aem_command_start_operation_write(&aem_cmd_start_operation,
79  cmd_frame.payload,
80  ETHER_HDR_SIZE,
81  sizeof(cmd_frame.payload));
82 
83  if (aem_cmd_start_operation_returned < 0)
84  {
85  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "aem_command_start_operation_write error\n");
86  return -1;
87  }
88 
89  aecp_controller_state_machine_ref->common_hdr_init(JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_COMMAND,
90  &cmd_frame,
91  base_end_station_imp_ref->entity_id(),
92  JDKSAVDECC_AEM_COMMAND_START_OPERATION_COMMAND_LEN -
93  JDKSAVDECC_COMMON_CONTROL_HEADER_LEN);
94  system_queue_tx(notification_id, CMD_WITH_NOTIFICATION, cmd_frame.payload, cmd_frame.length);
95 
96  return 0;
97 }

99 int memory_object_descriptor_imp::proc_start_operation_resp(void *& notification_id,
100  const uint8_t * frame,
101  size_t frame_len,
102  int & status,
103  uint16_t & operation_id,
104  uint16_t & operation_type)
105 {
106  struct jdksavdecc_frame cmd_frame;
107  struct jdksavdecc_aem_command_start_operation_response aem_cmd_start_operation_resp;
108 
109  memcpy(cmd_frame.payload, frame, frame_len);
110  memset(&aem_cmd_start_operation_resp, 0, sizeof(aem_cmd_start_operation_resp));
111 
112  ssize_t aem_cmd_start_operation_resp_returned = jdksavdecc_aem_command_start_operation_response_read(&aem_cmd_start_operation_resp,
113  frame,
114  ETHER_HDR_SIZE,
115  frame_len);
116 
117  if (aem_cmd_start_operation_resp_returned < 0)
118  {
119  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "aem_command_start_operation_response_read error");
120  return -1;
121  }
122 
123  uint32_t msg_type = aem_cmd_start_operation_resp.aem_header.aecpdu_header.header.message_type;
124  status = aem_cmd_start_operation_resp.aem_header.aecpdu_header.header.status;
125  bool u_field = aem_cmd_start_operation_resp.aem_header.command_type >> 15 & 0x01;
126  operation_id = aem_cmd_start_operation_resp.operation_id;
127  operation_type = aem_cmd_start_operation_resp.operation_type;
128 
129  aecp_controller_state_machine_ref->update_inflight_for_rcvd_resp(notification_id, msg_type, u_field, &cmd_frame);
130 
131  return 0;
132 }
