52 int adp_discovery_state_machine::ether_frame_init(struct jdksavdecc_frame * cmd_frame)
53 {
54  /*** Offset to write the field to ***/
55  size_t ether_frame_pos = 0;
56  jdksavdecc_frame_init(cmd_frame);
57 
58  /********************************************************** Ethernet Frame **********************************************************/
59  cmd_frame->ethertype = JDKSAVDECC_AVTP_ETHERTYPE;
60  utility::convert_uint64_to_eui48(net_interface_ref->mac_addr(), cmd_frame->src_address.value); // Send from the Controller MAC address
61  cmd_frame->dest_address = jdksavdecc_multicast_adp_acmp; // Send to the ADP multicast destination MAC address
62  cmd_frame->length = ADP_FRAME_LEN; // Length of ADP packet is 82 bytes
63 
64  /****************** Fill frame payload with Ethernet frame information ****************/
65  jdksavdecc_frame_write(cmd_frame, cmd_frame->payload, ether_frame_pos, ETHER_HDR_SIZE);
66 
67  return 0;
68 }

70 void adp_discovery_state_machine::common_hdr_init(struct jdksavdecc_frame * cmd_frame, uint64_t target_entity_id)
71 {
72  struct jdksavdecc_adpdu_common_control_header adpdu_common_ctrl_hdr;
73  ssize_t adpdu_common_ctrl_hdr_returned;
74 
75  /********************************** 1722 Protocol Header ***********************************/
76  adpdu_common_ctrl_hdr.cd = 1;
77  adpdu_common_ctrl_hdr.subtype = JDKSAVDECC_SUBTYPE_ADP;
78  adpdu_common_ctrl_hdr.sv = 0;
79  adpdu_common_ctrl_hdr.version = 0;
80  adpdu_common_ctrl_hdr.message_type = 2;
81  adpdu_common_ctrl_hdr.valid_time = 0;
82  adpdu_common_ctrl_hdr.control_data_length = 56;
83  jdksavdecc_uint64_write(target_entity_id, &adpdu_common_ctrl_hdr.entity_id, 0, sizeof(uint64_t));
84 
85  /******************** Fill frame payload with AECP Common Control Header information ********************/
86  adpdu_common_ctrl_hdr_returned = jdksavdecc_adpdu_common_control_header_write(&adpdu_common_ctrl_hdr,
87  cmd_frame->payload,
88  ETHER_HDR_SIZE,
89  sizeof(cmd_frame->payload));
90 
91  if (adpdu_common_ctrl_hdr_returned < 0)
92  {
93  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "adpdu_common_ctrl_hdr_write error");
94  assert(adpdu_common_ctrl_hdr_returned >= 0);
95  }
96 }

158 int adp_discovery_state_machine::state_avail(const uint8_t * frame, size_t frame_len)
159 {
160  struct jdksavdecc_adpdu_common_control_header adp_hdr;
161  uint64_t entity_entity_id;
162  uint32_t entity_index;
163 
164  entity_entity_id = jdksavdecc_uint64_get(frame, ETHER_HDR_SIZE + PROTOCOL_HDR_SIZE);
165  jdksavdecc_adpdu_common_control_header_read(&adp_hdr, frame, ETHER_HDR_SIZE, frame_len);
166 
167  if (have_entity(entity_entity_id, &entity_index))
168  {
169  update_entity_timeout(entity_index, adp_hdr.valid_time * 2 * 1000); // Valid time period is between 2 and 62 seconds
170  }
171  else
172  {
173  struct entities entity;
174  entity.entity_id = entity_entity_id;
175  entity.inflight_timer.start(adp_hdr.valid_time * 2 * 1000); // Valid time period is between 2 and 62 seconds
176  add_entity(entity);
177  notification_imp_ref->post_notification_msg(END_STATION_CONNECTED, entity_entity_id, 0, 0, 0, 0, 0);
178  }
179 
180  return 0;
181 }

195 bool adp_discovery_state_machine::tick(uint64_t & end_station_entity_id)
196 {
197  if (first_tick)
198  {
199  state_discover(0);
200  first_tick = false;
201  }
202 
203  for (uint32_t i = 0; i < entities_vec.size(); i++)
204  {
205  if (entities_vec.at(i).inflight_timer.timeout())
206  {
207  end_station_entity_id = entities_vec.at(i).entity_id;
208  state_timeout(i);
209  notification_imp_ref->post_notification_msg(END_STATION_DISCONNECTED, end_station_entity_id, 0, 0, 0, 0, 0);
210  return true;
211  }
212  }
213 
214  return false;
215 }
