418 class PathTraceTLV {
419  private:
420  uint16_t tlvType;
421  typedef std::list<ClockIdentity> IdentityList;
422  IdentityList identityList;
423  public:
428  PathTraceTLV() {
429  tlvType = PLAT_htons(PATH_TRACE_TLV_TYPE);
430  }
437  void parseClockIdentity(uint8_t *buffer, int size) {
438  int length = PLAT_ntohs(*(uint16_t*)buffer);
439 
440  buffer += sizeof(uint16_t);
441  size -= sizeof(uint16_t);
442 
443  if((unsigned)size < (unsigned)length) {
444  length = size;
445  }
446  length /= PTP_CLOCK_IDENTITY_LENGTH;
447 
448  for(; length > 0; --length) {
449  ClockIdentity add;
450  add.set(buffer);
451  identityList.push_back(add);
452  buffer += PTP_CLOCK_IDENTITY_LENGTH;
453  }
454  }
455 
461  void appendClockIdentity(ClockIdentity * id) {
462  identityList.push_back(*id);
463  }
464 
470  void toByteString(uint8_t * byte_str) {
471  IdentityList::iterator iter;
472  *((uint16_t *)byte_str) = tlvType; // tlvType already in network byte order
473  byte_str += sizeof(tlvType);
474  *((uint16_t *)byte_str) = PLAT_htons
475  ((uint16_t)identityList.size()*PTP_CLOCK_IDENTITY_LENGTH);
476  byte_str += sizeof(uint16_t);
477  for
478  (iter = identityList.begin();
479  iter != identityList.end(); ++iter) {
480  iter->getIdentityString(byte_str);
481  byte_str += PTP_CLOCK_IDENTITY_LENGTH;
482  }
483  }
484 
490  bool has(ClockIdentity *id) {
491  return std::find
492  (identityList.begin(), identityList.end(), *id) !=
493  identityList.end();
494  }
495 
503  int length() {
504  return (int)(2*sizeof(uint16_t) + PTP_CLOCK_IDENTITY_LENGTH*identityList.size());
505  }
506 };
