169 {
170  unsigned char* test_stream_id;
171  struct ethernet_header* eth_header;
172  uint32_t *buf;
173  uint32_t frame[2] = { 0 , 0 };
174  int i;

176  (void) packet_header; /* unused */
177 
178 #if DEBUG
179  fprintf(stdout,"Got packet.\n");

182  eth_header = (struct ethernet_header*)(packet);
183 
184 #if DEBUG
185  fprintf(stdout,"Ether Type: 0x%02x%02x\n", eth_header->type[0], eth_header->type[1]);
186 #endif /* DEBUG*/
187 
188  if (0 == memcmp(glob_ether_type,eth_header->type,sizeof(eth_header->type)))
189  {
190  test_stream_id = (unsigned char*)(packet + ETHERNET_HEADER_SIZE + SEVENTEEN22_HEADER_PART1_SIZE);
191 
192 #if DEBUG
193  fprintf(stderr, "Received stream id: %02x%02x%02x%02x%02x%02x%02x%02x\n ",
194  test_stream_id[0], test_stream_id[1],
195  test_stream_id[2], test_stream_id[3],
196  test_stream_id[4], test_stream_id[5],
197  test_stream_id[6], test_stream_id[7]);
198 #endif /* DEBUG*/
199 
200  if (0 == memcmp(test_stream_id, ctx->stream_id, sizeof(STREAM_ID_SIZE)))
201  {
202 
203 #if DEBUG
204  fprintf(stdout,"Stream ids matched.\n");
205 #endif /* DEBUG*/
206  buf = (uint32_t*) (packet + HEADER_SIZE);
207  for(i = 0; i < SAMPLES_PER_FRAME * CHANNELS; i += 2)
208  {
209  memcpy(&frame[0], &buf[i], sizeof(frame));
210 
211  frame[0] = ntohl(frame[0]); /* convert to host-byte order */
212  frame[1] = ntohl(frame[1]);
213  frame[0] &= 0x00ffffff; /* ignore leading label */
214  frame[1] &= 0x00ffffff;
215  frame[0] <<= 8; /* left-align remaining PCM-24 sample */
216  frame[1] <<= 8;
217 
218  sf_writef_int(glob_snd_file, (const int *)frame, 1);
219  }
220  }

242 int join_mcast(char *interface, struct mrp_listener_ctx *ctx)
243 {
244  struct sockaddr_ll addr;
245  struct ifreq if_request;
246  int rc;
247  struct packet_mreq multicast_req;
248 
249  if (NULL == ctx)
250  return -1;
251 
252  glob_sock = socket(PF_PACKET, SOCK_RAW, htons(0x2272));
253  if (glob_sock < 0)
254  return -1;
255 
256  memset(&if_request, 0, sizeof(if_request));
257  strncpy(if_request.ifr_name, interface, sizeof(if_request.ifr_name)-1);
258 
259  rc = ioctl(glob_sock, SIOCGIFINDEX, &if_request);
260  if (rc < 0) {
261  /* glob_sock will be closed by cleanup() */
262  return -1;
263  }
264 
265  memset(&addr, 0, sizeof(addr));
266  addr.sll_ifindex = if_request.ifr_ifindex;
267  addr.sll_family = AF_PACKET;
268  addr.sll_protocol = htons(0x2272);
269 
270  rc = bind(glob_sock, (struct sockaddr *)&addr, sizeof(addr));
271  if (0 != rc)
272  return -1;
273 
274  rc = setsockopt(glob_sock, SOL_SOCKET, SO_BINDTODEVICE, interface,
275  strlen(interface));
276  if (0 != rc)
277  return -1;
278 
279  memset(&multicast_req, 0, sizeof(multicast_req));
280  multicast_req.mr_ifindex = if_request.ifr_ifindex;
281  multicast_req.mr_type = PACKET_MR_MULTICAST;
282  multicast_req.mr_alen = 6;
283  memcpy(multicast_req.mr_address, (void*)ctx->dst_mac, 6);
284 
285  rc = setsockopt(glob_sock, SOL_PACKET, PACKET_ADD_MEMBERSHIP,
286  &multicast_req, sizeof(multicast_req));
287  if (0 != rc)
288  return -1;
289 
290  return 0;

293 int pci_connect(device_t *igb_dev)
294 {
295  char devpath[IGB_BIND_NAMESZ];
296  struct pci_access *pacc;
297  struct pci_dev *dev;
298  int err;
299 
300  memset(igb_dev, 0, sizeof(device_t));
301  pacc = pci_alloc();
302  pci_init(pacc);
303  pci_scan_bus(pacc);
304  for (dev = pacc->devices; dev; dev = dev->next) {
305  pci_fill_info(dev,
306  PCI_FILL_IDENT | PCI_FILL_BASES | PCI_FILL_CLASS);
307  igb_dev->pci_vendor_id = dev->vendor_id;
308  igb_dev->pci_device_id = dev->device_id;
309  igb_dev->domain = dev->domain;
310  igb_dev->bus = dev->bus;
311  igb_dev->dev = dev->dev;
312  igb_dev->func = dev->func;
313  snprintf(devpath, IGB_BIND_NAMESZ, "%04x:%02x:%02x.%d",
314  dev->domain, dev->bus, dev->dev, dev->func);
315 
316  err = igb_probe(igb_dev);
317 
318  if (err)
319  continue;
320 
321  printf("attaching to %s\n", devpath);
322 
323  if (igb_attach(devpath, igb_dev)) {
324  printf("attach failed! (%s)\n", strerror(errno));
325  continue;
326  }
327  if (igb_attach_rx(igb_dev)) {
328  printf("rx attach failed! (%s)\n", strerror(err));
329  continue;
330  }
331  goto out;
332  }
333  pci_cleanup(pacc);
334  return -ENXIO;
335 out:
336  pci_cleanup(pacc);
337  return 0;
338 }

340 int igb_start_rx(device_t *igb_dev, struct mrp_listener_ctx *ctx)
341 {
342  int rc = -1;
343  int err;
344 
345  uint32_t i;
346  uint32_t count = 0;
347 
348  struct myqelem *page_qelem = NULL;
349  struct myqelem *prev_page_qelem = NULL;
350  struct myqelem *pkt_qelem = NULL;
351  struct myqelem *prev_pkt_qelem = NULL;
352 
353  struct igb_dma_alloc *a_page = NULL;
354  struct igb_packet *a_packet = NULL;
355 
356  uint8_t test_filter[128];
357  uint8_t filter_mask[64];
358  uint32_t filter_len = 0;
359  struct ethhdr *ethhdr = NULL;
360  uint16_t *ethtype = NULL;
361 
362  if (0 != pci_connect(igb_dev))
363  goto out;
364 
365  if (0 != igb_init(igb_dev))
366  goto out;
367 
368  memset (glob_dma_pages, 0, sizeof(glob_dma_pages));
369  memset (glob_rx_packets, 0, sizeof(glob_rx_packets));
370 
371  while (count < PKT_NUM) {
372  page_qelem = calloc(1, sizeof(struct myqelem));
373  if (!page_qelem)
374  goto out;
375 
376  if (!glob_page_head)
377  glob_page_head = page_qelem;
378 
379  a_page = (struct igb_dma_alloc*)calloc(1, sizeof(struct igb_dma_alloc));
380  if (!a_page)
381  goto out;
382 
383  page_qelem->q_data = (void*)a_page;
384  insque(page_qelem, prev_page_qelem);
385  prev_page_qelem = page_qelem;
386 
387  err = igb_dma_malloc_page(igb_dev, a_page);
388  if (err) {
389  printf("malloc failed (%s) - out of memory?\n",
390  strerror(err));
391  goto out;
392  }
393 
394  if (a_page->mmap_size < PKT_SZ)
395  goto out;
396 
397  /* divide the dma page into buffers for packets */
398  for (i = 0; i < a_page->mmap_size/PKT_SZ; i++) {
399  pkt_qelem = calloc(1, sizeof(struct myqelem));
400  if (!pkt_qelem)
401  goto out;
402 
403  if (!glob_pkt_head)
404  glob_pkt_head = pkt_qelem;
405 
406  a_packet = (struct igb_packet*)calloc(1, sizeof(struct igb_packet));
407  if (a_packet == NULL) {
408  printf("failed to allocate igb_packet memory!\n");
409  err = -errno;
410  goto out;
411  }
412 
413  a_packet->dmatime = 0;
414  a_packet->attime = 0;
415  a_packet->flags = 0;
416  a_packet->map.paddr = a_page->dma_paddr;
417  a_packet->map.mmap_size = a_page->mmap_size;
418  a_packet->offset = (i * PKT_SZ);
419  a_packet->vaddr = a_page->dma_vaddr + a_packet->offset;
420  a_packet->len = PKT_SZ;
421  memset(a_packet->vaddr, 0, a_packet->len);
422  if (prev_pkt_qelem)
423  ((struct igb_packet*)prev_pkt_qelem->q_data)->next = a_packet;
424 
425  igb_refresh_buffers(igb_dev, IGB_QUEUE_0, &a_packet, 1);
426 
427  pkt_qelem->q_data = (void*)a_packet;
428  insque(pkt_qelem, prev_pkt_qelem);
429  prev_pkt_qelem = pkt_qelem;
430 
431  count++;
432  }
433  }
434 
435  /* filtering example */
436  memset(filter_mask, 0, sizeof(filter_mask));
437  memset(test_filter, 0, sizeof(test_filter));
438 
439  /* the filter for IEEE1722 with VLAN */
440 
441  /* accept packets toward the multicast dst mac given by MRP */
442  ethhdr = (struct ethhdr *)test_filter;
443  memcpy((void*)ethhdr->h_dest, (void*)ctx->dst_mac, ETH_ALEN);
444 
445  /* ethtype in ethernet frame = 802.1Q */
446  ethhdr->h_proto = htons(ETH_P_8021Q);
447 
448  /* real ethtype = IEEE1722 */
449  ethtype = (uint16_t*)(test_filter + ETH_HLEN + 2);
450  *ethtype = htons(ETH_P_IEEE1722);
451 
452  filter_mask[0] = 0x3F; /* 00111111b = dst mac */
453  filter_mask[1] = 0x30; /* 00110000b = ethtype */
454  filter_mask[2] = 0x03; /* 00000011b = ethtype after a vlan tag */
455 
456  /* length must be 8 byte-aligned */
457  filter_len = ((ETHERNET_HEADER_SIZE + (8u - 1u)) / 8u) * 8u;
458 
459  /* install the filter on queue0 */
460  rc = igb_setup_flex_filter(igb_dev, IGB_QUEUE_0, 0, filter_len,
461  test_filter, filter_mask);
462 out:
463  if (rc != 0)
464  igb_stop_rx(igb_dev);
465 
466  return rc;
467 }

470 {
471  struct myqelem *qelem = NULL;
472 
473  igb_clear_flex_filter(igb_dev, 0);
474 
475  while (glob_pkt_head) {
476  qelem = glob_pkt_head;
477  glob_pkt_head = qelem->q_forw;
478  if (qelem->q_data)
479  free(qelem->q_data);
480  free(qelem);
481  }
482 
483  while (glob_page_head) {
484  qelem = glob_page_head;
485  glob_page_head = qelem->q_forw;
486  if (qelem->q_data){
487  igb_dma_free_page(igb_dev, (struct igb_dma_alloc*)qelem->q_data);
488  free(qelem->q_data);
489  }
490  free(qelem);
491  }
492 
493  igb_detach(igb_dev);
494 }

496 void igb_process_rx(device_t *igb_dev, struct mrp_listener_ctx *ctx)
497 {
498  uint32_t i;
499  uint32_t count;
500  uint8_t *packet;
501  uint32_t *buf;
502  uint32_t frame[2] = { 0 , 0 };
503  uint8_t *test_stream_id;
504 
505  struct igb_packet *igb_pkt = NULL;
506  struct ethhdr* eth_hdr;
507  uint16_t *ethtype = NULL;
508 
509  halt_rx_sig = 1;
510 
511  while (halt_rx_sig) {
512 
513  /* put back */
514  if (igb_pkt) {
515  igb_refresh_buffers(igb_dev, IGB_QUEUE_0, &igb_pkt, 1);
516  igb_pkt = NULL;
517  }
518 
519  count = 1;
520  igb_receive(igb_dev, IGB_QUEUE_0, &igb_pkt, &count);
521  if (igb_pkt == NULL) {
522  continue;
523  }
524 
525  packet = igb_pkt->vaddr;
526  eth_hdr = (struct ethhdr*)(packet);
527  switch (htons(eth_hdr->h_proto)) {
528  case ETH_P_8021Q:
529  ethtype = (uint16_t*)(packet + ETH_HLEN + 2);
530  if (*ethtype != htons(ETH_P_IEEE1722))
531  continue; /* drop the packet */
532  break;
533  case ETH_P_IEEE1722:
534  break;
535  default:
536  continue; /* drop the packet */
537  break;
538  }
539 
540  test_stream_id = (uint8_t*)(packet + ETHERNET_HEADER_SIZE +
541  SEVENTEEN22_HEADER_PART1_SIZE);
542  buf = (uint32_t*)(packet + HEADER_SIZE);
543 
544  if (ETH_P_IEEE1722 == htons(eth_hdr->h_proto)) {
545  /* I210 would strip the VLAN tag field when CTRL.VME = 1b */
546  /* pull 4 bytes the VLAN tag size */
547  test_stream_id = ((uint8_t*)test_stream_id - 4);
548  buf = ((uint32_t*)buf - 1);
549  }
550 
551  if (0 == memcmp(test_stream_id, ctx->stream_id, STREAM_ID_SIZE))
552  {
553  for(i = 0; i < SAMPLES_PER_FRAME * CHANNELS; i += 2)
554  {
555  memcpy(&frame[0], &buf[i], sizeof(frame));
556 
557  frame[0] = ntohl(frame[0]); /* convert to host-byte order */
558  frame[1] = ntohl(frame[1]);
559  frame[0] &= 0x00ffffff; /* ignore leading label */
560  frame[1] &= 0x00ffffff;
561  frame[0] <<= 8; /* left-align remaining PCM-24 sample */
562  frame[1] <<= 8;
563 
564  sf_writef_int(glob_snd_file, (const int *)frame, 1);
565  }
566  }
567  }
568 }
