58 bool cli_command::run_command(cmd_line * cmd_ptr, std::queue<std::string> args, bool & done, std::string prefix)
59 {
60  if (args.size() > 0)
61  {
62  std::map<std::string, cli_command *>::iterator iter = m_sub_commands.find(args.front());
63  if (iter != m_sub_commands.end())
64  {
65  prefix += args.front() + " ";
66 
67  args.pop();
68  if (iter->second->run_command(cmd_ptr, args, done, prefix))
69  {
70  return true;
71  }
72  }
73  }
74 
75  for (std::vector<cli_command_format *>::iterator iter = m_formats.begin(); iter != m_formats.end(); ++iter)
76  {
77  if ((*iter)->run_command(cmd_ptr, args, done))
78  {
79  return true;
80  }
81  }
82 
83  if (m_formats.size())
84  {
85  // Only print help at leaf nodes
86  print_help_details(prefix);
87 
88  // Prevent printing of full list of commands
89  return true;
90  }
91 
92  return false;
93 }

95 void cli_command::print_help_all(std::string prefix, size_t depth) const

101  ++iter)

103  std::string sub_prefix(prefix);

107

120 {
121  while (cmd_path.size() && cmd_path.front().length() == 0)
122  {
123  // Ignore any blanks in the path
124  cmd_path.pop();
125  }
126 
127  if ((cmd_path.size() == 0) || (m_sub_commands.size() == 0))
128  {
129  return this;

131  else

134  if (iter != m_sub_commands.end())

136  prefix += cmd_path.front() + " ";
