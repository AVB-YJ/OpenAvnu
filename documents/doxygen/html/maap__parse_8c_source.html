45 int parse_text_cmd(char *buf, Maap_Cmd *cmd) {
46  char *argv[5];
47  int argc = 0;
48  char *p;
49  int set_cmd = 0;
50 
51  p = strtok(buf, " \r\n");
52  while ((p != NULL) && (argc < sizeof(argv) / sizeof(*argv))) {
53  argv[argc++] = p;
54  p = strtok(NULL, " \r\n");
55  }
56 
57  if (argc >= 1 && argc <= 3)
58  {
59  /* Give all parameters default values. */
60  cmd->kind = MAAP_CMD_INVALID;
61  cmd->id = -1; /* N/A */
62  cmd->start = 0; /* N/A */
63  cmd->count = 0; /* N/A */
64 
65  if (strncmp(argv[0], "init", 4) == 0) {
66  if (argc == 1) {
67  cmd->kind = MAAP_CMD_INIT;
68  cmd->start = MAAP_DYNAMIC_POOL_BASE;
69  cmd->count = MAAP_DYNAMIC_POOL_SIZE;
70  set_cmd = 1;
71  } else if (argc == 3) {
72  cmd->kind = MAAP_CMD_INIT;
73  cmd->start = strtoull(argv[1], NULL, 16);
74  cmd->count = strtoul(argv[2], NULL, 0);
75  set_cmd = 1;
76  }
77  } else if (strncmp(argv[0], "reserve", 7) == 0) {
78  if (argc == 2) {
79  cmd->kind = MAAP_CMD_RESERVE;
80  cmd->count = strtoul(argv[1], NULL, 0);
81  set_cmd = 1;
82  } else if (argc == 3) {
83  cmd->kind = MAAP_CMD_RESERVE;
84  cmd->start = strtoull(argv[1], NULL, 16);
85  cmd->count = strtoul(argv[2], NULL, 0);
86  set_cmd = 1;
87  }
88  } else if (strncmp(argv[0], "release", 7) == 0 && argc == 2) {
89  cmd->kind = MAAP_CMD_RELEASE;
90  cmd->id = (int)strtoul(argv[1], NULL, 0);
91  set_cmd = 1;
92  } else if (strncmp(argv[0], "status", 7) == 0 && argc == 2) {
93  cmd->kind = MAAP_CMD_STATUS;
94  cmd->id = (int)strtoul(argv[1], NULL, 0);
95  set_cmd = 1;
96  } else if (strncmp(argv[0], "yield", 7) == 0 && argc == 2) {
97  cmd->kind = MAAP_CMD_YIELD;
98  cmd->id = (int)strtoul(argv[1], NULL, 0);
99  set_cmd = 1;
100  } else if (strncmp(argv[0], "exit", 4) == 0 && argc == 1) {
101  cmd->kind = MAAP_CMD_EXIT;
102  set_cmd = 1;
103  }
104  }
105 
106  return set_cmd;
107  }

109 int parse_write(Maap_Client *mc, const void *sender, char *buf, int *input_is_text) {
110  Maap_Cmd *bufcmd, cmd;
111  int rv = 0;
112  int retVal = 0;
113 
114  bufcmd = (Maap_Cmd *)buf;
115 
116  switch (bufcmd->kind) {
117  case MAAP_CMD_INIT:
118  case MAAP_CMD_RESERVE:
119  case MAAP_CMD_RELEASE:
120  case MAAP_CMD_STATUS:
121  case MAAP_CMD_YIELD:
122  case MAAP_CMD_EXIT:
123  if (input_is_text) { *input_is_text = 0; }
124  memcpy(&cmd, bufcmd, sizeof (Maap_Cmd));
125  rv = 1;
126  break;
127  default:
128  if (input_is_text) { *input_is_text = 1; }
129  memset(&cmd, 0, sizeof (Maap_Cmd));
130  rv = parse_text_cmd(buf, &cmd);
131  if (!rv) {
132  /* Unrecognized command. */
133  retVal = -1;
134  }
135  break;
136  }
137 
138  if (rv) {
139  switch(cmd.kind) {
140  case MAAP_CMD_INIT:
141 #ifdef DEBUG_CMD_MSG
142  MAAP_LOGF_DEBUG("Got cmd MAAP_CMD_INIT, range_base: 0x%016llx, range_size: 0x%04x",
143  (unsigned long long)cmd.start, cmd.count);
144 #endif
145  rv = maap_init_client(mc, sender, cmd.start, cmd.count);
146  break;
147  case MAAP_CMD_RESERVE:
148 #ifdef DEBUG_CMD_MSG
149  if (cmd.start != 0) {
150  MAAP_LOGF_DEBUG("Got cmd MAAP_CMD_RESERVE, start: 0x%016llx, length: %u",
151  (unsigned long long)cmd.start, (unsigned) cmd.count);
152  } else {
153  MAAP_LOGF_DEBUG("Got cmd MAAP_CMD_RESERVE, length: %u", (unsigned) cmd.count);
154  }
155 #endif
156  rv = maap_reserve_range(mc, sender, cmd.start, cmd.count);
157  break;
158  case MAAP_CMD_RELEASE:
159 #ifdef DEBUG_CMD_MSG
160  MAAP_LOGF_DEBUG("Got cmd MAAP_CMD_RELEASE, id: %d", (int) cmd.id);
161 #endif
162  rv = maap_release_range(mc, sender, cmd.id);
163  break;
164  case MAAP_CMD_STATUS:
165 #ifdef DEBUG_CMD_MSG
166  MAAP_LOGF_DEBUG("Got cmd MAAP_CMD_STATUS, id: %d", (int) cmd.id);
167 #endif
168  maap_range_status(mc, sender, cmd.id);
169  break;
170  case MAAP_CMD_YIELD:
171 #ifdef DEBUG_CMD_MSG
172  MAAP_LOGF_DEBUG("Got cmd MAAP_CMD_YIELD, id: %d", (int) cmd.id);
173 #endif
174  rv = maap_yield_range(mc, sender, cmd.id);
175  break;
176  case MAAP_CMD_EXIT:
177 #ifdef DEBUG_CMD_MSG
178  MAAP_LOG_DEBUG("Got cmd MAAP_CMD_EXIT");
179 #endif
180  retVal = 1; /* Indicate that we should exit. */
181  break;
182  default:
183  MAAP_LOG_ERROR("Unrecognized input to parse_write");
184  retVal = -1;
185  break;
186  }
187  }
188 
189  return retVal;
190 }
