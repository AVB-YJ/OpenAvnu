58 // Returns TRUE, to say we're connected and registers tspec with FQTSS tspec should be initialized
59 bool openavbTLRunTalkerInit(tl_state_t *pTLState)
60 {
61  openavb_tl_cfg_t *pCfg = &pTLState->cfg;
62  talker_data_t *pTalkerData = pTLState->pPvtTalkerData;
63  //avtp_stream_t *pStream = (avtp_stream_t *)(pTalkerData->avtpHandle);
64 
65  strncpy(pTalkerData->ifname, pCfg->ifname, sizeof(pTalkerData->ifname) - 1);
66 
67  // CORE_TODO: It would be good to have some parts of endpoint moved into non-endpoint general code to handle some the stream
68  // configuration values.
69  if (pCfg->stream_addr.mac) {
70  memcpy(pTalkerData->streamID.addr, pCfg->stream_addr.mac, ETH_ALEN);
71  }else {
72  AVB_LOG_WARNING("Stream Address Not Set");
73  }
74 
75  pTalkerData->streamID.uniqueID = pCfg->stream_uid;
76  pTalkerData->srClass = pCfg->sr_class;
77  if (pCfg->sr_class == SR_CLASS_A) {
78  pTalkerData->classRate = 8000;
79  pTalkerData->vlanID = (pCfg->vlan_id == 0 ?
80  SR_CLASS_A_DEFAULT_VID : pCfg->vlan_id);
81  pTalkerData->vlanPCP = SR_CLASS_A_DEFAULT_PRIORITY;
82  }
83  else if (pCfg->sr_class == SR_CLASS_B) {
84  pTalkerData->classRate = 4000;
85  pTalkerData->vlanID = (pCfg->vlan_id == 0 ?
86  SR_CLASS_B_DEFAULT_VID : pCfg->vlan_id);
87  pTalkerData->vlanPCP = SR_CLASS_B_DEFAULT_PRIORITY;
88  }
89  memcpy(&pTalkerData->destAddr, &pCfg->dest_addr.mac->ether_addr_octet, ETH_ALEN);
90 
91  unsigned int maxBitrate = 0;
92  if (pCfg->intf_cb.intf_get_src_bitrate_cb != NULL) {
93  maxBitrate = pCfg->intf_cb.intf_get_src_bitrate_cb(pTLState->pMediaQ);
94  }
95  if (maxBitrate > 0) {
96  if (pCfg->map_cb.map_set_src_bitrate_cb != NULL) {
97  pCfg->map_cb.map_set_src_bitrate_cb(pTLState->pMediaQ, maxBitrate);
98  }
99 
100  if (pCfg->map_cb.map_get_max_interval_frames_cb != NULL) {
101  unsigned int map_intv_frames = pCfg->map_cb.map_get_max_interval_frames_cb(pTLState->pMediaQ, pTLState->cfg.sr_class);
102  pCfg->max_interval_frames = map_intv_frames > 0 ? map_intv_frames : pCfg->max_interval_frames;
103  }
104  }
105  pTalkerData->tSpec.maxIntervalFrames = pCfg->max_interval_frames;
106  pTalkerData->tSpec.maxFrameSize = pCfg->map_cb.map_max_data_size_cb(pTLState->pMediaQ);
107 
108  // TODO_COREAVB : This wakeRate should also be set in the endpoint case and removed from the tasker.c start stream
109  if (!pCfg->map_cb.map_transmit_interval_cb(pTLState->pMediaQ)) {
110  pTalkerData->wakeRate = pTalkerData->classRate / pCfg->batch_factor;
111  }
112  else {
113  // Override the class observation interval with the one provided by the mapping module.
114  pTalkerData->wakeRate = pCfg->map_cb.map_transmit_interval_cb(pTLState->pMediaQ) / pCfg->batch_factor;
115  }
116 
117  if_info_t ifinfo;
118  openavbCheckInterface(pTalkerData->ifname, &ifinfo);
119 
120  pTalkerData->fwmark = openavbQmgrAddStream((SRClassIdx_t)pCfg->sr_class,
121  pTalkerData->wakeRate,
122  pTalkerData->tSpec.maxIntervalFrames,
123  pTalkerData->tSpec.maxFrameSize);
124 
125  if (pTalkerData->fwmark == INVALID_FWMARK)
126  return FALSE;
127 
128  AVB_LOGF_INFO("Dest Addr: "ETH_FORMAT, ETH_OCTETS(pTalkerData->destAddr));
129  AVB_LOGF_INFO("Starting stream: "STREAMID_FORMAT, STREAMID_ARGS(&pTalkerData->streamID));
130  talkerStartStream(pTLState);
131 
132  // Let the AVDECC Msg server know our current stream ID, in case it is waiting for an update.
133  if (pTLState->avdeccMsgHandle != AVB_AVDECC_MSG_HANDLE_INVALID) {
134  if (!openavbAvdeccMsgClntTalkerStreamID(pTLState->avdeccMsgHandle,
135  pTalkerData->srClass, pTalkerData->streamID.addr, pTalkerData->streamID.uniqueID,
136  pTalkerData->destAddr, pTalkerData->vlanID)) {
137  AVB_LOG_ERROR("openavbAvdeccMsgClntTalkerStreamID() failed");
138  }
139  }
140 
141  return TRUE;
142 }

147  openavbQmgrRemoveStream(pTalkerData->fwmark);
