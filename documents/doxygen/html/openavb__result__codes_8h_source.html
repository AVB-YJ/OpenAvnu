39 // Result Code (success or error code)
40 // First (left) byte:
41 // bits 0 thru 6 reserved;
42 // bit 7 (rightmost) - 0 indicates success, 1 indicates failure.
43 // Second Byte:
44 // indicates specific software module - see enum openavbModules below
45 // Third & Forth (right) Bytes;
46 // defined independently by each module to indicate specific failure (or success)
47 typedef U32 openavbRC;
48 
49 // Example usage:
50 // openavbRC foo()
51 // {
52 // .
53 // .
54 // return AVB_RC(OPENAVB_AVTP_FAILURE | OPENAVB_RC_OUT_OF_MEMORY)
55 // }
56 //
57 // openavbRC result;
58 // result = foo();
59 // if (IS_OPENAVB_SUCCESS(result)) {
60 // ...
61 // }

200 #define IS_OPENAVB_FAILURE(_rc_) ((_rc_) & 0x01000000)
201 #define AVB_RC(_rc_) (openavbUtilRCRecord(_rc_))
202 #define AVB_RC_RET(_rc_) return (_rc_)
203 #define AVB_RC_MSG(_rc_) openavbUtilRCCodeToString(_rc_)
204 #define AVB_RC_LOG(_rc_) AVB_LOGF_ERROR("%s", openavbUtilRCCodeToString(_rc_))
205 #define AVB_RC_LOG_RET(_rc_) AVB_LOGF_ERROR("%s", openavbUtilRCCodeToString(_rc_)); return (_rc_)
206 #define AVB_RC_LOG_TRACE_RET(_rc_, _trace_) AVB_LOGF_ERROR("%s", openavbUtilRCCodeToString(_rc_)); AVB_TRACE_EXIT(_trace_); return (_rc_)
207 #define AVB_RC_TRACE_RET(_rc_, _trace_) AVB_TRACE_EXIT(_trace_); return (_rc_)
208 #define AVB_RC_FAIL_RET(_rc_) if (IS_OPENAVB_FAILURE(_rc_)) return (_rc_)
209 #define AVB_RC_FAIL_TRACE_RET(_rc_, _trace_) if (IS_OPENAVB_FAILURE(_rc_)) {AVB_TRACE_EXIT(_trace_); return (_rc_);}
