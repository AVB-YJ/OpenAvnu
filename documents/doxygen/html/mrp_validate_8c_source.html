35 /* This utility is designed to do three things:
36  * 1. Allow you to very simply exercise the MRP daemon between two
37  * end stations with or without an AVB Bridge between them.
38  * 2. Monitor the MxRPDUs sent by a third party device.
39  * 3. Provide an example of how to use the Open-AVB MRP daemon
40  * command interface.
41  *
42  * To run this program load the MRP daemon on two PCs. Run one in
43  * Monitor (mon) mode, and issue M++, V++, S++, etc commands on the
44  * other PC. Wireshark can also be used to verify that MRPDUs are
45  * formatted correctly for the various MxRP applications.
46  *
47  * Possible future enhancements:
48  * 1. Allow the various parameters to be modified (e.g.: change the
49  * STREAM_ID, VLAN_ID, etc).
50  * 2. Timestamp the Monitor output, including elapsed time between
51  * commands of the same type (e.g.: MMRP to next MMRP command).
52  * That way a user could validate that the periodic timer for
53  * MMRP is working correctly, or that the MSRP LeaveAll timer
54  * is working, or .... This would also allow the user to verify
55  * the timers of a third party device or operating correctly.
56  */

329 int fnMonitor(void) {
330  /* First we need to have the mrpd daemon put this application in the
331  * list of attached clients that wish to receive reports when any MMRP,
332  * MSRP or MVRP attributes are received.
333  *
334  * Note that any M*, S* or V* command will put this application in the
335  * mrpd attached clients list. However, the M??, S?? and V?? do not
336  * cause any MRP attribute declarations or registrations, and are
337  * therefore the preferred method for doing this for monitoring.
338  */
339  mrpdclient_sendto(mrpd_sock, M_QUESTION_QUESTION, sizeof(M_QUESTION_QUESTION));
340  mrpdclient_sendto(mrpd_sock, S_QUESTION_QUESTION, sizeof(S_QUESTION_QUESTION));
341  mrpdclient_sendto(mrpd_sock, V_QUESTION_QUESTION, sizeof(V_QUESTION_QUESTION));
342 
343  /* Now just wait for any MRPDUs to be received... */
344  while (1) {
345  mrpdclient_recv(mrpd_sock, dump_ctl_msg);
346  }
347 
348  return 0;
349 }

443  }
444 
445  mrpd_sock = mrpdclient_init();
446  if (mrpd_sock == SOCKET_ERROR) {
447  printf("mrpdclient_init failed\n");
448  goto out;
449  }
450 
451  saved[0]='\0';
452  while (!done) {
453  displayMenu();
454  printf("Option to run? ");
455  if (NULL == fgets(option, sizeof(option), stdin)) {
456  done = 1;
457  printf("\n");
458  continue;
459  }
460  option[strlen(option)-1] = '\0'; // remove assumed LF
461  if ('\0' == option[0]) {
462  // Repeat last cmd if nothing entered
463  strncpy(option, saved, sizeof(option));
464  }
465 
466  rc = runMenuFunc(option);
467  if (rc >= 0) {
468  // Save last command if valid
469  strncpy(saved, option, sizeof(option));
470  }
471  }
472 
473  free(msgbuf);
474  rc = mrpdclient_close(&mrpd_sock);
475 
476 out:
477  if (-1 == rc)
478  return EXIT_FAILURE;
479  else
480  return EXIT_SUCCESS;
481 }
