159  U8 *pBuf, *pFrame;
160  U32 offset, len;

162 
163  struct timespec now;
164  static U32 packetCnt = 0;
165  static U64 nextReportInterval = 0;
166 
167  clock_gettime(CLOCK_MONOTONIC, &now);
168  nextReportInterval = TIMESPEC_TO_NSEC(now) + (NANOSECONDS_PER_SECOND * reportSec);
169 
170  while (bRunning) {
171  pBuf = openavbRawsockGetRxFrame(rs, OPENAVB_RAWSOCK_BLOCK, &offset, &len);
172  pFrame = pBuf + offset;

174  if (dumpFlag) {

178 
179  packetCnt++;
180 
181  clock_gettime(CLOCK_MONOTONIC, &now);
182  U64 nowNSec = TIMESPEC_TO_NSEC(now);;
183 
184  if (reportSec > 0) {
185  if (nowNSec > nextReportInterval) {
186  printf("RX Packets: %d\n", packetCnt);
187  packetCnt = 0;
188  nextReportInterval = nowNSec + (NANOSECONDS_PER_SECOND * reportSec);
189  }
190  }
191  }
192
