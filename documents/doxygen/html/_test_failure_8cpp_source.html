34 static SimpleString removeAllPrintableCharactersFrom(const SimpleString& str)
35 {
36  size_t bufferSize = str.size()+1;
37  char* buffer = (char*) PlatformSpecificMalloc(bufferSize);
38  str.copyToBuffer(buffer, bufferSize);
39 
40  for (size_t i = 0; i < bufferSize-1; i++)
41  if (buffer[i] != '\t' && buffer[i] != '\n')
42  buffer[i] = ' ';
43 
44  SimpleString result(buffer);
45  PlatformSpecificFree(buffer);
46  return result;
47 }

138 SimpleString TestFailure::createDifferenceAtPosString(const SimpleString& actual, size_t position, DifferenceFormat format)
139 {
140  SimpleString result;
141  const size_t extraCharactersWindow = 20;
142  const size_t halfOfExtraCharactersWindow = extraCharactersWindow / 2;
143  const size_t actualOffset = (format == DIFFERENCE_STRING) ? position : (position * 3 + 1);
144 
145  SimpleString paddingForPreventingOutOfBounds (" ", halfOfExtraCharactersWindow);
146  SimpleString actualString = paddingForPreventingOutOfBounds + actual + paddingForPreventingOutOfBounds;
147  SimpleString differentString = StringFromFormat("difference starts at position %lu at: <", (unsigned long) position);
148 
149  result += "\n";
150  result += StringFromFormat("\t%s%s>\n", differentString.asCharString(), actualString.subString(actualOffset, extraCharactersWindow).asCharString());
151 
152  SimpleString markString = actualString.subString(actualOffset, halfOfExtraCharactersWindow+1);
153  markString = removeAllPrintableCharactersFrom(markString);
154  markString = addMarkerToString(markString, halfOfExtraCharactersWindow);
155 
156  result += StringFromFormat("\t%s%s", SimpleString(" ", differentString.size()).asCharString(), markString.asCharString());
157  return result;
158 }
