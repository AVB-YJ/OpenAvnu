32 /*
33 * MODULE SUMMARY :
34 *
35 * Stream clients (talkers or listeners) must connect to the central
36 * "endpoint" process to create a reservation for their traffic.
37 * This code provides the means for them to do so.
38 *
39 * This source file is compiled into the streamer executable.
40 *
41 * It provides proxy functions for the streamer to call. The arguments
42 * for those calls are packed into messages, which are unpacked in the
43 * endpoint process and then used to call the real functions.
44 *
45 * Current IPC uses unix sockets. Can change this by creating a new
46 * implementations in openavb_enpoint_client.c and openavb_endpoint_server.c
47 */

67 static bool openavbEptClntReceiveFromServer(int h, openavbEndpointMessage_t *msg)
68 {
69  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
70 
71  if (!msg || h == AVB_ENDPOINT_HANDLE_INVALID) {
72  AVB_LOG_ERROR("Client receive; invalid argument passed");
73  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
74  return FALSE;
75  }
76 
77  switch(msg->type) {
78  case OPENAVB_ENDPOINT_TALKER_CALLBACK:
79  openavbEptClntNotifyTlkrOfSrpCb(h,
80  &msg->streamID,
81  msg->params.talkerCallback.ifname,
82  msg->params.talkerCallback.destAddr,
83  msg->params.talkerCallback.lsnrDecl,
84  msg->params.talkerCallback.srClass,
85  msg->params.talkerCallback.classRate,
86  msg->params.talkerCallback.vlanID,
87  msg->params.talkerCallback.priority,
88  msg->params.talkerCallback.fwmark);
89  break;
90  case OPENAVB_ENDPOINT_LISTENER_CALLBACK:
91  openavbEptClntNotifyLstnrOfSrpCb(h,
92  &msg->streamID,
93  msg->params.listenerCallback.ifname,
94  msg->params.listenerCallback.destAddr,
95  msg->params.listenerCallback.tlkrDecl,
96  &msg->params.listenerCallback.tSpec,
97  msg->params.listenerCallback.srClass,
98  msg->params.listenerCallback.latency,
99  &msg->params.listenerCallback.failInfo);
100  break;
101  case OPENAVB_ENDPOINT_VERSION_CALLBACK:
102  openavbEptClntCheckVerMatchesSrvr(h, msg->params.versionCallback.AVBVersion);
103  break;
104  default:
105  AVB_LOG_ERROR("Client receive: unexpected message");
106  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
107  return FALSE;
108  }
109  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
110  return TRUE;
111 }

113 bool openavbEptClntRegisterStream(int h,
114  AVBStreamID_t *streamID,
115  U8 destAddr[],
116  U8 noMaapAllocate,
117  AVBTSpec_t *tSpec,
118  U8 srClass,
119  U8 srRank,
120  U32 latency,
121  U32 txRate)
122 {
123  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
124  openavbEndpointMessage_t msgBuf;
125 
126  // Check for valid parameters. destAddr is optional and checked later in this function before using.
127  if (!streamID || !tSpec) {
128  AVB_LOG_ERROR("Client register: invalid argument");
129  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
130  return FALSE;
131  }
132 
133  memset(&msgBuf, 0, OPENAVB_ENDPOINT_MSG_LEN);
134  msgBuf.type = OPENAVB_ENDPOINT_TALKER_REGISTER;
135  memcpy(&(msgBuf.streamID), streamID, sizeof(AVBStreamID_t));
136  if (destAddr) {
137  memcpy(msgBuf.params.talkerRegister.destAddr, destAddr, ETH_ALEN);
138  msgBuf.params.talkerRegister.noMaapAllocate = noMaapAllocate;
139  }
140  msgBuf.params.talkerRegister.tSpec.maxFrameSize = tSpec->maxFrameSize;
141  msgBuf.params.talkerRegister.tSpec.maxIntervalFrames = tSpec->maxIntervalFrames;
142  msgBuf.params.talkerRegister.srClass = srClass;
143  msgBuf.params.talkerRegister.srRank = srRank;
144  msgBuf.params.talkerRegister.latency = latency;
145  msgBuf.params.talkerRegister.txRate = txRate;
146  bool ret = openavbEptClntSendToServer(h, &msgBuf);
147 
148  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
149  return ret;
150 }
