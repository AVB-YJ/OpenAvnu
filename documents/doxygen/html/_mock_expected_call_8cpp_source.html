327 SimpleString MockCheckedExpectedCall::callToString()
328 {
329  SimpleString str;
330  if (objectPtr_)
331  str = StringFromFormat("(object address: %p)::", objectPtr_);
332 
333  str += getName();
334  str += " -> ";
335  if (expectedCallOrder_ != NO_EXPECTED_CALL_ORDER) {
336  str += StringFromFormat("expected call order: <%d> -> ", expectedCallOrder_);
337  }
338 
339  if (inputParameters_->begin() == NULL && outputParameters_->begin() == NULL) {
340  str += (ignoreOtherParameters_) ? "all parameters ignored" : "no parameters";
341  return str;
342  }
343 
344  MockNamedValueListNode* p;
345 
346  for (p = inputParameters_->begin(); p; p = p->next()) {
347  str += StringFromFormat("%s %s: <%s>", p->getType().asCharString(), p->getName().asCharString(), getInputParameterValueString(p->getName()).asCharString());
348  if (p->next()) str += ", ";
349  }
350 
351  if (inputParameters_->begin() && outputParameters_->begin())
352  {
353  str += ", ";
354  }
355 
356  for (p = outputParameters_->begin(); p; p = p->next()) {
357  str += StringFromFormat("%s %s: <output>", p->getType().asCharString(), p->getName().asCharString());
358  if (p->next()) str += ", ";
359  }
360 
361  if (ignoreOtherParameters_)
362  str += ", other parameters are ignored";
363  return str;
364 }

366 SimpleString MockCheckedExpectedCall::missingParametersToString()

370 
371  for (p = inputParameters_->begin(); p; p = p->next()) {

375  }
376  }
377  for (p = outputParameters_->begin(); p; p = p->next()) {
378  if (! item(p)->isMatchingActualCall()) {
379  if (str != "") str += ", ";

381  }
