54 bool talkerStartStream(tl_state_t *pTLState)
55 {
56  AVB_TRACE_ENTRY(AVB_TRACE_TL);
57 
58  if (!pTLState) {
59  AVB_LOG_ERROR("Invalid TLState");
60  AVB_TRACE_EXIT(AVB_TRACE_TL);
61  return FALSE;
62  }
63 
64  openavb_tl_cfg_t *pCfg = &pTLState->cfg;
65  talker_data_t *pTalkerData = pTLState->pPvtTalkerData;
66 
67  assert(!pTLState->bStreaming);
68 
69  pTalkerData->wakeFrames = pCfg->max_interval_frames * pCfg->batch_factor;
70 
71  // Set a max_transmit_deficit_usec default
72  if (pCfg->max_transmit_deficit_usec == 0)
73  pCfg->max_transmit_deficit_usec = 50000;
74 
75  U32 transmitInterval = pTalkerData->classRate;
76  if (pCfg->map_cb.map_transmit_interval_cb(pTLState->pMediaQ)) {
77  // Override the class observation interval with the one provided by the mapping module.
78  transmitInterval = pCfg->map_cb.map_transmit_interval_cb(pTLState->pMediaQ);
79  }
80 
81  if (pCfg->intf_cb.intf_enable_fixed_timestamp) {
82  pCfg->intf_cb.intf_enable_fixed_timestamp(pTLState->pMediaQ, pCfg->fixed_timestamp, transmitInterval, pCfg->batch_factor);
83  } else if (pCfg->fixed_timestamp) {
84  AVB_LOG_ERROR("Fixed timestamp enabled but interface doesn't support it");
85  }
86 
87  openavbRC rc = openavbAvtpTxInit(pTLState->pMediaQ,
88  &pCfg->map_cb,
89  &pCfg->intf_cb,
90  pTalkerData->ifname,
91  &pTalkerData->streamID,
92  pTalkerData->destAddr,
93  pCfg->max_transit_usec,
94  pTalkerData->fwmark,
95  pTalkerData->vlanID,
96  pTalkerData->vlanPCP,
97  pTalkerData->wakeFrames * pCfg->raw_tx_buffers,
98  &pTalkerData->avtpHandle);
99  if (IS_OPENAVB_FAILURE(rc)) {
100  AVB_LOG_ERROR("Failed to create AVTP stream");
101  AVB_TRACE_EXIT(AVB_TRACE_TL);
102  return FALSE;
103  }
104 
105  avtp_stream_t *pStream = (avtp_stream_t *)(pTalkerData->avtpHandle);
106 
107  pTalkerData->wakeRate = transmitInterval / pCfg->batch_factor;
108 
109  pTalkerData->sleepUsec = MICROSECONDS_PER_SECOND / pTalkerData->wakeRate;
110  pTalkerData->intervalNS = NANOSECONDS_PER_SECOND / pTalkerData->wakeRate;
111 
112  U32 SRKbps = ((unsigned long)pTalkerData->classRate * (unsigned long)pCfg->max_interval_frames * (unsigned long)pStream->frameLen * 8L) / 1000;
113  U32 DataKbps = ((unsigned long)pTalkerData->wakeRate * (unsigned long)pCfg->max_interval_frames * (unsigned long)pStream->frameLen * 8L) / 1000;
114 
115  AVB_LOGF_INFO(STREAMID_FORMAT", sr-rate=%" PRIu32 ", data-rate=%lu, frames=%" PRIu16 ", size=%" PRIu16 ", batch=%" PRIu32 ", sleep=%" PRIu64 "us, sr-Kbps=%d, data-Kbps=%d",
116  STREAMID_ARGS(&pTalkerData->streamID), pTalkerData->classRate, pTalkerData->wakeRate,
117  pTalkerData->tSpec.maxIntervalFrames, pTalkerData->tSpec.maxFrameSize,
118  pCfg->batch_factor, pTalkerData->intervalNS / 1000, SRKbps, DataKbps);
119 
120 
121  // number of intervals per report
122  pTalkerData->wakesPerReport = pCfg->report_seconds * NANOSECONDS_PER_SECOND / pTalkerData->intervalNS;
123  // counts of intervals and frames between reports
124  pTalkerData->cntFrames = 0;
125  pTalkerData->cntWakes = 0;
126 
127  // setup the initial times
128  U64 nowNS;
129 
130  if (!pCfg->spin_wait) {
131  CLOCK_GETTIME64(OPENAVB_TIMER_CLOCK, &nowNS);
132  } else {
133  CLOCK_GETTIME64(OPENAVB_CLOCK_WALLTIME, &nowNS);
134  }
135 
136  // Align clock : allows for some performance gain
137  nowNS = ((nowNS + (pTalkerData->intervalNS)) / pTalkerData->intervalNS) * pTalkerData->intervalNS;
138 
139  pTalkerData->nextReportNS = nowNS + (pCfg->report_seconds * NANOSECONDS_PER_SECOND);
140  pTalkerData->lastReportFrames = 0;
141  pTalkerData->nextSecondNS = nowNS + NANOSECONDS_PER_SECOND;
142  pTalkerData->nextCycleNS = nowNS + pTalkerData->intervalNS;
143 
144  // Clear stats
145  openavbTalkerClearStats(pTLState);
146 
147  // we're good to go!
148  pTLState->bStreaming = TRUE;
149 
150  AVB_TRACE_EXIT(AVB_TRACE_TL);
151  return TRUE;
152 }

180 
181  AVB_LOGF_INFO("TX "STREAMID_FORMAT", Totals: calls=%" PRIu64 ", frames=%" PRIu64 ", late=%" PRIu64 ", bytes=%" PRIu64 ", TXOutOfBuffs=%ld",
182  STREAMID_ARGS(&pTalkerData->streamID),
183  openavbTalkerGetStat(pTLState, TL_STAT_TX_CALLS),
184  openavbTalkerGetStat(pTLState, TL_STAT_TX_FRAMES),
185  openavbTalkerGetStat(pTLState, TL_STAT_TX_LATE),
186  openavbTalkerGetStat(pTLState, TL_STAT_TX_BYTES),
187  rawsock ? openavbRawsockGetTXOutOfBuffers(rawsock) : 0
188  );
189 
190  if (pTLState->bStreaming) {
191  openavbAvtpShutdownTalker(pTalkerData->avtpHandle);
192  pTLState->bStreaming = FALSE;
193  }
194 
195  AVB_TRACE_EXIT(AVB_TRACE_TL);
196 }

198 static inline void talkerShowStats(talker_data_t *pTalkerData, tl_state_t *pTLState)
199 {
200  S32 late = pTalkerData->wakesPerReport - pTalkerData->cntWakes;
201  U64 bytes = openavbAvtpBytes(pTalkerData->avtpHandle);
202  if (late < 0) late = 0;
203  U32 txbuf = openavbAvtpTxBufferLevel(pTalkerData->avtpHandle);
204  U32 mqbuf = openavbMediaQCountItems(pTLState->pMediaQ, TRUE);
205 
206  AVB_LOGRT_INFO(LOG_RT_BEGIN, LOG_RT_ITEM, FALSE, "TX UID:%d, ", LOG_RT_DATATYPE_U16, &pTalkerData->streamID.uniqueID);
207  AVB_LOGRT_INFO(FALSE, LOG_RT_ITEM, FALSE, "calls=%ld, ", LOG_RT_DATATYPE_U32, &pTalkerData->cntWakes);
208  AVB_LOGRT_INFO(FALSE, LOG_RT_ITEM, FALSE, "frames=%ld, ", LOG_RT_DATATYPE_U32, &pTalkerData->cntFrames);
209  AVB_LOGRT_INFO(FALSE, LOG_RT_ITEM, FALSE, "late=%d, ", LOG_RT_DATATYPE_U32, &late);
210  AVB_LOGRT_INFO(FALSE, LOG_RT_ITEM, FALSE, "bytes=%lld, ", LOG_RT_DATATYPE_U64, &bytes);
211  AVB_LOGRT_INFO(FALSE, LOG_RT_ITEM, FALSE, "txbuf=%d, ", LOG_RT_DATATYPE_U32, &txbuf);
212  AVB_LOGRT_INFO(FALSE, LOG_RT_ITEM, LOG_RT_END, "mqbuf=%d, ", LOG_RT_DATATYPE_U32, &mqbuf);
213 
214  openavbTalkerAddStat(pTLState, TL_STAT_TX_LATE, late);
215  openavbTalkerAddStat(pTLState, TL_STAT_TX_BYTES, bytes);
216 }

218 static inline bool talkerDoStream(tl_state_t *pTLState)
219 {
220  AVB_TRACE_ENTRY(AVB_TRACE_TL);
221 
222  if (!pTLState) {
223  AVB_LOG_ERROR("Invalid TLState");
224  AVB_TRACE_EXIT(AVB_TRACE_TL);
225  return FALSE;
226  }
227 
228  openavb_tl_cfg_t *pCfg = &pTLState->cfg;
229  talker_data_t *pTalkerData = pTLState->pPvtTalkerData;
230  bool bRet = FALSE;
231 
232  if (pTLState->bStreaming) {
233  U64 nowNS;
234 
235  if (!pCfg->tx_blocking_in_intf) {
236 
237  if (!pCfg->spin_wait) {
238  // sleep until the next interval
239  SLEEP_UNTIL_NSEC(pTalkerData->nextCycleNS);
240  } else {
241 #if !IGB_LAUNCHTIME_ENABLED
242  SPIN_UNTIL_NSEC(pTalkerData->nextCycleNS);
243 #endif
244  }
245 
246  //AVB_DBG_INTERVAL(8000, TRUE);
247 
248  // send the frames for this interval
249  int i;
250  for (i = pTalkerData->wakeFrames; i > 0; i--) {
251  if (IS_OPENAVB_SUCCESS(openavbAvtpTx(pTalkerData->avtpHandle, i == 1, pCfg->tx_blocking_in_intf)))
252  pTalkerData->cntFrames++;
253  else
254  break;
255  }
256  }
257  else {
258  // Interface module block option
259  if (IS_OPENAVB_SUCCESS(openavbAvtpTx(pTalkerData->avtpHandle, TRUE, pCfg->tx_blocking_in_intf)))
260  pTalkerData->cntFrames++;
261  }
262 
263  if (!pCfg->spin_wait) {
264  CLOCK_GETTIME64(OPENAVB_TIMER_CLOCK, &nowNS);
265  } else {
266  CLOCK_GETTIME64(OPENAVB_CLOCK_WALLTIME, &nowNS);
267  }
268 
269  if (pTalkerData->cntWakes++ % pTalkerData->wakeRate == 0) {
270  // time to service the endpoint IPC
271  bRet = TRUE;
272 
273  // Don't need to check again for another second.
274  pTalkerData->nextSecondNS = nowNS + NANOSECONDS_PER_SECOND;
275  }
276 
277  if (pCfg->report_seconds > 0) {
278  if (nowNS > pTalkerData->nextReportNS) {
279  talkerShowStats(pTalkerData, pTLState);
280 
281  openavbTalkerAddStat(pTLState, TL_STAT_TX_CALLS, pTalkerData->cntWakes);
282  openavbTalkerAddStat(pTLState, TL_STAT_TX_FRAMES, pTalkerData->cntFrames);
283 
284  pTalkerData->cntFrames = 0;
285  pTalkerData->cntWakes = 0;
286  pTalkerData->nextReportNS = nowNS + (pCfg->report_seconds * NANOSECONDS_PER_SECOND);
287  }
288  } else if (pCfg->report_frames > 0 && pTalkerData->cntFrames != pTalkerData->lastReportFrames) {
289  if (pTalkerData->cntFrames % pCfg->report_frames == 1) {
290  talkerShowStats(pTalkerData, pTLState);
291  pTalkerData->lastReportFrames = pTalkerData->cntFrames;
292  }
293  }
294 
295  if (nowNS > pTalkerData->nextSecondNS) {
296  pTalkerData->nextSecondNS = nowNS + NANOSECONDS_PER_SECOND;
297  bRet = TRUE;
298  }
299 
300  if (!pCfg->tx_blocking_in_intf) {
301  pTalkerData->nextCycleNS += pTalkerData->intervalNS;
302 
303  if ((pTalkerData->nextCycleNS + (pCfg->max_transmit_deficit_usec * 1000)) < nowNS) {
304  // Hit max deficit time. Something must be wrong. Reset the cycle timer.
305  // Align clock : allows for some performance gain
306  nowNS = ((nowNS + (pTalkerData->intervalNS)) / pTalkerData->intervalNS) * pTalkerData->intervalNS;
307  pTalkerData->nextCycleNS = nowNS + pTalkerData->intervalNS;
308  }
309  }
310  }
311  else {
312  SLEEP_MSEC(10);
313 
314  // time to service the endpoint IPC
315  bRet = TRUE;
316  }
317 
318  AVB_TRACE_EXIT(AVB_TRACE_TL);
319  return bRet;
320 }

323 // Called from openavbTLThreadFn() which is started from openavbTLRun()
324 void openavbTLRunTalker(tl_state_t *pTLState)
325 {
326  AVB_TRACE_ENTRY(AVB_TRACE_TL);
327 
328  if (!pTLState) {
329  AVB_LOG_ERROR("Invalid TLState");
330  AVB_TRACE_EXIT(AVB_TRACE_TL);
331  return;
332  }
333 
334  pTLState->pPvtTalkerData = calloc(1, sizeof(talker_data_t));
335  if (!pTLState->pPvtTalkerData) {
336  AVB_LOG_WARNING("Failed to allocate talker data.");
337  return;
338  }
339 
340  // Create Stats Mutex
341  {
342  MUTEX_ATTR_HANDLE(mta);
343  MUTEX_ATTR_INIT(mta);
344  MUTEX_ATTR_SET_TYPE(mta, MUTEX_ATTR_TYPE_DEFAULT);
345  MUTEX_ATTR_SET_NAME(mta, "TLStatsMutex");
346  MUTEX_CREATE_ERR();
347  MUTEX_CREATE(pTLState->statsMutex, mta);
348  MUTEX_LOG_ERR("Could not create/initialize 'TLStatsMutex' mutex");
349  }
350 
351  /* If using endpoint register talker,
352  else register with tpsec */
353  pTLState->bConnected = openavbTLRunTalkerInit(pTLState);
354 
355  if (pTLState->bConnected) {
356  bool bServiceIPC;
357 
358  // Notify AVDECC Msg of the state change.
359  openavbAvdeccMsgClntNotifyCurrentState(pTLState);
360 
361  // Do until we are stopped or lose connection to endpoint
362  while (pTLState->bRunning && pTLState->bConnected) {
363 
364  // Talk (or just sleep if not streaming.)
365  bServiceIPC = talkerDoStream(pTLState);
366 
367  // TalkerDoStream() returns TRUE occasionally,
368  // so that we can service our IPC at that low rate.
369  if (bServiceIPC) {
370  // Look for messages from endpoint. Don't block (timeout=0)
371  if (!openavbEptClntService(pTLState->endpointHandle, 0)) {
372  AVB_LOGF_WARNING("Lost connection to endpoint, will retry "STREAMID_FORMAT, STREAMID_ARGS(&(((talker_data_t *)pTLState->pPvtTalkerData)->streamID)));
373  pTLState->bConnected = FALSE;
374  pTLState->endpointHandle = 0;
375  }
376  }
377  }
378 
379  // Stop streaming
380  talkerStopStream(pTLState);
381 
382  {
383  MUTEX_CREATE_ERR();
384  MUTEX_DESTROY(pTLState->statsMutex); // Destroy Stats Mutex
385  MUTEX_LOG_ERR("Error destroying mutex");
386  }
387 
388  // withdraw our talker registration
389  if (pTLState->bConnected)
390  openavbEptClntStopStream(pTLState->endpointHandle, &(((talker_data_t *)pTLState->pPvtTalkerData)->streamID));
391 
392  openavbTLRunTalkerFinish(pTLState);
393 
394  // Notify AVDECC Msg of the state change.
395  openavbAvdeccMsgClntNotifyCurrentState(pTLState);
396  }
397  else {
398  AVB_LOGF_WARNING("Failed to connect to endpoint"STREAMID_FORMAT, STREAMID_ARGS(&(((talker_data_t *)pTLState->pPvtTalkerData)->streamID)));
399  }
400 
401  if (pTLState->pPvtTalkerData) {
402  free(pTLState->pPvtTalkerData);
403  pTLState->pPvtTalkerData = NULL;
404  }
405 
406  AVB_TRACE_EXIT(AVB_TRACE_TL);
407 }
