132 int igb_attach(char *dev_path, device_t *pdev)
133 {
134  struct adapter *adapter;
135  struct igb_bind_cmd bind = {0};
136  int error = 0;
137  bool locked = false;
138 
139  if (pdev == NULL)
140  return -EINVAL;
141 
142  adapter = (struct adapter *)pdev->private_data;
143 
144  if (adapter != NULL)
145  return -EBUSY;
146 
147  /* allocate an adapter */
148  pdev->private_data = malloc(sizeof(struct adapter));
149  if (pdev->private_data == NULL)
150  return -ENXIO;
151 
152  memset(pdev->private_data, 0, sizeof(struct adapter));
153 
154  adapter = (struct adapter *)pdev->private_data;
155 
156  adapter->ldev = open("/dev/igb_avb", O_RDWR);
157 
158  if (adapter->ldev < 0) {
159  error = -ENXIO;
160  goto err_prebind;
161  }
162 
163  if (igb_create_lock(adapter) != 0) {
164  error = -errno;
165  goto err_bind;
166  }
167 
168  adapter->active = 1;
169 
170  if (igb_lock(pdev) != 0) {
171  error = -errno;
172  goto err_bind;
173  }
174 
175  locked = true;
176 
177  /*
178  * dev_path should look something "0000:01:00.0"
179  */
180 
181  strncpy(bind.iface, dev_path, IGB_BIND_NAMESZ - 1);
182 
183  if (ioctl(adapter->ldev, IGB_BIND, &bind) < 0) {
184  error = -ENXIO;
185  goto err_bind;
186  }
187 
188  adapter->csr.paddr = 0;
189  adapter->csr.mmap_size = bind.mmap_size;
190 
191  /* Determine hardware and mac info */
192  adapter->hw.vendor_id = pdev->pci_vendor_id;
193  adapter->hw.device_id = pdev->pci_vendor_id;
194  adapter->hw.revision_id = 0;
195  adapter->hw.subsystem_vendor_id = 0;
196  adapter->hw.subsystem_device_id = 0;
197 
198  /* Set MAC type early for PCI setup */
199  adapter->hw.mac.type = e1000_i210;
200 
201  /* Setup PCI resources */
202  error = igb_allocate_pci_resources(adapter);
203  if (error)
204  goto err_pci;
205 
206  /*
207  * Set the frame limits assuming
208  * standard ethernet sized frames.
209  */
210  adapter->max_frame_size = 1518;
211  adapter->min_frame_size = 64;
212 
213  /*
214  * Copy the permanent MAC address out of the EEPROM
215  */
216  if (igb_read_mac_addr(&adapter->hw) < 0) {
217  error = -EIO;
218  goto err_late;
219  }
220 
221  if (igb_unlock(pdev) != 0) {
222  error = -errno;
223  goto err_gen;
224  }
225 
226  return 0;
227 
228 err_late:
229 err_pci:
230  igb_free_pci_resources(adapter);
231 err_bind:
232  if (locked)
233  (void) igb_unlock(pdev);
234  if (adapter && adapter->memlock) {
235  (void) munmap(adapter->memlock, sizeof(pthread_mutex_t));
236  adapter->memlock = NULL;
237  }
238  close(adapter->ldev);
239 err_prebind:
240  free(pdev->private_data);
241  pdev->private_data = NULL;
242 
243 err_gen:
244  return error;
245 }

318 int igb_detach(device_t *dev)
319 {
320  struct adapter *adapter;
321 
322  if (dev == NULL)
323  return -EINVAL;
324  adapter = (struct adapter *)dev->private_data;
325  if (adapter == NULL)
326  return -ENXIO;
327 
328  if (igb_lock(dev) != 0)
329  goto err_nolock;
330 
331  /*
332  * Prevent access to device after calling igb_detach since associated
333  * resources will be freed up from here thus in particular multi-thread
334  * application other thread must not access device if this flag is off.
335  */
336  adapter->active = 0;
337 
338  igb_reset(adapter);
339 
340  igb_unlock(dev);
341 
342  igb_free_pci_resources(adapter);
343 
344  if (adapter->tx_rings)
345  igb_free_transmit_structures(adapter);
346 
347  if (adapter->rx_rings)
348  igb_free_receive_structures(adapter);
349 
350 err_nolock:
351  if (adapter->memlock) {
352  /*
353  * Do not unmap the shared memory region holding the pthread mutex.
354  *
355  * (void) munmap(adapter->memlock, sizeof(pthread_mutex_t));
356  *
357  * The pthread mutex is configured as a robust type mutex so that
358  * it can automatically be unlocked on process termination if needed.
359  * In order to complete the cleanup, the memory region holding the
360  * mutex instance must be accessible until that cleanup timing.
361  * Therefore we should not unmap the memory region here, otherwise
362  * the cleanup may fail.
363  *
364  * The mapped regions will automatically be unmapped at the end of
365  * the process termination.
366  */
367  adapter->memlock = NULL;
368  }
369 
370  close(adapter->ldev);
371 
372  free(dev->private_data);
373  dev->private_data = NULL;
374  return 0;
375 }

377 int igb_suspend(device_t *dev)
378 {
379  struct adapter *adapter;
380  struct tx_ring *txr;
381  struct rx_ring *rxr;
382  struct e1000_hw *hw;
383  u32 txdctl, srrctl = 0;
384  int i;
385 
386  if (dev == NULL)
387  return -EINVAL;
388  adapter = (struct adapter *)dev->private_data;
389  if (adapter == NULL)
390  return -ENXIO;
391 
392  txr = adapter->tx_rings;
393  rxr = adapter->rx_rings;
394  hw = &adapter->hw;
395 
396  txdctl = 0;
397 
398  if (igb_lock(dev) != 0)
399  return errno;
400 
401  /* stop but don't reset the Tx Descriptor Rings */
402  for (i = 0; i < adapter->num_queues; i++, txr++) {
403  txdctl |= IGB_TX_PTHRESH;
404  txdctl |= IGB_TX_HTHRESH << 8;
405  txdctl |= IGB_TX_WTHRESH << 16;
406  E1000_WRITE_REG(hw, E1000_TXDCTL(i), txdctl);
407  txr->queue_status = IGB_QUEUE_IDLE;
408  }
409 
410  srrctl |= 2048 >> E1000_SRRCTL_BSIZEPKT_SHIFT;
411  srrctl |= E1000_SRRCTL_DESCTYPE_ADV_ONEBUF;
412 
413  for (i = 0; i < adapter->num_queues; i++, rxr++) {
414  u64 bus_addr = rxr->rxdma.paddr;
415  u32 rxdctl;
416 
417  E1000_WRITE_REG(hw, E1000_RDLEN(i),
418  adapter->num_rx_desc *
419  sizeof(union e1000_adv_rx_desc));
420  E1000_WRITE_REG(hw, E1000_RDBAH(i),
421  (uint32_t)(bus_addr >> 32));
422  E1000_WRITE_REG(hw, E1000_RDBAL(i),
423  (uint32_t)bus_addr);
424  E1000_WRITE_REG(hw, E1000_SRRCTL(i), srrctl);
425  /* Enable this Queue */
426  rxdctl = E1000_READ_REG(hw, E1000_RXDCTL(i));
427  rxdctl |= E1000_RXDCTL_QUEUE_ENABLE;
428  rxdctl &= 0xFFF00000;
429  rxdctl |= IGB_RX_PTHRESH;
430  rxdctl |= IGB_RX_HTHRESH << 8;
431  rxdctl |= IGB_RX_WTHRESH << 16;
432  E1000_WRITE_REG(hw, E1000_RXDCTL(i), rxdctl);
433  }
434 
435  if (igb_unlock(dev) != 0)
436  return errno;
437 
438  return 0;
439 }

540 static void
541 igb_reset(struct adapter *adapter)
542 {
543  struct tx_ring *txr = adapter->tx_rings;
544  struct rx_ring *rxr = adapter->rx_rings;
545  struct e1000_hw *hw = &adapter->hw;
546  u32 txdctl, srrctl;
547  int i;
548 
549  srrctl = 0;
550  txdctl = 0;
551 
552  /* Set up the Tx Descriptor Rings, leave queues idle */
553  if (adapter->tx_rings == NULL) {
554 #if DEBUG
555  printf("txr null\n");
556 #endif
557  } else {
558  for (i = 0; i < adapter->num_queues; i++, txr++) {
559  u64 bus_addr = txr->txdma.paddr;
560 
561  /* idle the queue */
562  txdctl |= IGB_TX_PTHRESH;
563  txdctl |= IGB_TX_HTHRESH << 8;
564  txdctl |= IGB_TX_WTHRESH << 16;
565  E1000_WRITE_REG(hw, E1000_TXDCTL(i), txdctl);
566 
567  /* reset the descriptor head/tail */
568  E1000_WRITE_REG(hw, E1000_TDLEN(i),
569  adapter->num_tx_desc *
570  sizeof(struct e1000_tx_desc));
571  E1000_WRITE_REG(hw, E1000_TDBAH(i),
572  (u_int32_t)(bus_addr >> 32));
573  E1000_WRITE_REG(hw, E1000_TDBAL(i),
574  (u_int32_t)bus_addr);
575 
576  /* Setup the HW Tx Head and Tail descriptor pointers */
577  E1000_WRITE_REG(hw, E1000_TDT(i), 0);
578  E1000_WRITE_REG(hw, E1000_TDH(i), 0);
579 
580  txr->queue_status = IGB_QUEUE_IDLE;
581  }
582  }
583 
584  srrctl |= 2048 >> E1000_SRRCTL_BSIZEPKT_SHIFT;
585  srrctl |= E1000_SRRCTL_DESCTYPE_ADV_ONEBUF;
586 
587  /* Setup the Base and Length of the Rx Descriptor Rings */
588  if (adapter->rx_rings == NULL) {
589 #if DEBUG
590  printf("rxr null\n");
591 #endif
592  } else {
593 
594  for (i = 0; i < adapter->num_queues; i++, rxr++) {
595  u64 bus_addr = rxr->rxdma.paddr;
596  u32 rxdctl;
597 
598  /* Disable this Queue */
599  E1000_WRITE_REG(hw, E1000_RXDCTL(i), 0);
600 
601  E1000_WRITE_REG(hw, E1000_RDLEN(i),
602  adapter->num_rx_desc *
603  sizeof(union e1000_adv_rx_desc));
604  E1000_WRITE_REG(hw, E1000_RDBAH(i),
605  (uint32_t)(bus_addr >> 32));
606  E1000_WRITE_REG(hw, E1000_RDBAL(i),
607  (uint32_t)bus_addr);
608  E1000_WRITE_REG(hw, E1000_SRRCTL(i), srrctl);
609 
610  /* Enable this Queue */
611  rxdctl = E1000_READ_REG(hw, E1000_RXDCTL(i));
612  rxdctl |= E1000_RXDCTL_QUEUE_ENABLE;
613  rxdctl &= 0xFFF00000;
614  rxdctl |= IGB_RX_PTHRESH;
615  rxdctl |= IGB_RX_HTHRESH << 8;
616  rxdctl |= IGB_RX_WTHRESH << 16;
617  E1000_WRITE_REG(hw, E1000_RXDCTL(i), rxdctl);
618  }
619  }

622 static int igb_read_mac_addr(struct e1000_hw *hw)
623 {
624  u32 rar_high;
625  u32 rar_low;
626  u16 i;
627 
628  rar_high = E1000_READ_REG(hw, E1000_RAH(0));
629  rar_low = E1000_READ_REG(hw, E1000_RAL(0));
630 
631  for (i = 0; i < E1000_RAL_MAC_ADDR_LEN; i++)
632  hw->mac.perm_addr[i] = (u8)(rar_low >> (i*8));
633 
634  for (i = 0; i < E1000_RAH_MAC_ADDR_LEN; i++)
635  hw->mac.perm_addr[i+4] = (u8)(rar_high >> (i*8));
636 
637  for (i = 0; i < ETH_ADDR_LEN; i++)
638  hw->mac.addr[i] = hw->mac.perm_addr[i];
639 
640  return 0;
641 
642 }

965 /*********************************************************************
966  *
967  * This routine maps a single buffer to an Advanced TX descriptor.
968  * returns ENOSPC if we run low on tx descriptors and the app needs to
969  * cleanup descriptors.
970  *
971  * this is a simplified routine which doesn't do LSO, checksum offloads,
972  * multiple fragments, etc. The provided buffers are assumed to have
973  * been previously mapped with the provided dma_malloc_page routines.
974  *
975  **********************************************************************/
976 int igb_xmit(device_t *dev, unsigned int queue_index, struct igb_packet *packet)
977 {
978  struct adapter *adapter;
979  struct tx_ring *txr;
980  struct igb_tx_buffer *tx_buffer;
981  union e1000_adv_tx_desc *txd = NULL;
982  u32 cmd_type_len, olinfo_status = 0;
983  int i, first, last = 0;
984  int error = 0;
985 
986  if (dev == NULL)
987  return -EINVAL;
988 
989  adapter = (struct adapter *)dev->private_data;
990  if (adapter == NULL)
991  return -ENXIO;
992 
993  txr = &adapter->tx_rings[queue_index];
994  if (!txr)
995  return -EINVAL;
996 
997  if (queue_index > adapter->num_queues)
998  return -EINVAL;
999 
1000  if (packet == NULL)
1001  return -EINVAL;
1002 
1003  if (igb_lock(dev) != 0)
1004  return errno;
1005 
1006  packet->next = NULL; /* used for cleanup */
1007 
1008  /* Set basic descriptor constants */
1009  cmd_type_len = E1000_ADVTXD_DTYP_DATA;
1010  cmd_type_len |= E1000_ADVTXD_DCMD_IFCS | E1000_ADVTXD_DCMD_DEXT;
1011 
1012  /* cmd_type_len |= E1000_ADVTXD_DCMD_VLE; to enable VLAN insertion */
1013 
1014  /*
1015  * Map the packet for DMA
1016  *
1017  * Capture the first descriptor index,
1018  * this descriptor will have the index
1019  * of the EOP which is the only one that
1020  * now gets a DONE bit writeback.
1021  */
1022  first = txr->next_avail_desc;
1023  tx_buffer = &txr->tx_buffers[first];
1024 
1025  /*
1026  * Make sure we don't overrun the ring,
1027  * we need nsegs descriptors and one for
1028  * the context descriptor used for the
1029  * offloads.
1030  */
1031  if (txr->tx_avail <= 2) {
1032  error = ENOSPC;
1033  goto unlock;
1034  }
1035 
1036  /*
1037  * Set up the context descriptor to specify
1038  * launchtimes for the packet.
1039  */
1040  igb_tx_ctx_setup(txr, packet);
1041 
1042  /*
1043  * for performance monitoring, report the DMA time of the tx desc wb
1044  */
1045  olinfo_status |= E1000_TXD_DMA_TXDWB;
1046 
1047  /* set payload length */
1048  olinfo_status |= packet->len << E1000_ADVTXD_PAYLEN_SHIFT;
1049 
1050  /* Set up our transmit descriptors */
1051  i = txr->next_avail_desc;
1052 
1053  /* we assume every packet is contiguous */
1054 
1055  tx_buffer = &txr->tx_buffers[i];
1056  txd = (union e1000_adv_tx_desc *)&txr->tx_base[i];
1057 
1058  txd->read.buffer_addr = htole64(packet->map.paddr + packet->offset);
1059  txd->read.cmd_type_len = htole32(cmd_type_len | packet->len);
1060  txd->read.olinfo_status = htole32(olinfo_status);
1061  last = i;
1062  if (++i == adapter->num_tx_desc)
1063  i = 0;
1064  tx_buffer->packet = NULL;
1065  tx_buffer->next_eop = -1;
1066 
1067  txr->next_avail_desc = i;
1068  txr->tx_avail--;
1069  tx_buffer->packet = packet;
1070 
1071  /*
1072  * Last Descriptor of Packet
1073  * needs End Of Packet (EOP)
1074  * and Report Status (RS)
1075  */
1076  txd->read.cmd_type_len |=
1077  htole32(E1000_ADVTXD_DCMD_EOP | E1000_ADVTXD_DCMD_RS);
1078 
1079  /*
1080  * Keep track in the first buffer which
1081  * descriptor will be written back
1082  */
1083  tx_buffer = &txr->tx_buffers[first];
1084  tx_buffer->next_eop = last;
1085 
1086  /*
1087  * Advance the Transmit Descriptor Tail (TDT), this tells the E1000
1088  * that this frame is available to transmit.
1089  */
1090 
1091  E1000_WRITE_REG(&adapter->hw, E1000_TDT(txr->me), i);
1092  ++txr->tx_packets;
1093 
1094 unlock:
1095  if (igb_unlock(dev) != 0)
1096  return errno;
1097 
1098  return error;
1099 }

1701 /**********************************************************************
1702  *
1703  * Examine each rx_buffer in the used queue. If the hardware is done
1704  * processing the packet then return the linked list of associated resources.
1705  *
1706  **********************************************************************/
1707 int igb_receive(device_t *dev, unsigned int queue_index,
1708  struct igb_packet **received_packets, u_int32_t *count)
1709 {
1710  struct adapter *adapter;
1711  struct rx_ring *rxr;
1712  union e1000_adv_rx_desc *cur;
1713  bool eop = FALSE;
1714  u_int32_t staterr = 0;
1715  u_int32_t desc = 0;
1716  u_int32_t max_pkt = 0;
1717  struct igb_packet *curr_pkt = NULL;
1718  struct igb_packet *prev_pkt = NULL;
1719 
1720  if (dev == NULL)
1721  return -EINVAL;
1722 
1723  adapter = (struct adapter *)dev->private_data;
1724  if (adapter == NULL)
1725  return -ENXIO;
1726 
1727  if (adapter->active != 1) // detach in progress
1728  return -ENXIO;
1729 
1730  if (queue_index > adapter->num_queues)
1731  return -EINVAL;
1732 
1733  rxr = &(adapter->rx_rings[queue_index]);
1734  if (!rxr)
1735  return -EINVAL;
1736 
1737  if (count == NULL)
1738  return -EINVAL;
1739 
1740  max_pkt = *count;
1741  *count = 0;
1742 
1743  if (received_packets == NULL)
1744  return -EINVAL;
1745 
1746  *received_packets = NULL; /* nothing reclaimed yet */
1747 
1748  if (sem_trywait(&rxr->lock) != 0)
1749  return errno; /* EAGAIN */
1750 
1751  if (adapter->active != 1) {
1752  sem_post(&rxr->lock);
1753  return -ENXIO;
1754  }
1755 
1756  /* Main clean loop - receive packets until no more
1757  * received_packets[]
1758  */
1759  for (desc = rxr->next_to_check; *count < max_pkt;) {
1760  cur = &(rxr->rx_base[desc]);
1761 #ifdef DEBUG
1762  if (i%2)
1763  printf("\033[2A");
1764 
1765  printf("desc.status_error=%x desc.length=%x desc.vlan=%x desc.rss=%x desc.pkt_info=%x desc.hdr_info=%x\n",
1766  cur->wb.upper.status_error,
1767  cur->wb.upper.length,
1768  cur->wb.upper.vlan,
1769  cur->wb.lower.hi_dword.rss,
1770  cur->wb.lower.lo_dword.hs_rss.pkt_info,
1771  cur->wb.lower.lo_dword.hs_rss.hdr_info);
1772 #endif
1773  staterr = le32toh(cur->wb.upper.status_error);
1774  if ((staterr & E1000_RXD_STAT_DD) == 0)
1775  break;
1776 
1777  cur->wb.upper.status_error = 0;
1778 
1779  eop = ((staterr & E1000_RXD_STAT_EOP) ==
1780  E1000_RXD_STAT_EOP);
1781 
1782  if (eop) {
1783  /*
1784  * Free the frame (all segments) if we're at EOP and
1785  * it's an error.
1786  *
1787  * The datasheet states that EOP + status is only valid
1788  * for the final segment in a multi-segment frame.
1789  */
1790  if (staterr & E1000_RXDEXT_ERR_FRAME_ERR_MASK) {
1791  ++rxr->rx_discarded;
1792  printf ("discard error packet\n");
1793  igb_refresh_buffers(dev, queue_index,
1794  &rxr->rx_buffers[desc].packet, 1);
1795  } else {
1796  /*
1797  * add new packet to list of received packets
1798  * to return
1799  */
1800  curr_pkt = rxr->rx_buffers[desc].packet;
1801  curr_pkt->len = cur->wb.upper.length;
1802 
1803  if (*received_packets == NULL)
1804  *received_packets = curr_pkt;
1805  if (prev_pkt)
1806  prev_pkt->next = curr_pkt;
1807  prev_pkt = curr_pkt;
1808  (*count)++;
1809 
1810  ++rxr->rx_packets;
1811  }
1812  } else {
1813  /* multi-segment frame is not supported yet */
1814  ++rxr->rx_discarded;
1815  printf ("discard non-eop packet\n");
1816  igb_refresh_buffers(dev, queue_index,
1817  &rxr->rx_buffers[desc].packet, 1);
1818  }
1819 next_desc:
1820  /* Advance our pointers to the next descriptor. */
1821  if (++desc == adapter->num_rx_desc)
1822  desc = 0;
1823  }
1824 
1825  rxr->next_to_check = desc;
1826 
1827  if (sem_post(&rxr->lock) != 0)
1828  return errno;
1829 
1830  if (*received_packets == NULL) {
1831  /* nothing reclaimed yet */
1832  errno = EAGAIN;
1833  return errno;
1834  }
1835 
1836  return 0;
1837 }

1928  /* borrow */
1929  a->tv_nsec += 1000000000;

1931  }
