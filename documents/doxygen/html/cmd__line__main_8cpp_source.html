125 char * command_generator(const char * text, int state)
126 {
127  static std::list<std::string> completion_options;
128  static int len;
129 
130  if (!current_command)
131  return NULL;
132 
133  if (!state)
134  {
135  // New word to complete then set up the options to match
136 
137  // Cache the len for efficiency
138  len = strlen(text);
139 
140  // Try the sub-commands of the current command
141  completion_options = current_command->get_sub_command_names();
142  if (!completion_options.size())
143  {
144  // If there are no sub-commands then try the arguments
145  std::vector<cli_argument *> args;
146  std::set<std::string> arg_options;
147 
148  // There can be multiple arguments at a given index as there
149  // can be multiple command formats
150  current_command->get_args(complettion_arg_index, args);
151  for (std::vector<cli_argument *>::iterator iter = args.begin();
152  iter != args.end();
153  ++iter)
154  {
155  (*iter)->get_completion_options(arg_options);
156  }
157  completion_options.insert(completion_options.end(),
158  arg_options.begin(), arg_options.end());
159  }
160  }
161 
162  // Return the next name which matches from the command list
163  while (completion_options.size())
164  {
165  std::string sub_command = completion_options.front();
166  completion_options.pop_front();
167 
168  if (strncmp(sub_command.c_str(), text, len) == 0)
169  return (strdup(sub_command.c_str()));
170  }
171 
172  // There are no matches
173  return NULL;
174 }

176 char ** command_completer(const char * text, int start, int end)
177 {
178  if (start == 0)
179  {
180  // Start of a new command
181  current_command = top_level_command;
182  }
183  else
184  {
185  // In the middle of a command line, use the rest of the line
186  // to find the right command to provide completion options
187  std::string cmd_path(rl_line_buffer);
188  cmd_path = cmd_path.substr(0, start);
189  std::queue<std::string, std::deque<std::string>> cmd_path_queue;
190  split(cmd_path, ' ', cmd_path_queue);
191 
192  std::string prefix;
193  current_command = top_level_command->get_sub_command(cmd_path_queue, prefix);
194 
195  // There can be remaining parts of the command which mean that an argument
196  // value is being completed instead
197  complettion_arg_index = cmd_path_queue.size();
198  }
199 
200  char ** matches = rl_completion_matches(text, command_generator);
201  return matches;
202 }
