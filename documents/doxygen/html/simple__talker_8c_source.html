151 uint16_t inet_checksum_sg( struct iovec *buf_iov, size_t buf_iovlen ){
152  size_t i;
153  uint32_t sum = 0; /* assume 32 bit long, 16 bit short */
154  uint8_t residual;
155  int has_residual = 0;
156 
157  for( i = 0; i < buf_iovlen; ++i,++buf_iov ) {
158  if( has_residual ) {
159  if( buf_iov->iov_len > 0 ) {
160  if(sum & 0x80000000) /* if high order bit set, fold */
161  sum = (sum & 0xFFFF) + (sum >> 16);
162  sum += residual | (*(( uint8_t *) buf_iov->iov_base) << 8);
163  buf_iov->iov_base += 1;
164  buf_iov->iov_len -= 1;
165  } else {
166  if(sum & 0x80000000) /* if high order bit set, fold */
167  sum = (sum & 0xFFFF) + (sum >> 16);
168  sum += (uint16_t) residual;
169  }
170  has_residual = 0;
171 
172  }
173  while(buf_iov->iov_len > 1){
174  if(sum & 0x80000000) /* if high order bit set, fold */
175  sum = (sum & 0xFFFF) + (sum >> 16);
176  sum += *(( uint16_t *) buf_iov->iov_base); buf_iov->iov_base += 2;
177  buf_iov->iov_len -= 2;
178  }
179  if( buf_iov->iov_len ) {
180  residual = *(( uint8_t *) buf_iov->iov_base);
181  has_residual = 1;
182  }
183  }
184  if( has_residual ) {
185  sum += (uint16_t) residual;
186  }
187 
188  while(sum>>16)
189  sum = (sum & 0xFFFF) + (sum >> 16);
190 
191  return ~sum;
192 }
