62 int STDCALL audio_unit_descriptor_imp::send_set_sampling_rate_cmd(void * notification_id, uint32_t new_sampling_rate)
63 {
64  struct jdksavdecc_frame cmd_frame;
65  struct jdksavdecc_aem_command_set_sampling_rate aem_cmd_set_sampling_rate;
66  ssize_t aem_cmd_set_sampling_rate_returned;
67  memset(&aem_cmd_set_sampling_rate, 0, sizeof(aem_cmd_set_sampling_rate));
68 
69  /******************************************* AECP Common Data **********************************************/
70  aem_cmd_set_sampling_rate.aem_header.aecpdu_header.controller_entity_id = base_end_station_imp_ref->get_adp()->get_controller_entity_id();
71  // Fill aem_cmd_get_sampling_rate.sequence_id in AEM Controller State Machine
72  aem_cmd_set_sampling_rate.aem_header.command_type = JDKSAVDECC_AEM_COMMAND_SET_SAMPLING_RATE;
73 
74  /******************** AECP Message Specific Data *******************/
75  aem_cmd_set_sampling_rate.descriptor_type = descriptor_type();
76  aem_cmd_set_sampling_rate.descriptor_index = descriptor_index();
77  aem_cmd_set_sampling_rate.sampling_rate = new_sampling_rate;
78 
79  /******************************** Fill frame payload with AECP data and send the frame ***************************/
80  aecp_controller_state_machine_ref->ether_frame_init(base_end_station_imp_ref->mac(), &cmd_frame,
81  ETHER_HDR_SIZE + JDKSAVDECC_AEM_COMMAND_SET_SAMPLING_RATE_COMMAND_LEN);
82  aem_cmd_set_sampling_rate_returned = jdksavdecc_aem_command_set_sampling_rate_write(&aem_cmd_set_sampling_rate,
83  cmd_frame.payload,
84  ETHER_HDR_SIZE,
85  sizeof(cmd_frame.payload));
86 
87  if (aem_cmd_set_sampling_rate_returned < 0)
88  {
89  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "aem_cmd_get_sampling_rate_write error\n");
90  assert(aem_cmd_set_sampling_rate_returned >= 0);
91  return -1;
92  }
93 
94  aecp_controller_state_machine_ref->common_hdr_init(JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_COMMAND,
95  &cmd_frame,
96  base_end_station_imp_ref->entity_id(),
97  JDKSAVDECC_AEM_COMMAND_SET_SAMPLING_RATE_COMMAND_LEN -
98  JDKSAVDECC_COMMON_CONTROL_HEADER_LEN);
99  system_queue_tx(notification_id, CMD_WITH_NOTIFICATION, cmd_frame.payload, cmd_frame.length);
100 
101  return 0;
102 }

104 int audio_unit_descriptor_imp::proc_set_sampling_rate_resp(void *& notification_id, const uint8_t * frame, size_t frame_len, int & status)
105 {
106  struct jdksavdecc_frame cmd_frame;
107  struct jdksavdecc_aem_command_set_sampling_rate_response aem_cmd_set_sampling_rate_resp;
108  ssize_t aem_cmd_set_sampling_rate_resp_returned;
109  uint32_t msg_type;
110  bool u_field;
111  uint8_t * buffer;
112 
113  memcpy(cmd_frame.payload, frame, frame_len);
114  memset(&aem_cmd_set_sampling_rate_resp, 0, sizeof(struct jdksavdecc_aem_command_set_sampling_rate_response));
115 
116  aem_cmd_set_sampling_rate_resp_returned = jdksavdecc_aem_command_set_sampling_rate_response_read(&aem_cmd_set_sampling_rate_resp,
117  frame,
118  ETHER_HDR_SIZE,
119  frame_len);
120 
121  if (aem_cmd_set_sampling_rate_resp_returned < 0)
122  {
123  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "aem_cmd_get_sampling_rate_resp_read error\n");
124  assert(aem_cmd_set_sampling_rate_resp_returned >= 0);
125  return -1;
126  }
127 
128  buffer = (uint8_t *)malloc(resp_ref->get_desc_size() * sizeof(uint8_t)); //fetch current desc frame
129  memcpy(buffer, resp_ref->get_desc_buffer(), resp_ref->get_desc_size());
130  jdksavdecc_descriptor_audio_unit_set_current_sampling_rate(aem_cmd_set_sampling_rate_resp.sampling_rate, buffer, resp_ref->get_desc_pos()); //set clk source
131 
132  replace_desc_frame(buffer, resp_ref->get_desc_pos(), resp_ref->get_desc_size()); //replace frame
133 
134  msg_type = aem_cmd_set_sampling_rate_resp.aem_header.aecpdu_header.header.message_type;
135  status = aem_cmd_set_sampling_rate_resp.aem_header.aecpdu_header.header.status;
136  u_field = aem_cmd_set_sampling_rate_resp.aem_header.command_type >> 15 & 0x01; // u_field = the msb of the uint16_t command_type
137 
138  aecp_controller_state_machine_ref->update_inflight_for_rcvd_resp(notification_id, msg_type, u_field, &cmd_frame);
139 
140  free(buffer);
141  return 0;
142 }
