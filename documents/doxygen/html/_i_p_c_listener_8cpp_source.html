38 DWORD WINAPI IPCListener::IPCListenerLoop( IPCSharedData *arg ) {
39  PeerList *list = arg->list;
40  LockableOffset *loffset = arg->offset;
41  HANDLE pipe = INVALID_HANDLE_VALUE;
42 
43  uint8_t tmp[NPIPE_MAX_MSG_SZ];
44  OVERLAPPED pipe_ol;
45  HANDLE ol_event;
46  enum { PIPE_CLOSED, PIPE_UNCONNECT, PIPE_CONNECT_PENDING, PIPE_CONNECT, PIPE_READ_PENDING } pipe_state;
47 
48  // Open named pipe
49  char pipename[64];
50  strncpy_s( pipename, 64, PIPE_PREFIX, 63 );
51  strncpy_s( pipename+strlen(pipename), 64-strlen(pipename), P802_1AS_PIPENAME, 63-strlen(pipename) );
52  ol_event = CreateEvent( NULL, true, false, NULL );
53  pipe_state = PIPE_CLOSED;
54 
55  DWORD retval = -1;
56 
57  while( !exit_waiting ) {
58  int err;
59  DWORD ret;
60  if( pipe_state < PIPE_UNCONNECT ) {
61  if( pipe != INVALID_HANDLE_VALUE ) {
62  DisconnectNamedPipe( pipe );
63  CloseHandle( pipe );
64  }
65  pipe = CreateNamedPipe( pipename, PIPE_ACCESS_DUPLEX | FILE_FLAG_OVERLAPPED, PIPE_TYPE_BYTE | PIPE_READMODE_BYTE , PIPE_UNLIMITED_INSTANCES,
66  OUTSTANDING_MESSAGES*NPIPE_MAX_SERVER_MSG_SZ, OUTSTANDING_MESSAGES*NPIPE_MAX_MSG_SZ, 0, NULL );
67  if( pipe == INVALID_HANDLE_VALUE ) {
68  GPTP_LOG_ERROR( "Open pipe error (%s): %d", pipename, GetLastError() );
69  goto do_error;
70  }
71  pipe_state = PIPE_UNCONNECT;
72  } else if( pipe_state < PIPE_CONNECT ) {
73  if( pipe_state != PIPE_CONNECT_PENDING ) {
74  memset( &pipe_ol, 0, sizeof( pipe_ol ));
75  pipe_ol.hEvent = ol_event;
76  if( ResetEvent( ol_event ) == 0 ) goto do_error;
77  if( ConnectNamedPipe( pipe, &pipe_ol ) != 0 ) {
78  // Successfully connected
79  pipe_state = PIPE_CONNECT;
80  continue;
81  } else {
82  err = GetLastError();
83  switch( err ) {
84  default:
85  GPTP_LOG_ERROR( "Attempt to connect on Pipe failed, %d", err );
86  goto do_error;
87  case ERROR_PIPE_CONNECTED:
88  pipe_state = PIPE_CONNECT;
89  continue;
90  case ERROR_IO_PENDING:
91  pipe_state = PIPE_CONNECT_PENDING;
92  }
93  }
94  }
95  ret = WaitForSingleObject( ol_event, 200 );
96  switch( ret ) {
97  case WAIT_OBJECT_0:
98  pipe_state = PIPE_CONNECT;
99  case WAIT_TIMEOUT:
100  continue;
101  default:
102  goto do_error;
103  }
104  } else {
105  // We're connected
106  long readlen;
107  if( pipe_state < PIPE_READ_PENDING ) {
108  // Wait for message - Read Base Message
109  ((WindowsNPipeMessage *)tmp)->init();
110  if( ResetEvent( ol_event ) == 0 ) goto do_error;
111  if(( readlen = ((WindowsNPipeMessage *)tmp)->read_ol( pipe, 0, ol_event )) == -1 ) {
112  err = GetLastError();
113  switch( err ) {
114  default:
115  GPTP_LOG_ERROR( "Failed to read from pipe @%u,%d", __LINE__, err );
116  goto do_error;
117  case ERROR_BROKEN_PIPE:
118  pipe_state = PIPE_CLOSED;
119  continue;
120  case ERROR_IO_PENDING:
121  ;
122  }
123  }
124  // Fall through to here whether data is available or not
125  pipe_state = PIPE_READ_PENDING;
126  }
127  ret = WaitForSingleObject( ol_event, 200 );
128  switch( ret ) {
129  default:
130  goto do_error;
131  case WAIT_TIMEOUT:
132  continue;
133  case WAIT_OBJECT_0:
134  if(( readlen = ((WinNPipeCtrlMessage *)tmp)->read_ol_complete( pipe )) == -1 ) {
135  err = GetLastError();
136  if( err == ERROR_BROKEN_PIPE ) {
137  pipe_state = PIPE_CLOSED;
138  continue;
139  }
140  GPTP_LOG_ERROR( "Failed to read from pipe @%u,%d", __LINE__, err );
141  goto do_error;
142  }
143  switch( ((WindowsNPipeMessage *)tmp)->getType() ) {
144  case CTRL_MSG:
145  ((WinNPipeCtrlMessage *)tmp)->init();
146  if(( readlen = ((WinNPipeCtrlMessage *)tmp)->read( pipe, readlen )) == -1 ) {
147  GPTP_LOG_ERROR( "Failed to read from pipe @%u", __LINE__ );
148  goto do_error;
149  }
150  //readlen may not be set properly ??
151  // Attempt to add or remove from the list
152  switch( ((WinNPipeCtrlMessage *)tmp)->getCtrlWhich() ) {
153  default:
154  GPTP_LOG_ERROR( "Recvd CTRL cmd specifying illegal operation @%u", __LINE__ );
155  goto do_error;
156  case ADD_PEER:
157  if( !list->IsReady() || !list->add( ((WinNPipeCtrlMessage *)tmp)->getPeerAddr() ) ) {
158  GPTP_LOG_ERROR( "Failed to add peer @%u", __LINE__ );
159  }
160  break;
161  case REMOVE_PEER:
162  if( !list->IsReady() || !list->remove( ((WinNPipeCtrlMessage *)tmp)->getPeerAddr() ) ) {
163  GPTP_LOG_ERROR( "Failed to remove peer @%u", __LINE__ );
164  }
165  break;
166  }
167  break;
168  case OFFSET_MSG:
169  ((WinNPipeQueryMessage *)tmp)->init();
170  if(( readlen = ((WinNPipeQueryMessage *)tmp)->read( pipe, readlen )) == -1 ) {
171  GPTP_LOG_ERROR( "Failed to read from pipe @%u", __LINE__ );
172  goto do_error;
173  }
174  // Create an offset message and send it
175  loffset->get();
176  if( loffset->isReady() ) ((WinNPipeOffsetUpdateMessage *)tmp)->init((Offset *)loffset);
177  else ((WinNPipeOffsetUpdateMessage *)tmp)->init();
178  loffset->put();
179  ((WinNPipeOffsetUpdateMessage *)tmp)->write(pipe);
180  break;
181  default:
182  GPTP_LOG_ERROR( "Recvd Unknown Message" );
183  // Is this recoverable?
184  goto do_error;
185  }
186  pipe_state = PIPE_CONNECT;
187  }
188  }
189  }
190 
191 
192  retval = 0; // Exit normally
193 do_error:
194  // Close Named Pipe
195  if( pipe != INVALID_HANDLE_VALUE ) CloseHandle( pipe );
196 
197  return retval;
198 }
