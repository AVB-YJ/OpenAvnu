262 /*
263  * When interesting StreamID tracking is enabled only a single
264  * client is supported correctly as there are no notifications
265  * sent to client B when client A removes an ID and client B is
266  * also "attached". Multiple clients are therefore not allowed
267  * when stream ID pruning is enabled and the second client
268  * will have an error returned.
269  *
270  * Note: mrpd should be started without the command line parameter
271  * to enable interesting stream tracking if multiple clients are
272  * going to be using mrpd. A notification mechanism between clients
273  * would need to be added if multiple client support was to be correctly
274  * implemented. And most likely either a reference count and/or a bitmask
275  * added to the database where the interesting stream IDs are stored.
276  * An API for dumping/reporting current interesting streamIDs might also
277  * be useful.
278  *
279  */
280 
281 TEST(MsrpPruningTestGroup, Prune_Multiple_Clients)
282 {
283  static struct sockaddr_in client1;
284  static struct sockaddr_in client2;
285 
286  memset(&client1, 0, sizeof(client1));
287  memset(&client2, 1, sizeof(client2));
288 
289  /* no error returned for first client */
290  msrp_recv_cmd("S??", strlen("S??") + 1, &client1);
291  CHECK(msrp_tests_cmd_ok(test_state.ctl_msg_data));
292 
293  /* error returned for second client */
294  msrp_recv_cmd("S??", strlen("S??") + 1, &client2);
295  CHECK(!msrp_tests_cmd_ok(test_state.ctl_msg_data));
296 }

422 /*
423  * This test checks behaviour when pruning is enabled and a
424  * talker stream that was interesting is now marked as not
425  * interesting. In this case there is no matching listener
426  * attribute.
427  *
428  * Because there is not matching listener attribute, the
429  * talker attribute (TA of TF) is deleted from the MRSP
430  * database. The ID is all deleted from the interesting IDs
431  * database. The delete is silent and there is no client
432  * notification.
433  */

435 // enabled.
436 // "I+S:ID"

443  int tx_flag_count = 0;
444  int ta_count_before = 0;
445  int ta_count_after = 0;
446  int rv;
447 
448  /* mark as interesting */
449  snprintf(cmd_string, sizeof(cmd_string), "I+S:S=%" PRIx64, id);
450  msrp_recv_cmd(cmd_string, strlen(cmd_string) + 1, &client);
451  CHECK(msrp_tests_cmd_ok(test_state.ctl_msg_data));
452  LONGS_EQUAL(1, msrp_interesting_id_count());
