193 uint64_t reverse_64(uint64_t val)
194 {
195  uint32_t low, high;
196 
197  low = val & 0xffffffff;
198  high = (val >> 32) & 0xffffffff;
199  low = htonl(low);
200  high = htonl(high);
201  val = 0;
202  val = val | low;
203  val = (val << 32) | high;
204  return val;
205 }

207 int start_feed_socket_init()
208 {
209  int sock, length, rc, on = 1, flags ;
210  struct sockaddr_in server;
211 
212  sock = socket(AF_INET, SOCK_DGRAM, 0);
213  if (sock < 0) {
214  perror("Opening socket");
215  return -1;
216  }
217 
218  g_start_feed_socket = sock;
219 
220  /* Allow socket descriptor to be reuseable */
221  rc = setsockopt(sock, SOL_SOCKET, SO_REUSEADDR,
222  (char *)&on, sizeof(on));
223  if (rc < 0) {
224  perror("setsockopt() failed");
225  close(sock);
226  exit(-1);
227  }
228 
229  /*
230  * Set socket to be non-blocking. All of the sockets for
231  * the incoming connections will also be non-blocking since
232  * they will inherit that state from the listening socket.
233  */
234  flags = fcntl(sock, F_GETFL, 0);
235  rc = fcntl(sock, F_SETFL, flags | O_NONBLOCK);
236  if (rc < 0) {
237  perror("ioctl() failed");
238  close(sock);
239  exit(-1);
240  }
241 
242  length = sizeof(server);
243  bzero(&server,length);
244  server.sin_family = AF_INET;
245  server.sin_addr.s_addr=INADDR_ANY;
246  server.sin_port=htons(7600);
247  if (bind(sock,(struct sockaddr *)&server,length) < 0) {
248  perror("binding");
249  exit(-1);
250  }
251 
252  return 0;
253 }

308 {
309  int lsock = socket(PF_PACKET, SOCK_RAW, htons(0x800));
310  struct ifreq if_request;
311  int err;
312 
313  if (lsock < 0)
314  return -1;
315 
316  memset(&if_request, 0, sizeof(if_request));
317  strncpy(if_request.ifr_name, (const char *)interface, sizeof(if_request.ifr_name));
318  err = ioctl(lsock, SIOCGIFHWADDR, &if_request);
319  if (err < 0) {
320  close(lsock);
321  return -1;
322  }
323  memcpy(STATION_ADDR, if_request.ifr_hwaddr.sa_data,

330 static GstFlowReturn
331 gst_avbsink_render (GstBaseSink * bsink, GstBuffer * buff)
332 {
333 
334  GstMapInfo info;
335  unsigned i;
336  int err;
337 
338  if(!avb_init)
339  {
340  struct igb_packet a_packet;
341  pthread_t tid;
342 
343  avb_init = 1;
344  pkt_sz = 1024;
345  payload_len = 1024;
346  pkt_sz += sizeof(six1883_header) + sizeof(seventeen22_header) + sizeof(eth_header);
347  if (pkt_sz > 1500) {
348  fprintf(stderr,"payload_len is > MAX_ETH_PACKET_LEN - not supported.\n");
349  return -EINVAL;
350  }
351  err = pci_connect(&igb_dev);
352  if (err) {
353  printf("connect failed (%s) - are you running as root?\n", strerror(errno));
354  return errno;
355  }
356 
357  err = igb_init(&igb_dev);
358  if (err) {
359  printf("init failed (%s) - is the driver really loaded?\n", strerror(errno));
360  return errno;
361  }
362  err = igb_dma_malloc_page(&igb_dev, &a_page);
363  if (err) {
364  printf("malloc failed (%s) - out of memory?\n", strerror(errno));
365  return errno;
366  }
367  signal(SIGINT, sigint_handler);
368  err = get_mac_addr(interface1);
369  if (err) {
370  printf("failed to open interface(%s)\n",interface1);
371  }
372  igb_set_class_bandwidth(&igb_dev, PACKET_IPG / 125000, 0, pkt_sz - 22, 0);
373  memset(STREAM_ID, 0, sizeof(STREAM_ID));
374  memcpy(STREAM_ID, STATION_ADDR, sizeof(STATION_ADDR));
375  a_packet.dmatime = a_packet.attime = a_packet.flags = 0;
376  a_packet.map.paddr = a_page.dma_paddr;
377  a_packet.map.mmap_size = a_page.mmap_size;
378  a_packet.offset = 0;
379  a_packet.vaddr = a_page.dma_vaddr + a_packet.offset;
380  a_packet.len = pkt_sz;
381  free_packets = NULL;
382  seqnum = 0;
383  frame_size = payload_len + sizeof(six1883_header) + sizeof(seventeen22_header) + sizeof(eth_header);
384  stream_packet = avb_create_packet(payload_len);
385  h1722 = (seventeen22_header *)((uint8_t*)stream_packet + sizeof(eth_header));
386  h61883 = (six1883_header *)((uint8_t*)stream_packet + sizeof(eth_header) +
387  sizeof(seventeen22_header));
388 
389  /*initialize h1722 header */
390  avb_initialize_h1722_to_defaults(h1722);
391  /* set the length */
392  avb_set_1722_length(h1722, htons(payload_len + sizeof(six1883_header)));
393  avb_set_1722_stream_id(h1722,reverse_64(STREAMID));
394  avb_set_1722_sid_valid(h1722, 0x1);
395 
396 
397  /*initialize h61883 header */
398  avb_initialize_61883_to_defaults(h61883);
399  avb_set_61883_format_tag(h61883, 0x1);
400  avb_set_61883_packet_channel(h61883, 0x1f);
401  avb_set_61883_packet_tcode(h61883, 0xa);
402  avb_set_61883_source_id(h61883 , 0x3f);
403  avb_set_61883_data_block_size(h61883, 0x1);
404  avb_set_61883_eoh(h61883, 0x2);
405  avb_set_61883_format_id(h61883, 0x10);
406  avb_set_61883_format_dependent_field(h61883, 0x2);
407  avb_set_61883_syt(h61883, 0xffff);
408 
409  /* initialize the source & destination mac address */
410  avb_eth_header_set_mac(stream_packet, DEST_ADDR, interface1);
411 
412  /* set 1772 eth type */
413  avb_1722_set_eth_type(stream_packet);
414 
415  /* divide the dma page into buffers for packets */
416  for (i = 1; i < ((a_page.mmap_size) / pkt_sz); i++) {
417  tmp_packet = malloc(sizeof(struct igb_packet));
418  if (!tmp_packet) {
419  printf("failed to allocate igb_packet memory!\n");
420  return errno;
421  }
422  *tmp_packet = a_packet;
423  tmp_packet->offset = (i * pkt_sz);
424  tmp_packet->vaddr += tmp_packet->offset;
425  tmp_packet->next = free_packets;
426  memset(tmp_packet->vaddr, 0, pkt_sz); /* MAC header at least */
427  memcpy(((char *)tmp_packet->vaddr), stream_packet, frame_size);
428  tmp_packet->len = frame_size;
429  free_packets = tmp_packet;
430  }
431 
432  start_feed_socket_init();
433  err = pthread_create(&tid, NULL, read_start_feed, NULL);
434  if (err != 0) {
435  printf("Failed to create thread :[%s] \n", strerror(err));
436  return -1;
437  }
438 
439  total_read_bytes = 0;
440  memset(&sched, 0 , sizeof (sched));
441  sched.sched_priority = 25;
442  sched_setscheduler(0, SCHED_RR, &sched);
443  }
444 
445  gst_buffer_map(buff,&info,GST_MAP_READ);
446  if(info.size>1024) {
447  g_print("Internal Data Flow Error: Expected blocksize=1024 for filesrc\n");
448  igb_cleanup();
449  }
450 
451  if (listeners && !halt_tx && avb_init) {
452 start:
453  if (g_start_feeding == 2)
454  goto exit_app;
455  tmp_packet = free_packets;
456  if (NULL == tmp_packet)
457  goto cleanup;
458 
459  stream_packet = ((char *)tmp_packet->vaddr);
460  free_packets = tmp_packet->next;
461  /* unfortuntely unless this thread is at rtprio
462  * you get pre-empted between fetching the time
463  * and programming the packet and get a late packet
464  */
465  h1722 = (seventeen22_header *)((uint8_t*)stream_packet + sizeof(eth_header));
466  avb_set_1722_seq_number(h1722, seqnum++);
467  if (seqnum % 4 == 0)
468  avb_set_1722_timestamp_valid(h1722, 0);
469  else
470  avb_set_1722_timestamp_valid(h1722, 1);
471 
472  data_ptr = (uint8_t *)((uint8_t*)stream_packet + sizeof(eth_header) +
473  sizeof(seventeen22_header) + sizeof(six1883_header));
474 
475  memcpy((void *)data_ptr,info.data,payload_len);
476  usleep(500);
477  total_read_bytes += payload_len;
478  total_samples += payload_len;
479  h61883 = (six1883_header *)((uint8_t*)stream_packet + sizeof(eth_header) + sizeof(seventeen22_header));
480  avb_set_61883_data_block_continuity(h61883, total_samples);
481 
482  err = igb_xmit(&igb_dev, 0, tmp_packet);
483  if (!err) {
484  fprintf(stderr,"frame sequence = %lld\n", frame_sequence++);
485  do{
486  usleep(100);
487  } while (g_start_feeding == 0);
488  return GST_FLOW_OK;
489  } else {
490  fprintf(stderr,"Failed frame sequence = %lld !!!!\n", frame_sequence++);
491  }
492 
493  if (ENOSPC == err) {
494  /* put back for now */
495  tmp_packet->next = free_packets;
496  free_packets = tmp_packet;
497  }
498  return GST_FLOW_OK;
499 
500 cleanup:
501  igb_clean(&igb_dev, &cleaned_packets);
502  while (cleaned_packets) {
503  tmp_packet = cleaned_packets;
504  cleaned_packets = cleaned_packets->next;
505  tmp_packet->next = free_packets;
506  free_packets = tmp_packet;
507  }
508  goto start;
509  }
510 
511  if (halt_tx == 0)
512  printf("listener left ...\n");
513 
514 exit_app:
515  halt_tx = 1;
516  igb_set_class_bandwidth(&igb_dev, 0, 0, 0, 0);
517  igb_dma_free_page(&igb_dev, &a_page);
518  igb_detach(&igb_dev);
519  pthread_exit(NULL);
520 
521  return 0;
522 }
