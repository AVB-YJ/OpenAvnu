89  int i;
90  _TCHAR *cur = macstr;

92  if( strnlen_s( macstr, MACSTR_LENGTH ) != 17 ) return -1;
93

251  parseMacAddr(argv[i++], addr_ostr);

255  if (!port->init_port()) {
256  printf("Failed to initialize port\n");

271  }

282 uint32_t findLinkSpeed( LinkLayerAddress *local_addr )
283 {
284  ULONG ret_sz;
285  char *buffer;
286  PIP_ADAPTER_ADDRESSES pAddr;
287  ULONG err;
288  uint32_t ret;
289 
290  buffer = (char *) malloc((size_t)15000);
291  ret_sz = 15000;
292  pAddr = (PIP_ADAPTER_ADDRESSES)buffer;
293  err = GetAdaptersAddresses( AF_UNSPEC, 0, NULL, pAddr, &ret_sz );
294  for (; pAddr != NULL; pAddr = pAddr->Next)
295  {
296  //fprintf(stderr, "** here : %p\n", pAddr);
297  if (pAddr->PhysicalAddressLength == ETHER_ADDR_OCTETS &&
298  *local_addr == LinkLayerAddress(pAddr->PhysicalAddress))
299  {
300  break;
301  }
302  }
303 
304  if (pAddr == NULL)
305  return INVALID_LINKSPEED;
306 
307  switch ( pAddr->ReceiveLinkSpeed / WIN_LINKSPEED_MULT )
308  {
309  default:
310  GPTP_LOG_ERROR("Can't find link speed, %llu", pAddr->ReceiveLinkSpeed);
311  ret = INVALID_LINKSPEED;
312  break;
313  case LINKSPEED_1G:
314  ret = LINKSPEED_1G;
315  break;
316  case LINKSPEED_100MB:
317  ret = LINKSPEED_100MB;
318  break;
319  }
320 
321  delete buffer;
322  return ret;
323 }
