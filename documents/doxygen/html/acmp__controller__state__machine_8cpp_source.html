53 int acmp_controller_state_machine::ether_frame_init(struct jdksavdecc_frame * cmd_frame)
54 {
55  /*** Offset to write the field to ***/
56  size_t ether_frame_pos = 0;
57  jdksavdecc_frame_init(cmd_frame);
58 
59  /************************************************************ Ethernet Frame ********************************************************/
60  cmd_frame->ethertype = JDKSAVDECC_AVTP_ETHERTYPE;
61  utility::convert_uint64_to_eui48(net_interface_ref->mac_addr(), cmd_frame->src_address.value); // Send from the Controller MAC address
62  cmd_frame->dest_address = jdksavdecc_multicast_adp_acmp; // Send to the ACMP multicast destination MAC address
63  cmd_frame->length = ACMP_FRAME_LEN; // Length of ACMP packet is 70 bytes
64 
65  /****************** Fill frame payload with Ethernet frame information ****************/
66  jdksavdecc_frame_write(cmd_frame, cmd_frame->payload, ether_frame_pos, ETHER_HDR_SIZE);
67 
68  return 0;

113 void acmp_controller_state_machine::state_timeout(uint32_t inflight_cmd_index)
114 {
115  struct jdksavdecc_frame frame = inflight_cmds.at(inflight_cmd_index).frame();
116  bool is_retried = inflight_cmds.at(inflight_cmd_index).retried();
117 
118  if (is_retried)
119  {
120  struct jdksavdecc_eui64 _end_station_entity_id = jdksavdecc_acmpdu_get_listener_entity_id(frame.payload, ETHER_HDR_SIZE);
121  uint64_t end_station_entity_id = jdksavdecc_uint64_get(&_end_station_entity_id, 0);
122  uint32_t msg_type = jdksavdecc_common_control_header_get_control_data(frame.payload, ETHER_HDR_SIZE);
123 
124  notification_imp_ref->post_notification_msg(RESPONSE_RECEIVED,
125  end_station_entity_id,
126  (uint16_t)msg_type + CMD_LOOKUP,
127  0,
128  0,
129  UINT_MAX,
130  inflight_cmds.at(inflight_cmd_index).cmd_notification_id);
131 
132  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR,
133  "Command Timeout, 0x%llx, %s, %s, %s, %d",
134  end_station_entity_id,
135  utility::acmp_cmd_value_to_name(msg_type),
136  "NULL",
137  "NULL",
138  inflight_cmds.at(inflight_cmd_index).cmd_seq_id);
139 
140  inflight_cmds.erase(inflight_cmds.begin() + inflight_cmd_index);
141  }
142  else
143  {
144  log_imp_ref->post_log_msg(LOGGING_LEVEL_DEBUG,
145  "Resend the command with sequence id = %d",
146  inflight_cmds.at(inflight_cmd_index).cmd_seq_id);
147 
148  tx_cmd(inflight_cmds.at(inflight_cmd_index).cmd_notification_id,
149  inflight_cmds.at(inflight_cmd_index).notification_flag(),
150  &frame,
151  true);
152  }

155 int acmp_controller_state_machine::tx_cmd(void * notification_id, uint32_t notification_flag, struct jdksavdecc_frame * cmd_frame, bool resend)
156 {
157  int send_frame_returned;
158 
159  if (!resend)
160  {
161  uint16_t this_seq_id = acmp_seq_id;
162  uint32_t msg_type = jdksavdecc_common_control_header_get_control_data(cmd_frame->payload, ETHER_HDR_SIZE);
163  uint32_t timeout_ms = utility::acmp_cmd_to_timeout(msg_type); // ACMP command timeout lookup
164  jdksavdecc_acmpdu_set_sequence_id(acmp_seq_id++, cmd_frame->payload, ETHER_HDR_SIZE);
165 
166  inflight in_flight = inflight(cmd_frame,
167  this_seq_id,
168  notification_id,
169  notification_flag,
170  timeout_ms);
171 
172  in_flight.start_timer();
173  inflight_cmds.push_back(in_flight);
174  }
175  else
176  {
177  uint16_t resend_with_seq_id = jdksavdecc_acmpdu_get_sequence_id(cmd_frame->payload, ETHER_HDR_SIZE);
178  std::vector<inflight>::iterator j =
179  std::find_if(inflight_cmds.begin(), inflight_cmds.end(), SeqIdComp(resend_with_seq_id));
180 
181  if (j != inflight_cmds.end()) // found?
182  {
183  (*j).start_timer();
184  }
185  }
186 
187  send_frame_returned = net_interface_ref->send_frame(cmd_frame->payload, cmd_frame->length);
188  if (send_frame_returned < 0)
189  {
190  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR, "netif_send_frame error");
191  assert(send_frame_returned >= 0);
192  }
193 
194  callback(notification_id, notification_flag, cmd_frame->payload);
195 
196  return 0;
197 }

199 int acmp_controller_state_machine::proc_resp(void *& notification_id, struct jdksavdecc_frame * cmd_frame)
200 {
201  uint16_t seq_id = jdksavdecc_acmpdu_get_sequence_id(cmd_frame->payload, ETHER_HDR_SIZE);
202  uint32_t notification_flag = 0;
203 
204  std::vector<inflight>::iterator j =
205  std::find_if(inflight_cmds.begin(), inflight_cmds.end(), SeqIdComp(seq_id));
206 
207  if (j != inflight_cmds.end()) // found?
208  {
209  notification_id = (*j).cmd_notification_id;
210  notification_flag = (*j).notification_flag();
211  callback(notification_id, notification_flag, cmd_frame->payload);
212  inflight_cmds.erase(j);
213  return 1;
214  }
215 
216  return -1;
217 }

243 int acmp_controller_state_machine::callback(void * notification_id, uint32_t notification_flag, uint8_t * frame)
244 {
245  uint32_t msg_type = jdksavdecc_common_control_header_get_control_data(frame, ETHER_HDR_SIZE);
246  uint16_t seq_id = jdksavdecc_acmpdu_get_sequence_id(frame, ETHER_HDR_SIZE);
247  uint32_t status = jdksavdecc_common_control_header_get_status(frame, ETHER_HDR_SIZE);
248  uint64_t end_station_entity_id;
249 
250  if ((notification_flag == CMD_WITH_NOTIFICATION) &&
251  ((msg_type == JDKSAVDECC_ACMP_MESSAGE_TYPE_GET_TX_STATE_RESPONSE) ||
252  (msg_type == JDKSAVDECC_ACMP_MESSAGE_TYPE_GET_TX_CONNECTION_RESPONSE)))
253  {
254  struct jdksavdecc_eui64 _end_station_entity_id = jdksavdecc_acmpdu_get_talker_entity_id(frame, ETHER_HDR_SIZE);
255  end_station_entity_id = jdksavdecc_uint64_get(&_end_station_entity_id, 0);
256 
257  notification_imp_ref->post_notification_msg(RESPONSE_RECEIVED,
258  end_station_entity_id,
259  (uint16_t)msg_type + CMD_LOOKUP,
260  0,
261  0,
262  status,
263  notification_id);
264 
265  if (status != ACMP_STATUS_SUCCESS)
266  {
267  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR,
268  "RESPONSE_RECEIVED, 0x%llx, %s, %s, %s, %s, %d",
269  end_station_entity_id,
270  utility::acmp_cmd_value_to_name(msg_type),
271  "NULL",
272  "NULL",
273  utility::acmp_cmd_status_value_to_name(status),
274  seq_id);
275  }
276  }
277  else if ((notification_flag == CMD_WITH_NOTIFICATION) &&
278  ((msg_type == JDKSAVDECC_ACMP_MESSAGE_TYPE_CONNECT_RX_RESPONSE) ||
279  (msg_type == JDKSAVDECC_ACMP_MESSAGE_TYPE_DISCONNECT_RX_RESPONSE) ||
280  (msg_type == JDKSAVDECC_ACMP_MESSAGE_TYPE_GET_RX_STATE_RESPONSE) ||
281  (msg_type == JDKSAVDECC_ACMP_MESSAGE_TYPE_GET_RX_STATE_RESPONSE)))
282  {
283  struct jdksavdecc_eui64 _end_station_entity_id = jdksavdecc_acmpdu_get_listener_entity_id(frame, ETHER_HDR_SIZE);
284  end_station_entity_id = jdksavdecc_uint64_get(&_end_station_entity_id, 0);
285  notification_imp_ref->post_notification_msg(RESPONSE_RECEIVED,
286  end_station_entity_id,
287  (uint16_t)msg_type + CMD_LOOKUP,
288  0,
289  0,
290  status,
291  notification_id);
292 
293  if (status != ACMP_STATUS_SUCCESS)
294  {
295  log_imp_ref->post_log_msg(LOGGING_LEVEL_ERROR,
296  "RESPONSE_RECEIVED, 0x%llx, %s, %s, %s, %s, %d",
297  end_station_entity_id,
298  utility::acmp_cmd_value_to_name(msg_type),
299  "NULL",
300  "NULL",
301  utility::acmp_cmd_status_value_to_name(status),
302  seq_id);
303  }
304  }
305  else if ((msg_type == JDKSAVDECC_ACMP_MESSAGE_TYPE_GET_TX_STATE_RESPONSE) ||
306  (msg_type == JDKSAVDECC_ACMP_MESSAGE_TYPE_GET_TX_CONNECTION_RESPONSE))

308  struct jdksavdecc_eui64 _end_station_entity_id = jdksavdecc_acmpdu_get_talker_entity_id(frame, ETHER_HDR_SIZE);

313  utility::acmp_cmd_value_to_name(msg_type),
314  "NULL",

317  seq_id);
318  }
319  else

322  end_station_entity_id = jdksavdecc_uint64_get(&_end_station_entity_id, 0);
