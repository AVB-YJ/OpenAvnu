44 static bool openavbEptClntSendToServer(int h, openavbEndpointMessage_t *msg)
45 {
46  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
47 
48  if (!msg || h == AVB_ENDPOINT_HANDLE_INVALID) {
49  AVB_LOG_ERROR("Client send: invalid argument passed");
50  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
51  return FALSE;
52  }
53 
54  ssize_t nWrite = write(h, msg, OPENAVB_ENDPOINT_MSG_LEN);
55  AVB_LOGF_VERBOSE("Sent message, len=%zu, nWrite=%zu", OPENAVB_ENDPOINT_MSG_LEN, nWrite);
56 
57  if (nWrite < OPENAVB_ENDPOINT_MSG_LEN) {
58  if (nWrite < 0) {
59  AVB_LOGF_ERROR("Client failed to write socket: %s", strerror(errno));
60  }
61  else if (nWrite == 0) {
62  AVB_LOG_ERROR("Client send: socket closed unexpectedly");
63  }
64  else {
65  AVB_LOG_ERROR("Client send: short write");
66  }
67  socketClose(h);
68  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
69  return FALSE;
70  }
71 
72  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
73  return TRUE;
74 }

76 int openavbEptClntOpenSrvrConnection(tl_state_t *pTLState)
77 {
78  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
79  struct sockaddr_un server;
80  server.sun_family = AF_UNIX;
81  snprintf(server.sun_path, UNIX_PATH_MAX, AVB_ENDPOINT_UNIX_PATH);
82 
83  int h = socket(AF_UNIX, SOCK_STREAM, 0);
84  if (h < 0) {
85  AVB_LOGF_DEBUG("Failed to open socket: %s", strerror(errno));
86  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
87  return AVB_ENDPOINT_HANDLE_INVALID;
88  }
89 
90  AVB_LOGF_DEBUG("Connecting to %s", server.sun_path);
91  int rslt = connect(h, (struct sockaddr*)&server, sizeof(struct sockaddr_un));
92  if (rslt < 0) {
93  AVB_LOGF_DEBUG("Failed to connect socket: %s", strerror(errno));
94  socketClose(h);
95  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
96  return AVB_ENDPOINT_HANDLE_INVALID;
97  }
98 
99  AVB_LOG_DEBUG("Connected to endpoint");
100  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
101  return h;
102 }

111 bool openavbEptClntService(int h, int timeout)
112 {
113  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
114  bool rc = FALSE;
115 
116  if (h == AVB_ENDPOINT_HANDLE_INVALID) {
117  AVB_LOG_ERROR("Client service: invalid socket");
118  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
119  return FALSE;
120  }
121 
122  struct pollfd fds[1];
123  memset(fds, 0, sizeof(struct pollfd));
124  fds[0].fd = h;
125  fds[0].events = POLLIN;
126 
127  AVB_LOG_VERBOSE("Waiting for event...");
128  int pRet = poll(fds, 1, timeout);
129 
130  if (pRet == 0) {
131  AVB_LOG_VERBOSE("Poll timeout");
132  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
133  return TRUE;
134  }
135  else if (pRet < 0) {
136  if (errno == EINTR) {
137  AVB_LOG_VERBOSE("Poll interrupted");
138  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
139  return TRUE;
140  }
141  else {
142  AVB_LOGF_ERROR("Poll error: %s", strerror(errno));
143  }
144  }
145  else {
146  AVB_LOGF_DEBUG("Poll returned %d events", pRet);
147  // only one fd, so it's readable.
148  openavbEndpointMessage_t msgBuf;
149  memset(&msgBuf, 0, OPENAVB_ENDPOINT_MSG_LEN);
150  ssize_t nRead = read(h, &msgBuf, OPENAVB_ENDPOINT_MSG_LEN);
151 
152  if (nRead < OPENAVB_ENDPOINT_MSG_LEN) {
153  // sock closed
154  if (nRead == 0) {
155  AVB_LOG_ERROR("Socket closed unexpectedly");
156  }
157  else if (nRead < 0) {
158  AVB_LOGF_ERROR("Socket read error: %s", strerror(errno));
159  }
160  else {
161  AVB_LOG_ERROR("Socket read to short");
162  }
163  socketClose(h);
164  }
165  else {
166  // got a message
167  if (openavbEptClntReceiveFromServer(h, &msgBuf)) {
168  rc = TRUE;
169  }
170  else {
171  AVB_LOG_ERROR("Invalid message received");
172  socketClose(h);
173  }
174  }
175  }
176  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
177  return rc;
178 }
