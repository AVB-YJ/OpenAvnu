156 // Element for linked list of declared / registered streams.
157 // Since both a talker and a listener can exist for the same stream on the same
158 // end-station, separate talker and listener lists are kept.
159 // IMPORTANT NOTE: This implementation assumes that on a given end-station, no
160 // more than one talker and no more than one listener exist for each stream; if
161 // this is not the case, individual stream list elements could be corrupted
162 // because they are not thread-safe; the overall lists are mutex protected.
163 //
164 // On talker:
165 // a list entry will exist only if the talker has a stream to declare
166 // (or is in the process of withdrawing)
167 // - declType is what this talker has most recently declared;
168 // - declSubType is not used;
169 // - regType / regSubType indicate what the talker has most recently received from
170 // the listener(s), if anything;
171 // - avtpHandle, streamId, DA, tSpec, SRClassId, Rank, and Latency
172 // are as received from AVTP via openavbSrpRegisterStream();
173 // - failInfo is populated only if this station has insufficient outbound
174 // bandwidth for this stream, so must send Talker Failed declaration;
175 // - kbpsReserved is the bandwidth currently reserved for the stream.
176 //
177 // On Listener:
178 // a list entry will exist if the listener has either expressed interest in
179 // receiving the stream (or is in the process of withdrawing interest) or has
180 // received a talker declaration for the stream, or both.
181 // - declType is what this listener has most recently declared (via
182 // openavbSrpAttachStream()), if anything;
183 // - declSubType is the listener declaration subtype, if any;
184 // - regType indicates what talker declaration the listener has most recently
185 // received for the stream, if anything; (openavbSrp_AtTyp_None indicates that
186 // the listener currently has no talker declaration for the stream);
187 // - regSubType is not used;
188 // - avtpHandle is as received from AVTP via openavbSrpAttachStream();
189 // - strmId is either as received from AVTP via openavbSrpAttachStream() or
190 // as received via a talker declaration packet (MSRPDU), whichever occurs first;
191 // - SRClassIdx is derived from priority received in talker declaration packet;
192 // - DA, tSpec, latency, and, if applicable, failInfo are
193 // as received in talker declaration packet
194 // - rank and kbpsReserved are not applicable on the listener;
195 //
196 typedef struct openavbSrpStrm {
197  struct openavbSrpStrm* prev;
198  struct openavbSrpStrm* next;
199  AVBStreamID_t strmId;
200  void* avtpHandle;
201  SRClassIdx_t SRClassIdx;
202  openavbSrpAttribType_t declType; // attribute type this station is declaring for this stream, if any;
203  // (declType == openavbSrp_AtTyp_None is redundant to appState == openavbSrp_ApSt_VO).
204  openavbSrpLsnrDeclSubtype_t declSubType; // listener subtype this station is declaring for this stream, if any;
205  // not used on talker; valid only if declType == openavbSrp_AtTyp_Listener.
206  openavbSrpAttribType_t regType; // attribute type this station has received (registered) for this stream, if any;
207  openavbSrpLsnrDeclSubtype_t regSubType; // listener subtype this station has received (registered) for this stream, if any;
208  // not used on listener; valid only if regType == openavbSrp_AtTyp_Listener.
209  U8 DA[ETH_MAC_ADDR_LEN];
210  AVBTSpec_t tSpec;
211  U32 kbpsReserved;
212  bool rank;
213  U32 latency;
214  openavbSrpFailInfo_t failInfo;
215  openavbSrpAppState_t appState; // current applicant state of the stream on this station
216  openavbSrpRegState_t regState;
217 } openavbSrpStrmElem_t;

248 // First Value content and size depend on the Declaration Type (openavbSrpAttribType_t)
249 // see IEEE 802.1Q-2011 Section 35.2.2.8.1
250 // (left column indicates byte index to start of parameter)
251 // FirstValue ::=
252 // 0 StreamID: 8 bytes
253 // 0 [Src] MAC: 6 bytes (ETH_MAC_ADDR_LEN)
254 // 6 UniqueID: 2 bytes, U16
255 // end of the Listener attribute
256 // 8 DataFrameParameters: 8 bytes
257 // 8 DA: 6 bytes (ETH_MAC_ADDR_LEN)
258 // 14 VID: 2 bytes
259 // 16 TSpec: 4 bytes
260 // 16 MaxFrameSize: 2 bytes, U16
261 // 18 MaxIntervalFrames: 2 bytes, U16
262 // 20 PriorityAndRank: 1 byte
263 // 20 Priority: 3 bits
264 // 20 Rank: 1 bit
265 // 20 Reserved: 4 bits
266 // 21 AccumulatedLatency: 4 bytes, U32
267 // end of the Talker Advertise attribute
268 // 25 FailureInformation: 9 bytes
269 // 25 BridgeID: 8 bytes
270 // 33 FailureCode: 1 byte
271 // end of the Talker Failed attribute
272 
273 
274 /* description of the limited MSRPDU being supported here
275  * (currently, only one declaration per packet is supported)
276  * openavbSrpDU ::= ProtocolVersion, Message
277  * ProtocolVersion BYTE ::= 0x00
278  * Message ::= AttributeType, AttributeLength [, AttributeListLength], AttributeList
279  * AttributeType BYTE ::= Talker Advertise | Talker Failed | Listener
280  * Talker Advertise::= 0x01
281  * Talker Failed::= 0x02
282  * Listener::= 0x03
283  * AttributeLength BYTE ::= 0x19 | 0x34 | 0x08
284  * AttributeListLength SHORT ::= AttributeLength + 3 | 4 (4 only if AttributeType = Listener)
285  * AttributeList ::= VectorAttribute {, VectorAttribute}, EndMark
286  * VectorAttribute ::= VectorHeader, FirstValue, Vector
287  * VectorHeader SHORT ::= 0x0001
288  * FirstValue ::=
289  * StreamID: 8 bytes
290  * [Src] MAC: 6 bytes
291  * UniqueID: 2 bytes, U16
292  * end of the Listener attribute
293  * DataFrameParameters: 8 bytes
294  * DA: 6 bytes
295  * VID: 2 bytes
296  * TSpec: 4 bytes
297  * MaxFrameSize: 2 bytes, U16
298  * MaxIntervalFrames: 2 bytes, U16
299  * PriorityAndRank: 1 byte
300  * Priority: 3 bits
301  * Rank: 1 bit
302  * Reserved: 4 bits
303  * AccumulatedLatency: 4 bytes, U32
304  * end of the Talker Advertise attribute
305  * FailureInformation: 9 bytes
306  * BridgeID: 8 bytes
307  * FailureCode: 1 byte
308  * end of the Talker Failed attribute
309  * Vector ::= ThreePackedEvent, [FourPackedEvent]
310  * ThreePackedEvents BYTE ::= New |Lv
311  * New ::= 0
312  * Lv ::= 5
313  * FourPackedEvents BYTE ::= Asking Failed | Ready | Ready Failed
314  * (FourPackedEvents is included only if AttributeType = Listener)
315  * Asking Failed ::= 1
316  * Ready ::= 2
317  * Ready Failed ::= 3
318  */
