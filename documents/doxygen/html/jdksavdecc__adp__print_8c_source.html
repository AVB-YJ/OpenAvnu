93 void jdksavdecc_adpdu_print_common_control_header( struct jdksavdecc_printer *self,
94  struct jdksavdecc_adpdu_common_control_header const *p )
95 {
96  jdksavdecc_printer_print_label( self, "message_type" );
97  jdksavdecc_printer_print_uint16_name( self, jdksavdecc_adpdu_print_message_type, p->message_type );
98  jdksavdecc_printer_print_eol( self );
99 
100  jdksavdecc_printer_print_label( self, "valid_time (seconds)" );
101  jdksavdecc_printer_print_uint16( self, p->valid_time * 2 );
102  jdksavdecc_printer_print_eol( self );
103 
104  jdksavdecc_printer_print_label( self, "control_data_length" );
105  jdksavdecc_printer_print_uint16( self, p->control_data_length );
106  jdksavdecc_printer_print_eol( self );
107 
108  jdksavdecc_printer_print_label( self, "entity_id" );
109  jdksavdecc_printer_print_eui64( self, p->entity_id );
110  jdksavdecc_printer_print_eol( self );
111 }

113 void jdksavdecc_adpdu_print( struct jdksavdecc_printer *self, struct jdksavdecc_adpdu const *p )
114 {
115  jdksavdecc_adpdu_print_common_control_header( self, &p->header );
116 
117  if ( p->header.message_type != JDKSAVDECC_ADP_MESSAGE_TYPE_ENTITY_DISCOVER )
118  {
119  if ( p->header.message_type != JDKSAVDECC_ADP_MESSAGE_TYPE_ENTITY_DEPARTING )
120  {
121  jdksavdecc_printer_print_label( self, "available_index" );
122  jdksavdecc_printer_print_uint32( self, p->available_index );
123  jdksavdecc_printer_print_eol( self );
124  }
125 
126  jdksavdecc_printer_print_label( self, "entity_model_id" );
127  jdksavdecc_printer_print_eui64( self, p->entity_model_id );
128  jdksavdecc_printer_print_eol( self );
129 
130  jdksavdecc_printer_print_label( self, "entity_capabilities" );
131  jdksavdecc_printer_print_32bit_names( self, jdksavdecc_adpdu_print_entity_capabilities, p->entity_capabilities );
132  jdksavdecc_printer_print_eol( self );
133 
134  if ( p->talker_stream_sources > 0 && ( p->talker_capabilities & JDKSAVDECC_ADP_TALKER_CAPABILITY_IMPLEMENTED ) != 0 )
135  {
136  jdksavdecc_printer_print_label( self, "talker_stream_sources" );
137  jdksavdecc_printer_print_uint16( self, p->talker_stream_sources );
138  jdksavdecc_printer_print_eol( self );
139 
140  jdksavdecc_printer_print_label( self, "talker_capabilities" );
141  jdksavdecc_printer_print_16bit_names( self, jdksavdecc_adpdu_print_talker_capabilities, p->talker_capabilities );
142  jdksavdecc_printer_print_eol( self );
143  }
144 
145  if ( p->listener_stream_sinks > 0 )
146  {
147  jdksavdecc_printer_print_label( self, "listener_stream_sinks" );
148  jdksavdecc_printer_print_uint16( self, p->listener_stream_sinks );
149  jdksavdecc_printer_print_eol( self );
150 
151  jdksavdecc_printer_print_label( self, "listener_capabilities" );
152  jdksavdecc_printer_print_16bit_names(
153  self, jdksavdecc_adpdu_print_listener_capabilities, p->listener_capabilities );
154  jdksavdecc_printer_print_eol( self );
155  }
156 
157  if ( ( p->controller_capabilities & JDKSAVDECC_ADP_CONTROLLER_CAPABILITY_IMPLEMENTED ) != 0 )
158  {
159  jdksavdecc_printer_print_label( self, "controller_capabilities" );
160  jdksavdecc_printer_print_32bit_names(
161  self, jdksavdecc_adpdu_print_controller_capabilities, p->controller_capabilities );
162  jdksavdecc_printer_print_eol( self );
163  }
164 
165  if ( ( p->entity_capabilities & JDKSAVDECC_ADP_ENTITY_CAPABILITY_GPTP_SUPPORTED ) != 0 )
166  {
167  jdksavdecc_printer_print_label( self, "gptp_grandmaster_id" );
168  jdksavdecc_printer_print_eui64( self, p->gptp_grandmaster_id );
169  jdksavdecc_printer_print_eol( self );
170 
171  jdksavdecc_printer_print_label( self, "gptp_domain_number" );
172  jdksavdecc_printer_print_uint16( self, p->gptp_domain_number );
173  jdksavdecc_printer_print_eol( self );
174  }
175 
176  if ( ( p->entity_capabilities & JDKSAVDECC_ADP_ENTITY_CAPABILITY_AEM_IDENTIFY_CONTROL_INDEX_VALID ) != 0 )
177  {
178  jdksavdecc_printer_print_label( self, "identify_control_index" );
179  jdksavdecc_printer_print_uint16( self, p->identify_control_index );
180  jdksavdecc_printer_print_eol( self );
181  }
182 
183  if ( ( p->entity_capabilities & JDKSAVDECC_ADP_ENTITY_CAPABILITY_AEM_INTERFACE_INDEX_VALID ) != 0 )
184  {
185  jdksavdecc_printer_print_label( self, "interface_index" );
186  jdksavdecc_printer_print_uint16( self, p->interface_index );
187  jdksavdecc_printer_print_eol( self );
188  }
189 
190  if ( ( p->entity_capabilities & JDKSAVDECC_ADP_ENTITY_CAPABILITY_ASSOCIATION_ID_VALID ) != 0 )
191  {
192  jdksavdecc_printer_print_label( self, "association_id" );
193  jdksavdecc_printer_print_eui64( self, p->association_id );
194  jdksavdecc_printer_print_eol( self );
195  }
196  }
197 }
