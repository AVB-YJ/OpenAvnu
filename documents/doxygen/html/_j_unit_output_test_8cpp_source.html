92 class FileSystemForJUnitTestOutputTests
93 {

95 
96 public:
97  FileSystemForJUnitTestOutputTests() : firstFile_(0) {}
98  ~FileSystemForJUnitTestOutputTests() { clear(); }
99 
100  void clear(void)
101  {
102  while (firstFile_) {
103  FileForJUnitOutputTests* fileToBeDeleted = firstFile_;
104  firstFile_ = firstFile_->nextFile();
105  delete fileToBeDeleted;
106  }
107  }
108 
109  FileForJUnitOutputTests* openFile(const SimpleString& filename)
110  {
111  firstFile_ = new FileForJUnitOutputTests(filename, firstFile_);
112  return firstFile_;
113  }
114 
115  int amountOfFiles() {
116  int totalAmountOfFiles = 0;
117  for (FileForJUnitOutputTests* current = firstFile_; current != NULL; current = current->nextFile())
118  totalAmountOfFiles++;
119  return totalAmountOfFiles;
120  }
121 
122  bool fileExists(const char* filename)
123  {
124  FileForJUnitOutputTests *searchedFile = file(filename);
125  return (searchedFile != NULL);
126  }
127 
128  FileForJUnitOutputTests* file(const char* filename)
129  {
130  for (FileForJUnitOutputTests* current = firstFile_; current != NULL; current = current->nextFile())
131  if (current->name() == filename)
132  return current;
133  return NULL;
134  }

152 class JUnitTestOutputTestRunner
153 {
154  TestResult result_;
155 
156  const char* currentGroupName_;
157  UtestShell* currentTest_;
158  bool firstTestInGroup_;
159  int timeTheTestTakes_;
160  unsigned int numberOfChecksInTest_;
161  TestFailure* testFailure_;
162 
163 public:
164 
165  JUnitTestOutputTestRunner(TestResult result) :
166  result_(result), currentGroupName_(0), currentTest_(0), firstTestInGroup_(true), timeTheTestTakes_(0), numberOfChecksInTest_(0), testFailure_(0)
167  {
168  millisTime = 0;
169  theTime = "1978-10-03T00:00:00";
170 
171  UT_PTR_SET(GetPlatformSpecificTimeInMillis, MockGetPlatformSpecificTimeInMillis);
172  UT_PTR_SET(GetPlatformSpecificTimeString, MockGetPlatformSpecificTimeString);
173  }
174 
175  JUnitTestOutputTestRunner& start()
176  {
177  result_.testsStarted();
178  return *this;
179  }
180 
181  JUnitTestOutputTestRunner& end()
182  {
183  endOfPreviousTestGroup();
184  delete currentTest_;
185  result_.testsEnded();
186  return *this;
187  }
188 
189  JUnitTestOutputTestRunner& endGroupAndClearTest()
190  {
191  endOfPreviousTestGroup();
192  delete currentTest_;
193  currentTest_ = 0;
194  return *this;
195  }
196 
197  void endOfPreviousTestGroup()
198  {
199  runPreviousTest();
200  if (currentTest_) {
201  result_.currentGroupEnded(currentTest_);
202  firstTestInGroup_ = true;
203  }
204 
205  currentGroupName_ = 0;
206  }
207 
208  JUnitTestOutputTestRunner& withGroup(const char* groupName)
209  {
210  runPreviousTest();
211  endOfPreviousTestGroup();
212 
213  currentGroupName_ = groupName;
214  return *this;
215  }
216 
217  JUnitTestOutputTestRunner& withTest(const char* testName)
218  {
219  runPreviousTest();
220  delete currentTest_;
221 
222  currentTest_ = new UtestShell(currentGroupName_, testName, "file", 1);
223  return *this;
224  }
225 
226  JUnitTestOutputTestRunner& withIgnoredTest(const char* testName)
227  {
228  runPreviousTest();
229  delete currentTest_;
230 
231  currentTest_ = new IgnoredUtestShell(currentGroupName_, testName, "file", 1);
232  return *this;
233  }
234 
235  JUnitTestOutputTestRunner& inFile(const char* fileName)
236  {
237  if(currentTest_) {
238  currentTest_->setFileName(fileName);
239  }
240  return *this;
241  }
242 
243  JUnitTestOutputTestRunner& onLine(int lineNumber)
244  {
245  if(currentTest_) {
246  currentTest_->setLineNumber(lineNumber);
247  }
248  return *this;
249  }
250 
251  void runPreviousTest()
252  {
253  if (currentTest_ == 0) return;
254 
255  if (firstTestInGroup_) {
256  result_.currentGroupStarted(currentTest_);
257  firstTestInGroup_ = false;
258  }
259  result_.currentTestStarted(currentTest_);
260 
261  millisTime += timeTheTestTakes_;
262  for(unsigned int i = 0; i < numberOfChecksInTest_; i++) {
263  result_.countCheck();
264  }
265  numberOfChecksInTest_ = 0;
266 
267  if (testFailure_) {
268  result_.addFailure(*testFailure_);
269  delete testFailure_;
270  testFailure_ = 0;
271  }
272 
273  result_.currentTestEnded(currentTest_);
274  }
275 
276  JUnitTestOutputTestRunner& thatHasChecks(unsigned int numOfChecks)
277  {
278  numberOfChecksInTest_ = numOfChecks;
279  return *this;
280  }
281 
282  JUnitTestOutputTestRunner& thatTakes(int timeElapsed)
283  {
284  timeTheTestTakes_ = timeElapsed;
285  return *this;
286  }
287 
288  JUnitTestOutputTestRunner& seconds()
289  {
290  return *this;
291  }
292 
293  JUnitTestOutputTestRunner& thatFails(const char* message, const char* file, int line)
294  {
295  testFailure_ = new TestFailure( currentTest_, file, line, message);
296  return *this;
297  }
298 
299  JUnitTestOutputTestRunner& atTime(const char* newTime)
300  {
301  theTime = newTime;
302  return *this;
303  }
304 
305  JUnitTestOutputTestRunner& thatPrints(const char* output)
306  {
307  runPreviousTest();
308  result_.print(output);
309  return *this;
310  }
