55 def fit_line(count, plots, file1):
56  # see http://en.wikipedia.org/wiki/Simple_linear_regression
57  SX = 0.0
58  SY = 0.0
59  SXX = 0.0
60  SXY = 0.0
61  n = 0.0
62  res = {}
63  with open(file1, 'rU') as csvfile:
64  vals = list(csv.reader(csvfile))
65  y_prev = float(vals[0][1])
66  for s in vals[0:count]:
67  x = float(s[0])
68  y = float(s[1])
69  y_delta = y - y_prev
70  if (y_delta > 200000):
71  print "### Error ### Timestamp step delta in %s and index %d of %f" % (file1, n, y_delta)
72  SX = SX + x
73  SXX = SXX + (x * x)
74  SY = SY + y
75  SXY = SXY + (x * y)
76  y_prev = y
77  n = n + 1
78  m = (n * SXY - SX * SY)/(n * SXX - SX * SX)
79  b = SY/n - m * SX/n
80  x_vals = np.array([float(vals[i][0]) for i in range(count)])
81  y_line_array = x_vals * m + b
82  y_array = np.array([float(vals[i][1]) for i in range(count)])
83  line_deviation = y_line_array - y_array
84  max_error = np.amax(abs(line_deviation))
85  rnd_max_error = np.int_(max_error)
86  histogram_bins = range((rnd_max_error + 1) * 2) - (rnd_max_error + 1);
87  hist, bins = np.histogram(line_deviation, histogram_bins)
88  res['histogram'] = hist
89  res['bins'] = bins
90  res['file'] = file1
91  res['line_slope'] = m
92  res['line_y_intercept'] = b
93  res['line_max deviataion'] = max_error
94  res['astimestamps'] = y_array
95  if plots == True:
96  if True: # Distribution bar graph, switch to True to run
97  figh = plt.figure()
98  center = (bins[:-1]+bins[1:])/2
99  plt.bar(center, hist, align = 'center', width = 0.7)
100  plt.xlabel('deviation from fitted line in ns')
101  plt.ylabel('count')
102  plt.title(file1 + ' AS timestamp distribution\nof deviations from fitted line')
103  res['hist_fig'] = figh
104 
105  if False: # Fitted line plot, switch to True to run
106  figl = plt.figure()
107  plt.plot(range(count), y_array, 'ro', range(count), y_line_array, 'b-')
108  plt.xlabel('AS timestamp count')
109  plt.ylabel('AS timestamp')
110  plt.title(file1 + ' fitted line')
111  res['line_fig'] = figl
112 
113  if True: # Deviation from fitted line plot, switch to True to run
114  figl = plt.figure()
115  plt.plot(range(count), line_deviation, 'ro')
116  plt.xlabel('AS timestamp count')
117  plt.ylabel('AS timestamp deviation')
118  plt.title(file1 + ' deviation from fitted line')
119  res['line_err'] = figl
120 
121  if count > 32768:
122  if False: # Frequency transform of deviation from fitted line, switch to True to run
123  figf = plt.figure()
124  fft_count = 32678
125  # fft bin spacing is Fs/N, where Fs = 48000/8 and N = fft_count
126  fft_bin_spacing = float(48000/8) / float(fft_count)
127  freq_bins = [float(i) * fft_bin_spacing for i in range(fft_count/2 + 1)]
128  FFT = abs(np.fft.rfft(line_deviation[range(fft_count)]))
129  plt.plot(freq_bins, np.log10(FFT))
130  plt.xlabel('Hz')
131  plt.title(file1 + ' deviation periodicity')
132  res['fft_fig'] = figf
133 
134  csvfile.close()
135  return res
