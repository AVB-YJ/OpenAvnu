208 static inline ssize_t jdksavdecc_jdks_log_control_generate( struct jdksavdecc_eui64 const *my_entity_id,
209  uint16_t descriptor_index,
210  uint16_t *sequence_id,
211  uint8_t log_detail,
212  uint8_t reserved,
213  const char *utf8_log_string,
214  void *buf,
215  size_t len )
216 {
217  int pos = 14;
218  ssize_t text_len = (ssize_t)strlen( utf8_log_string );
219  ssize_t expected_length = JDKSAVDECC_JDKS_LOG_CONTROL_HEADER_LEN + text_len;
220  ssize_t r = jdksavdecc_validate_range( pos, len, expected_length );
221  if ( r >= 0 )
222  {
223  struct jdksavdecc_aem_command_set_control_response setctrl;
224  setctrl.aem_header.aecpdu_header.header.cd = true;
225  setctrl.aem_header.aecpdu_header.header.subtype = JDKSAVDECC_SUBTYPE_AECP;
226  setctrl.aem_header.aecpdu_header.header.sv = false;
227  setctrl.aem_header.aecpdu_header.header.version = 0;
228  setctrl.aem_header.aecpdu_header.header.message_type = JDKSAVDECC_AECP_MESSAGE_TYPE_AEM_RESPONSE;
229  setctrl.aem_header.aecpdu_header.header.status = JDKSAVDECC_AEM_STATUS_SUCCESS;
230  setctrl.aem_header.aecpdu_header.header.control_data_length
231  = (uint16_t)text_len + JDKSAVDECC_JDKS_LOG_CONTROL_HEADER_LEN - JDKSAVDECC_COMMON_CONTROL_HEADER_LEN + 2;
232  setctrl.aem_header.aecpdu_header.header.target_entity_id = *my_entity_id;
233  setctrl.aem_header.aecpdu_header.controller_entity_id = jdksavdecc_jdks_notifications_controller_entity_id;
234  setctrl.aem_header.aecpdu_header.sequence_id = *sequence_id;
235  setctrl.aem_header.command_type = JDKSAVDECC_AEM_COMMAND_SET_CONTROL | 0x8000;
236  setctrl.descriptor_type = JDKSAVDECC_DESCRIPTOR_CONTROL;
237  setctrl.descriptor_index = descriptor_index;
238  jdksavdecc_aem_command_set_control_response_write( &setctrl, buf, pos, len );
239  jdksavdecc_eui64_write(
240  &jdksavdecc_jdks_aem_control_log_text, buf, pos + JDKSAVDECC_JDKS_LOG_CONTROL_OFFSET_VENDOR_EUI64, len );
241  jdksavdecc_uint32_write( (uint32_t)text_len + 2, buf, pos + JDKSAVDECC_JDKS_LOG_CONTROL_OFFSET_BLOB_SIZE, len );
242  jdksavdecc_uint8_write( log_detail, buf, pos + JDKSAVDECC_JDKS_LOG_CONTROL_OFFSET_LOG_DETAIL, len );
243  jdksavdecc_uint8_write( reserved, buf, pos + JDKSAVDECC_JDKS_LOG_CONTROL_OFFSET_RESERVED, len );
244  memcpy( ( (uint8_t *)buf ) + pos + JDKSAVDECC_JDKS_LOG_CONTROL_OFFSET_TEXT, utf8_log_string, text_len );
245  memcpy( ( (uint8_t *)buf ) + JDKSAVDECC_FRAME_HEADER_DA_OFFSET, jdksavdecc_jdks_multicast_log.value, 6 );
246  jdksavdecc_uint16_set( JDKSAVDECC_AVTP_ETHERTYPE, buf, JDKSAVDECC_FRAME_HEADER_ETHERTYPE_OFFSET );
247 
248  *sequence_id = *sequence_id + 1;
249  }
250  return r;
251 }

354 static inline bool jdksavdecc_jdks_is_log( struct jdksavdecc_jdks_log_control *p, void const *buf, size_t len )
355 {
356  bool is_log_response = false;
357  int pos = 14;
358  ssize_t r = jdksavdecc_validate_range( pos, len, JDKSAVDECC_JDKS_LOG_CONTROL_HEADER_LEN );
359  if ( r >= 0 )
360  {
361  jdksavdecc_aem_command_set_control_response_read( &p->cmd, buf, pos, len );
362  jdksavdecc_eui64_read( &p->vendor_eui64, buf, JDKSAVDECC_JDKS_LOG_CONTROL_OFFSET_VENDOR_EUI64 + pos, len );
363  if ( jdksavdecc_eui64_compare( &jdksavdecc_jdks_aem_control_log_text, &p->vendor_eui64 ) == 0 )
364  {
365  jdksavdecc_uint32_read( &p->blob_size, buf, JDKSAVDECC_JDKS_LOG_CONTROL_OFFSET_BLOB_SIZE + pos, len );
366  if ( p->blob_size > ( JDKSAVDECC_JDKS_LOG_CONTROL_OFFSET_TEXT - JDKSAVDECC_AEM_COMMAND_SET_CONTROL_RESPONSE_LEN )
367  && p->blob_size <= JDKSAVDECC_AEM_CONTROL_VALUE_TYPE_BLOB_MAX_SIZE )
368  {
369  is_log_response = true;
370  }
371  }
372  }
373  return is_log_response;
