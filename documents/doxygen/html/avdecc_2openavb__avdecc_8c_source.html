141 bool openavbAvdeccAddConfiguration(openavb_tl_data_cfg_t *stream)
142 {
143  bool first_time = 0;
144  // Create a new config to hold the configuration information.
145  openavb_avdecc_configuration_cfg_t *pCfg = malloc(sizeof(openavb_avdecc_configuration_cfg_t));
146  if (!pCfg) {
147  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
148  return FALSE;
149  }
150  memset(pCfg, 0, sizeof(openavb_avdecc_configuration_cfg_t));
151 
152  // Add a pointer to the supplied stream information.
153  pCfg->stream = stream;
154 
155  // Add the new config to the end of the list of configurations.
156  if (pFirstConfigurationCfg == NULL) {
157  pFirstConfigurationCfg = pCfg;
158  } else {
159  openavb_avdecc_configuration_cfg_t *pLast = pFirstConfigurationCfg;
160  while (pLast->next != NULL) {
161  pLast = pLast->next;
162  }
163  pLast->next = pCfg;
164  }
165 
166  // Create a new configuration.
167  U16 nConfigIdx = 0;
168  if (pConfiguration == NULL)
169  {
170  first_time = 1;
171  pConfiguration = openavbAemDescriptorConfigurationNew();
172  if (!openavbAemAddDescriptor(pConfiguration, OPENAVB_AEM_DESCRIPTOR_INVALID, &nConfigIdx)) {
173  AVB_LOG_ERROR("Error adding AVDECC configuration");
174  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
175  return FALSE;
176  }
177  }
178  // Specify a default user-friendly name to use.
179  // AVDECC_TODO - Allow the user to specify a friendly name, or use the name if the .INI file.
180  if (pCfg->friendly_name[0] == '\0') {
181  snprintf((char *) pCfg->friendly_name, OPENAVB_AEM_STRLEN_MAX, "Configuration %u", nConfigIdx);
182  }
183 
184  // Save the stream information in the configuration.
185  if (!openavbAemDescriptorConfigurationInitialize(pConfiguration, nConfigIdx, pCfg)) {
186  AVB_LOG_ERROR("Error initializing AVDECC configuration");
187  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
188  return FALSE;
189  }
190 
191  // Add the descriptors needed for both talkers and listeners.
192  U16 nResultIdx;
193  if (first_time)
194  {
195  openavb_aem_descriptor_avb_interface_t *pNewAvbInterface = openavbAemDescriptorAvbInterfaceNew();
196  if (!openavbAemAddDescriptor(pNewAvbInterface, nConfigIdx, &nResultIdx) ||
197  !openavbAemDescriptorAvbInterfaceInitialize(pNewAvbInterface, nConfigIdx, pCfg)) {
198  AVB_LOG_ERROR("Error adding AVDECC AVB Interface to configuration");
199  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
200  return FALSE;
201  }
202  openavb_aem_descriptor_audio_unit_t *pNewAudioUnit = openavbAemDescriptorAudioUnitNew();
203  if (!openavbAemAddDescriptor(pNewAudioUnit, nConfigIdx, &nResultIdx) ||
204  !openavbAemDescriptorAudioUnitInitialize(pNewAudioUnit, nConfigIdx, pCfg)) {
205  AVB_LOG_ERROR("Error adding AVDECC Audio Unit to configuration");
206  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
207  return FALSE;
208  }
209  }
210  else
211  {
212  openavb_aem_descriptor_audio_unit_t *pAudioUnitDescriptor =
213  (openavb_aem_descriptor_audio_unit_t *) openavbAemGetDescriptor(nConfigIdx, OPENAVB_AEM_DESCRIPTOR_AUDIO_UNIT, 0);
214  if (pAudioUnitDescriptor != NULL)
215  {
216  if (!openavbAemDescriptorAudioUnitInitialize(pAudioUnitDescriptor, nConfigIdx, pCfg)) {
217  AVB_LOG_ERROR("Error updating AVDECC Audio Unit to configuration");
218  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
219  return FALSE;
220  }
221  }
222  else
223  {
224  AVB_LOG_ERROR("Error getting AVDECC Audio Unit descriptor");
225  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
226  return FALSE;
227  }
228  }
229 
230  // AVDECC_TODO: Add other descriptors as needed. Future options include:
231  // VIDEO_UNIT
232  // SENSOR_UNIT
233  // CONTROL
234 
235  if (stream->role == AVB_ROLE_TALKER) {
236  gAvdeccCfg.bTalker = TRUE;
237 
238  openavb_aem_descriptor_stream_io_t *pNewStreamOutput = openavbAemDescriptorStreamOutputNew();
239  if (!openavbAemAddDescriptor(pNewStreamOutput, nConfigIdx, &nResultIdx) ||
240  !openavbAemDescriptorStreamOutputInitialize(pNewStreamOutput, nConfigIdx, pCfg)) {
241  AVB_LOG_ERROR("Error adding AVDECC Stream Output to configuration");
242  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
243  return FALSE;
244  }
245  if (first_talker)
246  {
247  first_talker = 0;
248  openavb_aem_descriptor_clock_source_t *pNewClockSource = openavbAemDescriptorClockSourceNew();
249  if (!openavbAemAddDescriptor(pNewClockSource, nConfigIdx, &nResultIdx) ||
250  !openavbAemDescriptorClockSourceInitialize(pNewClockSource, nConfigIdx, pCfg)) {
251  AVB_LOG_ERROR("Error adding AVDECC Clock Source to configuration");
252  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
253  return FALSE;
254  }
255  openavb_aem_descriptor_clock_domain_t *pNewClockDomain = openavbAemDescriptorClockDomainNew();
256  if (!openavbAemAddDescriptor(pNewClockDomain, nConfigIdx, &nResultIdx) ||
257  !openavbAemDescriptorClockDomainInitialize(pNewClockDomain, nConfigIdx, pCfg)) {
258  AVB_LOG_ERROR("Error adding AVDECC Clock Domain to configuration");
259  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
260  return FALSE;
261  }
262  }
263 
264  // AVDECC_TODO: Add other descriptors as needed. Future options include:
265  // JACK_INPUT
266  talker_stream_sources++;
267 
268  // Add the class specific to the talker.
269  if (stream->sr_class == SR_CLASS_A) { gAvdeccCfg.bClassASupported = TRUE; }
270  if (stream->sr_class == SR_CLASS_B) { gAvdeccCfg.bClassBSupported = TRUE; }
271 
272  AVB_LOG_DEBUG("AVDECC talker configuration added");
273  }
274  if (stream->role == AVB_ROLE_LISTENER) {
275  gAvdeccCfg.bListener = TRUE;
276 
277  openavb_aem_descriptor_stream_io_t *pNewStreamInput = openavbAemDescriptorStreamInputNew();
278  if (!openavbAemAddDescriptor(pNewStreamInput, nConfigIdx, &nResultIdx) ||
279  !openavbAemDescriptorStreamInputInitialize(pNewStreamInput, nConfigIdx, pCfg)) {
280  AVB_LOG_ERROR("Error adding AVDECC Stream Input to configuration");
281  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
282  return FALSE;
283  }
284  if (first_listener)
285  {
286  openavb_aem_descriptor_clock_source_t *pNewClockSource = openavbAemDescriptorClockSourceNew();
287  if (!openavbAemAddDescriptor(pNewClockSource, nConfigIdx, &nResultIdx) ||
288  !openavbAemDescriptorClockSourceInitialize(pNewClockSource, nConfigIdx, pCfg)) {
289  AVB_LOG_ERROR("Error adding AVDECC Clock Source to configuration");
290  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
291  return FALSE;
292  }
293  openavb_aem_descriptor_clock_domain_t *pNewClockDomain = openavbAemDescriptorClockDomainNew();
294  if (!openavbAemAddDescriptor(pNewClockDomain, nConfigIdx, &nResultIdx) ||
295  !openavbAemDescriptorClockDomainInitialize(pNewClockDomain, nConfigIdx, pCfg)) {
296  AVB_LOG_ERROR("Error adding AVDECC Clock Domain to configuration");
297  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
298  return FALSE;
299  }
300  }
301 
302  // AVDECC_TODO: Add other descriptors as needed. Future options include:
303  // JACK_OUTPUT
304  listener_stream_sources++;
305 
306  // Listeners support both Class A and Class B.
307  gAvdeccCfg.bClassASupported = TRUE;
308  gAvdeccCfg.bClassBSupported = TRUE;
309 
310  AVB_LOG_DEBUG("AVDECC listener configuration added");
311  }
312 
313  if (first_time)
314  {
315  // Add the localized strings to the configuration.
316  if (!openavbAemDescriptorLocaleStringsHandlerAddToConfiguration(gAvdeccCfg.pAemDescriptorLocaleStringsHandler, nConfigIdx)) {
317  AVB_LOG_ERROR("Error adding AVDECC locale strings to configuration");
318  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
319  return FALSE;
320  }
321  }
322 
323  return TRUE;
324 }

330 extern DLL_EXPORT bool openavbAvdeccInitialize()
331 {
332  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC);
333 
334  gAvdeccCfg.pDescriptorEntity = openavbAemDescriptorEntityNew();
335  if (!gAvdeccCfg.pDescriptorEntity) {
336  AVB_LOG_ERROR("Failed to allocate an AVDECC descriptor");
337  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
338  return FALSE;
339  }
340 
341  openavbAvdeccFindMacAddr();
342 
343  if (!openavbAemDescriptorEntitySet_entity_id(gAvdeccCfg.pDescriptorEntity, NULL, gAvdeccCfg.ifmac, gAvdeccCfg.avdeccId)) {
344  AVB_LOG_ERROR("Failed to set the AVDECC descriptor");
345  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
346  return FALSE;
347  }
348 
349  // Create the Entity Model
350  openavbRC rc = openavbAemCreate(gAvdeccCfg.pDescriptorEntity);
351  if (IS_OPENAVB_FAILURE(rc)) {
352  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
353  return FALSE;
354  }
355 
356  // Copy the supplied non-localized strings to the descriptor.
357  openavbAemDescriptorEntitySet_entity_model_id(gAvdeccCfg.pDescriptorEntity, gAvdeccCfg.entity_model_id);
358  openavbAemDescriptorEntitySet_entity_name(gAvdeccCfg.pDescriptorEntity, gAvdeccCfg.entity_name);
359  openavbAemDescriptorEntitySet_firmware_version(gAvdeccCfg.pDescriptorEntity, gAvdeccCfg.firmware_version);
360  openavbAemDescriptorEntitySet_group_name(gAvdeccCfg.pDescriptorEntity, gAvdeccCfg.group_name);
361  openavbAemDescriptorEntitySet_serial_number(gAvdeccCfg.pDescriptorEntity, gAvdeccCfg.serial_number);
362 
363  // Initialize the localized strings support.
364  gAvdeccCfg.pAemDescriptorLocaleStringsHandler = openavbAemDescriptorLocaleStringsHandlerNew();
365  if (gAvdeccCfg.pAemDescriptorLocaleStringsHandler) {
366  // Add the strings to the locale strings hander.
367  openavbAemDescriptorLocaleStringsHandlerSet_local_string(
368  gAvdeccCfg.pAemDescriptorLocaleStringsHandler, gAvdeccCfg.locale_identifier, gAvdeccCfg.vendor_name, LOCALE_STRING_VENDOR_NAME_INDEX);
369  openavbAemDescriptorLocaleStringsHandlerSet_local_string(
370  gAvdeccCfg.pAemDescriptorLocaleStringsHandler, gAvdeccCfg.locale_identifier, gAvdeccCfg.model_name, LOCALE_STRING_MODEL_NAME_INDEX);
371 
372  // Have the descriptor entity reference the locale strings.
373  openavbAemDescriptorEntitySet_vendor_name(gAvdeccCfg.pDescriptorEntity, 0, LOCALE_STRING_VENDOR_NAME_INDEX);
374  openavbAemDescriptorEntitySet_model_name(gAvdeccCfg.pDescriptorEntity, 0, LOCALE_STRING_MODEL_NAME_INDEX);
375  }
376 
377  gAvdeccCfg.bTalker = gAvdeccCfg.bListener = FALSE;
378 
379  // Add a configuration for each talker or listener stream.
380  openavb_tl_data_cfg_t *current_stream = streamList;
381  while (current_stream != NULL) {
382  // Create a new configuration with the information from this stream.
383  if (!openavbAvdeccAddConfiguration(current_stream)) {
384  AVB_LOG_ERROR("Error adding AVDECC configuration");
385  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
386  return FALSE;
387  }
388 
389  // Proceed to the next stream.
390  current_stream = current_stream->next;
391  }
392 
393  if (!gAvdeccCfg.bTalker && !gAvdeccCfg.bListener) {
394  AVB_LOG_ERROR("No AVDECC Configurations -- Aborting");
395  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
396  return FALSE;
397  }
398 
399  // Add non-top-level descriptors. These are independent of the configurations.
400  // STRINGS are handled by gAvdeccCfg.pAemDescriptorLocaleStringsHandler, so not included here.
401  U16 nResultIdx;
402  if (!openavbAemAddDescriptor(openavbAemDescriptorAudioClusterNew(), OPENAVB_AEM_DESCRIPTOR_INVALID, &nResultIdx)) {
403  AVB_LOG_ERROR("Error adding AVDECC Audio Cluster");
404  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
405  return FALSE;
406  }
407  if (gAvdeccCfg.bTalker) {
408  if (!openavbAemAddDescriptor(openavbAemDescriptorStreamPortOutputNew(), OPENAVB_AEM_DESCRIPTOR_INVALID, &nResultIdx)) {
409  AVB_LOG_ERROR("Error adding AVDECC Output Stream Port");
410  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
411  return FALSE;
412  }
413  }
414  if (gAvdeccCfg.bListener) {
415  if (!openavbAemAddDescriptor(openavbAemDescriptorStreamPortInputNew(), OPENAVB_AEM_DESCRIPTOR_INVALID, &nResultIdx)) {
416  AVB_LOG_ERROR("Error adding AVDECC Input Stream Port");
417  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
418  return FALSE;
419  }
420  }
421 
422  // AVDECC_TODO: Add other descriptors as needed. Future options include:
423  // EXTERNAL_PORT_INPUT
424  // EXTERNAL_PORT_OUTPUT
425  // INTERNAL_PORT_INPUT
426  // INTERNAL_PORT_OUTPUT
427  // VIDEO_CLUSTER
428  // SENSOR_CLUSTER
429  // AUDIO_MAP
430  // VIDEO_MAP
431  // SENSOR_MAP
432 
433  // Fill in the descriptor capabilities.
434  // AVDECC_TODO: Set these based on the available capabilities.
435  if (!gAvdeccCfg.bClassASupported && !gAvdeccCfg.bClassBSupported) {
436  // If the user didn't specify a traffic class, assume both are supported.
437  gAvdeccCfg.bClassASupported = gAvdeccCfg.bClassBSupported = TRUE;
438  }
439  openavbAemDescriptorEntitySet_entity_capabilities(gAvdeccCfg.pDescriptorEntity,
440  OPENAVB_ADP_ENTITY_CAPABILITIES_AEM_SUPPORTED |
441  (gAvdeccCfg.bClassASupported ? OPENAVB_ADP_ENTITY_CAPABILITIES_CLASS_A_SUPPORTED : 0) |
442  (gAvdeccCfg.bClassBSupported ? OPENAVB_ADP_ENTITY_CAPABILITIES_CLASS_B_SUPPORTED : 0) |
443  OPENAVB_ADP_ENTITY_CAPABILITIES_GPTP_SUPPORTED);
444 
445  if (gAvdeccCfg.bTalker) {
446  // AVDECC_TODO: Set these based on the available capabilities.
447  openavbAemDescriptorEntitySet_talker_capabilities(gAvdeccCfg.pDescriptorEntity, talker_stream_sources,
448  OPENAVB_ADP_TALKER_CAPABILITIES_IMPLEMENTED |
449  OPENAVB_ADP_TALKER_CAPABILITIES_AUDIO_SOURCE |
450  OPENAVB_ADP_TALKER_CAPABILITIES_MEDIA_CLOCK_SOURCE);
451  }
452  if (gAvdeccCfg.bListener) {
453  // AVDECC_TODO: Set these based on the available capabilities.
454  openavbAemDescriptorEntitySet_listener_capabilities(gAvdeccCfg.pDescriptorEntity, listener_stream_sources,
455  OPENAVB_ADP_LISTENER_CAPABILITIES_IMPLEMENTED |
456  OPENAVB_ADP_LISTENER_CAPABILITIES_AUDIO_SINK);
457  }
458 
459  AVB_TRACE_EXIT(AVB_TRACE_AVDECC);
460  return TRUE;
461 }
