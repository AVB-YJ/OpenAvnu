60  AVB_LOGF_ERROR("Failed to parse addr: %s", str);
61  return FALSE;

70 static int cfgCallback(void *user, const char *section, const char *name, const char *value)
71 {
72  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
73 
74  if (!user || !section || !name || !value) {
75  AVB_LOG_ERROR("Config: invalid arguments passed to callback");
76  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
77  return 0;
78  }
79 
80  openavb_endpoint_cfg_t *pCfg = (openavb_endpoint_cfg_t*)user;
81 
82  AVB_LOGF_DEBUG("name=[%s] value=[%s]", name, value);
83 
84  bool valOK = FALSE;
85  char *pEnd;
86 
87  if (MATCH(section, "network"))
88  {
89  if (MATCH(name, "ifname"))
90  {
91  if_info_t ifinfo;
92  if (openavbCheckInterface(value, &ifinfo)) {
93  strncpy(pCfg->ifname, value, sizeof(pCfg->ifname) - 1);
94  memcpy(pCfg->ifmac, &ifinfo.mac, ETH_ALEN);
95  pCfg->ifindex = ifinfo.index;
96  pCfg->mtu = ifinfo.mtu;
97  valOK = TRUE;
98  }
99  }
100  else if (MATCH(name, "link_kbit")) {
101  errno = 0;
102  pCfg->link_kbit = strtoul(value, &pEnd, 10);
103  if (*pEnd == '\0' && errno == 0)
104  valOK = TRUE;
105  }
106  else if (MATCH(name, "nsr_kbit")) {
107  errno = 0;
108  pCfg->nsr_kbit = strtoul(value, &pEnd, 10);
109  if (*pEnd == '\0' && errno == 0)
110  valOK = TRUE;
111  }
112  else {
113  // unmatched item, fail
114  AVB_LOGF_ERROR("Unrecognized configuration item: section=%s, name=%s", section, name);
115  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
116  return 0;
117  }
118  }
119  else if (MATCH(section, "ptp"))
120  {
121  if (MATCH(name, "start_options")) {
122  pCfg->ptp_start_opts = strdup(value);
123  valOK = TRUE;
124  }
125  else {
126  // unmatched item, fail
127  AVB_LOGF_ERROR("Unrecognized configuration item: section=%s, name=%s", section, name);
128  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
129  return 0;
130  }
131  }
132  else if (MATCH(section, "fqtss"))
133  {
134  if (MATCH(name, "mode")) {
135  errno = 0;
136  pCfg->fqtss_mode = strtoul(value, &pEnd, 10);
137  if (*pEnd == '\0' && errno == 0)
138  valOK = TRUE;
139  }
140  else {
141  // unmatched item, fail
142  AVB_LOGF_ERROR("Unrecognized configuration item: section=%s, name=%s", section, name);
143  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
144  return 0;
145  }
146  }
147  else if (MATCH(section, "srp"))
148  {
149  if (MATCH(name, "preconfigured")) {
150  errno = 0;
151  unsigned temp = strtoul(value, &pEnd, 10);
152  if (*pEnd == '\0' && errno == 0) {
153  valOK = TRUE;
154  if (temp == 1)
155  pCfg->noSrp = TRUE;
156  else
157  pCfg->noSrp = FALSE;
158  }
159  }
160  else if (MATCH(name, "gptp_asCapable_not_required")) {
161  errno = 0;
162  unsigned temp = strtoul(value, &pEnd, 10);
163  if (*pEnd == '\0' && errno == 0) {
164  valOK = TRUE;
165  if (temp == 1)
166  pCfg->bypassAsCapableCheck = TRUE;
167  else
168  pCfg->bypassAsCapableCheck = FALSE;
169  }
170  }
171  else {
172  // unmatched item, fail
173  AVB_LOGF_ERROR("Unrecognized configuration item: section=%s, name=%s", section, name);
174  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
175  return 0;
176  }
177  }
178  else if (MATCH(section, "maap"))
179  {
180  if (MATCH(name, "port")) {
181  errno = 0;
182  unsigned temp = strtoul(value, &pEnd, 10);
183  if (*pEnd == '\0' && errno == 0) {
184  if (temp >= 1 && temp <= 65535) {
185  pCfg->maapPort = temp;
186  valOK = TRUE;
187  }
188  }
189  }
190  else {
191  // unmatched item, fail
192  AVB_LOGF_ERROR("Unrecognized configuration item: section=%s, name=%s", section, name);
193  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
194  return 0;
195  }
196  }
197  else if (MATCH(section, "shaper"))
198  {
199  if (MATCH(name, "port")) {
200  errno = 0;
201  unsigned temp = strtoul(value, &pEnd, 10);
202  if (*pEnd == '\0' && errno == 0) {
203  if (temp >= 1 && temp <= 65535) {
204  pCfg->shaperPort = temp;
205  valOK = TRUE;
206  }
207  }
208  }
209  else {
210  // unmatched item, fail
211  AVB_LOGF_ERROR("Unrecognized configuration item: section=%s, name=%s", section, name);
212  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
213  return 0;
214  }
215  }
216 
217  // Special section to load saved settings.
218  else if (MATCH(section, "saved"))
219  {
220  if (MATCH(name, "maap_preferred"))
221  {
222  valOK = parse_mac(value, &(pCfg->maap_preferred));
223  }
224  else {
225  // unmatched item, fail
226  AVB_LOGF_ERROR("Unrecognized configuration item: section=%s, name=%s", section, name);
227  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
228  return 0;
229  }
230  }
231 
232  else {
233  // unmatched item, fail
234  AVB_LOGF_ERROR("Unrecognized configuration item: section=%s, name=%s", section, name);
235  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
236  return 0;
237  }
238 
239  if (!valOK) {
240  cfgValErr(section, name, value);
241  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
242  return 0;
243  }
244 
245  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
246  return 1; // OK
247 }

249 // Parse ini file, and create config data
250 //
251 int openavbReadConfig(const char *ini_file, const char *save_ini_file, openavb_endpoint_cfg_t *pCfg)
252 {
253  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
254 
255  // defaults - most are handled by setting everything to 0
256  memset(pCfg, 0, sizeof(openavb_endpoint_cfg_t));
257  pCfg->fqtss_mode = -1;
258 
259  int result = ini_parse(ini_file, cfgCallback, pCfg);
260  if (result < 0) {
261  AVB_LOGF_ERROR("Couldn't parse INI file: %s", ini_file);
262  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
263  return -1;
264  }
265  if (result > 0) {
266  AVB_LOGF_ERROR("Error in INI file: %s, line %d", ini_file, result);
267  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
268  return -1;
269  }
270 
271  result = ini_parse(save_ini_file, cfgCallback, pCfg);
272  if (result > 0) {
273  AVB_LOGF_ERROR("Error in INI file: %s, line %d", ini_file, result);
274  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
275  return -1;
276  }
277 
278  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
279 
280  // Yay, we did it.
281  return 0;
282 }

286 int openavbSaveConfig(const char *ini_file, const openavb_endpoint_cfg_t *pCfg)
287 {
288  AVB_TRACE_ENTRY(AVB_TRACE_ENDPOINT);
289 
290  FILE* file;
291 
292  char *pvtFilename = strdup(ini_file);
293  if (!pvtFilename) {
294  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
295  return -1;
296  }
297 
298  char *override = strchr(pvtFilename, ',');
299  if (override)
300  *override++ = '\0';
301 
302  file = fopen(pvtFilename, "w");
303  if (!file) {
304  AVB_LOGF_WARNING("Error saving to INI file: %s", ini_file);
305  free (pvtFilename);
306  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
307  return -1;
308  }
309 
310  if (fputs("[saved]\n", file) < 0) {
311  AVB_LOGF_ERROR("Error writing to INI file: %s", ini_file);
312  fclose(file);
313  free (pvtFilename);
314  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
315  return -1;
316  }
317  if (fprintf(file, "maap_preferred = " ETH_FORMAT "\n", ETH_OCTETS(pCfg->maap_preferred.ether_addr_octet)) < 0) {
318  AVB_LOGF_ERROR("Error writing to INI file: %s", ini_file);
319  fclose(file);
320  free (pvtFilename);
321  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
322  return -1;
323  }
324 
325  fclose(file);
326  free (pvtFilename);
327 
328  AVB_LOGF_DEBUG("Saved settings to INI file: %s", ini_file);
329  AVB_TRACE_EXIT(AVB_TRACE_ENDPOINT);
330 
331  // Yay, we did it.
332  return 0;
333 }
