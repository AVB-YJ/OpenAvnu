49 bool openavbTLRunListenerInit(int hnd, AVBStreamID_t *streamID)
50 {
51  tl_state_t *pTLState = TLHandleListGet(hnd);
52  openavb_tl_cfg_t *pCfg = &pTLState->cfg;
53  listener_data_t *pListenerData = pTLState->pPvtListenerData;
54 
55  strncpy(pListenerData->ifname, pCfg->ifname, sizeof(pListenerData->ifname) - 1);
56  memcpy(&pListenerData->streamID.addr, &pCfg->stream_addr.mac->ether_addr_octet, ETH_ALEN);
57  pListenerData->streamID.uniqueID = pCfg->stream_uid;
58  memcpy(&pListenerData->destAddr, &pCfg->dest_addr.mac->ether_addr_octet, ETH_ALEN);
59  pListenerData->tSpec.maxIntervalFrames = pCfg->max_interval_frames;
60  pListenerData->tSpec.maxFrameSize = pCfg->max_frame_size;
61 
62  AVB_LOGF_INFO("Dest Addr: "ETH_FORMAT, ETH_OCTETS(pListenerData->destAddr));
63  AVB_LOGF_INFO("Starting stream: "STREAMID_FORMAT, STREAMID_ARGS(streamID));
64  listenerStartStream(pTLState);
65 
66  return TRUE;
67 }
