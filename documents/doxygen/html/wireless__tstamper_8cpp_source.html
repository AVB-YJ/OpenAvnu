37 net_result WirelessTimestamper::requestTimingMeasurement
38 ( LinkLayerAddress *dest, uint16_t seq, WirelessDialog *prev_dialog,
39  uint8_t *follow_up, int followup_length )
40 {
41  WirelessDialog next_dialog;
42  net_result retval;
43  TIMINGMSMT_REQUEST *req;
44 
45  // Valid dialog token > 0 && < 256
46  next_dialog.dialog_token = (seq % MAX_DIALOG_TOKEN) + 1;
47  next_dialog.fwup_seq = seq;
48  next_dialog.action_devclk = 0;
49  req = (TIMINGMSMT_REQUEST *)follow_up;
50 
51  // File in request
52  req->DialogToken = next_dialog.dialog_token;
53  req->FollowUpDialogToken = prev_dialog->dialog_token;
54  req->Category = 0x0;
55  req->Action = 0x0;
56  req->WiFiVSpecHdr.ElementId = FWUP_VDEF_TAG;
57  req->WiFiVSpecHdr.Length = 0;
58  if( req->FollowUpDialogToken != 0 && prev_dialog->action_devclk != 0 )
59  {
60  req->T1 = (uint32_t)prev_dialog->action_devclk;
61  req->T4 = (uint32_t)prev_dialog->ack_devclk;
62  req->WiFiVSpecHdr.Length = followup_length +
63  (uint8_t)sizeof(FwUpLabel); // 1 byte type
64  req->PtpSpec.FwUpLabel = FwUpLabel;
65  }
66  dest->toOctetArray(req->PeerMACAddress);
67 
68  retval = _requestTimingMeasurement( req );
69  port->setPrevDialog(&next_dialog);
70 
71  return retval;
72 }

90 void WirelessTimestamper::timeMeasurementIndicationCB
91 ( LinkLayerAddress addr, WirelessDialog *current, WirelessDialog *previous,
92  uint8_t *buf, size_t buflen )
93 {
94  uint64_t link_delay;
95  WirelessDialog *prev_local;
96  PTPMessageCommon *msg;
97  PTPMessageFollowUp *fwup;
98  // Translate devclk scalar to Timestamp
99 
100  msg = buildPTPMessage((char *)buf, (int)buflen, &addr, port);
101  fwup = dynamic_cast<PTPMessageFollowUp *> (msg);
102  current->action_devclk *= 10;
103  current->ack_devclk *= 10;
104  current->action.set64(current->action_devclk);
105  prev_local = port->getPrevDialog();
106  if ( previous->dialog_token == prev_local->dialog_token &&
107  fwup != NULL && previous->action_devclk != 0)
108  {
109  previous->action_devclk *= 10;
110  previous->ack_devclk *= 10;
111  unsigned round_trip = (unsigned)
112  (previous->ack_devclk - previous->action_devclk);
113  unsigned turn_around = (unsigned)
114  (prev_local->ack_devclk - prev_local->action_devclk);
115  link_delay = (round_trip - turn_around) / 2;
116  port->setLinkDelay(link_delay);
117  GPTP_LOG_VERBOSE( "Link Delay = %llu(RT=%u,TA=%u,"
118  "T4=%llu,T1=%llu,DT=%hhu)",
119  link_delay, round_trip, turn_around,
120  previous->ack_devclk,
121  previous->action_devclk,
122  previous->dialog_token);
123  fwup->processMessage(port, prev_local->action);
124  }
125 
126  port->setPrevDialog(current);
127 }
