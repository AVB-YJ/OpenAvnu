65 Timestamp tsToTimestamp(struct timespec *ts)
66 {
67  Timestamp ret;
68  int seclen = sizeof(ts->tv_sec) - sizeof(ret.seconds_ls);
69  if (seclen > 0) {
70  ret.seconds_ms =
71  ts->tv_sec >> (sizeof(ts->tv_sec) - seclen) * 8;
72  ret.seconds_ls = ts->tv_sec & 0xFFFFFFFF;
73  } else {
74  ret.seconds_ms = 0;
75  ret.seconds_ls = ts->tv_sec;
76  }
77  ret.nanoseconds = ts->tv_nsec;
78  return ret;
79 }

173 static void x_readEvent
174 ( int sockint, EtherPort *pPort, int ifindex )
175 {
176  int status;
177  char buf[4096];
178  struct iovec iov = { buf, sizeof buf };
179  struct sockaddr_nl snl;
180  struct msghdr msg = { (void *) &snl, sizeof snl, &iov, 1, NULL, 0, 0 };
181  struct nlmsghdr *msgHdr;
182  struct ifinfomsg *ifi;
183 
184  status = recvmsg(sockint, &msg, 0);
185 
186  if (status < 0) {
187  GPTP_LOG_ERROR("read_netlink: Error recvmsg: %d", status);
188  return;
189  }
190 
191  if (status == 0) {
192  GPTP_LOG_ERROR("read_netlink: EOF");
193  return;
194  }
195 
196  // Process the NETLINK messages
197  for (msgHdr = (struct nlmsghdr *)buf; NLMSG_OK(msgHdr, (unsigned int)status); msgHdr = NLMSG_NEXT(msgHdr, status))
198  {
199  if (msgHdr->nlmsg_type == NLMSG_DONE)
200  return;
201 
202  if (msgHdr->nlmsg_type == NLMSG_ERROR) {
203  GPTP_LOG_ERROR("netlink message error");
204  return;
205  }
206 
207  if (msgHdr->nlmsg_type == RTM_NEWLINK) {
208  ifi = (struct ifinfomsg *)NLMSG_DATA(msgHdr);
209  if (ifi->ifi_index == ifindex) {
210  bool linkUp = ifi->ifi_flags & IFF_RUNNING;
211  if (linkUp != pPort->getLinkUpState()) {
212  pPort->setLinkUpState(linkUp);
213  if (linkUp) {
214  pPort->processEvent(LINKUP);
215  }
216  else {
217  pPort->processEvent(LINKDOWN);
218  }
219  }
220  else {
221  GPTP_LOG_DEBUG("False (repeated) %s event for the interface", linkUp ? "LINKUP" : "LINKDOWN");
222  }
223  }
224  }
225  }
226  return;
227 }

230 {

233  device.ifr_ifindex = ifindex;

255  pPort->setLinkUpState(true);

262 {

274

293  break;
294  case SPEED_1000:
295  *speed = LINKSPEED_1G;

297  case SPEED_2500:

299  break;

309 void LinuxNetworkInterface::watchNetLink( CommonPort *iPort )
310 {
311  fd_set netLinkFD;
312  int netLinkSocket;
313  int inetSocket;
314  struct sockaddr_nl addr;
315 
316  EtherPort *pPort =
317  dynamic_cast<EtherPort *>(iPort);
318  if( pPort == NULL )
319  {
320  GPTP_LOG_ERROR("NETLINK socket open error");
321  return;
322  }
323 
324  netLinkSocket = socket (AF_NETLINK, SOCK_RAW, NETLINK_ROUTE);
325  if (netLinkSocket < 0) {
326  GPTP_LOG_ERROR("NETLINK socket open error");
327  return;
328  }
329 
330  memset((void *) &addr, 0, sizeof (addr));
331 
332  addr.nl_family = AF_NETLINK;
333  addr.nl_pid = getpid ();
334  addr.nl_groups = RTMGRP_LINK;
335 
336  if (bind (netLinkSocket, (struct sockaddr *) &addr, sizeof (addr)) < 0) {
337  GPTP_LOG_ERROR("Socket bind failed");
338  close (netLinkSocket);
339  return;
340  }
341 
342  /*
343  * Open an INET family socket to be passed to getLinkSpeed() which calls
344  * ioctl() because NETLINK sockets do not support ioctl(). Since we will
345  * enter an infinite loop, there are no apparent close() calls for the
346  * open sockets, but they will be closed on process termination.
347  */
348  inetSocket = socket (AF_INET, SOCK_STREAM, 0);
349  if (inetSocket < 0) {
350  GPTP_LOG_ERROR("watchNetLink error opening socket: %s", strerror(errno));
351  close (netLinkSocket);
352  return;
353  }
354 
355  x_initLinkUpStatus(pPort, ifindex);
356  if( pPort->getLinkUpState() )
357  {
358  uint32_t link_speed;
359  getLinkSpeed( inetSocket, &link_speed );
360  pPort->setLinkSpeed((int32_t) link_speed );
361  } else
362  {
363  pPort->setLinkSpeed( INVALID_LINKSPEED );
364  }
365 
366  while (1) {
367  FD_ZERO(&netLinkFD);
368  FD_CLR(netLinkSocket, &netLinkFD);
369  FD_SET(netLinkSocket, &netLinkFD);
370 
371  // Wait forever for a net link event
372  int retval = select(FD_SETSIZE, &netLinkFD, NULL, NULL, NULL);
373  if (retval == -1)
374  ; // Error on select. We will ignore and keep going
375  else if (retval) {
376  bool prev_link_up = pPort->getLinkUpState();
377  x_readEvent(netLinkSocket, pPort, ifindex);
378 
379  // Don't do anything else if link state is the same
380  if( prev_link_up == pPort->getLinkUpState() )
381  continue;
382  if( pPort->getLinkUpState() )
383  {
384  uint32_t link_speed;
385  getLinkSpeed( inetSocket, &link_speed );
386  pPort->setLinkSpeed((int32_t) link_speed );
387  } else
388  {
389  pPort->setLinkSpeed( INVALID_LINKSPEED );
390  }
391  }
392  else {
393  ; // Would be timeout but Won't happen because we wait forever
394  }
395  }
396 }

424 void *LinuxTimerQueueHandler( void *arg ) {
425  LinuxTimerQueue *timerq = (LinuxTimerQueue *) arg;
426  sigset_t waitfor;
427  struct timespec timeout;
428  timeout.tv_sec = 0; timeout.tv_nsec = 100000000; /* 100 ms */
429 
430  sigemptyset( &waitfor );
431  GPTP_LOG_DEBUG("Signal thread started");
432  while( !timerq->stop ) {
433  siginfo_t info;
434  LinuxTimerQueueMap_t::iterator iter;
435  sigaddset( &waitfor, SIGUSR1 );
436  if( sigtimedwait( &waitfor, &info, &timeout ) == -1 ) {
437  if( errno == EAGAIN ) {
438  continue;
439  }
440  else {
441  GPTP_LOG_ERROR("Signal thread sigtimedwait error: %d", errno);
442  break;
443  }
444  }
445  if( timerq->lock->lock() != oslock_ok ) {
446  break;
447  }
448 
449  iter = timerq->timerQueueMap.find(info.si_value.sival_int);
450  if( iter != timerq->timerQueueMap.end() ) {
451  struct LinuxTimerQueueActionArg *arg = iter->second;
452  timerq->timerQueueMap.erase(iter);
453  timerq->LinuxTimerQueueAction( arg );
454  if( arg->rm ) {
455  delete arg->inner_arg;
456  }
457  timer_delete(arg->timer_handle);
458  delete arg;
459  }
460  if( timerq->lock->unlock() != oslock_ok ) {
461  break;
462  }
463  }
464  GPTP_LOG_DEBUG("Signal thread exit");
465  return NULL;
466 }

497 bool LinuxTimerQueue::addEvent
498 ( unsigned long micros, int type, ostimerq_handler func,
499  event_descriptor_t * arg, bool rm, unsigned *event) {
500  LinuxTimerQueueActionArg *outer_arg;
501  int err;
502  LinuxTimerQueueMap_t::iterator iter;
503 
504 
505  outer_arg = new LinuxTimerQueueActionArg;
506  outer_arg->inner_arg = arg;
507  outer_arg->rm = rm;
508  outer_arg->func = func;
509  outer_arg->type = type;
510 
511  // Find key that we can use
512  while( timerQueueMap.find( key ) != timerQueueMap.end() ) {
513  ++key;
514  }
515 
516  {
517  struct itimerspec its;
518  memset(&(outer_arg->sevp), 0, sizeof(outer_arg->sevp));
519  outer_arg->sevp.sigev_notify = SIGEV_SIGNAL;
520  outer_arg->sevp.sigev_signo = SIGUSR1;
521  outer_arg->sevp.sigev_value.sival_int = key;
522  if ( timer_create
523  (CLOCK_MONOTONIC, &outer_arg->sevp, &outer_arg->timer_handle)
524  == -1) {
525  GPTP_LOG_ERROR("timer_create failed - %s", strerror(errno));
526  return false;
527  }
528  timerQueueMap[key] = outer_arg;
529 
530  memset(&its, 0, sizeof(its));
531  its.it_value.tv_sec = micros / 1000000;
532  its.it_value.tv_nsec = (micros % 1000000) * 1000;
533  err = timer_settime( outer_arg->timer_handle, 0, &its, NULL );
534  if( err < 0 ) {
535  GPTP_LOG_ERROR("Failed to arm timer: %s", strerror(errno));
536  return false;
537  }
538  }
539 
540  return true;
541 }

600 bool TicketingLock::lock( bool *got ) {
601  uint8_t ticket;
602  bool yield = false;
603  bool ret = true;
604  if( !init_flag ) return false;
605 
606  if( pthread_mutex_lock( &_private->cond_lock ) != 0 ) {
607  ret = false;
608  goto done;
609  }
610  // Take a ticket
611  ticket = cond_ticket_issue++;
612  while( ticket != cond_ticket_serving ) {
613  if( got != NULL ) {
614  *got = false;
615  --cond_ticket_issue;
616  yield = true;
617  goto unlock;
618  }
619  if( pthread_cond_wait( &_private->condition, &_private->cond_lock ) != 0 ) {
620  ret = false;
621  goto unlock;
622  }
623  }
624 
625  if( got != NULL ) *got = true;
626 
627  unlock:
628  if( pthread_mutex_unlock( &_private->cond_lock ) != 0 ) {
629  ret = false;
630  goto done;
631  }
632 
633  if( yield ) pthread_yield();
634 
635  done:
636  return ret;
637 }

639 bool TicketingLock::unlock() {
640  bool ret = true;
641  if( !init_flag ) return false;
642 
643  if( pthread_mutex_lock( &_private->cond_lock ) != 0 ) {
644  ret = false;
645  goto done;
646  }
647  ++cond_ticket_serving;
648  if( pthread_cond_broadcast( &_private->condition ) != 0 ) {
649  ret = false;
650  goto unlock;
651  }
652 
653  unlock:
654  if( pthread_mutex_unlock( &_private->cond_lock ) != 0 ) {
655  ret = false;
656  goto done;
657  }
658 
659  done:
660  return ret;
661 }
