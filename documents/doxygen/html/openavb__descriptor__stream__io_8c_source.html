52 openavbRC openavbAemDescriptorStreamIOToBuf(void *pVoidDescriptor, U16 bufLength, U8 *pBuf, U16 *descriptorSize)
53 {
54  AVB_TRACE_ENTRY(AVB_TRACE_AEM);
55 
56  openavb_aem_descriptor_stream_io_t *pDescriptor = pVoidDescriptor;
57 
58  if (!pDescriptor || !pBuf || !descriptorSize) {
59  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVB_RC_INVALID_ARGUMENT), AVB_TRACE_AEM);
60  }
61 
62  if (bufLength < OPENAVB_DESCRIPTOR_STREAM_IO_BASE_LENGTH + (pDescriptor->number_of_formats * OPENAVB_DESCRIPTOR_STREAM_IO_FORMAT_LENGTH)) {
63  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVBAVDECC_RC_BUFFER_TOO_SMALL), AVB_TRACE_AEM);
64  }
65 
66  *descriptorSize = 0;
67 
68  openavb_aem_descriptor_stream_io_t *pSrc = pDescriptor;
69  U8 *pDst = pBuf;
70 
71  OCT_D2BHTONS(pDst, pSrc->descriptor_type);
72  OCT_D2BHTONS(pDst, pSrc->descriptor_index);
73  OCT_D2BMEMCP(pDst, pSrc->object_name);
74  BIT_D2BHTONS(pDst, pSrc->localized_description.offset, 3, 0);
75  BIT_D2BHTONS(pDst, pSrc->localized_description.index, 0, 2);
76  OCT_D2BHTONS(pDst, pSrc->clock_domain_index);
77  OCT_D2BHTONS(pDst, pSrc->stream_flags);
78  openavbAemStreamFormatToBuf(&pSrc->current_format, pDst);
79  pDst += 8;
80  OCT_D2BHTONS(pDst, pSrc->formats_offset);
81  OCT_D2BHTONS(pDst, pSrc->number_of_formats);
82  OCT_D2BMEMCP(pDst, pSrc->backup_talker_entity_id_0);
83  OCT_D2BHTONS(pDst, pSrc->backup_talker_unique_id_0);
84  OCT_D2BMEMCP(pDst, pSrc->backup_talker_entity_id_1);
85  OCT_D2BHTONS(pDst, pSrc->backup_talker_unique_id_1);
86  OCT_D2BMEMCP(pDst, pSrc->backup_talker_entity_id_2);
87  OCT_D2BHTONS(pDst, pSrc->backup_talker_unique_id_2);
88  OCT_D2BMEMCP(pDst, pSrc->backedup_talker_entity_id);
89  OCT_D2BHTONS(pDst, pSrc->backedup_talker_unique_id);
90  OCT_D2BHTONS(pDst, pSrc->avb_interface_index);
91  OCT_D2BHTONL(pDst, pSrc->buffer_length);
92 
93  int i1;
94  for (i1 = 0; i1 < pSrc->number_of_formats; i1++) {
95  openavbAemStreamFormatToBuf(&pSrc->stream_formats[i1], pDst);
96  pDst += 8;
97  }
98 
99  *descriptorSize = pDst - pBuf;
100 
101  AVB_RC_TRACE_RET(OPENAVB_AVDECC_SUCCESS, AVB_TRACE_AEM);
102 }

105 void x_openavbAemStreamFormatFromBuf(openavb_aem_stream_format_t *pDst, U8 *pBuf)
106 {
107  U8 *pSrc = pBuf;
108 
109  // Tightly bound to caller not safety checks needed.
110 
111  BIT_B2DNTOHB(pDst->v, pSrc, 0x80, 7, 0);
112  if (pDst->v == 0) {
113  BIT_B2DNTOHB(pDst->subtype, pSrc, 0x7f, 0, 1);
114  switch (pDst->subtype) {
115  case OPENAVB_AEM_STREAM_FORMAT_61883_IIDC_SUBTYPE:
116  {
117  BIT_B2DNTOHB(pDst->subtypes.iec_61883_iidc.sf, pSrc, 0x80, 7, 0);
118  if (pDst->subtypes.iec_61883_iidc.sf == OPENAVB_AEM_STREAM_FORMAT_SF_IIDC) {
119  pSrc += 4;
120  OCT_B2DNTOHB(pDst->subtypes.iidc.iidc_format, pSrc);
121  OCT_B2DNTOHB(pDst->subtypes.iidc.iidc_mode, pSrc);
122  OCT_B2DNTOHB(pDst->subtypes.iidc.iidc_rate, pSrc);
123  }
124  else {
125  BIT_B2DNTOHB(pDst->subtypes.iec_61883.sf, pSrc, 0x80, 7, 0);
126  BIT_B2DNTOHB(pDst->subtypes.iec_61883.fmt, pSrc, 0x7e, 1, 1);
127  if (pDst->subtypes.iec_61883.fmt == OPENAVB_AEM_STREAM_FORMAT_FMT_61883_4) {
128  // Nothingn to do
129  }
130  else if (pDst->subtypes.iec_61883.fmt == OPENAVB_AEM_STREAM_FORMAT_FMT_61883_6) {
131  BIT_B2DNTOHB(pDst->subtypes.iec_61883_6.fdf_evt, pSrc, 0xf8, 3, 0);
132  BIT_B2DNTOHB(pDst->subtypes.iec_61883_6.fdf_sfc, pSrc, 0x07, 0, 1);
133  OCT_B2DNTOHB(pDst->subtypes.iec_61883_6.dbs, pSrc);
134  BIT_B2DNTOHB(pDst->subtypes.iec_61883_6.b, pSrc, 0x80, 7, 0);
135  BIT_B2DNTOHB(pDst->subtypes.iec_61883_6.nb, pSrc, 0x40, 6, 1);
136 
137  if (pDst->subtypes.iec_61883_6.fdf_evt == OPENAVB_AEM_STREAM_FORMAT_FDF_EVT_61883_6_32BITS) {
138  // Nothing to do
139  }
140  else if (pDst->subtypes.iec_61883_6.fdf_evt == OPENAVB_AEM_STREAM_FORMAT_FDF_EVT_61883_6_FLOAT) {
141  // Nothing to do
142  }
143  else if (pDst->subtypes.iec_61883_6.fdf_evt == OPENAVB_AEM_STREAM_FORMAT_FDF_EVT_61883_6_AM824) {
144  OCT_B2DNTOHB(pDst->subtypes.iec_61883_6_am824.label_iec_60958_cnt, pSrc);
145  OCT_B2DNTOHB(pDst->subtypes.iec_61883_6_am824.label_mbla_cnt, pSrc);
146  BIT_B2DNTOHB(pDst->subtypes.iec_61883_6_am824.label_midi_cnt, pSrc, 0xf0, 4, 0);
147  BIT_B2DNTOHB(pDst->subtypes.iec_61883_6_am824.label_smptecnt, pSrc, 0x0f, 0, 1);
148  }
149  }
150 
151  else if (pDst->subtypes.iec_61883.fmt == OPENAVB_AEM_STREAM_FORMAT_FMT_61883_8) {
152  pSrc += 3;
153  OCT_B2DNTOHB(pDst->subtypes.iec_61883_8.video_mode, pSrc);
154  OCT_B2DNTOHB(pDst->subtypes.iec_61883_8.compress_mode, pSrc);
155  OCT_B2DNTOHB(pDst->subtypes.iec_61883_8.color_space, pSrc);
156  }
157  }
158  }
159 
160  break;
161  case OPENAVB_AEM_STREAM_FORMAT_MMA_SUBTYPE:
162  // Nothing to do
163  break;
164  case OPENAVB_AEM_STREAM_FORMAT_AVTP_AUDIO_SUBTYPE:
165  BIT_B2DNTOHB(pDst->subtypes.avtp_audio.nominal_sample_rate, pSrc, 0x0f, 0, 1);
166  OCT_B2DNTOHB(pDst->subtypes.avtp_audio.format, pSrc);
167  OCT_B2DNTOHB(pDst->subtypes.avtp_audio.bit_depth, pSrc);
168  BIT_B2DNTOHL(pDst->subtypes.avtp_audio.channels_per_frame, pSrc, 0xffc00000, 22, 0);
169  BIT_B2DNTOHL(pDst->subtypes.avtp_audio.samples_per_frame, pSrc, 0x003ff000, 12, 4);
170  break;
171  case OPENAVB_AEM_STREAM_FORMAT_AVTP_VIDEO_SUBTYPE:
172  OCT_B2DNTOHB(pDst->subtypes.avtp_video.format, pSrc);
173  break;
174  case OPENAVB_AEM_STREAM_FORMAT_AVTP_CONTROL_SUBTYPE:
175  BIT_B2DNTOHB(pDst->subtypes.avtp_control.protocol_type, pSrc, 0xf0, 4, 1);
176  OCT_B2DMEMCP(pDst->subtypes.avtp_control.format_id, pSrc);
177  break;
178  case OPENAVB_AEM_STREAM_FORMAT_VENDOR_SUBTYPE:
179  pSrc++;
180  OCT_B2DMEMCP(pDst->subtypes.vendor.format_id, pSrc);
181  break;
182  case OPENAVB_AEM_STREAM_FORMAT_EXPERIMENTAL_SUBTYPE:
183  // Nothing to do
184  break;
185  }
186  }
187  else {
188  pSrc++;
189  OCT_B2DMEMCP(pDst->subtypes.vendor_specific.format_id, pSrc);
190  }
191 }
