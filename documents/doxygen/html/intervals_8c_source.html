53  }
54  current = *root;
55  while (1) {

81 Interval *remove_interval(Interval **root, Interval *node) {
82  Interval *snip, *child;
83 
84  /* If the node to remove does not have two children, we will snip it,
85  otherwise we will swap it with its successor and snip that one */
86  if (!node->left_child || !node->right_child) {
87  snip = node;
88  } else {
89  snip = next_interval(node);
90  }
91 
92  /* If the node to snip has a child, make its parent link point to the snipped
93  node's parent */
94  if (snip->left_child) {
95  child = snip->left_child;
96  } else {
97  child = snip->right_child;
98  }
99  if (child) {
100  child->parent = snip->parent;
101  }
102 
103  /* If the snipped node has no parent, it is the root node and we use the
104  provided root pointer to point to the child. Otherwise, we find if the
105  snipped node was a left or right child and set the appropriate link in the
106  parent node */
107  if (!snip->parent) {
108  *root = child;
109  } else if (snip == snip->parent->left_child) {
110  snip->parent->left_child = child;
111  } else {
112  snip->parent->right_child = child;
113  }
114 
115  /* Swap the contents of the node passed in to remove with the one chosen to be
116  snipped */
117  if (snip != node) {
118  void *old_data = node->data;
119  node->low = snip->low;
120  node->high = snip->high;
121  node->data = snip->data;
122  snip->data = old_data;
123  }
124 
125  return snip;
126 }

184 Interval *search_interval(Interval *root, uint32_t start, uint32_t count) {
185  Interval *current, *test;
186  Interval i;
187 
188  i.low = start;
189  i.high = start + count - 1;
190  current = root;
191 
192  while (current && !check_overlap(current, &i)) {
193  if (current->low > i.low) {
194  current = current->left_child;
195  } else {
196  current = current->right_child;
197  }
198  }
199 
200  /* Make sure we really are returning the first (lowest) matching interval. */
201  if (current) {
202  for (test = prev_interval(current);
203  test != NULL && check_overlap(test, &i);
204  test = prev_interval(test))
205  {
206  /* We found a lower interval that also matches the search parameters. */
207  current = test;
208  }
209  }
210 
211  return current;
212 }
