95 // Open a rawsock for TX or RX
96 void* sendmmsgRawsockOpen(sendmmsg_rawsock_t* rawsock, const char *ifname, bool rx_mode, bool tx_mode, U16 ethertype, U32 frame_size, U32 num_frames)
97 {
98  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK);
99 
100  AVB_LOGF_DEBUG("Open, ifname=%s, rx=%d, tx=%d, ethertype=%x size=%d, num=%d",
101  ifname, rx_mode, tx_mode, ethertype, frame_size, num_frames);
102 
103  baseRawsockOpen(&rawsock->base, ifname, rx_mode, tx_mode, ethertype, frame_size, num_frames);
104 
105  rawsock->sock = -1;
106 
107  // Get info about the network device
108  if (!simpleAvbCheckInterface(ifname, &(rawsock->base.ifInfo))) {
109  AVB_LOGF_ERROR("Creating rawsock; bad interface name: %s", ifname);
110  free(rawsock);
111  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
112  return NULL;
113  }
114 
115  // Deal with frame size.
116  if (rawsock->base.frameSize == 0) {
117  // use interface MTU as max frames size, if none specified
118  rawsock->base.frameSize = rawsock->base.ifInfo.mtu + ETH_HLEN + VLAN_HLEN;
119  }
120  else if (rawsock->base.frameSize > rawsock->base.ifInfo.mtu + ETH_HLEN + VLAN_HLEN) {
121  AVB_LOG_ERROR("Creating raswsock; requested frame size exceeds MTU");
122  free(rawsock);
123  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
124  return NULL;
125  }
126  rawsock->base.frameSize = TPACKET_ALIGN(rawsock->base.frameSize);
127  assert(rawsock->base.frameSize <= MAX_FRAME_SIZE);
128 
129  // Prepare default Ethernet header.
130  rawsock->base.ethHdrLen = sizeof(eth_hdr_t);
131  memset(&(rawsock->base.ethHdr.notag.dhost), 0xFF, ETH_ALEN);
132  memcpy(&(rawsock->base.ethHdr.notag.shost), &(rawsock->base.ifInfo.mac), ETH_ALEN);
133  rawsock->base.ethHdr.notag.ethertype = htons(rawsock->base.ethertype);
134 
135  // Create socket
136  rawsock->sock = socket(PF_PACKET, SOCK_RAW, htons(rawsock->base.ethertype));
137  if (rawsock->sock == -1) {
138  AVB_LOGF_ERROR("Creating rawsock; opening socket: %s", strerror(errno));
139  sendmmsgRawsockClose(rawsock);
140  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
141  return NULL;
142  }
143 
144  // Allow address reuse
145  int temp = 1;
146  if(setsockopt(rawsock->sock, SOL_SOCKET, SO_REUSEADDR, &temp, sizeof(int)) < 0) {
147  AVB_LOG_ERROR("Creating rawsock; failed to set reuseaddr");
148  sendmmsgRawsockClose(rawsock);
149  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
150  return NULL;
151  }
152 
153  // Bind to interface
154  struct sockaddr_ll my_addr;
155  memset(&my_addr, 0, sizeof(my_addr));
156  my_addr.sll_family = PF_PACKET;
157  my_addr.sll_protocol = htons(rawsock->base.ethertype);
158  my_addr.sll_ifindex = rawsock->base.ifInfo.index;
159 
160  if (bind(rawsock->sock, (struct sockaddr*)&my_addr, sizeof(my_addr)) == -1) {
161  AVB_LOGF_ERROR("Creating rawsock; bind socket: %s", strerror(errno));
162  sendmmsgRawsockClose(rawsock);
163  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
164  return NULL;
165  }
166 
167  // Clear our buffers and other tracking data
168  memset(rawsock->mmsg, 0, sizeof(rawsock->mmsg));
169  memset(rawsock->miov, 0, sizeof(rawsock->miov));
170  memset(rawsock->pktbuf, 0, sizeof(rawsock->pktbuf));
171 #if USE_LAUNCHTIME
172  memset(rawsock->cmsgbuf, 0, sizeof(rawsock->cmsgbuf));
173 #endif
174 
175  rawsock->buffersOut = 0;
176  rawsock->buffersReady = 0;
177  rawsock->frameCount = MSG_COUNT;
178 
179  // fill virtual functions table
180  rawsock_cb_t *cb = &rawsock->base.cb;
181  cb->close = sendmmsgRawsockClose;
182  cb->getTxFrame = sendmmsgRawsockGetTxFrame;
183  cb->txSetMark = sendmmsgRawsockTxSetMark;
184  cb->txSetHdr = sendmmsgRawsockTxSetHdr;
185  cb->txFrameReady = sendmmsgRawsockTxFrameReady;
186  cb->send = sendmmsgRawsockSend;
187  cb->getRxFrame = sendmmsgRawsockGetRxFrame;
188  cb->rxMulticast = sendmmsgRawsockRxMulticast;
189  cb->getSocket = sendmmsgRawsockGetSocket;
190 
191  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK);
192  return rawsock;
193 }

331 // Send all packets that are ready (i.e. tell kernel to send them)
332 int sendmmsgRawsockSend(void *pvRawsock)
333 {
334  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK_DETAIL);
335  sendmmsg_rawsock_t *rawsock = (sendmmsg_rawsock_t*)pvRawsock;
336  int sz, bytes;
337 
338  if (!VALID_TX_RAWSOCK(rawsock)) {
339  AVB_LOG_ERROR("Marking TX frame ready; invalid argument");
340  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
341  return -1;
342  }
343 
344  if (rawsock->buffersOut != rawsock->buffersReady) {
345  AVB_LOGF_ERROR("Tried to send with %d bufs out, %d bufs ready", rawsock->buffersOut, rawsock->buffersReady);
346  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
347  return -1;
348  }
349 
350  IF_LOG_INTERVAL(1000) AVB_LOGF_DEBUG("Send with %d of %d buffers ready", rawsock->buffersReady, rawsock->frameCount);
351  sz = sendmmsg(rawsock->sock, rawsock->mmsg, rawsock->buffersReady, 0);
352  if (sz < 0) {
353  AVB_LOGF_ERROR("Call to sendmmsg failed! Error code was %d", sz);
354  bytes = sz;
355  } else {
356  int i;
357  for (i = 0, bytes = 0; i < sz; i++) {
358  bytes += rawsock->mmsg[i].msg_len;
359  }
360  if (sz < rawsock->buffersReady) {
361  AVB_LOGF_WARNING("Only sent %ld of %d messages; dropping others", sz, rawsock->buffersReady);
362  }
363  }
364 
365  rawsock->buffersOut = rawsock->buffersReady = 0;
366 
367  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
368  return bytes;
369 }

404 // Setup the rawsock to receive multicast packets
405 bool sendmmsgRawsockRxMulticast(void *pvRawsock, bool add_membership, const U8 addr[ETH_ALEN])
406 {
407  AVB_TRACE_ENTRY(AVB_TRACE_RAWSOCK_DETAIL);
408 
409  sendmmsg_rawsock_t *rawsock = (sendmmsg_rawsock_t*)pvRawsock;
410  if (!VALID_RX_RAWSOCK(rawsock)) {
411  AVB_LOG_ERROR("Setting multicast; invalid arguments");
412  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
413  return FALSE;
414  }
415 
416  struct ether_addr mcast_addr;
417  memcpy(mcast_addr.ether_addr_octet, addr, ETH_ALEN);
418 
419  // Fill in the structure for the multicast ioctl
420  struct packet_mreq mreq;
421  memset(&mreq, 0, sizeof(struct packet_mreq));
422  mreq.mr_ifindex = rawsock->base.ifInfo.index;
423  mreq.mr_type = PACKET_MR_MULTICAST;
424  mreq.mr_alen = ETH_ALEN;
425  memcpy(&mreq.mr_address, &mcast_addr.ether_addr_octet, ETH_ALEN);
426 
427  // And call the ioctl to add/drop the multicast address
428  int action = (add_membership ? PACKET_ADD_MEMBERSHIP : PACKET_DROP_MEMBERSHIP);
429  if (setsockopt(rawsock->sock, SOL_PACKET, action,
430  (void*)&mreq, sizeof(struct packet_mreq)) < 0) {
431  AVB_LOGF_ERROR("Setting multicast; setsockopt(%s) failed: %s",
432  (add_membership ? "PACKET_ADD_MEMBERSHIP" : "PACKET_DROP_MEMBERSHIP"),
433  strerror(errno));
434 
435  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
436  return FALSE;
437  }
438 
439  // In addition to adding the multicast membership, we also want to
440  // add a packet filter to restrict the packets that we'll receive
441  // on this socket. Multicast memeberships are global - not
442  // per-socket, so without the filter, this socket would receieve
443  // packets for all the multicast addresses added by all other
444  // sockets.
445  //
446  if (add_membership)
447  {
448  // Here's the template packet filter code.
449  // It was produced by running:
450  // tcpdump -dd ether dest host 91:e0:01:02:03:04
451  struct sock_filter bpfCode[] = {
452  { 0x20, 0, 0, 0x00000002 },
453  { 0x15, 0, 3, 0x01020304 }, // last 4 bytes of dest mac
454  { 0x28, 0, 0, 0x00000000 },
455  { 0x15, 0, 1, 0x000091e0 }, // first 2 bytes of dest mac
456  { 0x06, 0, 0, 0x0000ffff },
457  { 0x06, 0, 0, 0x00000000 }
458  };
459 
460  // We need to graft our multicast dest address into bpfCode
461  U32 tmp; U8 *buf = (U8*)&tmp;
462  memcpy(buf, mcast_addr.ether_addr_octet + 2, 4);
463  bpfCode[1].k = ntohl(tmp);
464  memset(buf, 0, 4);
465  memcpy(buf + 2, mcast_addr.ether_addr_octet, 2);
466  bpfCode[3].k = ntohl(tmp);
467 
468  // Now wrap the filter code in the appropriate structure
469  struct sock_fprog filter;
470  memset(&filter, 0, sizeof(filter));
471  filter.len = 6;
472  filter.filter = bpfCode;
473 
474  // And attach it to our socket
475  if (setsockopt(rawsock->sock, SOL_SOCKET, SO_ATTACH_FILTER,
476  &filter, sizeof(filter)) < 0) {
477  AVB_LOGF_ERROR("Setting multicast; setsockopt(SO_ATTACH_FILTER) failed: %s", strerror(errno));
478  }
479  }
480  else {
481  if (setsockopt(rawsock->sock, SOL_SOCKET, SO_DETACH_FILTER, NULL, 0) < 0) {
482  AVB_LOGF_ERROR("Setting multicast; setsockopt(SO_DETACH_FILTER) failed: %s", strerror(errno));
483  }
484  }
485 
486  AVB_TRACE_EXIT(AVB_TRACE_RAWSOCK_DETAIL);
487  return TRUE;
488 }
