73 openavbRC openavbAdpStart()
74 {
75  AVB_TRACE_ENTRY(AVB_TRACE_ADP);
76 
77  openavb_avdecc_entity_model_t *pAem = openavbAemGetModel();
78  if (!pAem) {
79  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVBAVDECC_RC_ENTITY_MODEL_MISSING), AVB_TRACE_ADP);
80  }
81 
82  {
83  MUTEX_ATTR_HANDLE(mta);
84  MUTEX_ATTR_INIT(mta);
85  MUTEX_ATTR_SET_TYPE(mta, MUTEX_ATTR_TYPE_DEFAULT);
86  MUTEX_ATTR_SET_NAME(mta, "openavbAdpMutex");
87  MUTEX_CREATE_ERR();
88  MUTEX_CREATE(openavbAdpMutex, mta);
89  MUTEX_LOG_ERR("Could not create/initialize 'openavbAdpMutex' mutex");
90  }
91 
92  ADP_LOCK();
93  {
94  // Populate global ADP vars
95  openavb_adp_control_header_t *pHeader = &openavbAdpSMGlobalVars.entityInfo.header;
96 
97  pHeader->cd = 1;
98  pHeader->subtype = OPENAVB_ADP_AVTP_SUBTYPE;
99  pHeader->sv = 0;
100  pHeader->version = 0;
101  pHeader->message_type = 0; // Set later depending on message type
102  pHeader->valid_time = gAvdeccCfg.valid_time;
103  pHeader->control_data_length = 56;
104  memcpy(pHeader->entity_id, pAem->pDescriptorEntity->entity_id, sizeof(pAem->pDescriptorEntity->entity_id));
105  }
106 
107  {
108  // Populate global ADP PDU vars
109  openavb_adp_data_unit_t *pPdu = &openavbAdpSMGlobalVars.entityInfo.pdu;
110 
111 #ifdef AVB_PTP_AVAILABLE
112  openavbRC retCode = OPENAVB_PTP_FAILURE;
113  retCode = openavbPtpInitializeSharedMemory();
114  if (IS_OPENAVB_FAILURE(retCode)) {
115  AVB_LOG_WARNING("Failed to init PTP shared memory");
116  }
117 #endif // AVB_PTP_AVAILABLE
118 
119  memcpy(pPdu->entity_model_id, pAem->pDescriptorEntity->entity_model_id, sizeof(pPdu->entity_model_id));
120  pPdu->entity_capabilities = pAem->pDescriptorEntity->entity_capabilities;
121  pPdu->talker_stream_sources = pAem->pDescriptorEntity->talker_stream_sources;
122  pPdu->talker_capabilities = pAem->pDescriptorEntity->talker_capabilities;
123  pPdu->listener_stream_sinks = pAem->pDescriptorEntity->listener_stream_sinks;
124  pPdu->listener_capabilities = pAem->pDescriptorEntity->listener_capabilities;
125  pPdu->controller_capabilities = pAem->pDescriptorEntity->controller_capabilities;
126  pPdu->available_index = pAem->pDescriptorEntity->available_index;
127  // The pPdu->gptp_grandmaster_id and pPdu->gptp_domain_number will be filled in when the ADPDU is transmitted.
128  // pPdu->reserved0;
129  // pPdu->identify_control_index = ???; // AVDECC_TODO
130  // pPdu->interface_index = ???; // AVDECC_TODO
131  memcpy(pPdu->association_id, pAem->pDescriptorEntity->association_id, sizeof(pPdu->association_id));
132  // pPdu->reserved1;
133  }
134  ADP_UNLOCK();
135 
136  // Wait until we are notified that we have a Talker or Listener before supporting discover.
137  s_bPreviousHaveTL = false;
138 
139  AVB_RC_TRACE_RET(OPENAVB_AVDECC_SUCCESS, AVB_TRACE_ADP);
140 }

142 void openavbAdpStop()
143 {
144  AVB_TRACE_ENTRY(AVB_TRACE_ADP);
145 
146 #ifdef AVB_PTP_AVAILABLE
147  openavbPtpReleaseSharedMemory();
148 #endif // AVB_PTP_AVAILABLE
149 
150  // Stop Advertising and supporting Discovery.
151  openavbAdpHaveTL(false);
152 
153  AVB_TRACE_EXIT(AVB_TRACE_ADP);
154 }

157 openavbRC openavbAdpHaveTL(bool bHaveTL)
158 {
159  AVB_TRACE_ENTRY(AVB_TRACE_ADP);
160 
161  if (bHaveTL && !s_bPreviousHaveTL) {
162  // Start Advertising and supporting Discovery.
163  openavbRC rc = openavbAdpMessageHandlerStart();
164  if (IS_OPENAVB_FAILURE(rc)) {
165  AVB_RC_TRACE_RET(rc, AVB_TRACE_ADP);
166  }
167 
168  openavbAdpSMAdvertiseInterfaceStart();
169  openavbAdpSMAdvertiseEntityStart();
170  s_bPreviousHaveTL = true;
171  }
172  else if (!bHaveTL && s_bPreviousHaveTL) {
173  // Stop Advertising and supporting Discovery.
174  openavbAdpSMAdvertiseInterfaceStop();
175  openavbAdpSMAdvertiseEntityStop();
176  openavbAdpMessageHandlerStop();
177  s_bPreviousHaveTL = false;
178  }
179 
180  AVB_RC_TRACE_RET(OPENAVB_AVDECC_SUCCESS, AVB_TRACE_ADP);
181 }
