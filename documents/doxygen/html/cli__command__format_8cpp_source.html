53 bool cli_command_format::run_command(cmd_line * cmd_ptr, std::queue<std::string> args, bool & done)
54 {
55  clear_args();
56 
57  bool args_ok = true;
58  size_t m_i = 0;
59  int match_count = 0;
60  int total_matched = 0;
61  while (args.size() && (m_i < m_args.size()) && args_ok)
62  {
63  bool ok = m_args[m_i]->set_value(args.front());
64 
65  const int match_max = m_args[m_i]->get_match_max();
66  const int match_min = m_args[m_i]->get_match_min();
67  if (ok)
68  {
69  args.pop();
70  match_count++;
71  total_matched++;
72 
73  if ((match_max != UINT_MAX) && (match_count >= match_max))
74  {
75  // Move to next argument - completed this one
76  m_i++;
77  match_count = 0;
78  }
79  }
80  else if (match_count >= match_min)
81  {
82  // Move to next argument - matched enough of this one
83  m_i++;
84  match_count = 0;
85  }
86  else
87  {
88  args_ok = false;
89  }
90  }
91 
92  if (args.size())
93  {
94  // Unused arguments left
95  args_ok = false;
96  }
97 
98  while (m_i < m_args.size() && args_ok)
99  {
100  if (m_args[m_i]->get_match_min() > m_args[m_i]->get_all_value_count())
101  {
102  // Left over arguments that are not optional
103  args_ok = false;
104  }
105  m_i++;
106  }
107 
108  if (args_ok && m_action)
109  {
110  done = ((cmd_ptr->*(m_action))(total_matched, m_args) != 0);
111  return true;
112  }
113 
114  return false;
115 }
