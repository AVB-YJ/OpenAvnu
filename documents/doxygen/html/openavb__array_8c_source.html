110 openavb_array_elem_t openavbArrayAdd(openavb_array_t array, void *data)
111 {
112  openavb_array_elem_t retElem = NULL;
113 
114  if (array && data) {
115  if (array->elemArray && (array->count < array->size)) {
116  // Find the empty slot
117  S32 i1;
118  for (i1 = 0; i1 < array->size; i1++) {
119  if (!array->elemArray[i1].data) {
120  retElem = &array->elemArray[i1];
121  break;
122  }
123  }
124  }
125  else {
126  // Need to make room for new element
127  openavb_array_elem_t newElemArray = realloc(array->elemArray, (array->size + 1) * sizeof(struct openavb_array_elem));
128  if (newElemArray) {
129  array->elemArray = newElemArray;
130  retElem = &array->elemArray[array->size];
131  retElem->data = NULL;
132  retElem->managed = FALSE;
133  retElem->idx = array->size;
134  array->size++;
135  }
136  }
137 
138  if (retElem) {
139  retElem->data = data;
140  array->count++;
141  }
142  }
143 
144  return retElem;
145 }
