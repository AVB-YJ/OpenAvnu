49 static inline uint64_t getCpuFrequency(void)
50 {
51  uint64_t freq = 0;
52  std::string line;
53  std::ifstream cpuinfo("/proc/cpuinfo");
54  if (cpuinfo.is_open())
55  {
56  while ( getline (cpuinfo,line) )
57  {
58  if(line.find("MHz") != line.npos)
59  break;
60  }
61  cpuinfo.close();
62  }
63  else std::cout << "Unable to open file";
64 
65  size_t pos = line.find(":");
66  if (pos != line.npos) {
67  std::string mhz_str = line.substr(pos+2, line.npos - pos);
68  double freq1 = strtod(mhz_str.c_str(), NULL);
69  freq = freq1 * 1000000ULL;
70  }
71  return freq;
72 }
