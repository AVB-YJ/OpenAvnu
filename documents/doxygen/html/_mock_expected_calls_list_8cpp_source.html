250 void MockExpectedCallsList::pruneEmptyNodeFromList()
251 {
252  MockExpectedCallsListNode* current = head_;
253  MockExpectedCallsListNode* previous = NULL;
254  MockExpectedCallsListNode* toBeDeleted = NULL;
255 
256  while (current) {
257  if (current->expectedCall_ == NULL) {
258  toBeDeleted = current;
259  if (previous == NULL)
260  head_ = current = current->next_;
261  else
262  current = previous->next_ = current->next_;
263  delete toBeDeleted;
264  }
265  else {
266  previous = current;
267  current = current->next_;
268  }
269  }
270 }
