127 int main(int argc, char* argv[])
128 {
129  GError *error = NULL;
130  GOptionContext *context;
131 
132  context = g_option_context_new("- rawsock listenr");
133  g_option_context_add_main_entries(context, entries, NULL);
134  if (!g_option_context_parse(context, &argc, &argv, &error))
135  {
136  printf("error: %s\n", error->message);
137  exit(1);
138  }
139 
140  if (interface == NULL || ethertype < 0) {
141  printf("error: must specify network interface and ethertype\n");
142  exit(2);
143  }
144 
145  void* rs = openavbRawsockOpen(interface, FALSE, TRUE, ethertype, 0, MAX_NUM_FRAMES);
146  if (!rs) {
147  printf("error: failed to open raw socket (are you root?)\n");
148  exit(3);
149  }
150 
151  U8 *pBuf, *pData;
152  U32 buflen, hdrlen, datalen;
153  hdr_info_t hdr;
154 
155  avbLogInit();
156 
157  memset(&hdr, 0, sizeof(hdr_info_t));
158  openavbRawsockTxSetHdr(rs, &hdr);
159 
160  struct timespec now;
161  static U64 packetIntervalNSec = 0;
162  static U64 nextCycleNSec = 0;
163  static U32 packetCnt = 0;
164  static U64 nextReportInterval = 0;
165 
166  packetIntervalNSec = NANOSECONDS_PER_SECOND / txRate;
167  clock_gettime(CLOCK_MONOTONIC, &now);
168  nextCycleNSec = TIMESPEC_TO_NSEC(now);
169  nextReportInterval = TIMESPEC_TO_NSEC(now) + (NANOSECONDS_PER_SECOND * reportSec);
170 
171  while (1) {
172  int remainder = 0;
173  pBuf = (U8*)openavbRawsockGetTxFrame(rs, TRUE, &buflen);
174  if (!pBuf) {
175  printf("failed to get TX frame buffer\n");
176  exit(4);
177  }
178 
179  if (buflen < txlen || txlen == -1)
180  txlen = buflen;
181 
182  openavbRawsockTxFillHdr(rs, pBuf, &hdrlen);
183  pData = pBuf + hdrlen;
184  datalen = txlen - hdrlen;
185 
186  if (mode == RAWSOCK_TX_MODE_FILL) {
187  int i;
188  for (i=0; i<datalen; i++)
189  pData[i] = (i & 0xff);
190  }
191  else { // RAWSOCK_TX_MODE_sEQ
192  static unsigned char seq = 0;
193  pData[0] = 0x7F; // Experimental subtype
194  pData[1] = 0x00;
195  pData[2] = seq++;
196  txlen = hdrlen + 3;
197  }
198 
199  openavbRawsockTxFrameReady(rs, pBuf, txlen, 0);
200 
201  packetCnt++;
202 
203  if ((packetCnt % chunkSize) == 0) {
204  openavbRawsockSend(rs);
205  }
206 
207  nextCycleNSec += packetIntervalNSec;
208  clock_gettime(CLOCK_MONOTONIC, &now);
209  U64 nowNSec = TIMESPEC_TO_NSEC(now);;
210 
211  if (nowNSec > nextReportInterval) {
212  printf("TX Packets: %d\n", packetCnt-remainder);
213  packetCnt %= chunkSize;
214  remainder = packetCnt;
215  nextReportInterval = nowNSec + (NANOSECONDS_PER_SECOND * reportSec);
216  }
217 
218  if (nowNSec < nextCycleNSec) {
219  usleep((nextCycleNSec - nowNSec) / 1000);
220  }
221  }
222 
223  openavbRawsockClose(rs);
224  avbLogExit();
225  return 0;
226 }
