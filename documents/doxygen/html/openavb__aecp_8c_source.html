57 openavbRC openavbAecpStart()
58 {
59  AVB_TRACE_ENTRY(AVB_TRACE_AECP);
60 
61  openavb_avdecc_entity_model_t *pAem = openavbAemGetModel();
62  if (!pAem) {
63  AVB_RC_LOG_TRACE_RET(AVB_RC(OPENAVB_AVDECC_FAILURE | OPENAVBAVDECC_RC_ENTITY_MODEL_MISSING), AVB_TRACE_AECP);
64  }
65 
66  MUTEX_ATTR_HANDLE(mta);
67  MUTEX_ATTR_INIT(mta);
68  MUTEX_ATTR_SET_TYPE(mta, MUTEX_ATTR_TYPE_DEFAULT);
69  MUTEX_ATTR_SET_NAME(mta, "openavbAecpMutex");
70  MUTEX_CREATE_ERR();
71  MUTEX_CREATE(openavbAecpMutex, mta);
72  MUTEX_LOG_ERR("Could not create/initialize 'openavbAecpMutex' mutex");
73 
74  // Set AECP global vars
75  memcpy(openavbAecpSMGlobalVars.myEntityID, pAem->pDescriptorEntity->entity_id, sizeof(openavbAecpSMGlobalVars.myEntityID));
76 
77  openavbRC rc = openavbAecpMessageHandlerStart();
78  if (IS_OPENAVB_FAILURE(rc)) {
79  openavbAecpStop();
80  AVB_RC_TRACE_RET(rc, AVB_TRACE_AECP);
81  }
82 
83  openavbAecpSMEntityModelEntityStart();
84 
85  AVB_RC_TRACE_RET(OPENAVB_AVDECC_SUCCESS, AVB_TRACE_AECP);
86 }
