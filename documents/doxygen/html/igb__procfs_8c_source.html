40 bool igb_thermal_present(struct igb_adapter *adapter)
41 {
42  s32 status;
43  struct e1000_hw *hw;
44 
45  if (adapter == NULL)
46  return false;
47  hw = &adapter->hw;
48 
49  /*
50  * Only set I2C bit-bang mode if an external thermal sensor is
51  * supported on this device.
52  */
53  if (adapter->ets) {
54  status = e1000_set_i2c_bb(hw);
55  if (status != E1000_SUCCESS)
56  return false;
57  }
58 
59  status = hw->mac.ops.init_thermal_sensor_thresh(hw);
60  if (status != E1000_SUCCESS)
61  return false;
62 
63  return true;
64 }

213 void igb_del_proc_entries(struct igb_adapter *adapter)
214 {
215  int index, i;
216  char buf[16]; /* much larger than the sensor number will ever be */
217 
218  if (igb_top_dir == NULL)
219  return;
220 
221  for (i = 0; i < E1000_MAX_SENSORS; i++) {
222  if (adapter->therm_dir[i] == NULL)
223  continue;
224 
225  for (index = 0; ; index++) {
226  if (igb_internal_entries[index].read == NULL)
227  break;
228 
229  remove_proc_entry(igb_internal_entries[index].name,
230  adapter->therm_dir[i]);
231  }
232  snprintf(buf, sizeof(buf), "sensor_%d", i);
233  remove_proc_entry(buf, adapter->info_dir);
234  }
235 
236  if (adapter->info_dir != NULL) {
237  for (index = 0; ; index++) {
238  if (igb_proc_entries[index].read == NULL)
239  break;
240  remove_proc_entry(igb_proc_entries[index].name,
241  adapter->info_dir);
242  }
243  remove_proc_entry("info", adapter->eth_dir);
244  }
245 
246  if (adapter->eth_dir != NULL)
247  remove_proc_entry(pci_name(adapter->pdev), igb_top_dir);
248 }

270 /* called from igb_main.c */
271 int igb_procfs_init(struct igb_adapter *adapter)
272 {
273  int rc = 0;
274  int i;
275  int index;
276  char buf[16]; /* much larger than the sensor number will ever be */
277 
278  adapter->eth_dir = NULL;
279  adapter->info_dir = NULL;
280  for (i = 0; i < E1000_MAX_SENSORS; i++)
281  adapter->therm_dir[i] = NULL;
282 
283  if (igb_top_dir == NULL) {
284  rc = -ENOMEM;
285  goto fail;
286  }
287 
288  adapter->eth_dir = proc_mkdir(pci_name(adapter->pdev), igb_top_dir);
289  if (adapter->eth_dir == NULL) {
290  rc = -ENOMEM;
291  goto fail;
292  }
293 
294  adapter->info_dir = proc_mkdir("info", adapter->eth_dir);
295  if (adapter->info_dir == NULL) {
296  rc = -ENOMEM;
297  goto fail;
298  }
299  for (index = 0; ; index++) {
300  if (igb_proc_entries[index].read == NULL)
301  break;
302  if (!(create_proc_read_entry(igb_proc_entries[index].name,
303  0444,
304  adapter->info_dir,
305  igb_proc_entries[index].read,
306  adapter))) {
307 
308  rc = -ENOMEM;
309  goto fail;
310  }
311  }
312  if (igb_thermal_present(adapter) == false)
313  goto exit;
314 
315  for (i = 0; i < E1000_MAX_SENSORS; i++) {
316  if (adapter->hw.mac.thermal_sensor_data.sensor[i].location == 0)
317  continue;
318 
319  snprintf(buf, sizeof(buf), "sensor_%d", i);
320  adapter->therm_dir[i] = proc_mkdir(buf, adapter->info_dir);
321  if (adapter->therm_dir[i] == NULL) {
322  rc = -ENOMEM;
323  goto fail;
324  }
325  for (index = 0; ; index++) {
326  if (igb_internal_entries[index].read == NULL)
327  break;
328  /*
329  * therm_data struct contains pointer the read func
330  * will be needing
331  */
332  adapter->therm_data[i].hw = &adapter->hw;
333  adapter->therm_data[i].sensor_data =
334  &adapter->hw.mac.thermal_sensor_data.sensor[i];
335 
336  if (!(create_proc_read_entry(
337  igb_internal_entries[index].name,
338  0444,
339  adapter->therm_dir[i],
340  igb_internal_entries[index].read,
341  &adapter->therm_data[i]))) {
342  rc = -ENOMEM;
343  goto fail;
344  }
345  }
346  }
347  goto exit;
348 
349 fail:
350  igb_del_proc_entries(adapter);
351 exit:
352  return rc;
353 }
