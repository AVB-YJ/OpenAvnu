79 void openavbAdpSMAdvertiseEntityStateMachine()
80 {
81  AVB_TRACE_ENTRY(AVB_TRACE_ADP);
82  bool bRunning = TRUE;
83 
84  openavb_adp_sm_advertise_entity_state_t state = OPENAVB_ADP_SM_ADVERTISE_ENTITY_STATE_INITIALIZE;
85 
86  while (bRunning) {
87  switch (state) {
88  case OPENAVB_ADP_SM_ADVERTISE_ENTITY_STATE_INITIALIZE:
89  {
90  AVB_TRACE_LINE(AVB_TRACE_ADP);
91  AVB_LOG_DEBUG("State: OPENAVB_ADP_SM_ADVERTISE_ENTITY_STATE_INITIALIZE");
92 
93  ADP_LOCK();
94  openavbAdpSMGlobalVars.entityInfo.pdu.available_index = 0;
95  ADP_UNLOCK();
96 
97  // The advertise interface will send the first advertisement on startup.
98  // This entity will be responsible for sending subsequent advertisements.
99  // This allows the advertise entity and advertise interface to startup without any inter-dependencies.
100  state = OPENAVB_ADP_SM_ADVERTISE_ENTITY_STATE_RESET_WAIT;
101  }
102  break;
103  case OPENAVB_ADP_SM_ADVERTISE_ENTITY_STATE_ADVERTISE:
104  {
105  AVB_TRACE_LINE(AVB_TRACE_ADP);
106  AVB_LOG_DEBUG("State: OPENAVB_ADP_SM_ADVERTISE_ENTITY_STATE_ADVERTISE");
107 
108  openavbAdpSMAdvertiseEntity_sendAvailable();
109  ADP_LOCK();
110  openavbAdpSMAdvertiseEntityVars.needsAdvertise = FALSE;
111  ADP_UNLOCK();
112  state = OPENAVB_ADP_SM_ADVERTISE_ENTITY_STATE_RESET_WAIT;
113  }
114  break;
115  case OPENAVB_ADP_SM_ADVERTISE_ENTITY_STATE_RESET_WAIT:
116  {
117  AVB_TRACE_LINE(AVB_TRACE_ADP);
118  AVB_LOG_DEBUG("State: OPENAVB_ADP_SM_ADVERTISE_ENTITY_STATE_RESET_WAIT");
119 
120  ADP_LOCK();
121  CLOCK_GETTIME(OPENAVB_CLOCK_REALTIME, &openavbAdpSMAdvertiseEntityVars.reannounceTimerTimeout);
122  /* The advertisements should be sent at intervals of 1/4 valid_time, where valid_time is in 2-second units.
123  * See IEEE Std 1722.1-2013 clauses 6.2.1.6 and 6.2.4. */
124  U32 advDelayUsec = openavbAdpSMGlobalVars.entityInfo.header.valid_time / 2 * MICROSECONDS_PER_SECOND;
125  if (advDelayUsec < MICROSECONDS_PER_SECOND) {
126  advDelayUsec = MICROSECONDS_PER_SECOND;
127  }
128  openavbTimeTimespecAddUsec(&openavbAdpSMAdvertiseEntityVars.reannounceTimerTimeout, advDelayUsec);
129  ADP_UNLOCK();
130  state = OPENAVB_ADP_SM_ADVERTISE_ENTITY_STATE_WAITING;
131  }
132  break;
133  case OPENAVB_ADP_SM_ADVERTISE_ENTITY_STATE_WAITING:
134  {
135  AVB_TRACE_LINE(AVB_TRACE_ADP);
136  AVB_LOG_DEBUG("State: OPENAVB_ADP_SM_ADVERTISE_ENTITY_STATE_WAITING");
137 
138  ADP_LOCK();
139  openavbAdpSMAdvertiseInterfaceSet_rcvdDiscover(FALSE);
140  openavbAdpSMGlobalVars.entityInfo.pdu.available_index++;
141  ADP_UNLOCK();
142 
143  // Wait for change in state
144  while (state == OPENAVB_ADP_SM_ADVERTISE_ENTITY_STATE_WAITING && bRunning) {
145  U32 timeoutMSec;
146  struct timespec now;
147  CLOCK_GETTIME(OPENAVB_CLOCK_REALTIME, &now);
148  timeoutMSec = openavbTimeUntilMSec(&now, &openavbAdpSMAdvertiseEntityVars.reannounceTimerTimeout);
149 
150  if (timeoutMSec == 0) {
151  /* No need to wait. */
152  state = OPENAVB_ADP_SM_ADVERTISE_ENTITY_STATE_ADVERTISE;
153  }
154  else {
155  SEM_ERR_T(err);
156  SEM_TIMEDWAIT(openavbAdpSMAdvertiseEntitySemaphore, timeoutMSec, err);
157  if (!SEM_IS_ERR_NONE(err) && !SEM_IS_ERR_TIMEOUT(err)) { AVB_LOGF_WARNING("Semaphore error %d", err); }
158 
159  if (openavbAdpSMAdvertiseEntityVars.doTerminate) {
160  bRunning = FALSE;
161  } else if (SEM_IS_ERR_TIMEOUT(err) ||
162  openavbAdpSMAdvertiseEntityVars.needsAdvertise) {
163  state = OPENAVB_ADP_SM_ADVERTISE_ENTITY_STATE_ADVERTISE;
164  }
165  }
166  }
167  }
168  break;
169 
170  default:
171  AVB_LOG_ERROR("State: Unexpected!");
172  bRunning = FALSE; // Unexpected
173  break;
174  }
175  }
176  AVB_TRACE_EXIT(AVB_TRACE_ADP);
177 }

190 
191  SEM_ERR_T(err);
192  SEM_INIT(openavbAdpSMAdvertiseEntitySemaphore, 1, err);
193  SEM_LOG_ERR(err);
194 
195  ADP_LOCK();
196  openavbAdpSMAdvertiseEntityVars.needsAdvertise = FALSE;
197  openavbAdpSMAdvertiseEntityVars.doTerminate = FALSE;
198  ADP_UNLOCK();
199 
200  // Start the Advertise Entity State Machine
201  bool errResult;
202  THREAD_CREATE(openavbAdpSmAdvertiseEntityThread, openavbAdpSmAdvertiseEntityThread, NULL, openavbAdpSMAdvertiseEntityThreadFn, NULL);
203  THREAD_CHECK_ERROR(openavbAdpSmAdvertiseEntityThread, "Thread / task creation failed", errResult);
204  if (errResult); // Already reported
205 
206  AVB_TRACE_EXIT(AVB_TRACE_ADP);
207 }
