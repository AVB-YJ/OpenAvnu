43 GstAlBuf* gst_al_pull_buffer(GstAppSink *sink)
44 {
45  GstAlBuf *buf = g_new0(GstAlBuf,1);
46  GstMemory *memory;
47  GstBuffer * buffer;
48  GstSample *sample = gst_app_sink_pull_sample(sink);
49  buf->m_sample = sample;
50  if(sample)
51  {
52  buffer = gst_sample_get_buffer(sample);
53  buf->m_buffer = buffer;
54  if(buffer)
55  {
56  memory = gst_buffer_get_memory(buffer, 0);
57  buf->m_memory = memory;
58  if(memory)
59  {
60  GstMapInfo *info = &buf->m_info;
61  if(gst_memory_map(memory, info, GST_MAP_READ))
62  {
63  buf->m_dptr = info->data;
64  buf->m_dlen = info->size;
65  goto pull_success;
66  }
67  gst_memory_unref(memory);
68  }
69  }
70  gst_sample_unref(sample);
71  }
72  g_free(buf);
73  buf = NULL;
74 pull_success:
75  return buf;
76 }
