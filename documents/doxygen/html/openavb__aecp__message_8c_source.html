97 bool openavbAecpOpenSocket(const char* ifname, U16 vlanID, U8 vlanPCP)
98 {
99  AVB_TRACE_ENTRY(AVB_TRACE_AECP);
100 
101  hdr_info_t hdr;
102 
103 #ifndef UBUNTU
104  // This is the normal case for most of our supported platforms
105  rxSock = openavbRawsockOpen(ifname, TRUE, FALSE, ETHERTYPE_8021Q, AECP_FRAME_LEN, AECP_NUM_BUFFERS);
106 #else
107  rxSock = openavbRawsockOpen(ifname, TRUE, FALSE, ETHERTYPE_AVTP, AECP_FRAME_LEN, AECP_NUM_BUFFERS);
108 #endif
109  txSock = openavbRawsockOpen(ifname, FALSE, TRUE, ETHERTYPE_AVTP, AECP_FRAME_LEN, AECP_NUM_BUFFERS);
110 
111  if (txSock && rxSock
112  && openavbRawsockGetAddr(txSock, ADDR_PTR(&intfAddr))
113  && openavbRawsockRxMulticast(rxSock, TRUE, ADDR_PTR(&intfAddr))) // Only accept packets sent directly to this interface
114  {
115  if (!openavbRawsockRxAVTPSubtype(rxSock, OPENAVB_AECP_AVTP_SUBTYPE | 0x80)) {
116  AVB_LOG_DEBUG("RX AVTP Subtype not supported");
117  }
118 
119  memset(&hdr, 0, sizeof(hdr_info_t));
120  hdr.shost = ADDR_PTR(&intfAddr);
121  // hdr.dhost; // Set at tx time.
122  hdr.ethertype = ETHERTYPE_AVTP;
123  if (vlanID != 0 || vlanPCP != 0) {
124  hdr.vlan = TRUE;
125  hdr.vlan_pcp = vlanPCP;
126  hdr.vlan_vid = vlanID;
127  AVB_LOGF_DEBUG("VLAN pcp=%d vid=%d", hdr.vlan_pcp, hdr.vlan_vid);
128  }
129  if (!openavbRawsockTxSetHdr(txSock, &hdr)) {
130  AVB_LOG_ERROR("TX socket Header Failure");
131  openavbAecpCloseSocket();
132  AVB_TRACE_EXIT(AVB_TRACE_AECP);
133  return false;
134  }
135 
136  AVB_TRACE_EXIT(AVB_TRACE_AECP);
137  return true;
138  }
139 
140  AVB_LOG_ERROR("Invalid socket");
141  openavbAecpCloseSocket();
142 
143  AVB_TRACE_EXIT(AVB_TRACE_AECP);
144  return false;
145 }

524 static void openavbAecpMessageRxFrameReceive(U32 timeoutUsec)
525 {
526  AVB_TRACE_ENTRY(AVB_TRACE_AECP);
527 
528  hdr_info_t hdrInfo;
529  unsigned int offset, len;
530  U8 *pBuf, *pFrame;
531 
532  memset(&hdrInfo, 0, sizeof(hdr_info_t));
533 
534  pBuf = (U8 *)openavbRawsockGetRxFrame(rxSock, timeoutUsec, &offset, &len);
535  if (pBuf) {
536  pFrame = pBuf + offset;
537 
538  offset = openavbRawsockRxParseHdr(rxSock, pBuf, &hdrInfo);
539  {
540 #ifndef UBUNTU
541  if (hdrInfo.ethertype == ETHERTYPE_8021Q) {
542  // Oh! Need to look past the VLAN tag
543  U16 vlan_bits = ntohs(*(U16 *)(pFrame + offset));
544  hdrInfo.vlan = TRUE;
545  hdrInfo.vlan_vid = vlan_bits & 0x0FFF;
546  hdrInfo.vlan_pcp = (vlan_bits >> 13) & 0x0007;
547  offset += 2;
548  hdrInfo.ethertype = ntohs(*(U16 *)(pFrame + offset));
549  offset += 2;
550  }
551 #endif
552 
553  // Make sure that this is an AVTP packet
554  // (Should always be AVTP if it's to our AVTP-specific multicast address)
555  if (hdrInfo.ethertype == ETHERTYPE_AVTP) {
556  // parse the PDU only for AECP messages
557  if (*(pFrame + offset) == (0x80 | OPENAVB_AECP_AVTP_SUBTYPE)) {
558  openavbAecpMessageRxFrameParse(pFrame + offset, len - offset, &hdrInfo);
559  }
560  }
561  else {
562  AVB_LOG_WARNING("Received non-AVTP frame!");
563  AVB_LOGF_DEBUG("Unexpected packet data (length %d):", len);
564  AVB_LOG_BUFFER(AVB_LOG_LEVEL_DEBUG, pFrame, len, 16);
565  }
566  }
567 
568  // Release the frame
569  openavbRawsockRelRxFrame(rxSock, pBuf);
570  }
571 
572  AVB_TRACE_EXIT(AVB_TRACE_AECP);
573 }
