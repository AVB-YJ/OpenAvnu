44 static bool openavbAvdeccMsgClntSendToServer(int socketHandle, openavbAvdeccMessage_t *msg)
45 {
46  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC_MSG);
47 
48  if (!msg || socketHandle == AVB_AVDECC_MSG_HANDLE_INVALID) {
49  AVB_LOG_ERROR("Client send: invalid argument passed");
50  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
51  return FALSE;
52  }
53 
54  ssize_t nWrite = write(socketHandle, msg, OPENAVB_AVDECC_MSG_LEN);
55  AVB_LOGF_VERBOSE("Sent message, len=%zu, nWrite=%zu", OPENAVB_AVDECC_MSG_LEN, nWrite);
56 
57  if (nWrite < OPENAVB_AVDECC_MSG_LEN) {
58  if (nWrite < 0) {
59  AVB_LOGF_ERROR("Client failed to write socket: %s", strerror(errno));
60  }
61  else if (nWrite == 0) {
62  AVB_LOG_ERROR("Client send: socket closed unexpectedly");
63  }
64  else {
65  AVB_LOG_ERROR("Client send: short write");
66  }
67  socketClose(socketHandle);
68  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
69  return FALSE;
70  }
71 
72  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);

76 int openavbAvdeccMsgClntOpenSrvrConnection(void)
77 {
78  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC_MSG);
79  struct sockaddr_un server;
80  server.sun_family = AF_UNIX;
81  snprintf(server.sun_path, UNIX_PATH_MAX, AVB_AVDECC_MSG_UNIX_PATH);
82 
83  int h = socket(AF_UNIX, SOCK_STREAM, 0);
84  if (h < 0) {
85  AVB_LOGF_ERROR("Failed to open socket: %s", strerror(errno));
86  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
87  return AVB_AVDECC_MSG_HANDLE_INVALID;
88  }
89 
90  AVB_LOGF_DEBUG("Connecting to %s", server.sun_path);
91  int rslt = connect(h, (struct sockaddr*)&server, sizeof(struct sockaddr_un));
92  if (rslt < 0) {
93  AVB_LOGF_DEBUG("Failed to connect socket: %s", strerror(errno));
94  socketClose(h);
95  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
96  return AVB_AVDECC_MSG_HANDLE_INVALID;
97  }
98 
99  AVB_LOG_DEBUG("Connected to AVDECC Msg");
100  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
101  return h;

112 bool openavbAvdeccMsgClntService(int socketHandle, int timeout)
113 {
114  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC_MSG);
115  bool rc = FALSE;
116 
117  if (socketHandle == AVB_AVDECC_MSG_HANDLE_INVALID) {
118  AVB_LOG_ERROR("Client service: invalid socket");
119  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
120  return FALSE;
121  }
122 
123  struct pollfd fds[1];
124  memset(fds, 0, sizeof(struct pollfd));
125  fds[0].fd = socketHandle;
126  fds[0].events = POLLIN;
127 
128  AVB_LOG_VERBOSE("Waiting for event...");
129  int pRet = poll(fds, 1, timeout);
130 
131  if (pRet == 0) {
132  AVB_LOG_VERBOSE("Poll timeout");
133  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
134  return TRUE;
135  }
136  else if (pRet < 0) {
137  if (errno == EINTR) {
138  AVB_LOG_VERBOSE("Poll interrupted");
139  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
140  return TRUE;
141  }
142  else {
143  AVB_LOGF_ERROR("Poll error: %s", strerror(errno));
144  }
145  }
146  else {
147  AVB_LOGF_DEBUG("Poll returned %d events", pRet);
148  // only one fd, so it's readable.
149  openavbAvdeccMessage_t msgBuf;
150  memset(&msgBuf, 0, OPENAVB_AVDECC_MSG_LEN);
151  ssize_t nRead = read(socketHandle, &msgBuf, OPENAVB_AVDECC_MSG_LEN);
152 
153  if (nRead < OPENAVB_AVDECC_MSG_LEN) {
154  // sock closed
155  if (nRead == 0) {
156  AVB_LOG_ERROR("Socket closed unexpectedly");
157  }
158  else if (nRead < 0) {
159  AVB_LOGF_ERROR("Socket read error: %s", strerror(errno));
160  }
161  else {
162  AVB_LOG_ERROR("Socket read to short");
163  }
164  socketClose(socketHandle);
165  }
166  else {
167  // got a message
168  if (openavbAvdeccMsgClntReceiveFromServer(socketHandle, &msgBuf)) {
169  rc = TRUE;
170  }
171  else {
172  AVB_LOG_ERROR("Invalid message received");
173  socketClose(socketHandle);
174  }
175  }
176  }
177  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
178  return rc;
179 }
