89 int process_mrp_msg(char *buf, int buflen, struct mrp_talker_ctx *ctx)
90 {
91 
92  /*
93  * 1st character indicates application
94  * [MVS] - MAC, VLAN or STREAM
95  */
96  unsigned int id;
97  unsigned int priority;
98  unsigned int vid;
99  int i, j, k;
100  unsigned int substate;
101  unsigned char recovered_streamid[8];
102  k = 0;
103  next_line:if (k >= buflen)
104  return 0;
105  switch (buf[k]) {
106  case 'E':
107  printf("%s from mrpd\n", buf);
108  fflush(stdout);
109  mrp_error = 1;
110  break;
111  case 'O':
112  mrp_okay = 1;
113  break;
114  case 'M':
115  case 'V':
116  printf("%s unhandled from mrpd\n", buf);
117  fflush(stdout);
118 
119  /* unhandled for now */
120  break;
121  case 'L':
122 
123  /* parse a listener attribute - see if it matches our monitor_stream_id */
124  i = k;
125  while (buf[i] != 'D')
126  i++;
127  i += 2; /* skip the ':' */
128  sscanf(&(buf[i]), "%d", &substate);
129  while (buf[i] != 'S')
130  i++;
131  i += 2; /* skip the ':' */
132  for (j = 0; j < 8; j++) {
133  sscanf(&(buf[i + 2 * j]), "%02x", &id);
134  recovered_streamid[j] = (unsigned char)id;
135  } printf
136  ("FOUND STREAM ID=%02x%02x%02x%02x%02x%02x%02x%02x ",
137  recovered_streamid[0], recovered_streamid[1],
138  recovered_streamid[2], recovered_streamid[3],
139  recovered_streamid[4], recovered_streamid[5],
140  recovered_streamid[6], recovered_streamid[7]);
141  switch (substate) {
142  case 0:
143  printf("with state ignore\n");
144  break;
145  case 1:
146  printf("with state askfailed\n");
147  break;
148  case 2:
149  printf("with state ready\n");
150  break;
151  case 3:
152  printf("with state readyfail\n");
153  break;
154  default:
155  printf("with state UNKNOWN (%d)\n", substate);
156  break;
157  }
158  if (substate > MSRP_LISTENER_ASKFAILED) {
159  if (memcmp
160  (recovered_streamid, ctx->monitor_stream_id,
161  sizeof(recovered_streamid)) == 0) {
162  ctx->listeners = 1;
163  printf("added listener\n");
164  }
165  }
166  fflush(stdout);
167 
168  /* try to find a newline ... */
169  while ((i < buflen) && (buf[i] != '\n') && (buf[i] != '\0'))
170  i++;
171  if (i == buflen)
172  return 0;
173  if (buf[i] == '\0')
174  return 0;
175  i++;
176  k = i;
177  goto next_line;
178  break;
179  case 'D':
180  i = k + 4;
181 
182  /* save the domain attribute */
183  sscanf(&(buf[i]), "%d", &id);
184  while (buf[i] != 'P')
185  i++;
186  i += 2; /* skip the ':' */
187  sscanf(&(buf[i]), "%d", &priority);
188  while (buf[i] != 'V')
189  i++;
190  i += 2; /* skip the ':' */
191  sscanf(&(buf[i]), "%x", &vid);
192  if (id == 6) {
193  ctx->domain_class_a_id = id;
194  ctx->domain_class_a_priority = priority;
195  ctx->domain_class_a_vid = vid;
196  ctx->domain_a_valid = 1;
197  } else {
198  ctx->domain_class_b_id = id;
199  ctx->domain_class_b_priority = priority;
200  ctx->domain_class_b_vid = vid;
201  ctx->domain_b_valid = 1;
202  }
203  while ((i < buflen) && (buf[i] != '\n') && (buf[i] != '\0'))
204  i++;
205  if ((i == buflen) || (buf[i] == '\0'))
206  return 0;
207  i++;
208  k = i;
209  goto next_line;
210  break;
211  case 'T':
212 
213  /* as simple_talker we don't care about other talkers */
214  i = k;
215  while ((i < buflen) && (buf[i] != '\n') && (buf[i] != '\0'))
216  i++;
217  if (i == buflen)
218  return 0;
219  if (buf[i] == '\0')
220  return 0;
221  i++;
222  k = i;
223  goto next_line;
224  break;
225  case 'S':
226 
227  /* handle the leave/join events */
228  switch (buf[k + 4]) {
229  case 'L':
230  i = k + 5;
231  while (buf[i] != 'D')
232  i++;
233  i += 2; /* skip the ':' */
234  sscanf(&(buf[i]), "%d", &substate);
235  while (buf[i] != 'S')
236  i++;
237  i += 2; /* skip the ':' */
238  for (j = 0; j < 8; j++) {
239  sscanf(&(buf[i + 2 * j]), "%02x", &id);
240  recovered_streamid[j] = (unsigned char)id;
241  } printf
242  ("EVENT on STREAM ID=%02x%02x%02x%02x%02x%02x%02x%02x ",
243  recovered_streamid[0], recovered_streamid[1],
244  recovered_streamid[2], recovered_streamid[3],
245  recovered_streamid[4], recovered_streamid[5],
246  recovered_streamid[6], recovered_streamid[7]);
247  switch (substate) {
248  case 0:
249  printf("with state ignore\n");
250  break;
251  case 1:
252  printf("with state askfailed\n");
253  break;
254  case 2:
255  printf("with state ready\n");
256  break;
257  case 3:
258  printf("with state readyfail\n");
259  break;
260  default:
261  printf("with state UNKNOWN (%d)\n", substate);
262  break;
263  }
264  switch (buf[k + 1]) {
265  case 'L':
266  printf("got a leave indication\n");
267  if (memcmp
268  (recovered_streamid, ctx->monitor_stream_id,
269  sizeof(recovered_streamid)) == 0) {
270  ctx->listeners = 0;
271  printf("listener left\n");
272  }
273  break;
274  case 'J':
275  case 'N':
276  printf("got a new/join indication\n");
277  if (substate > MSRP_LISTENER_ASKFAILED) {
278  if (memcmp
279  (recovered_streamid,
280  ctx->monitor_stream_id,
281  sizeof(recovered_streamid)) == 0)
282  ctx->listeners = 1;
283  }
284  break;
285  }
286 
287  /* only care about listeners ... */
288  default:
289  return 0;
290  break;
291  }
292  break;
293  case '\0':
294  break;
295  }
296  return 0;
297 }

299 void *mrp_monitor_thread(void *arg)
300 {
301  char *msgbuf;
302  struct sockaddr_in client_addr;
303  struct msghdr msg;
304  struct iovec iov;
305  int bytes = 0;
306  struct pollfd fds;
307  int rc;
308  struct mrp_talker_ctx *ctx = (struct mrp_talker_ctx*) arg;
309 
310  msgbuf = (char *)malloc(MAX_MRPD_CMDSZ);
311  if (NULL == msgbuf)
312  return NULL;
313  while (!ctx->halt_tx) {
314  fds.fd = ctx->control_socket;
315  fds.events = POLLIN;
316  fds.revents = 0;
317  rc = poll(&fds, 1, 100);
318  if (rc < 0) {
319  free(msgbuf);
320  pthread_exit(NULL);
321  }
322  if (rc == 0)
323  continue;
324  if ((fds.revents & POLLIN) == 0) {
325  free(msgbuf);
326  pthread_exit(NULL);
327  }
328  memset(&msg, 0, sizeof(msg));
329  memset(&client_addr, 0, sizeof(client_addr));
330  memset(msgbuf, 0, MAX_MRPD_CMDSZ);
331  iov.iov_len = MAX_MRPD_CMDSZ;
332  iov.iov_base = msgbuf;
333  msg.msg_name = &client_addr;
334  msg.msg_namelen = sizeof(client_addr);
335  msg.msg_iov = &iov;
336  msg.msg_iovlen = 1;
337  bytes = recvmsg(ctx->control_socket, &msg, 0);
338  if (bytes < 0)
339  continue;
340  process_mrp_msg(msgbuf, bytes, ctx);
341  }
342  free(msgbuf);
343  pthread_exit(NULL);
344 }

350 int mrp_connect(struct mrp_talker_ctx *ctx)
351 {
352  struct sockaddr_in addr;
353  int sock_fd = -1;
354  int rc;
355  sock_fd = socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP);
356  if (sock_fd < 0)
357  goto out;
358  memset(&addr, 0, sizeof(addr));
359  addr.sin_family = AF_INET;
360  addr.sin_port = htons(MRPD_PORT_DEFAULT);
361  inet_aton("127.0.0.1", &addr.sin_addr);
362  memset(&addr, 0, sizeof(addr));
363  ctx->control_socket = sock_fd;
364  rc = pthread_attr_init(&monitor_attr);
365  rc |= pthread_create(&monitor_thread, NULL, mrp_monitor_thread, ctx);
366  return rc;
367  return 0;
368  out: if (sock_fd != -1)
369  close(sock_fd);
370  sock_fd = -1;
371  return -1;
372 }
