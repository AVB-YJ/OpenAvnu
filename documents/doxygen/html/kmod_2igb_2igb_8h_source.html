165 /* RX descriptor control thresholds.
166  * PTHRESH - MAC will consider prefetch if it has fewer than this number of
167  * descriptors available in its onboard memory.
168  * Setting this to 0 disables RX descriptor prefetch.
169  * HTHRESH - MAC will only prefetch if there are at least this many descriptors
170  * available in host memory.
171  * If PTHRESH is 0, this should also be 0.
172  * WTHRESH - RX descriptor writeback threshold - MAC will delay writing back
173  * descriptors until either it has this many to write back, or the
174  * ITR timer expires.
175  */
176 #define IGB_RX_PTHRESH ((hw->mac.type == e1000_i354) ? 12 : 8)
177 #define IGB_RX_HTHRESH 8
178 #define IGB_TX_PTHRESH ((hw->mac.type == e1000_i354) ? 20 : 8)
179 #define IGB_TX_HTHRESH 1
180 #define IGB_RX_WTHRESH ((hw->mac.type == e1000_82576 && \
181  adapter->msix_entries) ? 1 : 4)
