70  LinkLayerAddress(uint64_t address_scalar) {
71  uint8_t *ptr;
72  address_scalar <<= 16;
73  if(addr == NULL)
74  return;
75 
76  for (ptr = addr; ptr < addr + ETHER_ADDR_OCTETS; ++ptr) {
77  *ptr = (address_scalar & 0xFF00000000000000ULL) >> 56;
78  address_scalar <<= 8;
79  }
