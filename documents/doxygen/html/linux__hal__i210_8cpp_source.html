55 pci_connect( device_t *igb_dev )
56 {
57  char devpath[IGB_BIND_NAMESZ];
58  struct pci_access *pacc;
59  struct pci_dev *dev;
60  int err;
61 
62  memset(igb_dev, 0, sizeof(device_t));
63  pacc = pci_alloc();
64  pci_init(pacc);
65  pci_scan_bus(pacc);
66 
67  for (dev = pacc->devices; dev; dev = dev->next) {
68  pci_fill_info(dev, PCI_FILL_IDENT | PCI_FILL_BASES | PCI_FILL_CLASS);
69  igb_dev->pci_vendor_id = dev->vendor_id;
70  igb_dev->pci_device_id = dev->device_id;
71  igb_dev->domain = dev->domain;
72  igb_dev->bus = dev->bus;
73  igb_dev->dev = dev->dev;
74  igb_dev->func = dev->func;
75  snprintf(devpath, IGB_BIND_NAMESZ, "%04x:%02x:%02x.%d",
76  dev->domain, dev->bus, dev->dev, dev->func);
77  err = igb_probe(igb_dev);
78  if (err) {
79  continue;
80  }
81  GPTP_LOG_INFO("attaching to %s", devpath);
82  err = igb_attach(devpath, igb_dev);
83  if (err) {
84  GPTP_LOG_ERROR("attach failed! (%s)", strerror(err));
85  continue;
86  }
87  /*igb_attach_tx missing here ???*/
88  goto out;
89  }
90  pci_cleanup(pacc);
91  return ENXIO;
92 out:
93  pci_cleanup(pacc);
94  return 0;
95 }

98 bool LinuxTimestamperGeneric::HWTimestamper_PPS_start( ) {
99  unsigned tssdp;
100  unsigned freqout;
101  unsigned ctrl;
102  unsigned tsauxc;
103  unsigned trgttimh;
104 
105  if( igb_private == NULL ) {
106  igb_private = new LinuxTimestamperIGBPrivate;
107  }
108 
109  if( pci_connect( &igb_private->igb_dev ) != 0 ) {
110  return false;
111  }
112 
113  if( igb_init( &igb_private->igb_dev ) != 0 ) {
114  return false;
115  }
116 
117  igb_private->igb_initd = true;
118 
119  igb_lock( &igb_private->igb_dev );
120 
121  // Edges must be second aligned
122  igb_readreg( &igb_private->igb_dev, SYSTIMH, &trgttimh );
123  trgttimh += 2; // First edge in 1-2 seconds
124  igb_writereg(&igb_private->igb_dev, TRGTTIMH0, trgttimh );
125  igb_writereg(&igb_private->igb_dev, TRGTTIML0, 0 );
126 
127  freqout = 500000000;
128  igb_writereg(&igb_private->igb_dev, FREQOUT0, freqout );
129 
130  igb_readreg(&igb_private->igb_dev, IGB_CTRL, &ctrl );
131  ctrl |= 0x400000; // set bit 22 SDP0 enabling output
132  igb_writereg(&igb_private->igb_dev, IGB_CTRL, ctrl );
133 
134  igb_readreg(&igb_private->igb_dev, TSSDP, &tssdp);
135  tssdp &= ~0x40; // Set SDP0 output to freq clock 0
136  tssdp |= 0x80;
137  igb_writereg(&igb_private->igb_dev, TSSDP, tssdp);
138 
139  igb_readreg(&igb_private->igb_dev, TSSDP, &tssdp);
140  tssdp |= 0x100; // set bit 8 -> SDP0 Time Sync Output
141  igb_writereg(&igb_private->igb_dev, TSSDP, tssdp);
142 
143  igb_readreg( &igb_private->igb_dev, TSAUXC, &tsauxc );
144  tsauxc |= 0x14;
145  igb_writereg( &igb_private->igb_dev, TSAUXC, tsauxc );
146 
147  igb_unlock( &igb_private->igb_dev );
148 
149  return true;
150 }
