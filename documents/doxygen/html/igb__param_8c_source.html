246 struct igb_option {
247  enum { enable_option, range_option, list_option } type;
248  const char *name;
249  const char *err;
250  int def;
251  union {
252  struct { /* range_option info */
253  int min;
254  int max;
255  } r;
256  struct { /* list_option info */
257  int nr;
258  struct igb_opt_list *p;
259  } l;
260  } arg;
261 };

263 static int igb_validate_option(unsigned int *value,
264  struct igb_option *opt,
265  struct igb_adapter *adapter)
266 {
267  if (*value == OPTION_UNSET) {
268  *value = opt->def;
269  return 0;
270  }
271 
272  switch (opt->type) {
273  case enable_option:
274  switch (*value) {
275  case OPTION_ENABLED:
276  DPRINTK(PROBE, INFO, "%s Enabled\n", opt->name);
277  return 0;
278  case OPTION_DISABLED:
279  DPRINTK(PROBE, INFO, "%s Disabled\n", opt->name);
280  return 0;
281  }
282  break;
283  case range_option:
284  if (*value >= opt->arg.r.min && *value <= opt->arg.r.max) {
285  DPRINTK(PROBE, INFO,
286  "%s set to %d\n", opt->name, *value);
287  return 0;
288  }
289  break;
290  case list_option: {
291  int i;
292  struct igb_opt_list *ent;
293 
294  for (i = 0; i < opt->arg.l.nr; i++) {
295  ent = &opt->arg.l.p[i];
296  if (*value == ent->i) {
297  if (ent->str[0] != '\0')
298  DPRINTK(PROBE, INFO, "%s\n", ent->str);
299  return 0;
300  }
301  }
302  }
303  break;
304  default:
305  BUG();
306  }
307 
308  DPRINTK(PROBE, INFO, "Invalid %s value specified (%d) %s\n",
309  opt->name, *value, opt->err);
310  *value = opt->def;
311  return -1;
312 }

324 void igb_check_options(struct igb_adapter *adapter)
325 {
326  int bd = adapter->bd_number;
327  struct e1000_hw *hw = &adapter->hw;
328 
329  if (bd >= IGB_MAX_NIC) {
330  DPRINTK(PROBE, NOTICE,
331  "Warning: no configuration for board #%d\n", bd);
332  DPRINTK(PROBE, NOTICE, "Using defaults for all values\n");
333 #ifndef module_param_array
334  bd = IGB_MAX_NIC;
335 #endif
336  }
337 
338  { /* Interrupt Throttling Rate */
339  struct igb_option opt = {
340  .type = range_option,
341  .name = "Interrupt Throttling Rate (ints/sec)",
342  .err = "using default of "__MODULE_STRING(DEFAULT_ITR),
343  .def = DEFAULT_ITR,
344  .arg = { .r = { .min = MIN_ITR,
345  .max = MAX_ITR } }
346  };
347 
348 #ifdef module_param_array
349  if (num_InterruptThrottleRate > bd) {
350 #endif
351  unsigned int itr = InterruptThrottleRate[bd];
352 
353  switch (itr) {
354  case 0:
355  DPRINTK(PROBE, INFO, "%s turned off\n",
356  opt.name);
357  if (hw->mac.type >= e1000_i350)
358  adapter->dmac = IGB_DMAC_DISABLE;
359  adapter->rx_itr_setting = itr;
360  break;
361  case 1:
362  DPRINTK(PROBE, INFO, "%s set to dynamic mode\n",
363  opt.name);
364  adapter->rx_itr_setting = itr;
365  break;
366  case 3:
367  DPRINTK(PROBE, INFO,
368  "%s set to dynamic conservative mode\n",
369  opt.name);
370  adapter->rx_itr_setting = itr;
371  break;
372  default:
373  igb_validate_option(&itr, &opt, adapter);
374  /* Save the setting, because the dynamic bits
375  * change itr. In case of invalid user value,
376  * default to conservative mode, else need to
377  * clear the lower two bits because they are
378  * used as control */
379  if (itr == 3) {
380  adapter->rx_itr_setting = itr;
381  } else {
382  adapter->rx_itr_setting = 1000000000
383  / (itr * 256);
384  adapter->rx_itr_setting &= ~3;
385  }
386  break;
387  }
388 #ifdef module_param_array
389  } else {
390  adapter->rx_itr_setting = opt.def;
391  }
392 #endif
393  adapter->tx_itr_setting = adapter->rx_itr_setting;
394  }
395  { /* Interrupt Mode */
396  struct igb_option opt = {
397  .type = range_option,
398  .name = "Interrupt Mode",
399  .err = "defaulting to 2 (MSI-X)",
400  .def = IGB_INT_MODE_MSIX,
401  .arg = { .r = { .min = MIN_INTMODE,
402  .max = MAX_INTMODE } }
403  };
404 
405 #ifdef module_param_array
406  if (num_IntMode > bd) {
407 #endif
408  unsigned int int_mode = IntMode[bd];
409  igb_validate_option(&int_mode, &opt, adapter);
410  adapter->int_mode = int_mode;
411 #ifdef module_param_array
412  } else {
413  adapter->int_mode = opt.def;
414  }
415 #endif
416  }

537  .def = DEFAULT_VMDQ,
538  .arg = { .r = { .min = MIN_VMDQ,
539  .max = (MAX_VMDQ

546 #endif

553  }

557 #ifdef module_param_array

563  adapter->vmdq_pools = 1;
564  }
565 #endif

582  .name = "RSS - RSS multiqueue receive count",

584  .def = DEFAULT_RSS,

608  break;
609  case e1000_i211:
610  opt.arg.r.max = 2;
611  break;
612  case e1000_82576:
613 #ifndef CONFIG_IGB_VMDQ_NETDEV

615  opt.arg.r.max = 2;
616  break;
617 #endif /* CONFIG_IGB_VMDQ_NETDEV */

619  case e1000_i350:
620  case e1000_i354:
621  default:
622  if (!!adapter->vmdq_pools)
623  opt.arg.r.max = 1;
624  break;
625  }
626 
627  if (adapter->int_mode != IGB_INT_MODE_MSIX) {
628  DPRINTK(PROBE, INFO,
629  "RSS is not supported when in MSI/Legacy Interrupt mode, %s\n",
630  opt.err);
631  opt.arg.r.max = 1;
632  }
633

635  if (num_RSS > bd) {
636 #endif
637  adapter->rss_queues = RSS[bd];
638  switch (adapter->rss_queues) {
639  case 1:
640  break;
641  default:
642  igb_validate_option(&adapter->rss_queues, &opt,
643  adapter);
644  if (adapter->rss_queues)
645  break;
646  case 0:
647  adapter->rss_queues = min_t(u32, opt.arg.r.max,
648  num_online_cpus());
649  break;
650  }
651 #ifdef module_param_array
652  } else {
653  adapter->rss_queues = opt.def;
654  }
655 #endif
656  }
657  { /* QueuePairs - Enable Tx/Rx queue pairs for interrupt handling */
658  struct igb_option opt = {
659  .type = enable_option,
660  .name =
661  "QueuePairs - Tx/Rx queue pairs for interrupt handling",
662  .err = "defaulting to Enabled",
663  .def = OPTION_ENABLED
664  };
665 #ifdef module_param_array
666  if (num_QueuePairs > bd) {
667 #endif
668  unsigned int qp = QueuePairs[bd];
669  /*
670  * We must enable queue pairs if the number of queues
671  * exceeds the number of available interrupts. We are
672  * limited to 10, or 3 per unallocated vf. On I210 and
673  * I211 devices, we are limited to 5 interrupts.
674  * However, since I211 only supports 2 queues, we do not
675  * need to check and override the user option.
676  */
677  if (qp == OPTION_DISABLED) {
678  if (adapter->rss_queues > 4)
679  qp = OPTION_ENABLED;
680 
681  if (adapter->vmdq_pools > 4)
682  qp = OPTION_ENABLED;
683 
684  if (adapter->rss_queues > 1 &&
685  (adapter->vmdq_pools > 3 ||
686  adapter->vfs_allocated_count > 6))
687  qp = OPTION_ENABLED;
688 
689  if (hw->mac.type == e1000_i210 &&
690  adapter->rss_queues > 2)
691  qp = OPTION_ENABLED;
692 
693  if (qp == OPTION_ENABLED)
694  DPRINTK(PROBE, INFO,
695  "Number of queues exceeds available interrupts, %s\n",
696  opt.err);
697  }
698  igb_validate_option(&qp, &opt, adapter);
699  adapter->flags |= qp ? IGB_FLAG_QUEUE_PAIRS : 0;
700 #ifdef module_param_array
701  } else {
