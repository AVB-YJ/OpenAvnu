558 bool PTPMessageCommon::getTxTimestamp( EtherPort *port, uint32_t link_speed )
559 {
560  OSTimer *timer = port->getTimerFactory()->createTimer();
561  int ts_good;
562  Timestamp tx_timestamp;
563  uint32_t unused;
564  unsigned req = TX_TIMEOUT_BASE;
565  int iter = TX_TIMEOUT_ITER;
566 
567  ts_good = port->getTxTimestamp
568  ( this, tx_timestamp, unused, false );
569  while( ts_good != GPTP_EC_SUCCESS && iter-- != 0 )
570  {
571  timer->sleep(req);
572  if (ts_good != GPTP_EC_EAGAIN && iter < 1)
573  GPTP_LOG_ERROR(
574  "Error (TX) timestamping PDelay request "
575  "(Retrying-%d), error=%d", iter, ts_good);
576  ts_good = port->getTxTimestamp
577  ( this, tx_timestamp, unused , iter == 0 );
578  req *= 2;
579  }
580 
581  if( ts_good == GPTP_EC_SUCCESS )
582  {
583  Timestamp phy_compensation = port->getTxPhyDelay( link_speed );
584  GPTP_LOG_DEBUG( "TX PHY compensation: %s sec",
585  phy_compensation.toString().c_str() );
586  phy_compensation._version = tx_timestamp._version;
587  _timestamp = tx_timestamp + phy_compensation;
588  } else
589  {
590  char msg[HWTIMESTAMPER_EXTENDED_MESSAGE_SIZE];
591  port->getExtendedError(msg);
592  GPTP_LOG_ERROR(
593  "Error (TX) timestamping PDelay request, error=%d\t%s",
594  ts_good, msg);
595  _timestamp = INVALID_TIMESTAMP;
596  }
597 
598  delete timer;
599  return ts_good == GPTP_EC_SUCCESS;
600 }
