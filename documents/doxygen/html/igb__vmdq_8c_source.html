33 int igb_vmdq_open(struct net_device *dev)
34 {
35  struct igb_vmdq_adapter *vadapter = netdev_priv(dev);
36  struct igb_adapter *adapter = vadapter->real_adapter;
37  struct net_device *main_netdev = adapter->netdev;
38  int hw_queue = vadapter->rx_ring->queue_index +
39  adapter->vfs_allocated_count;
40 
41  if (test_bit(__IGB_DOWN, &adapter->state)) {
42  DPRINTK(DRV, WARNING,
43  "Open %s before opening this device.\n",
44  main_netdev->name);
45  return -EAGAIN;
46  }
47  netif_carrier_off(dev);
48  vadapter->tx_ring->vmdq_netdev = dev;
49  vadapter->rx_ring->vmdq_netdev = dev;
50  if (is_valid_ether_addr(dev->dev_addr)) {
51  igb_del_mac_filter(adapter, dev->dev_addr, hw_queue);
52  igb_add_mac_filter(adapter, dev->dev_addr, hw_queue);
53  }
54  netif_carrier_on(dev);
55  return 0;
56 }

116 static int igb_write_vm_addr_list(struct net_device *netdev)
117 {
118  struct igb_vmdq_adapter *vadapter = netdev_priv(netdev);
119  struct igb_adapter *adapter = vadapter->real_adapter;
120  int count = 0;
121  int hw_queue = vadapter->rx_ring->queue_index +
122  adapter->vfs_allocated_count;
123 
124  /* return ENOMEM indicating insufficient memory for addresses */
125  if (netdev_uc_count(netdev) > igb_available_rars(adapter))
126  return -ENOMEM;
127 
128  if (!netdev_uc_empty(netdev)) {
129 #ifdef NETDEV_HW_ADDR_T_UNICAST
130  struct netdev_hw_addr *ha;
131 #else
132  struct dev_mc_list *ha;
133 #endif
134  netdev_for_each_uc_addr(ha, netdev) {
135 #ifdef NETDEV_HW_ADDR_T_UNICAST
136  igb_del_mac_filter(adapter, ha->addr, hw_queue);
137  igb_add_mac_filter(adapter, ha->addr, hw_queue);
138 #else
139  igb_del_mac_filter(adapter, ha->da_addr, hw_queue);
140  igb_add_mac_filter(adapter, ha->da_addr, hw_queue);
141 #endif
142  count++;
143  }
144  }
145  return count;
146 }

149 #define E1000_VMOLR_UPE 0x20000000 /* Unicast promiscuous mode */
150 void igb_vmdq_set_rx_mode(struct net_device *dev)
151 {
152  struct igb_vmdq_adapter *vadapter = netdev_priv(dev);
153  struct igb_adapter *adapter = vadapter->real_adapter;
154  struct e1000_hw *hw = &adapter->hw;
155  u32 vmolr, rctl;
156  int hw_queue = vadapter->rx_ring->queue_index +
157  adapter->vfs_allocated_count;
158 
159  /* Check for Promiscuous and All Multicast modes */
160  vmolr = E1000_READ_REG(hw, E1000_VMOLR(hw_queue));
161 
162  /* clear the affected bits */
163  vmolr &= ~(E1000_VMOLR_UPE | E1000_VMOLR_MPME |
164  E1000_VMOLR_ROPE | E1000_VMOLR_ROMPE);
165 
166  if (dev->flags & IFF_PROMISC) {
167  vmolr |= E1000_VMOLR_UPE;
168  rctl = E1000_READ_REG(hw, E1000_RCTL);
169  rctl |= E1000_RCTL_UPE;
170  E1000_WRITE_REG(hw, E1000_RCTL, rctl);
171  } else {
172  rctl = E1000_READ_REG(hw, E1000_RCTL);
173  rctl &= ~E1000_RCTL_UPE;
174  E1000_WRITE_REG(hw, E1000_RCTL, rctl);
175  if (dev->flags & IFF_ALLMULTI) {
176  vmolr |= E1000_VMOLR_MPME;
177  } else {
178  /*
179  * Write addresses to the MTA, if the attempt fails
180  * then we should just turn on promiscous mode so
181  * that we can at least receive multicast traffic
182  */
183  if (igb_write_mc_addr_list(adapter->netdev) != 0)
184  vmolr |= E1000_VMOLR_ROMPE;
185  }
186 #ifdef HAVE_SET_RX_MODE
187  /*
188  * Write addresses to available RAR registers, if there is not
189  * sufficient space to store all the addresses then enable
190  * unicast promiscous mode
191  */
192  if (igb_write_vm_addr_list(dev) < 0)
193  vmolr |= E1000_VMOLR_UPE;
194 #endif
195  }
196  E1000_WRITE_REG(hw, E1000_VMOLR(hw_queue), vmolr);
197 
198  return;
199 }

289 static int igb_vmdq_get_settings(struct net_device *netdev,
290  struct ethtool_cmd *ecmd)
291 {
292  struct igb_vmdq_adapter *vadapter = netdev_priv(netdev);
293  struct igb_adapter *adapter = vadapter->real_adapter;
294  struct e1000_hw *hw = &adapter->hw;
295  u32 status;
296 
297  if (hw->phy.media_type == e1000_media_type_copper) {
298 
299  ecmd->supported = (SUPPORTED_10baseT_Half |
300  SUPPORTED_10baseT_Full |
301  SUPPORTED_100baseT_Half |
302  SUPPORTED_100baseT_Full |
303  SUPPORTED_1000baseT_Full|
304  SUPPORTED_Autoneg |
305  SUPPORTED_TP);
306  ecmd->advertising = ADVERTISED_TP;
307 
308  if (hw->mac.autoneg == 1) {
309  ecmd->advertising |= ADVERTISED_Autoneg;
310  /* the e1000 autoneg seems to match ethtool nicely */
311  ecmd->advertising |= hw->phy.autoneg_advertised;
312  }
313 
314  ecmd->port = PORT_TP;
315  ecmd->phy_address = hw->phy.addr;
316  } else {
317  ecmd->supported = (SUPPORTED_1000baseT_Full |
318  SUPPORTED_FIBRE |
319  SUPPORTED_Autoneg);
320 
321  ecmd->advertising = (ADVERTISED_1000baseT_Full |
322  ADVERTISED_FIBRE |
323  ADVERTISED_Autoneg);
324 
325  ecmd->port = PORT_FIBRE;
326  }
327 
328  ecmd->transceiver = XCVR_INTERNAL;
329 
330  status = E1000_READ_REG(hw, E1000_STATUS);
331 
332  if (status & E1000_STATUS_LU) {
333 
334  if ((status & E1000_STATUS_SPEED_1000) ||
335  hw->phy.media_type != e1000_media_type_copper)
336  ethtool_cmd_speed_set(ecmd, SPEED_1000);
337  else if (status & E1000_STATUS_SPEED_100)
338  ethtool_cmd_speed_set(ecmd, SPEED_100);
339  else
340  ethtool_cmd_speed_set(ecmd, SPEED_10);
341 
342  if ((status & E1000_STATUS_FD) ||
343  hw->phy.media_type != e1000_media_type_copper)
344  ecmd->duplex = DUPLEX_FULL;
345  else
346  ecmd->duplex = DUPLEX_HALF;
347  } else {
348  ethtool_cmd_speed_set(ecmd, SPEED_UNKNOWN);
349  ecmd->duplex = -1;
350  }
351 
352  ecmd->autoneg = hw->mac.autoneg ? AUTONEG_ENABLE : AUTONEG_DISABLE;
353  return 0;
354 }
