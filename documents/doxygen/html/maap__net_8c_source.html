85 void *Net_getPacketBuffer(Net *net)
86 {
87  struct maap_buffer *pNew, *pLast;
88  int buffer_index;
89 
90  assert(net);
91 
92  /* See if one of the preallocated buffers is available. */
93  for (buffer_index = 0; buffer_index < NUM_BUFFERS; ++buffer_index)
94  {
95  if (net->net_buffer[buffer_index].state == BUFFER_FREE)
96  {
97  net->net_buffer[buffer_index].state = BUFFER_SUPPLIED;
98 #ifdef DEBUG_NET_MSG
99  MAAP_LOGF_DEBUG("Allocated buffer %d", buffer_index);
100 #endif
101  return (void *)(net->net_buffer[buffer_index].buffer);
102  }
103  }
104 
105  /* We need to create an overflow buffer. */
106  pNew = malloc(sizeof(struct maap_buffer));
107  if (pNew) {
108  memset(pNew, 0, sizeof(struct maap_buffer));
109  if (net->pFirst == NULL) {
110  net->pFirst = pNew;
111  } else {
112  pLast = net->pFirst;
113  while (pLast->pNext) { pLast = pLast->pNext; }
114  pLast->pNext = pNew;
115  }
116 
117  pNew->state = BUFFER_SUPPLIED;
118 #ifdef DEBUG_NET_MSG
119  MAAP_LOGF_DEBUG("Allocated buffer 0x%llx", (long long int)(uintptr_t) pNew);
120 #endif
121  return (void *)(pNew->buffer);
122  }
123 
124  assert(0);
125  return NULL;
126 }

128 int Net_queuePacket(Net *net, void *buffer)
129 {
130  struct maap_buffer *pTest;
131  int buffer_index;
132 
133  assert(net);
134 
135  /* See if the supplied buffer is one of the preallocated buffers. */
136  for (buffer_index = 0; buffer_index < NUM_BUFFERS; ++buffer_index)
137  {
138  if (net->net_buffer[buffer_index].buffer == buffer)
139  {
140  /* We found the buffer provided. */
141  assert(net->net_buffer[buffer_index].state == BUFFER_SUPPLIED);
142  net->net_buffer[buffer_index].state = BUFFER_QUEUED;
143 #ifdef DEBUG_NET_MSG
144  MAAP_LOGF_DEBUG("Queuing buffer %d", buffer_index);
145 #endif
146  return 0;
147  }
148  }
149 
150  /* See if the supplied buffer is one of the allocated buffers. */
151  for (pTest = net->pFirst; pTest != NULL; pTest = pTest->pNext)
152  {
153  if (pTest->buffer == buffer)
154  {
155  /* We found the the buffer provided. */
156  assert(pTest->state == BUFFER_SUPPLIED);

160 #endif
161  return 0;

208 int Net_freeQueuedPacket(Net *net, void *buffer)
209 {
210  struct maap_buffer *pTest, *pPrevious;
211  int buffer_index;
212 
213  assert(net);
214 
215  /* See if the supplied buffer is one of the preallocated buffers. */
216  for (buffer_index = 0; buffer_index < NUM_BUFFERS; ++buffer_index)
217  {
218  if (net->net_buffer[buffer_index].buffer == buffer)
219  {
220  /* We found the buffer provided. */
221  assert(net->net_buffer[buffer_index].state == BUFFER_SENDING);
222  net->net_buffer[buffer_index].state = BUFFER_FREE;
223 #ifdef DEBUG_NET_MSG
224  MAAP_LOGF_DEBUG("Freed buffer %d", buffer_index);
225 #endif
226  return 0;
227  }
228  }
229 
230  /* See if the supplied buffer is one of the allocated buffers. */
231  for (pPrevious = NULL, pTest = net->pFirst; pTest != NULL; pPrevious = pTest, pTest = pTest->pNext)
232  {
233  if (pTest->buffer == buffer)
234  {
235  /* We found the the buffer provided. */
236  assert(pTest->state == BUFFER_SENDING);
237  pTest->state = BUFFER_FREE;
238 #ifdef DEBUG_NET_MSG
239  MAAP_LOGF_DEBUG("Freed buffer 0x%llx", (long long int)(uintptr_t) pTest);
240 #endif
241 
242  /* Free the buffer. */
243  if (pPrevious) {
244  /* Remove this item from the middle (or end) of the queue. */
245  pPrevious->pNext = pTest->pNext;
246  } else {
247  /* Remove this item from the start of the queue. */
248  net->pFirst = pTest->pNext;
249  }
250  free(pTest);
251 
252  return 0;
253  }
254  }
255 
256  assert(0);
257  return -1;
258 }
