135 int system_layer2_multithreaded_callback::queue_tx_frame(
136  void * notification_id,
137  uint32_t notification_flag,
138  uint8_t * frame,
139  size_t mem_buf_len)
140 {
141  struct tx_data t;
142 
143  t.frame = new uint8_t[2048];
144  if (!t.frame)
145  {
146  perror("malloc");
147  exit(EXIT_FAILURE);
148  }
149  t.mem_buf_len = mem_buf_len;
150  memcpy(t.frame, frame, mem_buf_len);
151  t.notification_id = notification_id;
152  t.notification_flag = notification_flag;
153  write(tx_pipe[PIPE_WR], &t, sizeof(t));
154 
155  // Check for conditions that cause wait for completion.
156  if (wait_mgr->primed_state() &&
157  wait_mgr->match_id(notification_id) &&
158  (notification_flag == CMD_WITH_NOTIFICATION))
159  {
160  int status = 0;
161 
162  status = wait_mgr->set_active_state();
163  assert(status == 0);
164  sem_wait(waiting_sem);
165  resp_status_for_cmd = wait_mgr->get_completion_status();
166  status = wait_mgr->set_idle_state();
167  assert(status == 0);
168  }
169 
170  return 0;
171 }

214 int system_layer2_multithreaded_callback::fn_timer(struct epoll_priv * priv)
215 {
216  uint64_t timer_exp_count;
217  read(priv->fd, &timer_exp_count, sizeof(timer_exp_count));
218 
219  bool notification_id_incomplete = false;
220 
221  if (wait_mgr->active_state())
222  {
223  if (controller_ref_in_system->is_inflight_cmd_with_notification_id(wait_mgr->get_notify_id()) ||
224  controller_ref_in_system->is_active_operation_with_notification_id(wait_mgr->get_notify_id()))
225  notification_id_incomplete = true;
226  }
227 
228  // If waiting on a command to complete and the command was "inflight" prior to calling the
229  // timer tick update (ie notification_id_incomplete == true) AND after calling the timer tick
230  // update the command is no longer "inflight", timeout processing has removed it, so signal the
231  // waiting app thread.
232  controller_ref_in_system->time_tick_event();
233 
234  bool is_timeout_for_waiting_notify_id = wait_mgr->active_state() && notification_id_incomplete &&
235  !controller_ref_in_system->is_inflight_cmd_with_notification_id(wait_mgr->get_notify_id()) &&
236  !controller_ref_in_system->is_active_operation_with_notification_id(wait_mgr->get_notify_id());
237  if (is_timeout_for_waiting_notify_id)
238  {
239  int status = wait_mgr->set_completion_status(AVDECC_LIB_STATUS_TICK_TIMEOUT);
240  assert(status == 0);
241  sem_post(waiting_sem);
242  }
243 
244  return 0;
245 }

267 int system_layer2_multithreaded_callback::fn_netif(struct epoll_priv * priv)
268 {
269  uint16_t length = 0;
270  const uint8_t * rx_frame;
271  int status = 0;
272 
273  status = netif_obj_in_system->capture_frame(&rx_frame, &length);
274 
275  if (status > 0)
276  {
277  bool is_notification_id_valid = false;
278  int rx_status = -1;
279  void * notification_id = NULL;
280  uint16_t operation_id = 0;
281  bool is_operation_id_valid = false;
282 
283  controller_ref_in_system->rx_packet_event(notification_id,
284  is_notification_id_valid,
285  rx_frame,
286  length,
287  rx_status,
288  operation_id,
289  is_operation_id_valid);
290 
291  if (
292  wait_mgr->active_state() &&
293  is_notification_id_valid &&
294  wait_mgr->match_id(notification_id) &&
295  !controller_ref_in_system->is_inflight_cmd_with_notification_id(wait_mgr->get_notify_id()) &&
296  !controller_ref_in_system->is_active_operation_with_notification_id(wait_mgr->get_notify_id()))
297  {
298  int status = wait_mgr->set_completion_status(rx_status);
299  assert(status == 0);
300  sem_post(waiting_sem);
301  }
302  }
303  return 0;

319 int system_layer2_multithreaded_callback::proc_poll_loop()
320 {
321 
322  int epollfd;
323  struct epoll_event ev, epoll_evt[POLL_COUNT];
324  struct epoll_priv fd_fns[POLL_COUNT];
325 
326  epollfd = epoll_create(POLL_COUNT);
327 
328  prep_evt_desc(timerfd_create(CLOCK_MONOTONIC, 0), &system_layer2_multithreaded_callback::fn_timer_cb, &fd_fns[0], &ev);
329  epoll_ctl(epollfd, EPOLL_CTL_ADD, fd_fns[0].fd, &ev);
330 
331  prep_evt_desc(netif_obj_in_system->get_fd(), &system_layer2_multithreaded_callback::fn_netif_cb, &fd_fns[1], &ev);
332  epoll_ctl(epollfd, EPOLL_CTL_ADD, fd_fns[1].fd, &ev);
333 
334  prep_evt_desc(tx_pipe[PIPE_RD], &system_layer2_multithreaded_callback::fn_tx_cb, &fd_fns[2],
335  &ev);
336  epoll_ctl(epollfd, EPOLL_CTL_ADD, fd_fns[2].fd, &ev);
337 
338  fcntl(fd_fns[0].fd, F_SETFL, O_NONBLOCK);
339  timer_start_interval(fd_fns[0].fd);
340 
341  do
342  {
343  int i, res;
344  struct epoll_priv * priv;
345  res = epoll_wait(epollfd, epoll_evt, POLL_COUNT, -1);
346 
347  if (local_system == NULL)
348  {
349  // System has been shut down
350  sem_post(shutdown_sem);
351  return 0;
352  }
353 
354  /* exit on error */
355  if (-1 == res)
356  return -errno;
357 
358  for (i = 0; i < res; i++)
359  {
360  priv = (struct epoll_priv *)epoll_evt[i].data.ptr;
361  if (priv->fn(priv) < 0)
362  return -1;
363  }
364  } while (1);
365  return 0;
366 }
