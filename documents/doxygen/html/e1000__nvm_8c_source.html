137 static void e1000_shift_out_eec_bits(struct e1000_hw *hw, u16 data, u16 count)
138 {
139  struct e1000_nvm_info *nvm = &hw->nvm;
140  u32 eecd = E1000_READ_REG(hw, E1000_EECD);
141  u32 mask;
142 
143  DEBUGFUNC("e1000_shift_out_eec_bits");
144 
145  mask = 0x01 << (count - 1);
146  if (nvm->type == e1000_nvm_eeprom_spi)
147  eecd |= E1000_EECD_DO;
148 
149  do {
150  eecd &= ~E1000_EECD_DI;
151 
152  if (data & mask)
153  eecd |= E1000_EECD_DI;
154 
155  E1000_WRITE_REG(hw, E1000_EECD, eecd);
156  E1000_WRITE_FLUSH(hw);
157 
158  usec_delay(nvm->delay_usec);
159 
160  e1000_raise_eec_clk(hw, &eecd);
161  e1000_lower_eec_clk(hw, &eecd);
162 
163  mask >>= 1;
164  } while (mask);
165 
166  eecd &= ~E1000_EECD_DI;
167  E1000_WRITE_REG(hw, E1000_EECD, eecd);
168 }

181 static u16 e1000_shift_in_eec_bits(struct e1000_hw *hw, u16 count)
182 {
183  u32 eecd;
184  u32 i;
185  u16 data;
186 
187  DEBUGFUNC("e1000_shift_in_eec_bits");
188 
189  eecd = E1000_READ_REG(hw, E1000_EECD);
190 
191  eecd &= ~(E1000_EECD_DO | E1000_EECD_DI);
192  data = 0;
193 
194  for (i = 0; i < count; i++) {
195  data <<= 1;
196  e1000_raise_eec_clk(hw, &eecd);
197 
198  eecd = E1000_READ_REG(hw, E1000_EECD);
199 
200  eecd &= ~E1000_EECD_DI;
201  if (eecd & E1000_EECD_DO)
202  data |= 1;
203 
204  e1000_lower_eec_clk(hw, &eecd);
205  }
206 
207  return data;
208 }

218 s32 e1000_poll_eerd_eewr_done(struct e1000_hw *hw, int ee_reg)
219 {
220  u32 attempts = 100000;
221  u32 i, reg = 0;
222 
223  DEBUGFUNC("e1000_poll_eerd_eewr_done");
224 
225  for (i = 0; i < attempts; i++) {
226  if (ee_reg == E1000_NVM_POLL_READ)
227  reg = E1000_READ_REG(hw, E1000_EERD);
228  else
229  reg = E1000_READ_REG(hw, E1000_EEWR);
230 
231  if (reg & E1000_NVM_RW_REG_DONE)
232  return E1000_SUCCESS;
233 
234  usec_delay(5);
235  }
236 
237  return -E1000_ERR_NVM;

248 s32 e1000_acquire_nvm_generic(struct e1000_hw *hw)
249 {
250  u32 eecd = E1000_READ_REG(hw, E1000_EECD);
251  s32 timeout = E1000_NVM_GRANT_ATTEMPTS;
252 
253  DEBUGFUNC("e1000_acquire_nvm_generic");
254 
255  E1000_WRITE_REG(hw, E1000_EECD, eecd | E1000_EECD_REQ);
256  eecd = E1000_READ_REG(hw, E1000_EECD);
257 
258  while (timeout) {
259  if (eecd & E1000_EECD_GNT)
260  break;
261  usec_delay(5);
262  eecd = E1000_READ_REG(hw, E1000_EECD);
263  timeout--;
264  }
265 
266  if (!timeout) {
267  eecd &= ~E1000_EECD_REQ;
268  E1000_WRITE_REG(hw, E1000_EECD, eecd);
269  DEBUGOUT("Could not acquire NVM grant\n");
270  return -E1000_ERR_NVM;
271  }
272 
273  return E1000_SUCCESS;
274 }

347 static s32 e1000_ready_nvm_eeprom(struct e1000_hw *hw)
348 {
349  struct e1000_nvm_info *nvm = &hw->nvm;
350  u32 eecd = E1000_READ_REG(hw, E1000_EECD);
351  u8 spi_stat_reg;
352 
353  DEBUGFUNC("e1000_ready_nvm_eeprom");
354 
355  if (nvm->type == e1000_nvm_eeprom_spi) {
356  u16 timeout = NVM_MAX_RETRY_SPI;
357 
358  /* Clear SK and CS */
359  eecd &= ~(E1000_EECD_CS | E1000_EECD_SK);
360  E1000_WRITE_REG(hw, E1000_EECD, eecd);
361  E1000_WRITE_FLUSH(hw);
362  usec_delay(1);
363 
364  /* Read "Status Register" repeatedly until the LSB is cleared.
365  * The EEPROM will signal that the command has been completed
366  * by clearing bit 0 of the internal status register. If it's
367  * not cleared within 'timeout', then error out.
368  */
369  while (timeout) {
370  e1000_shift_out_eec_bits(hw, NVM_RDSR_OPCODE_SPI,
371  hw->nvm.opcode_bits);
372  spi_stat_reg = (u8)e1000_shift_in_eec_bits(hw, 8);
373  if (!(spi_stat_reg & NVM_STATUS_RDY_SPI))
374  break;
375 
376  usec_delay(5);
377  e1000_standby_nvm(hw);
378  timeout--;
379  }
380 
381  if (!timeout) {
382  DEBUGOUT("SPI NVM Status error\n");
383  return -E1000_ERR_NVM;
384  }
385  }
386 
387  return E1000_SUCCESS;
388 }

399 s32 e1000_read_nvm_spi(struct e1000_hw *hw, u16 offset, u16 words, u16 *data)
400 {
401  struct e1000_nvm_info *nvm = &hw->nvm;
402  u32 i = 0;
403  s32 ret_val;
404  u16 word_in;
405  u8 read_opcode = NVM_READ_OPCODE_SPI;
406 
407  DEBUGFUNC("e1000_read_nvm_spi");
408 
409  /* A check for invalid values: offset too large, too many words,
410  * and not enough words.
411  */
412  if ((offset >= nvm->word_size) || (words > (nvm->word_size - offset)) ||
413  (words == 0)) {
414  DEBUGOUT("nvm parameter(s) out of bounds\n");
415  return -E1000_ERR_NVM;
416  }
417 
418  ret_val = nvm->ops.acquire(hw);
419  if (ret_val)
420  return ret_val;
421 
422  ret_val = e1000_ready_nvm_eeprom(hw);
423  if (ret_val)
424  goto release;
425 
426  e1000_standby_nvm(hw);
427 
428  if ((nvm->address_bits == 8) && (offset >= 128))
429  read_opcode |= NVM_A8_OPCODE_SPI;
430 
431  /* Send the READ command (opcode + addr) */
432  e1000_shift_out_eec_bits(hw, read_opcode, nvm->opcode_bits);
433  e1000_shift_out_eec_bits(hw, (u16)(offset*2), nvm->address_bits);
434 
435  /* Read the data. SPI NVMs increment the address with each byte
436  * read and will roll over if reading beyond the end. This allows
437  * us to read the whole NVM from any offset
438  */
439  for (i = 0; i < words; i++) {
440  word_in = e1000_shift_in_eec_bits(hw, 16);
441  data[i] = (word_in >> 8) | (word_in << 8);
442  }
443 
444 release:
445  nvm->ops.release(hw);
446 
447  return ret_val;
448 }

459 s32 e1000_read_nvm_eerd(struct e1000_hw *hw, u16 offset, u16 words, u16 *data)
460 {
461  struct e1000_nvm_info *nvm = &hw->nvm;
462  u32 i, eerd = 0;
463  s32 ret_val = E1000_SUCCESS;
464 
465  DEBUGFUNC("e1000_read_nvm_eerd");
466 
467  /* A check for invalid values: offset too large, too many words,
468  * too many words for the offset, and not enough words.
469  */
470  if ((offset >= nvm->word_size) || (words > (nvm->word_size - offset)) ||
471  (words == 0)) {
472  DEBUGOUT("nvm parameter(s) out of bounds\n");
473  return -E1000_ERR_NVM;
474  }
475 
476  for (i = 0; i < words; i++) {
477  eerd = ((offset+i) << E1000_NVM_RW_ADDR_SHIFT) +
478  E1000_NVM_RW_REG_START;
479 
480  E1000_WRITE_REG(hw, E1000_EERD, eerd);
481  ret_val = e1000_poll_eerd_eewr_done(hw, E1000_NVM_POLL_READ);
482  if (ret_val)
483  break;
484 
485  data[i] = (E1000_READ_REG(hw, E1000_EERD) >>
486  E1000_NVM_RW_REG_DATA);
487  }
488 
489  if (ret_val)
490  DEBUGOUT1("NVM read error: %d\n", ret_val);
491 
492  return ret_val;
493 }

507 s32 e1000_write_nvm_spi(struct e1000_hw *hw, u16 offset, u16 words, u16 *data)
508 {
509  struct e1000_nvm_info *nvm = &hw->nvm;
510  s32 ret_val = -E1000_ERR_NVM;
511  u16 widx = 0;
512 
513  DEBUGFUNC("e1000_write_nvm_spi");
514 
515  /* A check for invalid values: offset too large, too many words,
516  * and not enough words.
517  */
518  if ((offset >= nvm->word_size) || (words > (nvm->word_size - offset)) ||
519  (words == 0)) {
520  DEBUGOUT("nvm parameter(s) out of bounds\n");
521  return -E1000_ERR_NVM;
522  }
523 
524  while (widx < words) {
525  u8 write_opcode = NVM_WRITE_OPCODE_SPI;
526 
527  ret_val = nvm->ops.acquire(hw);
528  if (ret_val)
529  return ret_val;
530 
531  ret_val = e1000_ready_nvm_eeprom(hw);
532  if (ret_val) {
533  nvm->ops.release(hw);
534  return ret_val;
535  }
536 
537  e1000_standby_nvm(hw);
538 
539  /* Send the WRITE ENABLE command (8 bit opcode) */
540  e1000_shift_out_eec_bits(hw, NVM_WREN_OPCODE_SPI,
541  nvm->opcode_bits);
542 
543  e1000_standby_nvm(hw);
544 
545  /* Some SPI eeproms use the 8th address bit embedded in the
546  * opcode
547  */
548  if ((nvm->address_bits == 8) && (offset >= 128))
549  write_opcode |= NVM_A8_OPCODE_SPI;
550 
551  /* Send the Write command (8-bit opcode + addr) */
552  e1000_shift_out_eec_bits(hw, write_opcode, nvm->opcode_bits);
553  e1000_shift_out_eec_bits(hw, (u16)((offset + widx) * 2),
554  nvm->address_bits);
555 
556  /* Loop to allow for up to whole page write of eeprom */
557  while (widx < words) {
558  u16 word_out = data[widx];
559  word_out = (word_out >> 8) | (word_out << 8);
560  e1000_shift_out_eec_bits(hw, word_out, 16);
561  widx++;
562 
563  if ((((offset + widx) * 2) % nvm->page_size) == 0) {
564  e1000_standby_nvm(hw);
565  break;
566  }
567  }
568  msec_delay(10);
569  nvm->ops.release(hw);
570  }
571 
572  return ret_val;
573 }

584 s32 e1000_read_pba_string_generic(struct e1000_hw *hw, u8 *pba_num,
585  u32 pba_num_size)
586 {
587  s32 ret_val;
588  u16 nvm_data;
589  u16 pba_ptr;
590  u16 offset;
591  u16 length;
592 
593  DEBUGFUNC("e1000_read_pba_string_generic");
594 
595  if ((hw->mac.type >= e1000_i210) &&
596  !e1000_get_flash_presence_i210(hw)) {
597  DEBUGOUT("Flashless no PBA string\n");
598  return -E1000_ERR_NVM_PBA_SECTION;
599  }
600 
601  if (pba_num == NULL) {
602  DEBUGOUT("PBA string buffer was null\n");
603  return -E1000_ERR_INVALID_ARGUMENT;
604  }
605 
606  ret_val = hw->nvm.ops.read(hw, NVM_PBA_OFFSET_0, 1, &nvm_data);
607  if (ret_val) {
608  DEBUGOUT("NVM Read Error\n");
609  return ret_val;
610  }
611 
612  ret_val = hw->nvm.ops.read(hw, NVM_PBA_OFFSET_1, 1, &pba_ptr);
613  if (ret_val) {
614  DEBUGOUT("NVM Read Error\n");
615  return ret_val;
616  }
617 
618  /* if nvm_data is not ptr guard the PBA must be in legacy format which
619  * means pba_ptr is actually our second data word for the PBA number
620  * and we can decode it into an ascii string
621  */
622  if (nvm_data != NVM_PBA_PTR_GUARD) {
623  DEBUGOUT("NVM PBA number is not stored as string\n");
624 
625  /* make sure callers buffer is big enough to store the PBA */
626  if (pba_num_size < E1000_PBANUM_LENGTH) {
627  DEBUGOUT("PBA string buffer too small\n");
628  return E1000_ERR_NO_SPACE;
629  }
630 
631  /* extract hex string from data and pba_ptr */
632  pba_num[0] = (nvm_data >> 12) & 0xF;
633  pba_num[1] = (nvm_data >> 8) & 0xF;
634  pba_num[2] = (nvm_data >> 4) & 0xF;
635  pba_num[3] = nvm_data & 0xF;
636  pba_num[4] = (pba_ptr >> 12) & 0xF;
637  pba_num[5] = (pba_ptr >> 8) & 0xF;
638  pba_num[6] = '-';
639  pba_num[7] = 0;
640  pba_num[8] = (pba_ptr >> 4) & 0xF;
641  pba_num[9] = pba_ptr & 0xF;
642 
643  /* put a null character on the end of our string */
644  pba_num[10] = '\0';
645 
646  /* switch all the data but the '-' to hex char */
647  for (offset = 0; offset < 10; offset++) {
648  if (pba_num[offset] < 0xA)
649  pba_num[offset] += '0';
650  else if (pba_num[offset] < 0x10)
651  pba_num[offset] += 'A' - 0xA;
652  }
653 
654  return E1000_SUCCESS;
655  }
656 
657  ret_val = hw->nvm.ops.read(hw, pba_ptr, 1, &length);
658  if (ret_val) {
659  DEBUGOUT("NVM Read Error\n");
660  return ret_val;
661  }
662 
663  if (length == 0xFFFF || length == 0) {
664  DEBUGOUT("NVM PBA number section invalid length\n");
665  return -E1000_ERR_NVM_PBA_SECTION;
666  }
667  /* check if pba_num buffer is big enough */
668  if (pba_num_size < (((u32)length * 2) - 1)) {
669  DEBUGOUT("PBA string buffer too small\n");
670  return -E1000_ERR_NO_SPACE;
671  }
672 
673  /* trim pba length from start of string */
674  pba_ptr++;
675  length--;
676 
677  for (offset = 0; offset < length; offset++) {
678  ret_val = hw->nvm.ops.read(hw, pba_ptr + offset, 1, &nvm_data);
679  if (ret_val) {
680  DEBUGOUT("NVM Read Error\n");
681  return ret_val;
682  }
683  pba_num[offset * 2] = (u8)(nvm_data >> 8);
684  pba_num[(offset * 2) + 1] = (u8)(nvm_data & 0xFF);
685  }
686  pba_num[offset * 2] = '\0';
687 
688  return E1000_SUCCESS;
689 }

758 s32 e1000_read_mac_addr_generic(struct e1000_hw *hw)
759 {
760  u32 rar_high;
761  u32 rar_low;
762  u16 i;
763 
764  rar_high = E1000_READ_REG(hw, E1000_RAH(0));
765  rar_low = E1000_READ_REG(hw, E1000_RAL(0));
766 
767  for (i = 0; i < E1000_RAL_MAC_ADDR_LEN; i++)
768  hw->mac.perm_addr[i] = (u8)(rar_low >> (i*8));
769 
770  for (i = 0; i < E1000_RAH_MAC_ADDR_LEN; i++)

870 void e1000_get_fw_version(struct e1000_hw *hw, struct e1000_fw_version *fw_vers)
871 {
872  u16 eeprom_verh, eeprom_verl, etrack_test, fw_version;
873  u8 q, hval, rem, result;
874  u16 comb_verh, comb_verl, comb_offset;
875 
876  memset(fw_vers, 0, sizeof(struct e1000_fw_version));
877 
878  /* basic eeprom version numbers, bits used vary by part and by tool
879  * used to create the nvm images */
880  /* Check which data format we have */
881  switch (hw->mac.type) {
882  case e1000_i211:
883  e1000_read_invm_version(hw, fw_vers);
884  return;
885  case e1000_82575:
886  case e1000_82576:
887  case e1000_82580:
888  hw->nvm.ops.read(hw, NVM_ETRACK_HIWORD, 1, &etrack_test);
889  /* Use this format, unless EETRACK ID exists,
890  * then use alternate format
891  */
892  if ((etrack_test & NVM_MAJOR_MASK) != NVM_ETRACK_VALID) {
893  hw->nvm.ops.read(hw, NVM_VERSION, 1, &fw_version);
894  fw_vers->eep_major = (fw_version & NVM_MAJOR_MASK)
895  >> NVM_MAJOR_SHIFT;
896  fw_vers->eep_minor = (fw_version & NVM_MINOR_MASK)
897  >> NVM_MINOR_SHIFT;
898  fw_vers->eep_build = (fw_version & NVM_IMAGE_ID_MASK);
899  goto etrack_id;
900  }
901  break;
902  case e1000_i210:
903  if (!(e1000_get_flash_presence_i210(hw))) {
904  e1000_read_invm_version(hw, fw_vers);
905  return;
906  }
907  /* fall through */
908  case e1000_i350:
909  hw->nvm.ops.read(hw, NVM_ETRACK_HIWORD, 1, &etrack_test);
910  /* find combo image version */
911  hw->nvm.ops.read(hw, NVM_COMB_VER_PTR, 1, &comb_offset);
912  if ((comb_offset != 0x0) &&
913  (comb_offset != NVM_VER_INVALID)) {
914 
915  hw->nvm.ops.read(hw, (NVM_COMB_VER_OFF + comb_offset
916  + 1), 1, &comb_verh);
917  hw->nvm.ops.read(hw, (NVM_COMB_VER_OFF + comb_offset),
918  1, &comb_verl);
919 
920  /* get Option Rom version if it exists and is valid */
921  if ((comb_verh && comb_verl) &&
922  ((comb_verh != NVM_VER_INVALID) &&
923  (comb_verl != NVM_VER_INVALID))) {
924 
925  fw_vers->or_valid = true;
926  fw_vers->or_major =
927  comb_verl >> NVM_COMB_VER_SHFT;
928  fw_vers->or_build =
929  (comb_verl << NVM_COMB_VER_SHFT)
930  | (comb_verh >> NVM_COMB_VER_SHFT);
931  fw_vers->or_patch =
932  comb_verh & NVM_COMB_VER_MASK;
933  }
934  }
935  break;
936  default:
937  hw->nvm.ops.read(hw, NVM_ETRACK_HIWORD, 1, &etrack_test);
938  return;
939  }
940  hw->nvm.ops.read(hw, NVM_VERSION, 1, &fw_version);
941  fw_vers->eep_major = (fw_version & NVM_MAJOR_MASK)
942  >> NVM_MAJOR_SHIFT;
943 
944  /* check for old style version format in newer images*/
945  if ((fw_version & NVM_NEW_DEC_MASK) == 0x0) {
946  eeprom_verl = (fw_version & NVM_COMB_VER_MASK);
947  } else {
948  eeprom_verl = (fw_version & NVM_MINOR_MASK)
949  >> NVM_MINOR_SHIFT;
950  }
951  /* Convert minor value to hex before assigning to output struct
952  * Val to be converted will not be higher than 99, per tool output
953  */
954  q = eeprom_verl / NVM_HEX_CONV;
955  hval = q * NVM_HEX_TENS;
956  rem = eeprom_verl % NVM_HEX_CONV;
957  result = hval + rem;
958  fw_vers->eep_minor = result;
959 
960 etrack_id:
961  if ((etrack_test & NVM_MAJOR_MASK) == NVM_ETRACK_VALID) {
962  hw->nvm.ops.read(hw, NVM_ETRACK_WORD, 1, &eeprom_verl);
963  hw->nvm.ops.read(hw, (NVM_ETRACK_WORD + 1), 1, &eeprom_verh);
964  fw_vers->etrack_id = (eeprom_verh << NVM_ETRACK_SHIFT)
965  | eeprom_verl;
966  } else if ((etrack_test & NVM_ETRACK_VALID) == 0) {
967  hw->nvm.ops.read(hw, NVM_ETRACK_WORD, 1, &eeprom_verh);
968  hw->nvm.ops.read(hw, (NVM_ETRACK_WORD + 1), 1, &eeprom_verl);
969  fw_vers->etrack_id = (eeprom_verh << NVM_ETRACK_SHIFT) |
970  eeprom_verl;
971  }
972 }
