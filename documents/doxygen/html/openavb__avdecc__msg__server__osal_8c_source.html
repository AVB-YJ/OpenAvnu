43 static void socketClose(int h)
44 {
45  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC_MSG);
46 
47  if (h < 0 || h >= POLL_FD_COUNT) {
48  AVB_LOG_ERROR("Closing socket; invalid handle");
49  }
50  else {
51  if (h != AVB_AVDECC_LISTEN_FDS) {
52  openavbAvdeccMsgSrvrCloseClientConnection(h);
53  }
54  close(fds[h].fd);
55  fds[h].fd = SOCK_INVALID;
56  fds[h].events = 0;
57  }
58 
59  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
60 }

62 static bool openavbAvdeccMsgSrvrSendToClient(int h, openavbAvdeccMessage_t *msg)
63 {
64  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC_MSG);
65 
66  if (h < 0 || h >= POLL_FD_COUNT) {
67  AVB_LOG_ERROR("Sending message; invalid handle");
68  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
69  return FALSE;
70  }
71  if (!msg) {
72  AVB_LOG_ERROR("Sending message; invalid argument passed");
73  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
74  return FALSE;
75  }
76 
77  int csock = fds[h].fd;
78  if (csock == SOCK_INVALID) {
79  AVB_LOG_ERROR("Socket closed unexpectedly");
80  return FALSE;
81  }
82 
83  ssize_t nWrite = write(csock, msg, OPENAVB_AVDECC_MSG_LEN);
84  AVB_LOGF_VERBOSE("Sent message, len=%zu, nWrite=%zu", OPENAVB_AVDECC_MSG_LEN, nWrite);
85  if (nWrite < OPENAVB_AVDECC_MSG_LEN) {
86  if (nWrite < 0) {
87  AVB_LOGF_ERROR("Failed to write socket: %s", strerror(errno));
88  }
89  else if (nWrite == 0) {
90  AVB_LOG_ERROR("Socket closed unexpectedly");
91  }
92  else {
93  AVB_LOG_ERROR("Socket write too short");
94  }
95  socketClose(h);
96  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC_MSG);
97  return FALSE;
98  }
99 
100  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
101  return TRUE;
102 }

104 bool openavbAvdeccMsgServerOpen(void)
105 {
106  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC_MSG);
107  int i;
108 
109  // Perform the base initialization.
110  openavbAvdeccMsgInitialize();
111 
112  for (i=0; i < POLL_FD_COUNT; i++) {
113  fds[i].fd = SOCK_INVALID;
114  fds[i].events = 0;
115  }
116 
117  lsock = socket(AF_UNIX, SOCK_STREAM, 0);
118  if (lsock < 0) {
119  AVB_LOGF_ERROR("Failed to open socket: %s", strerror(errno));
120  goto error;
121  }
122  // serverAddr is file static
123  serverAddr.sun_family = AF_UNIX;
124  snprintf(serverAddr.sun_path, UNIX_PATH_MAX, AVB_AVDECC_MSG_UNIX_PATH);
125 
126  // try remove old socket
127  if (unlink(serverAddr.sun_path) == -1 && errno != ENOENT) {
128  AVB_LOGF_ERROR("Failed to remove %s: %s", serverAddr.sun_path, strerror(errno));
129  }
130 
131  int rslt = bind(lsock, (struct sockaddr*)&serverAddr, sizeof(struct sockaddr_un));
132  if (rslt != 0) {
133  AVB_LOGF_ERROR("Failed to create %s: %s", serverAddr.sun_path, strerror(errno));
134  AVB_LOG_WARNING("** If AVDECC Msg process crashed, run the cleanup script **");
135  goto error;
136  }
137 
138  rslt = listen(lsock, 5);
139  if (rslt != 0) {
140  AVB_LOGF_ERROR("Failed to listen on socket: %s", strerror(errno));
141  goto error;
142  }
143  AVB_LOGF_DEBUG("Listening on socket: %s", serverAddr.sun_path);
144 
145  fds[AVB_AVDECC_LISTEN_FDS].fd = lsock;
146  fds[AVB_AVDECC_LISTEN_FDS].events = POLLIN;
147 
148  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
149  return TRUE;
150 
151  error:
152  if (lsock >= 0) {
153  close(lsock);
154  lsock = -1;
155  }
156  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
157  return FALSE;
158 }

160 void openavbAvdeccMsgSrvrService(void)
161 {
162  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC_MSG);
163  struct sockaddr_un addrClient;
164  socklen_t lenAddr;
165  int i, j;
166  int csock;
167 
168  int nfds = POLL_FD_COUNT;
169  int pRet;
170 
171  AVB_LOG_VERBOSE("Waiting for event...");
172  pRet = poll(fds, nfds, 1000);
173 
174  if (pRet == 0) {
175  AVB_LOG_VERBOSE("poll timeout");
176  }
177  else if (pRet < 0) {
178  if (errno == EINTR) {
179  AVB_LOG_VERBOSE("Poll interrupted");
180  }
181  else {
182  AVB_LOGF_ERROR("Poll error: %s", strerror(errno));
183  }
184  }
185  else {
186  AVB_LOGF_VERBOSE("Poll returned %d events", pRet);
187  for (i=0; i<nfds; i++) {
188  if (fds[i].revents != 0) {
189  AVB_LOGF_VERBOSE("%d sock=%d, event=0x%x, revent=0x%x", i, fds[i].fd, fds[i].events, fds[i].revents);
190 
191  if (i == AVB_AVDECC_LISTEN_FDS) {
192  // listen sock - indicates new connection from client
193  lenAddr = sizeof(addrClient);
194  csock = accept(lsock, (struct sockaddr*)&addrClient, &lenAddr);
195  if (csock < 0) {
196  AVB_LOGF_ERROR("Failed to accept connection: %s", strerror(errno));
197  }
198  else {
199  for (j = 0; j < POLL_FD_COUNT; j++) {
200  if (fds[j].fd == SOCK_INVALID) {
201  fds[j].fd = csock;
202  fds[j].events = POLLIN;
203  break;
204  }
205  }
206  if (j >= POLL_FD_COUNT) {
207  AVB_LOG_ERROR("Too many client connections");
208  close(csock);
209  }
210  }
211 
212  AVB_LOG_INFO("New AVDECC Msg client connection detected");
213  }
214  else {
215  csock = fds[i].fd;
216  openavbAvdeccMessage_t msgBuf;
217  memset(&msgBuf, 0, OPENAVB_AVDECC_MSG_LEN);
218  ssize_t nRead = read(csock, &msgBuf, OPENAVB_AVDECC_MSG_LEN);
219  AVB_LOGF_VERBOSE("Socket read h=%d,fd=%d: read=%zu, expect=%zu", i, csock, nRead, OPENAVB_AVDECC_MSG_LEN);
220 
221  if (nRead < OPENAVB_AVDECC_MSG_LEN) {
222  // sock closed
223  if (nRead == 0) {
224  AVB_LOGF_DEBUG("Socket closed, h=%d", i);
225  }
226  else if (nRead < 0) {
227  AVB_LOGF_ERROR("Socket read, h=%d: %s", i, strerror(errno));
228  }
229  else {
230  AVB_LOGF_ERROR("Short read, h=%d", i);
231  }
232  socketClose(i);
233  }
234  else {
235  // got a message
236  if (!openavbAvdeccMsgSrvrReceiveFromClient(i, &msgBuf)) {
237  AVB_LOG_ERROR("Failed to handle message");
238  socketClose(i);
239  }
240  }
241  }
242  }
243  }
244  }
245  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
246 }

248 void openavbAvdeccMsgServerClose(void)
249 {
250  AVB_TRACE_ENTRY(AVB_TRACE_AVDECC_MSG);
251  int i;
252  for (i = 0; i < POLL_FD_COUNT; i++) {
253  if (fds[i].fd != SOCK_INVALID) {
254  socketClose(i);
255  }
256  }
257  if (lsock != SOCK_INVALID) {
258  close(lsock);
259  }
260 
261  if (unlink(serverAddr.sun_path) != 0) {
262  AVB_LOGF_ERROR("Failed to unlink %s: %s", serverAddr.sun_path, strerror(errno));
263  }
264 
265  // Perform the base cleanup.
266  openavbAvdeccMsgCleanup();
267 
268  AVB_TRACE_EXIT(AVB_TRACE_AVDECC_MSG);
269 }
