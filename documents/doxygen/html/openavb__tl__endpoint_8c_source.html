69 /* Talker Listener thread function that talks primarily with the endpoint
70  */
71 void* openavbTLThreadFn(void *pv)
72 {
73  AVB_TRACE_ENTRY(AVB_TRACE_TL);
74 
75  tl_state_t *pTLState = (tl_state_t *)pv;
76 
77  while (pTLState->bRunning) {
78  AVB_TRACE_LINE(AVB_TRACE_TL_DETAIL);
79 
80  int endpointHandle = openavbEptClntOpenSrvrConnection(pTLState);
81 
82  if (endpointHandle == AVB_ENDPOINT_HANDLE_INVALID) {
83  // error connecting to endpoint, already logged
84  }
85  else {
86  pTLState->endpointHandle = endpointHandle;
87 
88  // Validate the AVB version for TL and Endpoint are the same before continuing
89  pTLState->AVBVerState = OPENAVB_TL_AVB_VER_UNKNOWN;
90  pTLState->bConnected = openavbEptClntRequestVersionFromServer(pTLState->endpointHandle);
91  while (pTLState->bRunning && pTLState->bConnected && pTLState->AVBVerState == OPENAVB_TL_AVB_VER_UNKNOWN) {
92  // Check for endpoint version message. Timeout in 50 msec.
93  if (!openavbEptClntService(pTLState->endpointHandle, 50)) {
94  AVB_LOG_WARNING("Lost connection to endpoint, will retry");
95  pTLState->bConnected = FALSE;
96  pTLState->endpointHandle = 0;
97  }
98  }
99  if (pTLState->AVBVerState == OPENAVB_TL_AVB_VER_INVALID) {
100  AVB_LOG_ERROR("AVB core version is different than Endpoint AVB core version. Streams will not be started. Will reconnect to the endpoint and check again.");
101  }
102 
103  if (pTLState->bConnected && pTLState->AVBVerState == OPENAVB_TL_AVB_VER_VALID) {
104  if (pTLState->cfg.role == AVB_ROLE_TALKER) {
105  openavbTLRunTalker(pTLState);
106  }
107  else {
108  openavbTLRunListener(pTLState);
109  }
110  }
111 
112  // Close the endpoint connection. unless connection already gone in which case the socket could already be reused.
113  if (pTLState->bConnected) {
114  openavbEptClntCloseSrvrConnection(endpointHandle);
115  pTLState->bConnected = FALSE;
116  pTLState->endpointHandle = 0;
117  }
118  }
119 
120  if (pTLState->bRunning) {
121  SLEEP_MSEC(1);
122  }
123  }
124 
125  THREAD_JOINABLE(pTLState->TLThread);
126 
127  AVB_TRACE_EXIT(AVB_TRACE_TL);
128  return NULL;
129 }

132 tl_handle_t TLHandleListGet(int endpointHandle)
133 {
134  AVB_TRACE_ENTRY(AVB_TRACE_TL);
135 
136  if (!endpointHandle || !gTLHandleList) {
137  AVB_TRACE_EXIT(AVB_TRACE_TL);
138  return NULL;
139  }
140 
141  TL_LOCK();
142  int i1;
143  for (i1 = 0; i1 < gMaxTL; i1++) {
144  if (gTLHandleList[i1]) {
145  tl_state_t *pTLState = (tl_state_t *)gTLHandleList[i1];
146  if (pTLState->endpointHandle == endpointHandle) {
147  TL_UNLOCK();
148  AVB_TRACE_EXIT(AVB_TRACE_TL);
149  return pTLState;
150  }
151  }
152  }
153  TL_UNLOCK();
154  AVB_TRACE_EXIT(AVB_TRACE_TL);
155  return NULL;
156 }
