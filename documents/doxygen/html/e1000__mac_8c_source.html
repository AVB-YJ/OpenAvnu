301 s32 e1000_check_alt_mac_addr_generic(struct e1000_hw *hw)
302 {
303  u32 i;
304  s32 ret_val;
305  u16 offset, nvm_alt_mac_addr_offset, nvm_data;
306  u8 alt_mac_addr[ETH_ADDR_LEN];
307 
308  DEBUGFUNC("e1000_check_alt_mac_addr_generic");
309 
310  ret_val = hw->nvm.ops.read(hw, NVM_COMPAT, 1, &nvm_data);
311  if (ret_val)
312  return ret_val;
313 
314  /* Alternate MAC address is handled by the option ROM for 82580
315  * and newer. SW support not required.
316  */
317  if (hw->mac.type >= e1000_82580)
318  return E1000_SUCCESS;
319 
320  ret_val = hw->nvm.ops.read(hw, NVM_ALT_MAC_ADDR_PTR, 1,
321  &nvm_alt_mac_addr_offset);
322  if (ret_val) {
323  DEBUGOUT("NVM Read Error\n");
324  return ret_val;
325  }
326 
327  if ((nvm_alt_mac_addr_offset == 0xFFFF) ||
328  (nvm_alt_mac_addr_offset == 0x0000))
329  /* There is no Alternate MAC Address */
330  return E1000_SUCCESS;
331 
332  if (hw->bus.func == E1000_FUNC_1)
333  nvm_alt_mac_addr_offset += E1000_ALT_MAC_ADDRESS_OFFSET_LAN1;
334  if (hw->bus.func == E1000_FUNC_2)
335  nvm_alt_mac_addr_offset += E1000_ALT_MAC_ADDRESS_OFFSET_LAN2;
336 
337  if (hw->bus.func == E1000_FUNC_3)
338  nvm_alt_mac_addr_offset += E1000_ALT_MAC_ADDRESS_OFFSET_LAN3;
339  for (i = 0; i < ETH_ADDR_LEN; i += 2) {
340  offset = nvm_alt_mac_addr_offset + (i >> 1);
341  ret_val = hw->nvm.ops.read(hw, offset, 1, &nvm_data);
342  if (ret_val) {
343  DEBUGOUT("NVM Read Error\n");
344  return ret_val;
345  }
346 
347  alt_mac_addr[i] = (u8)(nvm_data & 0xFF);
348  alt_mac_addr[i + 1] = (u8)(nvm_data >> 8);
349  }
350 
351  /* if multicast bit is set, the alternate address will not be used */
352  if (alt_mac_addr[0] & 0x01) {
353  DEBUGOUT("Ignoring Alternate Mac Address with MC bit set\n");
354  return E1000_SUCCESS;
355  }
356 
357  /* We have a valid alternate MAC address, and we want to treat it the
358  * same as the normal permanent MAC address stored by the HW into the
359  * RAR. Do this by mapping this address into RAR0.
360  */
361  hw->mac.ops.rar_set(hw, alt_mac_addr, 0);
362 
363  return E1000_SUCCESS;
364 }

375 static int e1000_rar_set_generic(struct e1000_hw *hw, u8 *addr, u32 index)
376 {
377  u32 rar_low, rar_high;
378 
379  DEBUGFUNC("e1000_rar_set_generic");
380 
381  /* HW expects these in little endian so we reverse the byte order
382  * from network order (big endian) to little endian
383  */
384  rar_low = ((u32) addr[0] | ((u32) addr[1] << 8) |
385  ((u32) addr[2] << 16) | ((u32) addr[3] << 24));
386 
387  rar_high = ((u32) addr[4] | ((u32) addr[5] << 8));
388 
389  /* If MAC address zero, no need to set the AV bit */
390  if (rar_low || rar_high)
391  rar_high |= E1000_RAH_AV;
392 
393  /* Some bridges will combine consecutive 32-bit writes into
394  * a single burst write, which will malfunction on some parts.
395  * The flushes avoid this.
396  */
397  E1000_WRITE_REG(hw, E1000_RAL(index), rar_low);
398  E1000_WRITE_FLUSH(hw);
399  E1000_WRITE_REG(hw, E1000_RAH(index), rar_high);
400  E1000_WRITE_FLUSH(hw);
401 
402  return E1000_SUCCESS;
403 }

413 u32 e1000_hash_mc_addr_generic(struct e1000_hw *hw, u8 *mc_addr)
414 {
415  u32 hash_value, hash_mask;
416  u8 bit_shift = 0;
417 
418  DEBUGFUNC("e1000_hash_mc_addr_generic");
419 
420  /* Register count multiplied by bits per register */
421  hash_mask = (hw->mac.mta_reg_count * 32) - 1;
422 
423  /* For a mc_filter_type of 0, bit_shift is the number of left-shifts
424  * where 0xFF would still fall within the hash mask.
425  */
426  while (hash_mask >> bit_shift != 0xFF)
427  bit_shift++;
428 
429  /* The portion of the address that is used for the hash table
430  * is determined by the mc_filter_type setting.
431  * The algorithm is such that there is a total of 8 bits of shifting.
432  * The bit_shift for a mc_filter_type of 0 represents the number of
433  * left-shifts where the MSB of mc_addr[5] would still fall within
434  * the hash_mask. Case 0 does this exactly. Since there are a total
435  * of 8 bits of shifting, then mc_addr[4] will shift right the
436  * remaining number of bits. Thus 8 - bit_shift. The rest of the
437  * cases are a variation of this algorithm...essentially raising the
438  * number of bits to shift mc_addr[5] left, while still keeping the
439  * 8-bit shifting total.
440  *
441  * For example, given the following Destination MAC Address and an
442  * mta register count of 128 (thus a 4096-bit vector and 0xFFF mask),
443  * we can see that the bit_shift for case 0 is 4. These are the hash
444  * values resulting from each mc_filter_type...
445  * [0] [1] [2] [3] [4] [5]
446  * 01 AA 00 12 34 56
447  * LSB MSB
448  *
449  * case 0: hash_value = ((0x34 >> 4) | (0x56 << 4)) & 0xFFF = 0x563
450  * case 1: hash_value = ((0x34 >> 3) | (0x56 << 5)) & 0xFFF = 0xAC6
451  * case 2: hash_value = ((0x34 >> 2) | (0x56 << 6)) & 0xFFF = 0x163
452  * case 3: hash_value = ((0x34 >> 0) | (0x56 << 8)) & 0xFFF = 0x634
453  */
454  switch (hw->mac.mc_filter_type) {
455  default:
456  case 0:
457  break;
458  case 1:
459  bit_shift += 1;
460  break;
461  case 2:
462  bit_shift += 2;
463  break;
464  case 3:
465  bit_shift += 4;
466  break;
467  }
468 
469  hash_value = hash_mask & (((mc_addr[4] >> (8 - bit_shift)) |
470  (((u16) mc_addr[5]) << bit_shift)));
471 
472  return hash_value;
473 }

484 void e1000_update_mc_addr_list_generic(struct e1000_hw *hw,
485  u8 *mc_addr_list, u32 mc_addr_count)
486 {
487  u32 hash_value, hash_bit, hash_reg;
488  int i;
489 
490  DEBUGFUNC("e1000_update_mc_addr_list_generic");
491 
492  /* clear mta_shadow */
493  memset(&hw->mac.mta_shadow, 0, sizeof(hw->mac.mta_shadow));
494 
495  /* update mta_shadow from mc_addr_list */
496  for (i = 0; (u32) i < mc_addr_count; i++) {
497  hash_value = e1000_hash_mc_addr_generic(hw, mc_addr_list);
498 
499  hash_reg = (hash_value >> 5) & (hw->mac.mta_reg_count - 1);
500  hash_bit = hash_value & 0x1F;
501 
502  hw->mac.mta_shadow[hash_reg] |= (1 << hash_bit);
503  mc_addr_list += (ETH_ADDR_LEN);
504  }
505 
506  /* replace the entire MTA table */
507  for (i = hw->mac.mta_reg_count - 1; i >= 0; i--)
508  E1000_WRITE_REG_ARRAY(hw, E1000_MTA, i, hw->mac.mta_shadow[i]);
509  E1000_WRITE_FLUSH(hw);
510 }

607 {

609  s32 ret_val;
610  bool link;
611 
612  DEBUGFUNC("e1000_check_for_copper_link");
613 
614  /* We only want to go out to the PHY registers to see if Auto-Neg
615  * has completed and/or if our link status has changed. The
616  * get_link_status flag is set upon receiving a Link Status
617  * Change or Rx Sequence Error interrupt.
618  */
619  if (!mac->get_link_status)
620  return E1000_SUCCESS;
621 
622  /* First we want to see if the MII Status Register reports
623  * link. If so, then we want to get the current speed/duplex
624  * of the PHY.
625  */
626  ret_val = e1000_phy_has_link_generic(hw, 1, 0, &link);
627  if (ret_val)
628  return ret_val;
629 
630  if (!link)
631  return E1000_SUCCESS; /* No link detected */
632 
633  mac->get_link_status = false;
634 
635  /* Check if there was DownShift, must be checked
636  * immediately after link-up
637  */
638  e1000_check_downshift_generic(hw);
639 
640  /* If we are forcing speed/duplex, then we simply return since
641  * we have already determined whether we have link or not.
642  */
643  if (!mac->autoneg)
644  return -E1000_ERR_CONFIG;
645 
646  /* Auto-Neg is enabled. Auto Speed Detection takes care
647  * of MAC speed/duplex configuration. So we only need to
648  * configure Collision Distance in the MAC.
649  */
650  mac->ops.config_collision_dist(hw);
651 
652  /* Configure Flow Control now that Auto-Neg has completed.
653  * First, we need to restore the desired flow control
654  * settings because we may have had to re-autoneg with a
655  * different link partner.
656  */
657  ret_val = e1000_config_fc_after_link_up_generic(hw);
658  if (ret_val)
659  DEBUGOUT("Error configuring flow control\n");
660 
661  return ret_val;
662 }

671 s32 e1000_check_for_fiber_link_generic(struct e1000_hw *hw)
672 {
673  struct e1000_mac_info *mac = &hw->mac;
674  u32 rxcw;
675  u32 ctrl;
676  u32 status;
677  s32 ret_val;
678 
679  DEBUGFUNC("e1000_check_for_fiber_link_generic");
680 
681  ctrl = E1000_READ_REG(hw, E1000_CTRL);
682  status = E1000_READ_REG(hw, E1000_STATUS);
683  rxcw = E1000_READ_REG(hw, E1000_RXCW);
684 
685  /* If we don't have link (auto-negotiation failed or link partner
686  * cannot auto-negotiate), the cable is plugged in (we have signal),
687  * and our link partner is not trying to auto-negotiate with us (we
688  * are receiving idles or data), we need to force link up. We also
689  * need to give auto-negotiation time to complete, in case the cable
690  * was just plugged in. The autoneg_failed flag does this.
691  */
692  /* (ctrl & E1000_CTRL_SWDPIN1) == 1 == have signal */
693  if ((ctrl & E1000_CTRL_SWDPIN1) && !(status & E1000_STATUS_LU) &&
694  !(rxcw & E1000_RXCW_C)) {
695  if (!mac->autoneg_failed) {
696  mac->autoneg_failed = true;
697  return E1000_SUCCESS;
698  }
699  DEBUGOUT("NOT Rx'ing /C/, disable AutoNeg and force link.\n");
700 
701  /* Disable auto-negotiation in the TXCW register */
702  E1000_WRITE_REG(hw, E1000_TXCW, (mac->txcw & ~E1000_TXCW_ANE));
703 
704  /* Force link-up and also force full-duplex. */
705  ctrl = E1000_READ_REG(hw, E1000_CTRL);
706  ctrl |= (E1000_CTRL_SLU | E1000_CTRL_FD);
707  E1000_WRITE_REG(hw, E1000_CTRL, ctrl);
708 
709  /* Configure Flow Control after forcing link up. */

738 s32 e1000_check_for_serdes_link_generic(struct e1000_hw *hw)
739 {
740  struct e1000_mac_info *mac = &hw->mac;
741  u32 rxcw;
742  u32 ctrl;
743  u32 status;
744  s32 ret_val;
745 
746  DEBUGFUNC("e1000_check_for_serdes_link_generic");
747 
748  ctrl = E1000_READ_REG(hw, E1000_CTRL);
749  status = E1000_READ_REG(hw, E1000_STATUS);
750  rxcw = E1000_READ_REG(hw, E1000_RXCW);
751 
752  /* If we don't have link (auto-negotiation failed or link partner
753  * cannot auto-negotiate), and our link partner is not trying to
754  * auto-negotiate with us (we are receiving idles or data),
755  * we need to force link up. We also need to give auto-negotiation
756  * time to complete.
757  */

776  if (ret_val) {

778  return ret_val;
779  }

781  /* If we are forcing link and we are receiving /C/ ordered
782  * sets, re-enable auto-negotiation in the TXCW register
783  * and disable forced link in the Device Control register
784  * in an attempt to auto-negotiate with our link partner.
785  */

787  E1000_WRITE_REG(hw, E1000_TXCW, mac->txcw);
788  E1000_WRITE_REG(hw, E1000_CTRL, (ctrl & ~E1000_CTRL_SLU));

791  } else if (!(E1000_TXCW_ANE & E1000_READ_REG(hw, E1000_TXCW))) {

793  * link status based on MAC synchronization for internal
794  * serdes media type.
795  */

804  } else {

806  DEBUGOUT("SERDES: Link down - force failed.\n");

808  }

812  if (status & E1000_STATUS_LU) {

815  rxcw = E1000_READ_REG(hw, E1000_RXCW);
816  if (rxcw & E1000_RXCW_SYNCH) {
817  if (!(rxcw & E1000_RXCW_IV)) {
818  mac->serdes_has_link = true;
819  DEBUGOUT("SERDES: Link up - autoneg completed successfully.\n");
820  } else {
821  mac->serdes_has_link = false;
822  DEBUGOUT("SERDES: Link down - invalid codewords detected in autoneg.\n");
823  }
824  } else {
825  mac->serdes_has_link = false;
826  DEBUGOUT("SERDES: Link down - no sync.\n");
827  }
828  } else {
829  mac->serdes_has_link = false;
830  DEBUGOUT("SERDES: Link down - autoneg failed\n");
831  }
832  }
833 
834  return E1000_SUCCESS;

954 static s32 e1000_commit_fc_settings_generic(struct e1000_hw *hw)
955 {
956  struct e1000_mac_info *mac = &hw->mac;
957  u32 txcw;
958 
959  DEBUGFUNC("e1000_commit_fc_settings_generic");
960 
961  /* Check for a software override of the flow control settings, and
962  * setup the device accordingly. If auto-negotiation is enabled, then
963  * software will have to set the "PAUSE" bits to the correct value in
964  * the Transmit Config Word Register (TXCW) and re-start auto-
965  * negotiation. However, if auto-negotiation is disabled, then
966  * software will have to manually configure the two flow control enable
967  * bits in the CTRL register.
968  *
969  * The possible values of the "fc" parameter are:
970  * 0: Flow control is completely disabled
971  * 1: Rx flow control is enabled (we can receive pause frames,
972  * but not send pause frames).
973  * 2: Tx flow control is enabled (we can send pause frames but we
974  * do not support receiving pause frames).
975  * 3: Both Rx and Tx flow control (symmetric) are enabled.
976  */
977  switch (hw->fc.current_mode) {
978  case e1000_fc_none:
979  /* Flow control completely disabled by a software over-ride. */
980  txcw = (E1000_TXCW_ANE | E1000_TXCW_FD);
981  break;
982  case e1000_fc_rx_pause:
983  /* Rx Flow control is enabled and Tx Flow control is disabled
984  * by a software over-ride. Since there really isn't a way to
985  * advertise that we are capable of Rx Pause ONLY, we will
986  * advertise that we support both symmetric and asymmetric Rx
987  * PAUSE. Later, we will disable the adapter's ability to send
988  * PAUSE frames.
989  */
990  txcw = (E1000_TXCW_ANE | E1000_TXCW_FD | E1000_TXCW_PAUSE_MASK);
991  break;
992  case e1000_fc_tx_pause:
993  /* Tx Flow control is enabled, and Rx Flow control is disabled,
994  * by a software over-ride.
995  */
996  txcw = (E1000_TXCW_ANE | E1000_TXCW_FD | E1000_TXCW_ASM_DIR);
997  break;
998  case e1000_fc_full:
999  /* Flow control (both Rx and Tx) is enabled by a software
1000  * over-ride.
1001  */
1002  txcw = (E1000_TXCW_ANE | E1000_TXCW_FD | E1000_TXCW_PAUSE_MASK);
1003  break;
1004  default:
1005  DEBUGOUT("Flow control param set incorrectly\n");
1006  return -E1000_ERR_CONFIG;
1007  break;
1008  }
1009 
1010  E1000_WRITE_REG(hw, E1000_TXCW, txcw);
1011  mac->txcw = txcw;
1012 
1013  return E1000_SUCCESS;
1014 }

1247 s32 e1000_config_fc_after_link_up_generic(struct e1000_hw *hw)
1248 {
1249  struct e1000_mac_info *mac = &hw->mac;
1250  s32 ret_val = E1000_SUCCESS;
1251  u32 pcs_status_reg, pcs_adv_reg, pcs_lp_ability_reg, pcs_ctrl_reg;
1252  u16 mii_status_reg, mii_nway_adv_reg, mii_nway_lp_ability_reg;
1253  u16 speed, duplex;
1254 
1255  DEBUGFUNC("e1000_config_fc_after_link_up_generic");
1256 
1257  /* Check for the case where we have fiber media and auto-neg failed
1258  * so we had to force link. In this case, we need to force the
1259  * configuration of the MAC to match the "fc" parameter.
1260  */
1261  if (mac->autoneg_failed) {
1262  if (hw->phy.media_type == e1000_media_type_fiber ||
1263  hw->phy.media_type == e1000_media_type_internal_serdes)
1264  ret_val = e1000_force_mac_fc_generic(hw);
1265  } else {
1266  if (hw->phy.media_type == e1000_media_type_copper)
1267  ret_val = e1000_force_mac_fc_generic(hw);
1268  }
1269 
1270  if (ret_val) {
1271  DEBUGOUT("Error forcing flow control settings\n");
1272  return ret_val;
1273  }
1274 
1275  /* Check for the case where we have copper media and auto-neg is
1276  * enabled. In this case, we need to check and see if Auto-Neg
1277  * has completed, and if so, how the PHY and link partner has
1278  * flow control configured.
1279  */
1280  if ((hw->phy.media_type == e1000_media_type_copper) && mac->autoneg) {
1281  /* Read the MII Status Register and check to see if AutoNeg
1282  * has completed. We read this twice because this reg has
1283  * some "sticky" (latched) bits.
1284  */
1285  ret_val = hw->phy.ops.read_reg(hw, PHY_STATUS, &mii_status_reg);
1286  if (ret_val)
1287  return ret_val;
1288  ret_val = hw->phy.ops.read_reg(hw, PHY_STATUS, &mii_status_reg);
1289  if (ret_val)
1290  return ret_val;
1291 
1292  if (!(mii_status_reg & MII_SR_AUTONEG_COMPLETE)) {
1293  DEBUGOUT("Copper PHY and Auto Neg has not completed.\n");
1294  return ret_val;
1295  }
1296 
1297  /* The AutoNeg process has completed, so we now need to
1298  * read both the Auto Negotiation Advertisement
1299  * Register (Address 4) and the Auto_Negotiation Base
1300  * Page Ability Register (Address 5) to determine how
1301  * flow control was negotiated.
1302  */
1303  ret_val = hw->phy.ops.read_reg(hw, PHY_AUTONEG_ADV,
1304  &mii_nway_adv_reg);
1305  if (ret_val)
1306  return ret_val;
1307  ret_val = hw->phy.ops.read_reg(hw, PHY_LP_ABILITY,
1308  &mii_nway_lp_ability_reg);
1309  if (ret_val)
1310  return ret_val;
1311 
1312  /* Two bits in the Auto Negotiation Advertisement Register
1313  * (Address 4) and two bits in the Auto Negotiation Base
1314  * Page Ability Register (Address 5) determine flow control
1315  * for both the PHY and the link partner. The following
1316  * table, taken out of the IEEE 802.3ab/D6.0 dated March 25,
1317  * 1999, describes these PAUSE resolution bits and how flow
1318  * control is determined based upon these settings.
1319  * NOTE: DC = Don't Care
1320  *
1321  * LOCAL DEVICE | LINK PARTNER
1322  * PAUSE | ASM_DIR | PAUSE | ASM_DIR | NIC Resolution
1323  *-------|---------|-------|---------|--------------------
1324  * 0 | 0 | DC | DC | e1000_fc_none
1325  * 0 | 1 | 0 | DC | e1000_fc_none
1326  * 0 | 1 | 1 | 0 | e1000_fc_none
1327  * 0 | 1 | 1 | 1 | e1000_fc_tx_pause
1328  * 1 | 0 | 0 | DC | e1000_fc_none
1329  * 1 | DC | 1 | DC | e1000_fc_full
1330  * 1 | 1 | 0 | 0 | e1000_fc_none
1331  * 1 | 1 | 0 | 1 | e1000_fc_rx_pause
1332  *
1333  * Are both PAUSE bits set to 1? If so, this implies
1334  * Symmetric Flow Control is enabled at both ends. The
1335  * ASM_DIR bits are irrelevant per the spec.
1336  *
1337  * For Symmetric Flow Control:
1338  *
1339  * LOCAL DEVICE | LINK PARTNER
1340  * PAUSE | ASM_DIR | PAUSE | ASM_DIR | Result
1341  *-------|---------|-------|---------|--------------------
1342  * 1 | DC | 1 | DC | E1000_fc_full
1343  *
1344  */
1345  if ((mii_nway_adv_reg & NWAY_AR_PAUSE) &&
1346  (mii_nway_lp_ability_reg & NWAY_LPAR_PAUSE)) {
1347  /* Now we need to check if the user selected Rx ONLY
1348  * of pause frames. In this case, we had to advertise
1349  * FULL flow control because we could not advertise Rx
1350  * ONLY. Hence, we must now check to see if we need to
1351  * turn OFF the TRANSMISSION of PAUSE frames.
1352  */
1353  if (hw->fc.requested_mode == e1000_fc_full) {
1354  hw->fc.current_mode = e1000_fc_full;
1355  DEBUGOUT("Flow Control = FULL.\n");
1356  } else {
1357  hw->fc.current_mode = e1000_fc_rx_pause;
1358  DEBUGOUT("Flow Control = Rx PAUSE frames only.\n");
1359  }
1360  }
1361  /* For receiving PAUSE frames ONLY.
1362  *
1363  * LOCAL DEVICE | LINK PARTNER
1364  * PAUSE | ASM_DIR | PAUSE | ASM_DIR | Result
1365  *-------|---------|-------|---------|--------------------
1366  * 0 | 1 | 1 | 1 | e1000_fc_tx_pause
1367  */
1368  else if (!(mii_nway_adv_reg & NWAY_AR_PAUSE) &&
1369  (mii_nway_adv_reg & NWAY_AR_ASM_DIR) &&
1370  (mii_nway_lp_ability_reg & NWAY_LPAR_PAUSE) &&
1371  (mii_nway_lp_ability_reg & NWAY_LPAR_ASM_DIR)) {
1372  hw->fc.current_mode = e1000_fc_tx_pause;
1373  DEBUGOUT("Flow Control = Tx PAUSE frames only.\n");
1374  }
1375  /* For transmitting PAUSE frames ONLY.
1376  *
1377  * LOCAL DEVICE | LINK PARTNER
1378  * PAUSE | ASM_DIR | PAUSE | ASM_DIR | Result
1379  *-------|---------|-------|---------|--------------------
1380  * 1 | 1 | 0 | 1 | e1000_fc_rx_pause
1381  */
1382  else if ((mii_nway_adv_reg & NWAY_AR_PAUSE) &&
1383  (mii_nway_adv_reg & NWAY_AR_ASM_DIR) &&
1384  !(mii_nway_lp_ability_reg & NWAY_LPAR_PAUSE) &&
1385  (mii_nway_lp_ability_reg & NWAY_LPAR_ASM_DIR)) {
1386  hw->fc.current_mode = e1000_fc_rx_pause;
1387  DEBUGOUT("Flow Control = Rx PAUSE frames only.\n");
1388  } else {
1389  /* Per the IEEE spec, at this point flow control
1390  * should be disabled.
1391  */
1392  hw->fc.current_mode = e1000_fc_none;
1393  DEBUGOUT("Flow Control = NONE.\n");
1394  }
1395 
1396  /* Now we need to do one last check... If we auto-
1397  * negotiated to HALF DUPLEX, flow control should not be
1398  * enabled per IEEE 802.3 spec.
1399  */
1400  ret_val = mac->ops.get_link_up_info(hw, &speed, &duplex);
1401  if (ret_val) {
1402  DEBUGOUT("Error getting link speed and duplex\n");
1403  return ret_val;
1404  }
1405 
1406  if (duplex == HALF_DUPLEX)
1407  hw->fc.current_mode = e1000_fc_none;
1408 
1409  /* Now we call a subroutine to actually force the MAC
1410  * controller to use the correct flow control settings.
1411  */
1412  ret_val = e1000_force_mac_fc_generic(hw);
1413  if (ret_val) {
1414  DEBUGOUT("Error forcing flow control settings\n");
1415  return ret_val;
1416  }
1417  }
1418 
1419  /* Check for the case where we have SerDes media and auto-neg is
1420  * enabled. In this case, we need to check and see if Auto-Neg
1421  * has completed, and if so, how the PHY and link partner has
1422  * flow control configured.
1423  */
1424  if ((hw->phy.media_type == e1000_media_type_internal_serdes) &&
1425  mac->autoneg) {
1426  /* Read the PCS_LSTS and check to see if AutoNeg
1427  * has completed.
1428  */
1429  pcs_status_reg = E1000_READ_REG(hw, E1000_PCS_LSTAT);
1430 
1431  if (!(pcs_status_reg & E1000_PCS_LSTS_AN_COMPLETE)) {
1432  DEBUGOUT("PCS Auto Neg has not completed.\n");
1433  return ret_val;
1434  }
1435 
1436  /* The AutoNeg process has completed, so we now need to
1437  * read both the Auto Negotiation Advertisement
1438  * Register (PCS_ANADV) and the Auto_Negotiation Base
1439  * Page Ability Register (PCS_LPAB) to determine how
1440  * flow control was negotiated.
1441  */
1442  pcs_adv_reg = E1000_READ_REG(hw, E1000_PCS_ANADV);
1443  pcs_lp_ability_reg = E1000_READ_REG(hw, E1000_PCS_LPAB);
1444 
1445  /* Two bits in the Auto Negotiation Advertisement Register
1446  * (PCS_ANADV) and two bits in the Auto Negotiation Base
1447  * Page Ability Register (PCS_LPAB) determine flow control
1448  * for both the PHY and the link partner. The following
1449  * table, taken out of the IEEE 802.3ab/D6.0 dated March 25,
1450  * 1999, describes these PAUSE resolution bits and how flow
1451  * control is determined based upon these settings.
1452  * NOTE: DC = Don't Care
1453  *
1454  * LOCAL DEVICE | LINK PARTNER
1455  * PAUSE | ASM_DIR | PAUSE | ASM_DIR | NIC Resolution
1456  *-------|---------|-------|---------|--------------------
1457  * 0 | 0 | DC | DC | e1000_fc_none
1458  * 0 | 1 | 0 | DC | e1000_fc_none
1459  * 0 | 1 | 1 | 0 | e1000_fc_none
1460  * 0 | 1 | 1 | 1 | e1000_fc_tx_pause
1461  * 1 | 0 | 0 | DC | e1000_fc_none
1462  * 1 | DC | 1 | DC | e1000_fc_full
1463  * 1 | 1 | 0 | 0 | e1000_fc_none
1464  * 1 | 1 | 0 | 1 | e1000_fc_rx_pause
1465  *
1466  * Are both PAUSE bits set to 1? If so, this implies
1467  * Symmetric Flow Control is enabled at both ends. The
1468  * ASM_DIR bits are irrelevant per the spec.
1469  *
1470  * For Symmetric Flow Control:
1471  *
1472  * LOCAL DEVICE | LINK PARTNER
1473  * PAUSE | ASM_DIR | PAUSE | ASM_DIR | Result
1474  *-------|---------|-------|---------|--------------------
1475  * 1 | DC | 1 | DC | e1000_fc_full
1476  *
1477  */
1478  if ((pcs_adv_reg & E1000_TXCW_PAUSE) &&
1479  (pcs_lp_ability_reg & E1000_TXCW_PAUSE)) {
1480  /* Now we need to check if the user selected Rx ONLY
1481  * of pause frames. In this case, we had to advertise
1482  * FULL flow control because we could not advertise Rx
1483  * ONLY. Hence, we must now check to see if we need to
1484  * turn OFF the TRANSMISSION of PAUSE frames.
1485  */
1486  if (hw->fc.requested_mode == e1000_fc_full) {
1487  hw->fc.current_mode = e1000_fc_full;
1488  DEBUGOUT("Flow Control = FULL.\n");
1489  } else {
1490  hw->fc.current_mode = e1000_fc_rx_pause;
1491  DEBUGOUT("Flow Control = Rx PAUSE frames only.\n");
1492  }
1493  }
1494  /* For receiving PAUSE frames ONLY.
1495  *
1496  * LOCAL DEVICE | LINK PARTNER
1497  * PAUSE | ASM_DIR | PAUSE | ASM_DIR | Result
1498  *-------|---------|-------|---------|--------------------
1499  * 0 | 1 | 1 | 1 | e1000_fc_tx_pause
1500  */
1501  else if (!(pcs_adv_reg & E1000_TXCW_PAUSE) &&
1502  (pcs_adv_reg & E1000_TXCW_ASM_DIR) &&
1503  (pcs_lp_ability_reg & E1000_TXCW_PAUSE) &&
1504  (pcs_lp_ability_reg & E1000_TXCW_ASM_DIR)) {
1505  hw->fc.current_mode = e1000_fc_tx_pause;
1506  DEBUGOUT("Flow Control = Tx PAUSE frames only.\n");
1507  }
1508  /* For transmitting PAUSE frames ONLY.
1509  *
1510  * LOCAL DEVICE | LINK PARTNER
1511  * PAUSE | ASM_DIR | PAUSE | ASM_DIR | Result
1512  *-------|---------|-------|---------|--------------------
1513  * 1 | 1 | 0 | 1 | e1000_fc_rx_pause
1514  */
1515  else if ((pcs_adv_reg & E1000_TXCW_PAUSE) &&
1516  (pcs_adv_reg & E1000_TXCW_ASM_DIR) &&
1517  !(pcs_lp_ability_reg & E1000_TXCW_PAUSE) &&
1518  (pcs_lp_ability_reg & E1000_TXCW_ASM_DIR)) {
1519  hw->fc.current_mode = e1000_fc_rx_pause;
1520  DEBUGOUT("Flow Control = Rx PAUSE frames only.\n");
1521  } else {
1522  /* Per the IEEE spec, at this point flow control
1523  * should be disabled.
1524  */
1525  hw->fc.current_mode = e1000_fc_none;
1526  DEBUGOUT("Flow Control = NONE.\n");
1527  }
1528 
1529  /* Now we call a subroutine to actually force the MAC
1530  * controller to use the correct flow control settings.
1531  */

1533  pcs_ctrl_reg |= E1000_PCS_LCTL_FORCE_FCTRL;
1534  E1000_WRITE_REG(hw, E1000_PCS_LCTL, pcs_ctrl_reg);
1535

1537  if (ret_val) {
1538  DEBUGOUT("Error forcing flow control settings\n");
1539  return ret_val;
1540  }
1541  }
1542 
1543  return E1000_SUCCESS;
